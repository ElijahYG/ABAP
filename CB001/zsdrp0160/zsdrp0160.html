*    SELECT * INTO TABLE LT_VBAK_2 FROM VBAK
*      FOR ALL ENTRIES IN LT_VBFA_2
*      WHERE VBELN = LT_VBFA_2-VBELV.
*  ENDIF.

*取相关字段
  CLEAR LW_ALV.
  LOOP AT IT_ALV INTO LW_ALV.
    CLEAR LS_KNA1.
    SORT LT_KNA1 BY KUNNR ASCENDING.
    READ TABLE LT_KNA1 INTO LS_KNA1 WITH KEY KUNNR  = LW_ALV-KUNAG
                                             BINARY SEARCH.
    IF SY-SUBRC = 0.

      READ TABLE LT_T005U INTO LS_T005U WITH KEY BLAND  = LS_KNA1-REGIO.
      IF SY-SUBRC = 0.
        LW_ALV-REGIO = LS_T005U-BLAND. "地区编码
        LW_ALV-ORT01 = LS_T005U-BEZEI. "加盟门店区域

        LW_ALV-AUART   = LW_VBAK-AUART. "凭证类型
      ENDIF.
    ENDIF.
    MODIFY IT_ALV FROM LW_ALV TRANSPORTING REGIO ORT01 AUART NAME1.
  ENDLOOP.

  SORT IT_ALV BY VBELN KUNAG ASCENDING.

  LOOP AT IT_ALV INTO LW_ALV.
