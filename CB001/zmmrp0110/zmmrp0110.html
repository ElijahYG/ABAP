FORM FM_CHECK_DATA.

    L_COUNT = L_COUNT + 1.

  ENDIF.

    L_COUNT = L_COUNT + 1.
  ENDIF.

    L_COUNT = L_COUNT + 1.
  ENDIF.
* 计数器等于0表示没有输入搜索条件

  ENDIF.
ENDFORM.
*&---------------------------------------------------------------------*
*&取数 可以将文字描述带出来
*&---------------------------------------------------------------------*
FORM FM_GET_DATA.

  SELECT *
    INTO CORRESPONDING FIELDS OF TABLE GT_OUT_TAB
        FROM MSEG  AS A
  INNER JOIN MKPF  AS G ON A~MBLNR EQ G~MBLNR AND A~MJAHR EQ G~MJAHR
  INNER JOIN T001W AS B ON A~WERKS EQ B~WERKS
  INNER JOIN T001L AS C ON A~WERKS EQ C~WERKS AND A~LGORT EQ C~LGORT
  INNER JOIN MAKT  AS E ON A~MATNR EQ E~MATNR AND E~SPRAS EQ '1'
  INNER JOIN T157E AS D ON A~BWART EQ D~BWART AND D~SPRAS EQ '1'

       WHERE
          A~MATNR      IN S_MATNR  AND
          A~MBLNR      IN S_MBLNR  AND
          A~BUDAT_MKPF IN S_BUDAT  AND
          A~BWART      EQ '311'.

*按条件区分同一字段
  LOOP AT GT_OUT_TAB INTO WA_TEM_TAB.
    IF WA_TEM_TAB-SHKZG = 'H'.
      WA_TEM_TAB-LGORT_FH = WA_TEM_TAB-LGORT.
      WA_TEM_TAB-LGOBE1   = WA_TEM_TAB-LGOBE.

    ELSEIF WA_TEM_TAB-SHKZG = 'S'.
      WA_TEM_TAB-LGORT_SH = WA_TEM_TAB-LGORT.
      WA_TEM_TAB-LGOBE2   = WA_TEM_TAB-LGOBE.
    ENDIF.
    MODIFY GT_OUT_TAB FROM WA_TEM_TAB TRANSPORTING LGORT_FH LGORT_SH LGOBE1 LGOBE2.
  ENDLOOP.

  SORT GT_OUT_TAB BY MBLNR LGORT_FH LGORT_SH."EBELP.
ENDFORM.
*&---------------------------------------------------------------------*
*&排重
*&---------------------------------------------------------------------*
FORM FM_DEL_DATA.

  DATA:WA_TMP_TAB TYPE TY_OUT_TAB.
  MOVE-CORRESPONDING GT_OUT_TAB TO IT_OUT_TAB.

  SORT IT_OUT_TAB BY MBLNR BUDAT_MKPF SHKZG.

  DELETE ADJACENT DUPLICATES FROM IT_OUT_TAB COMPARING MBLNR BUDAT_MKPF SHKZG.
*将ALV中2列合并为一列
  LOOP AT IT_OUT_TAB INTO WA_OUT_TAB WHERE SHKZG ='H'.

    CLEAR:WA_TMP_TAB.
    READ TABLE IT_OUT_TAB INTO WA_TMP_TAB WITH KEY MBLNR = WA_OUT_TAB-MBLNR
                                                   SHKZG = 'S'.
    IF SY-SUBRC = 0.
      WA_ALV-LGOBE1     = WA_OUT_TAB-LGOBE1.
      WA_ALV-LGOBE2     = WA_TMP_TAB-LGOBE2 .
      WA_ALV-MBLNR      = WA_OUT_TAB-MBLNR.
      WA_ALV-BUDAT_MKPF = WA_OUT_TAB-BUDAT_MKPF.
      WA_ALV-MJAHR      = WA_TMP_TAB-MJAHR .
      APPEND WA_ALV TO IT_ALV.

    ENDIF.
  ENDLOOP.
ENDFORM.

  INTO CORRESPONDING FIELDS OF TABLE T_TAB
