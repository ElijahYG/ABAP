FORM FM_GET_DOC.
  DATA:
    LT_DOC    TYPE ZFIRP0121_DOC OCCURS 0.
* 取已归档数据
  SELECT *
    FROM ZFIRP0121_DOC
    INTO TABLE LT_DOC
   WHERE ZGROUP     IN S_ZGROUP
     AND BUDAT      IN S_BUDAT
     AND DOC_NUMBER IN S_DOCNUM
     AND BELNR      IN S_BELNR
    .
* 加入到OUT表
  MOVE-CORRESPONDING LT_DOC TO GT_OUT_TAB.

ENDFORM.
*&---------------------------------------------------------------------*
*&获取BKPF中符合条件的数据
*&---------------------------------------------------------------------*
FORM FM_GET_BKPF.
  FIELD-SYMBOLS <FS>  TYPE TY_OUT_TAB.
  DATA:
    BEGIN OF LW_BKPF,
      BUKRS TYPE BKPF-BUKRS,
      BELNR TYPE BKPF-BELNR,
      GJAHR TYPE BKPF-GJAHR,

      BLART TYPE BKPF-BLART,

    END OF LW_BKPF,
    LT_BKPF    LIKE LW_BKPF OCCURS 0,
    LT_OUT_TAB LIKE GT_OUT_TAB.

*取符合条件的BKPF数据
  LT_OUT_TAB = GT_OUT_TAB.
  SORT LT_OUT_TAB BY BUKRS BELNR GJAHR.
  DELETE ADJACENT DUPLICATES FROM LT_OUT_TAB COMPARING BUKRS BELNR GJAHR.
  IF LT_OUT_TAB IS NOT INITIAL.
    SELECT
      BUKRS
      BELNR
      GJAHR
      BLART
   FROM BKPF
      INTO TABLE LT_BKPF
      FOR ALL ENTRIES IN LT_OUT_TAB
       WHERE BUKRS = LT_OUT_TAB-BUKRS
         AND BELNR = LT_OUT_TAB-BELNR
         AND GJAHR = LT_OUT_TAB-GJAHR.
  ENDIF.
*将数据填充GT_OUT_TAB
  SORT LT_BKPF BY BUKRS BELNR GJAHR.
  LOOP AT GT_OUT_TAB ASSIGNING <FS>.
    CLEAR LW_BKPF.
    READ TABLE LT_BKPF INTO LW_BKPF WITH KEY  BUKRS = <FS>-BUKRS
                                              BELNR = <FS>-BELNR
                                              GJAHR = <FS>-GJAHR
                                              BINARY SEARCH.
    IF SY-SUBRC = 0.
      <FS>-BLART = LW_BKPF-BLART.
    ENDIF.
  ENDLOOP.
ENDFORM.

*&---------------------------------------------------------------------*
*&获取faglflexa中符合条件的数据
*&---------------------------------------------------------------------*
FORM FM_GET_FAGA.
  FIELD-SYMBOLS <FS> TYPE TY_OUT_TAB.
  DATA:
    BEGIN OF LW_FAGA,
      BUKRS TYPE BSEG-BUKRS,
      BELNR TYPE BSEG-BELNR,
      GJAHR TYPE BSEG-GJAHR,

      BUZEI TYPE FAGLFLEXA-DOCLN,
      RCNTR TYPE FAGLFLEXA-RCNTR,
      PRCTR TYPE BSEG-PRCTR,
      SAKNR TYPE BSEG-SAKNR,
      DMBTR TYPE FAGLFLEXA-HSL,

    END OF LW_FAGA,
    LT_FAGA    LIKE LW_FAGA OCCURS 0,
    LT_OUT_TAB LIKE  GT_OUT_TAB.

**取符合条件的faglflexa数据
  LT_OUT_TAB = GT_OUT_TAB.
  SORT LT_OUT_TAB BY BUKRS BELNR GJAHR.
  DELETE ADJACENT DUPLICATES FROM LT_OUT_TAB COMPARING BUKRS BELNR GJAHR.
  IF LT_OUT_TAB IS NOT INITIAL.
    SELECT RBUKRS
           DOCNR
           RYEAR
           DOCLN

           RCNTR
           PRCTR
           RACCT
           HSL
     FROM FAGLFLEXA
     INTO TABLE LT_FAGA
     FOR ALL ENTRIES IN LT_OUT_TAB
     WHERE RBUKRS = LT_OUT_TAB-BUKRS
      AND     DOCNR  = LT_OUT_TAB-BELNR
      AND     RYEAR  = LT_OUT_TAB-GJAHR.
  ENDIF.
*将数据填充GT_OUT_TAB
  SORT LT_FAGA BY BUKRS BELNR GJAHR.
  LOOP AT GT_OUT_TAB ASSIGNING <FS>.
    CLEAR LW_FAGA.
    READ TABLE LT_FAGA INTO LW_FAGA WITH KEY  BUKRS = <FS>-BUKRS
                                              BELNR = <FS>-BELNR
                                              GJAHR = <FS>-GJAHR
                                              BINARY SEARCH.
    IF SY-SUBRC = 0.
      <FS>-PRCTR  = LW_FAGA-PRCTR.
      <FS>-KOSTL  = LW_FAGA-RCNTR.
      <FS>-SAKNR  = LW_FAGA-SAKNR.
      <FS>-DMBTR  = ABS( LW_FAGA-DMBTR ).

    ENDIF.
  ENDLOOP.
ENDFORM.
