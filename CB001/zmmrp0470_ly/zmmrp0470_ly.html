FORM FM_GET_DATA.

  IF P_1 EQ ABAP_TRUE.    "ZB00

    SELECT A~KAPPL
           A~KSCHL
           A~DATBI
           A~DATAB
           A~KBSTAT
           A~KNUMH
           A~EX_SITE
           A~EXTWG
           A~RAUBE
           A~EX_GJS
           A~KFRST
           B~KBETR
      FROM A909 AS A
      INNER JOIN KONP AS B ON A~KNUMH EQ B~KNUMH
      INTO CORRESPONDING FIELDS OF TABLE IT_OUT_TAB
      WHERE A~EX_SITE IN S_WERKS
        AND A~EXTWG   IN S_EXTWG
        AND A~RAUBE   IN S_RAUBE
        AND A~EX_GJS  IN S_GJS
        AND A~DATAB   <= S_DATEB-LOW
        AND A~DATBI   >= S_DATEB-LOW
        .

  ELSEIF P_2 EQ ABAP_TRUE. "ZKSG

    SELECT A~KAPPL
           A~KSCHL
           A~DATBI
           A~DATAB
           A~KNUMH
           A~ZXSKGFD
           B~KBETR
      FROM A911 AS A
      INNER JOIN KONP AS B ON A~KNUMH EQ B~KNUMH
      INTO CORRESPONDING FIELDS OF TABLE IT_OUT_TAB
      WHERE A~ZXSKGFD IN S_ZKGFD
        AND A~DATAB   <= S_DATEB-LOW
        AND A~DATBI   >= S_DATEB-LOW
        AND A~KSCHL   EQ 'ZKSG'
        .

  ENDIF.

  IF IT_OUT_TAB IS INITIAL.
    MESSAGE S000(ZCB) WITH '未找到相应数据' DISPLAY LIKE 'E'.
    LEAVE LIST-PROCESSING.
  ENDIF.

ENDFORM.

FORM FM_SET_VISIBLE USING L_NUM. "0 表示不可编辑,1 标识可编辑
  DATA:
    LW_STYTLE  TYPE LVC_S_STYL.

  LOOP AT IT_OUT_TAB ASSIGNING <WA_OUT_TAB>.
    CLEAR <WA_OUT_TAB>-SEL.
    IF L_NUM = '1' AND <WA_OUT_TAB>-FLAG EQ ABAP_TRUE.
      CLEAR LW_STYTLE.
      LW_STYTLE-FIELDNAME = 'KBETR'.
      LW_STYTLE-STYLE = CL_GUI_ALV_GRID=>MC_STYLE_ENABLED.
      CLEAR <WA_OUT_TAB>-STYTLE.
      INSERT LW_STYTLE INTO TABLE <WA_OUT_TAB>-STYTLE.
      <WA_OUT_TAB>-SEL = ABAP_TRUE.       "将系统标记转移到自定义标记中,确保在计算中范围不会变.
    ELSEIF L_NUM = '0'.
      CLEAR LW_STYTLE.
      LW_STYTLE-FIELDNAME = 'KBETR'.
      LW_STYTLE-STYLE = CL_GUI_ALV_GRID=>MC_STYLE_DISABLED.
      CLEAR <WA_OUT_TAB>-STYTLE.
      INSERT LW_STYTLE INTO TABLE <WA_OUT_TAB>-STYTLE.
      <WA_OUT_TAB>-SEL = ABAP_FALSE.      "清除选择标记,如果不清楚会导致第二次编辑时没选中也更新.
    ENDIF.
  ENDLOOP.

  IF L_NUM = '1'.
    GL_CHECK = ABAP_TRUE.
  ELSE.
    GL_CHECK = ABAP_FALSE.
  ENDIF.

ENDFORM.

*&---------------------------------------------------------------------*
*& 更新数据
*&---------------------------------------------------------------------*
FORM FM_UPDATE.

* 1.定义结构与内表.
  DATA: LT_TABLE_OLD TYPE STANDARD TABLE OF STRU_OUT,
*        LT_TABLE_DEL TYPE STANDARD TABLE OF STRU_OUT,
*        LT_TABLE_ADD TYPE STANDARD TABLE OF STRU_OUT,
        LT_TABLE_MOD TYPE STANDARD TABLE OF STRU_OUT
*        LT_TABLE_NCH TYPE STANDARD TABLE OF STRU_OUT
        .
*
  DATA  LS_OUT_TAB   TYPE                   STRU_OUT.

* 2.检查ALV内表是否有更改
*  CALL FUNCTION 'CTVB_COMPARE_TABLES'     "比较内表差异
*    EXPORTING
*      TABLE_OLD  = LT_TABLE_OLD
*      TABLE_NEW  = IT_OUT_TAB
*      KEY_LENGTH = 35
**     IF_SORTED  =
*    IMPORTING
*      TABLE_DEL  = LT_TABLE_DEL
*      TABLE_ADD  = LT_TABLE_ADD
*      TABLE_MOD  = LT_TABLE_MOD
*      NO_CHANGES = LT_TABLE_NCH.
  IF P_1 EQ ABAP_TRUE.
    SORT IT_TABLE_OLD BY EX_SITE EXTWG RAUBE EX_GJS.
    SORT IT_OUT_TAB   BY EX_SITE EXTWG RAUBE EX_GJS.
    LOOP AT IT_OUT_TAB INTO LS_OUT_TAB.
      READ TABLE IT_TABLE_OLD ASSIGNING <WA_OUT_TAB> WITH KEY EX_SITE = LS_OUT_TAB-EX_SITE
                                                              EXTWG   = LS_OUT_TAB-EXTWG
                                                              RAUBE   = LS_OUT_TAB-RAUBE
                                                              EX_GJS  = LS_OUT_TAB-EX_GJS
                                                              BINARY SEARCH.
      IF SY-SUBRC = 0.
        IF LS_OUT_TAB-KBETR <> <WA_OUT_TAB>-KBETR.
          APPEND LS_OUT_TAB TO LT_TABLE_MOD.
        ENDIF.
      ENDIF.
    ENDLOOP.

  ELSE.
    SORT LT_TABLE_OLD BY ZXSKGFD.
    SORT IT_OUT_TAB   BY ZXSKGFD.
    LOOP AT IT_OUT_TAB INTO LS_OUT_TAB.
      READ TABLE IT_TABLE_OLD ASSIGNING <WA_OUT_TAB> WITH KEY ZXSKGFD = LS_OUT_TAB-ZXSKGFD
                                                              BINARY SEARCH.
      IF SY-SUBRC = 0.
        IF LS_OUT_TAB-KBETR <> <WA_OUT_TAB>-KBETR.
          APPEND LS_OUT_TAB TO LT_TABLE_MOD.
        ENDIF.
      ENDIF.
    ENDLOOP.

  ENDIF.

  IF LT_TABLE_MOD IS INITIAL.
    MESSAGE S000(ZCB) WITH '没有价格被更改,请更改后再保存' DISPLAY LIKE 'E'.
    GL_CHECK = 'E'.
    RETURN.
  ELSE.
    CLEAR IT_TABLE_OLD.
    IT_TABLE_OLD = LT_TABLE_MOD.
  ENDIF.

* 3.如果有更改执行BDC更新
  PERFORM FM_BDC.

* 4.更新成功后更改字段可编辑状态
  PERFORM FM_SET_VISIBLE USING '0'.

* 5.更新编辑状态检查变量
  GL_EDIT = ABAP_FALSE.

ENDFORM.
