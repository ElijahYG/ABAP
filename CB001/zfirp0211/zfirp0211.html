FORM FRM_AUTH_CHECK .
  DATA: BEGIN OF LS_WERKS,
          WERKS TYPE WERKS_D,
        END OF LS_WERKS,
        LT_WERKS LIKE TABLE OF LS_WERKS .

* 按地点检查权限
  " 先按用户输入的地点查询
  SELECT WERKS
    INTO TABLE LT_WERKS
    FROM T001W
    WHERE WERKS IN S_WERKS .
  IF SY-SUBRC <> 0 .
    MESSAGE S000(ZCB) WITH '指定的地点不存在！' DISPLAY LIKE 'E' .
    STOP .
  ENDIF.

  REFRESH: GRT_WERKS .
  LOOP AT LT_WERKS INTO LS_WERKS .
    AUTHORITY-CHECK OBJECT 'M_MSEG_WMB'
             ID 'ACTVT' FIELD '03'
             ID 'WERKS' FIELD LS_WERKS-WERKS .
    IF SY-SUBRC = 0.
      CLEAR: GRS_WERKS .
      GRS_WERKS-SIGN = 'I' .
      GRS_WERKS-OPTION = 'EQ' .
      GRS_WERKS-LOW = LS_WERKS-WERKS .
      GRS_WERKS-HIGH = '' .
      APPEND GRS_WERKS TO GRT_WERKS .
    ENDIF.
  ENDLOOP.

  IF GRT_WERKS[] IS INITIAL .
    MESSAGE E000(ZCB) WITH '您没有指定地点的权限！'  .
    STOP .
  ENDIF.
ENDFORM.

*&---------------------------------------------------------------------*
*&      Form  FRM_GET_BILLING
*&---------------------------------------------------------------------*
FORM FRM_GET_BILLING .
  SELECT
*     K~FKDAT,
     P~WERKS,
     P~VBELN,
     P~POSNR,
     P~MATNR,
     P~CHARG,
     K~EX_POSNUM,
     K~EX_POSTYP,
         "非以旧换新小票的
         CASE WHEN ( K~EX_POSTYP <> '20' AND K~EX_POSTYP <> '21' ) AND P~SHKZG =  'X' THEN  ABS( P~KZWI3 ) * -1
              WHEN ( K~EX_POSTYP <> '20' AND K~EX_POSTYP <> '21' ) AND P~SHKZG <> 'X' THEN  ABS( P~KZWI3 ) END AS NORM_KZWI3,
         CASE WHEN ( K~EX_POSTYP <> '20' AND K~EX_POSTYP <> '21' ) AND P~SHKZG =  'X' THEN  ABS( P~NETWR ) * -1
              WHEN ( K~EX_POSTYP <> '20' AND K~EX_POSTYP <> '21' ) AND P~SHKZG <> 'X' THEN  ABS( P~NETWR ) END AS NORM_NETWR,
         CASE WHEN ( K~EX_POSTYP <> '20' AND K~EX_POSTYP <> '21' ) AND P~SHKZG =  'X' THEN  ABS( P~KZWI2 ) * -1
              WHEN ( K~EX_POSTYP <> '20' AND K~EX_POSTYP <> '21' ) AND P~SHKZG <> 'X' THEN  ABS( P~KZWI2 ) END AS NORM_WAVWR,
         "以旧换新小票的
         CASE WHEN ( K~EX_POSTYP = '20' OR K~EX_POSTYP = '21' ) AND P~SHKZG =  'X' THEN  ABS( P~KZWI6 ) * -1
              WHEN ( K~EX_POSTYP = '20' OR K~EX_POSTYP = '21' ) AND P~SHKZG <> 'X' THEN  ABS( P~KZWI6 ) END AS YJHX_KZWI6,
         CASE WHEN ( K~EX_POSTYP = '20' OR K~EX_POSTYP = '21' ) AND P~SHKZG =  'X' THEN  ABS( P~KZWI1 ) * -1
              WHEN ( K~EX_POSTYP = '20' OR K~EX_POSTYP = '21' ) AND P~SHKZG <> 'X' THEN  ABS( P~KZWI1 ) END AS YJHX_KZWI1,
         CASE WHEN ( K~EX_POSTYP = '20' OR K~EX_POSTYP = '21' ) AND P~SHKZG =  'X' THEN  ABS( P~KZWI3 ) * -1
              WHEN ( K~EX_POSTYP = '20' OR K~EX_POSTYP = '21' ) AND P~SHKZG <> 'X' THEN  ABS( P~KZWI3 ) END AS YJHX_KZWI3,
         CASE WHEN ( K~EX_POSTYP = '20' OR K~EX_POSTYP = '21' ) AND P~SHKZG =  'X' THEN  ABS( P~NETWR ) * -1
              WHEN ( K~EX_POSTYP = '20' OR K~EX_POSTYP = '21' ) AND P~SHKZG <> 'X' THEN  ABS( P~NETWR ) END AS YJHX_NETWR,
         CASE WHEN ( K~EX_POSTYP = '20' OR K~EX_POSTYP = '21' ) AND P~SHKZG =  'X' THEN  ABS( P~KZWI2 ) * -1
              WHEN ( K~EX_POSTYP = '20' OR K~EX_POSTYP = '21' ) AND P~SHKZG <> 'X' THEN  ABS( P~KZWI2 ) END AS YJHX_WAVWR,
         "所有的
         CASE WHEN P~SHKZG =  'X' THEN  ABS( P~KZWI3 ) * -1
              WHEN P~SHKZG <> 'X' THEN  ABS( P~KZWI3 ) END AS SUM_KZWI3,
         CASE WHEN P~SHKZG =  'X' THEN  ABS( P~NETWR ) * -1
              WHEN P~SHKZG <> 'X' THEN  ABS( P~NETWR ) END AS SUM_NETWR
*         A~EXTWG
    INTO CORRESPONDING FIELDS OF TABLE @GT_BILLING
    FROM VBRK AS K
    INNER JOIN VBRP AS P ON P~VBELN = K~VBELN
    INNER JOIN MARA AS A ON A~MATNR = P~MATNR
    WHERE K~FKDAT IN @S_FKDAT
      AND K~BUKRS IN @S_BUKRS
      AND K~VBELN IN @S_VBELN
      AND K~FKART = 'FP'
      AND P~WERKS IN @GRT_WERKS .
ENDFORM.

*&---------------------------------------------------------------------*
*&      Form  FRM_GET_JSRK
*&---------------------------------------------------------------------*
FORM FRM_GET_JSRK .
  SELECT K~BKTXT AS EX_POSNUM, K~BUDAT AS FKDAT, K~MBLNR AS BMLNR, K~MJAHR AS MJAHR,
         SUM( CASE WHEN G~SHKZG = 'H' THEN G~MENGE * -1
                   WHEN G~SHKZG = 'S' THEN G~MENGE END ) AS MENGE,
         SUM( CASE WHEN G~SHKZG = 'H' THEN G~DMBTR * -1
                   WHEN G~SHKZG = 'S' THEN G~DMBTR END ) AS DMBTR,
         SUM( CASE WHEN G~SHKZG = 'H' THEN G~DMBTR * -1
                   WHEN G~SHKZG = 'S' THEN G~DMBTR END ) AS DMBTR_REMAIN
    INTO TABLE @GT_JSRK
    FROM MKPF AS K
    INNER JOIN MSEG AS G ON  G~MBLNR = K~MBLNR
                         AND G~MJAHR = K~MJAHR
    WHERE K~BUDAT IN @S_FKDAT
      AND G~BUKRS IN @S_BUKRS
      AND G~WERKS IN @S_WERKS
      AND G~BWART IN ('Z56','Z57')
    GROUP BY K~BKTXT, K~BUDAT, K~MBLNR, K~MJAHR .
ENDFORM.

*&      Form  FRM_PROCESS_DATA
*&---------------------------------------------------------------------*
FORM FRM_PROCESS_DATA .
  DATA: BEGIN OF LS_WERKS,
          WERKS TYPE T001W-WERKS,
          SORT1 TYPE ADRC-SORT1,
        END OF LS_WERKS,
        LT_WERKS LIKE TABLE OF LS_WERKS .
  DATA: LS_TWEWT TYPE TWEWT,
        LT_TWEWT LIKE TABLE OF LS_TWEWT .
  DATA: LV_JSCB TYPE VBRP-MWSBP . " 旧饰成本
  SELECT W~WERKS C~SORT1
    INTO TABLE LT_WERKS
    FROM T001W AS W
    INNER JOIN ADRC AS C ON C~ADDRNUMBER = W~ADRNR.

  "定义计算毛利率的宏
  DEFINE CAL_PERCENTR .
    " &1 销售收入，&2 毛利额，&3 毛利率
    IF &2 = 0 .
      &3 = 0 .
    ELSEIF &1 = 0 .
      IF &2 > 0 .
        &3 = '99999.99' .
      ELSE.
        &3 = '-99999.99' .
      ENDIF.
    ELSE.
      &3 = &2 / &1 * 100.
    ENDIF.
  END-OF-DEFINITION .

  SELECT *
    INTO TABLE LT_TWEWT
    FROM TWEWT
    WHERE SPRAS = SY-LANGU .

  SORT GT_JSRK BY EX_POSNUM FKDAT ."日期
  SORT LT_WERKS BY WERKS ."地点
  SORT LT_TWEWT BY EXTWG  ."大类
*--------------------------------------------------------------------------------------------------
*--------------------------------------------------------------------------------------------------
  LOOP AT GT_BILLING ASSIGNING FIELD-SYMBOL(<FS_BILLING>) .
    READ TABLE LT_WERKS INTO LS_WERKS WITH KEY WERKS = <FS_BILLING>-WERKS BINARY SEARCH .
    IF SY-SUBRC = 0 .
      <FS_BILLING>-NAME1 = LS_WERKS-SORT1.    "地点简称
    ENDIF.

*    READ TABLE LT_TWEWT INTO LS_TWEWT WITH KEY EXTWG = <FS_BILLING>-EXTWG BINARY SEARCH .
*    IF SY-SUBRC = 0 .
*      <FS_BILLING>-EWBEZ = LS_TWEWT-EWBEZ .    "商品大类名称
*    ENDIF.

    IF <FS_BILLING>-EX_POSTYP <> '20' AND <FS_BILLING>-EX_POSTYP <> '21' .
      " 非以旧换新销售
      <FS_BILLING>-NORM_GPVAL = <FS_BILLING>-NORM_NETWR - <FS_BILLING>-NORM_WAVWR . " 非旧新-毛利金额
      CAL_PERCENTR: <FS_BILLING>-NORM_NETWR
                    <FS_BILLING>-NORM_GPVAL
                    <FS_BILLING>-NORM_GPPER . " 非旧新-毛利率
      <FS_BILLING>-SUM_WAVWR = <FS_BILLING>-NORM_WAVWR . "合计-销售成本
      <FS_BILLING>-SUM_GPVAL = <FS_BILLING>-NORM_GPVAL . "合计-毛利额
      <FS_BILLING>-SUM_GPPER = <FS_BILLING>-NORM_GPPER . "合计-毛利率
    ELSE.
      " 以旧换新
      CLEAR: LV_JSCB .
      IF <FS_BILLING>-YJHX_WAVWR = 0 .
        LV_JSCB = 0 . " 如果商品销售成本为0，就不用分摊旧饰成本了
      ELSE.
        READ TABLE GT_JSRK ASSIGNING FIELD-SYMBOL(<FS_JSRK>) WITH KEY EX_POSNUM = <FS_BILLING>-EX_POSNUM
                                                                      FKDAT     = <FS_BILLING>-FKDAT
                                                                      BINARY SEARCH .
        IF SY-SUBRC <> 0 .
          LV_JSCB = 0 . " 没有找到对应旧饰入库的单据，也就不用分摊旧饰成本了
        ELSE.
          IF <FS_BILLING>-YJHX_KZWI6 = 0 .
            LV_JSCB = 0 .   " 如果原售价为0，那么分摊旧饰成本也为0
          ELSE.
            IF <FS_BILLING>-YJHX_KZWI1 = 0 .
              LV_JSCB = <FS_BILLING>-YJHX_WAVWR . " 如果原售价不为0，现售价为0，那么抵值比例是100%，分摊旧饰成本等于商品成本
            ELSE.
              " 旧饰分摊成本 = （（原售价 - 现售价） / 原售价  ） * 商品销售成本
              LV_JSCB = ( ABS( <FS_BILLING>-YJHX_KZWI6 ) - ABS( <FS_BILLING>-YJHX_KZWI1 ) )
                        / ABS( <FS_BILLING>-YJHX_KZWI6 )
                        * <FS_BILLING>-YJHX_WAVWR .
            ENDIF.
          ENDIF.
          " 如果计算出来的旧饰分摊成本 大于 旧饰入库单据中的剩余未分摊成本，则旧饰分摊成本 = 剩余未分摊成本
          IF ABS( LV_JSCB ) > ABS( <FS_JSRK>-DMBTR_REMAIN ) .
            LV_JSCB = <FS_JSRK>-DMBTR_REMAIN .
          ENDIF.
          " 更新旧饰入库单据中的剩余未分摊成本
          <FS_JSRK>-DMBTR_REMAIN = <FS_JSRK>-DMBTR_REMAIN - LV_JSCB .
        ENDIF.
      ENDIF.

      <FS_BILLING>-YJHX_WAVWR_JS = LV_JSCB .

      <FS_BILLING>-YJHX_WAVWR_FINAL = <FS_BILLING>-YJHX_WAVWR - <FS_BILLING>-YJHX_WAVWR_JS .
      <FS_BILLING>-YJHX_GPVAL = <FS_BILLING>-YJHX_NETWR - <FS_BILLING>-YJHX_WAVWR_FINAL .
      CAL_PERCENTR: <FS_BILLING>-YJHX_NETWR
                    <FS_BILLING>-YJHX_GPVAL
                    <FS_BILLING>-YJHX_GPPER .
      <FS_BILLING>-SUM_WAVWR = <FS_BILLING>-YJHX_WAVWR_FINAL . "合计 - 销售成本
      <FS_BILLING>-SUM_GPVAL = <FS_BILLING>-YJHX_GPVAL . "合计-毛利额
      <FS_BILLING>-SUM_GPPER = <FS_BILLING>-YJHX_GPPER . "合计-毛利率
    ENDIF.

    CLEAR: GS_ALV.
    GS_ALV = <FS_BILLING> .
    CLEAR: GS_ALV-VBELN, GS_ALV-POSNR, GS_ALV-EX_POSNUM,
           GS_ALV-EX_POSTYP, GS_ALV-MATNR, GS_ALV-CHARG,
           GS_ALV-NORM_GPPER, GS_ALV-YJHX_GPPER,GS_ALV-SUM_GPPER .
    COLLECT GS_ALV INTO GT_ALV .
  ENDLOOP.
*---------------------------------------------------------------------------------------------------
  SORT GT_ALV BY FKDAT WERKS EXTWG .
  " 如果旧饰入库凭证中的金额没有分摊完（四舍五入的原因），将未分摊完的金额继续分摊到Billing中
  LOOP AT GT_JSRK ASSIGNING <FS_JSRK> WHERE DMBTR_REMAIN <> 0 .
    LOOP AT GT_BILLING ASSIGNING <FS_BILLING> WHERE EX_POSNUM = <FS_JSRK>-EX_POSNUM
                                                AND FKDAT     = <FS_JSRK>-FKDAT
                                                AND ( EX_POSTYP = '20' OR EX_POSTYP = '21' )
                                                AND YJHX_WAVWR <> 0
                                                AND YJHX_WAVWR_FINAL <> 0 .
      DATA(LV_WAVWR_REMAIN) = ABS( <FS_BILLING>-YJHX_WAVWR ) -  ABS( <FS_BILLING>-YJHX_WAVWR_JS ) .
      IF LV_WAVWR_REMAIN = 0 .
        CONTINUE .
      ENDIF.
      IF ABS( <FS_JSRK>-DMBTR_REMAIN ) <= LV_WAVWR_REMAIN .
        " 如果旧饰入库剩余奖金额小于等于该行发票成本未分摊金额，则调整金额等于旧饰入库剩余金额
        DATA(LV_WAVWR_ADJUST) = <FS_JSRK>-DMBTR_REMAIN .
      ELSE.
        " 如果旧饰入库剩余奖金额大于该行发票成本未分摊金额，则调整金额等于该行发票成本未分摊金额
        IF <FS_JSRK>-DMBTR_REMAIN < 0 .
          LV_WAVWR_ADJUST = LV_WAVWR_REMAIN * -1 .
        ELSE.
          LV_WAVWR_ADJUST = LV_WAVWR_REMAIN .
        ENDIF.
      ENDIF.
      " 更新该行发票的旧饰成本、毛利额、毛利率等
      <FS_BILLING>-YJHX_WAVWR_JS = <FS_BILLING>-YJHX_WAVWR_JS + LV_WAVWR_ADJUST .
      <FS_BILLING>-YJHX_WAVWR_FINAL = <FS_BILLING>-YJHX_WAVWR - <FS_BILLING>-YJHX_WAVWR_JS .
      <FS_BILLING>-YJHX_GPVAL = <FS_BILLING>-YJHX_NETWR - <FS_BILLING>-YJHX_WAVWR_FINAL .
      CAL_PERCENTR: <FS_BILLING>-YJHX_NETWR
                    <FS_BILLING>-YJHX_GPVAL
                    <FS_BILLING>-YJHX_GPPER .

      <FS_BILLING>-SUM_WAVWR = <FS_BILLING>-YJHX_WAVWR_FINAL . "合计 - 销售成本
      <FS_BILLING>-SUM_GPVAL = <FS_BILLING>-YJHX_GPVAL . "合计-毛利额
      <FS_BILLING>-SUM_GPPER = <FS_BILLING>-YJHX_GPPER . "合计-毛利率
      "将继续分摊的金额累加到ALV中
      READ TABLE GT_ALV ASSIGNING FIELD-SYMBOL(<FS_ALV>) WITH KEY FKDAT = <FS_BILLING>-FKDAT
                                                                  WERKS = <FS_BILLING>-WERKS
                                                                  EXTWG = <FS_BILLING>-EXTWG BINARY SEARCH .
      IF SY-SUBRC = 0 .
        <FS_ALV>-YJHX_WAVWR_JS = <FS_ALV>-YJHX_WAVWR_JS + LV_WAVWR_ADJUST .
        <FS_ALV>-YJHX_WAVWR_FINAL = <FS_ALV>-YJHX_WAVWR - <FS_ALV>-YJHX_WAVWR_JS .
        <FS_ALV>-YJHX_GPVAL = <FS_ALV>-YJHX_NETWR - <FS_ALV>-YJHX_WAVWR_FINAL .
        <FS_ALV>-SUM_WAVWR = <FS_ALV>-SUM_WAVWR +  LV_WAVWR_ADJUST. "合计 - 销售成本
        <FS_ALV>-SUM_GPVAL = <FS_ALV>-SUM_NETWR -  <FS_ALV>-SUM_WAVWR . "合计-毛利额
      ENDIF.
      " 更新旧饰入库单据中的未分摊金额
      <FS_JSRK>-DMBTR_REMAIN = <FS_JSRK>-DMBTR_REMAIN - LV_WAVWR_ADJUST .
      IF <FS_JSRK>-DMBTR_REMAIN = 0 .
        EXIT . " 如果旧饰入库单据中的未分摊金额为0，则退出该循环，继续下一张未分摊完的就是入库单据
      ENDIF.
    ENDLOOP .
  ENDLOOP.
  " 计算各种比率
  LOOP AT GT_ALV ASSIGNING <FS_ALV> .
    " 非以旧换新-毛利率
    CAL_PERCENTR: <FS_ALV>-NORM_NETWR
                  <FS_ALV>-NORM_GPVAL
                  <FS_ALV>-NORM_GPPER .
    " 以旧换新-毛利率
    CAL_PERCENTR: <FS_ALV>-YJHX_NETWR
                  <FS_ALV>-YJHX_GPVAL
                  <FS_ALV>-YJHX_GPPER .
    " 合计-毛利率
    CAL_PERCENTR: <FS_ALV>-SUM_NETWR
                  <FS_ALV>-SUM_GPVAL
                  <FS_ALV>-SUM_GPPER .
  ENDLOOP.

  SORT GT_ALV BY FKDAT WERKS EXTWG .
  SORT GT_BILLING BY FKDAT WERKS VBELN POSNR .
ENDFORM.
