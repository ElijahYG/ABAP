*&获取BKPF中符合条件的数据
*&---------------------------------------------------------------------*
FORM FM_GET_BKPF.
  FIELD-SYMBOLS <FS>  TYPE TY_OUT_TAB.
  DATA:
    BEGIN OF LW_BKPF,
      BUKRS TYPE BKPF-BUKRS,
      BELNR TYPE BKPF-BELNR,
      GJAHR TYPE BKPF-GJAHR,

      BLART TYPE BKPF-BLART,
      NUMPG TYPE BKPF-NUMPG,

      PPNAM TYPE BKPF-PPNAM,
      USNAM TYPE BKPF-USNAM,
    END OF LW_BKPF,
    LT_BKPF    LIKE LW_BKPF OCCURS 0,
    LT_OUT_TAB LIKE GT_OUT_TAB.
* 取符合条件的bkpf数据
  LT_OUT_TAB   = GT_OUT_TAB.
  SORT LT_OUT_TAB BY BUKRS BELNR GJAHR.
  DELETE ADJACENT DUPLICATES FROM LT_OUT_TAB COMPARING BUKRS BELNR GJAHR.
  IF LT_OUT_TAB IS NOT INITIAL.
    SELECT BUKRS
           BELNR
           GJAHR
           BLART

           NUMPG
           PPNAM
           USNAM
      FROM BKPF
      INTO TABLE LT_BKPF
       FOR ALL ENTRIES IN LT_OUT_TAB
     WHERE BUKRS = LT_OUT_TAB-BUKRS
       AND BELNR = LT_OUT_TAB-BELNR
       AND GJAHR = LT_OUT_TAB-GJAHR
      .
  ENDIF.
* 将数据填充到OUT表
  SORT LT_BKPF BY BUKRS BELNR GJAHR.
  LOOP AT GT_OUT_TAB ASSIGNING <FS>.
    CLEAR LW_BKPF.
    READ TABLE LT_BKPF INTO LW_BKPF
                   WITH KEY BUKRS = <FS>-BUKRS
                            BELNR = <FS>-BELNR
                            GJAHR = <FS>-GJAHR
                     BINARY SEARCH.
    <FS>-BLART = LW_BKPF-BLART.
    <FS>-NUMPG = LW_BKPF-NUMPG.
    <FS>-PPNAM = LW_BKPF-PPNAM.
    <FS>-USNAM = LW_BKPF-USNAM.
  ENDLOOP.
ENDFORM.

FORM FM_GET_FAGA.
  FIELD-SYMBOLS <FS> TYPE TY_OUT_TAB.
  DATA:
    BEGIN OF LW_FAGA,
      BUKRS TYPE BSEG-BUKRS,
      BELNR TYPE BSEG-BELNR,
      GJAHR TYPE BSEG-GJAHR,
      BUZEI TYPE FAGLFLEXA-DOCLN,

      RCNTR TYPE FAGLFLEXA-RCNTR,
      PRCTR TYPE BSEG-PRCTR,
      SAKNR TYPE BSEG-SAKNR,
      DMBTR TYPE FAGLFLEXA-HSL,
    END OF LW_FAGA,
    LT_OUT_TAB LIKE GT_OUT_TAB,
    LT_FAGA    LIKE LW_FAGA OCCURS 0.
* 取符合条件的bkpf数据
  LOOP AT GT_OUT_TAB ASSIGNING <FS>.
    <FS>-DOCLN = '000' && <FS>-BUZEI.
  ENDLOOP.
  LT_OUT_TAB   = GT_OUT_TAB.

  SORT LT_OUT_TAB BY BUKRS BELNR GJAHR DOCLN.
  DELETE ADJACENT DUPLICATES FROM LT_OUT_TAB COMPARING BUKRS BELNR GJAHR DOCLN.
  IF LT_OUT_TAB IS NOT INITIAL.
    SELECT RBUKRS
           DOCNR
           RYEAR
           DOCLN

           RCNTR
           PRCTR
           RACCT
           HSL
      FROM FAGLFLEXA
      INTO TABLE LT_FAGA
       FOR ALL ENTRIES IN LT_OUT_TAB
     WHERE RBUKRS = LT_OUT_TAB-BUKRS
       AND DOCNR  = LT_OUT_TAB-BELNR
       AND RYEAR  = LT_OUT_TAB-GJAHR
       AND DOCLN  = LT_OUT_TAB-DOCLN
      .
  ENDIF.
* 将数据填充到OUT表
  SORT LT_FAGA BY BUKRS BELNR GJAHR BUZEI.
  LOOP AT GT_OUT_TAB ASSIGNING <FS>.
    CLEAR LW_FAGA.
    READ TABLE LT_FAGA INTO LW_FAGA
                   WITH KEY BUKRS = <FS>-BUKRS
                            BELNR = <FS>-BELNR
                            GJAHR = <FS>-GJAHR
                            BUZEI = <FS>-DOCLN
                     BINARY SEARCH.
    IF SY-SUBRC = 0.
      <FS>-PRCTR  = LW_FAGA-PRCTR.
      <FS>-KOSTL  = LW_FAGA-RCNTR.
      <FS>-SAKNR  = LW_FAGA-SAKNR.
      <FS>-DMBTR  = ABS( LW_FAGA-DMBTR ).
    ENDIF.
  ENDLOOP.
ENDFORM.
