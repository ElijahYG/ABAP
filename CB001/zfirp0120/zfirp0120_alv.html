FORM USER_COMMAND USING UCOMM LIKE SY-UCOMM
        SELFIELD TYPE SLIS_SELFIELD.

  DATA:
    L_GRID     TYPE REF TO CL_GUI_ALV_GRID.
* 刷新缺
  CALL FUNCTION 'GET_GLOBALS_FROM_SLVC_FULLSCR'
    IMPORTING
      E_GRID = L_GRID.
  CALL METHOD L_GRID->CHECK_CHANGED_DATA.
    SELFIELD-REFRESH    = 'X'. "刷新屏幕
  SELFIELD-COL_STABLE = 'X'. "行锁定
  SELFIELD-ROW_STABLE = 'X'. "列锁定

* 得到用户已经选择的数据记录
  CLEAR: IT_SELECTED[].

  LOOP AT IT_ALV INTO WA_ALV WHERE SEL_BOX = 'X'.

    CLEAR: WA_SELECTED.

    MOVE-CORRESPONDING WA_ALV TO WA_SELECTED.

    APPEND WA_SELECTED TO IT_SELECTED.

  ENDLOOP.
* 读取用户双击的当前行
  CLEAR: WA_ALV.
  READ TABLE IT_ALV INTO WA_ALV INDEX SELFIELD-TABINDEX.

* 判断用户具体做了什么动作
  CASE UCOMM.
    WHEN '&IC1'.                      "ALV中双击事件代码
*      IF SY-SUBRC = 0.        "双击每一个字段都行
         IF SELFIELD-SEL_TAB_FIELD = 'IT_ALV-BELNR'. "表示单击某一单元格
        SET PARAMETER ID 'BLN' FIELD WA_ALV-BELNR. "PARAMETER ID通过F1->technical information->Parameter id可以获得
        SET PARAMETER ID 'BUK' FIELD WA_ALV-BUKRS.
        SET PARAMETER ID 'GJR' FIELD WA_ALV-GJAHR.
        CALL TRANSACTION 'FB03' AND SKIP FIRST SCREEN.
      ENDIF.

    WHEN 'BTN_PRINT'.                 "调用打印功能

      IF IT_SELECTED[] IS NOT INITIAL.
      PERFORM PROCESS_PRINT.
      ENDIF.

       WHEN '&BACK'.                       "退出
      LEAVE TO SCREEN 0.

    WHEN 'ALL'.
      LOOP AT IT_ALV INTO WA_ALV.
        WA_ALV-SEL_BOX = 'X'.
        MODIFY IT_ALV FROM WA_ALV TRANSPORTING SEL_BOX.
      ENDLOOP.

    WHEN 'SAL'.
      LOOP AT IT_ALV INTO WA_ALV.
        WA_ALV-SEL_BOX = ' '.
        MODIFY IT_ALV FROM WA_ALV TRANSPORTING SEL_BOX.
      ENDLOOP.
        ENDCASE.
  CLEAR UCOMM.

ENDFORM. "USER_COMMAND
