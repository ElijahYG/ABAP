FORM GET_DATA.

  RP-PROVIDE-FROM-LAST P0001 SPACE PN-BEGDA PN-ENDDA.
  RP-PROVIDE-FROM-LAST P0002 SPACE PN-BEGDA PN-ENDDA.
  RP-PROVIDE-FROM-LAST P0041 SPACE PN-BEGDA PN-ENDDA.

*  RP-PROVIDE-FROM-LAST P0009 '0' PN-BEGDA PN-ENDDA.


  IF SY-SUBRC = 0.                    "NO PAYROLL RESULTS

    ITAB-YEARMONTH = PN-BEGDA+0(6)  . "发工资月
    ITAB-PERNR =  PERNR-PERNR.    "证章号
    CONCATENATE P0002-NACHN P0002-VORNA INTO ITAB-NAME. "姓名
    ITAB-GBDAT_NL = SY-DATUM+0(4) - P0002-GBDAT+0(4).                            "年龄
    ITAB-PERSG   = P0001-PERSG.


    IF ITAB-PERSG = '1' .

      ITAB-YGZ = '合同制'.

    ENDIF.

    IF ITAB-PERSG = '2' .

      ITAB-YGZ = '退休'.

    ENDIF.


     IF ITAB-PERSG = '3' .

      ITAB-YGZ = '劳务派遣'.

    ENDIF.

      IF ITAB-PERSG > '3' .

      EXIT.

    ENDIF.



    DO 12 TIMES VARYING V_DARXX FROM P0041-DAR01 NEXT P0041-DAR02                "取入职时间
               VARYING V_DATXX FROM P0041-DAT01 NEXT P0041-DAT02.
      IF V_DARXX = '02'.
        ITAB-DAT02 = V_DATXX.
      ENDIF.
    ENDDO.


   IF ITAB-DAT02 <> '00000000'.
    CALL FUNCTION 'HR_99S_INTERVAL_BETWEEN_DATES' "获得两个日期的年数
      EXPORTING
        BEGDA   = ITAB-DAT02
        ENDDA   = SY-DATUM
        TAB_MODE = ''
      IMPORTING
        C_YEARS = L_AGE1.
   ENDIF.




    IF P0001-PERSG = '1'  OR P0001-PERSG = '3'.

      ITAB-N = L_AGE1.
      IF ITAB-N < '5'  .
      ITAB-ZJE = '0'.
      ELSEIF ITAB-N >= '5' AND ITAB-N < '10' .
        ITAB-ZJE = '200'.
      ELSEIF  ITAB-N >= '10' AND ITAB-N < '15' .
        ITAB-ZJE = '300'.
      ELSEIF  ITAB-N >= '15' AND ITAB-N < '20' .
        ITAB-ZJE = '400'.
      ELSEIF  ITAB-N >= '20' .
        ITAB-ZJE = '480'.
      ENDIF.

    ENDIF.

    IF P0001-PERSG = '2' AND P0002-GESCH = '1' . "1 为男
      ITAB-XB = '男'.
      IF ITAB-GBDAT_NL < '60'.
        ITAB-ZJE = '480'.
      ENDIF.
      IF ITAB-GBDAT_NL >= '60' AND ITAB-GBDAT_NL < '70' .
        ITAB-ZJE = '600'.
      ENDIF.
      IF ITAB-GBDAT_NL >= '70' AND ITAB-GBDAT_NL < '80' .
        ITAB-ZJE = '720'.
      ENDIF.
      IF ITAB-GBDAT_NL >= '80'  .
        ITAB-ZJE = '840'.
      ENDIF.

    ENDIF.


    IF P0001-PERSG = '2' AND P0002-GESCH = '2' . "2为女
      ITAB-XB = '女'.

      IF ITAB-GBDAT_NL < '50'.
        ITAB-ZJE = '480'.
      ENDIF.
      IF ITAB-GBDAT_NL >= '50' AND ITAB-GBDAT_NL < '60' .
        ITAB-ZJE = '600'.
      ENDIF.
      IF ITAB-GBDAT_NL >= '60' AND ITAB-GBDAT_NL < '70' .
        ITAB-ZJE = '720'.
      ENDIF.
      IF ITAB-GBDAT_NL >= '70'  .
        ITAB-ZJE = '840'.
      ENDIF.


    ENDIF.


    IF TSQK1 <> 'X' AND TSQK2 <> 'X' AND TSQK3 <> 'X'.
    ITAB-QJE = ITAB-ZJE.
    ENDIF.

    ENDIF.

      ITAB-QJE = ITAB-ZJE + ITAB-ZJE * '0.3'.
      ITAB-ZTSQK2 = '建国前参加革命工作的老工人'.

    ENDIF.

    ENDIF.


     IF TSQK1 = 'X' AND TSQK2 = 'X' .

      ITAB-QJE = ITAB-ZJE + ITAB-ZJE * '0.3'.
      ITAB-QJE = ITAB-QJE + ITAB-QJE * '0.3'.
      ITAB-ZTSQK1 = '职工、退休职工 大病'.
