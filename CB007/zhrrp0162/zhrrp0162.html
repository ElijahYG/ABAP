*&***定义这个内表主要是处理金额小于0时，会自动扩大的问题


  DATA: BEGIN OF LT_E_DATA OCCURS 0,

        END OF LT_E_DATA.

  CLEAR:LT_E_DATA[],GT_E_DATA[].

  DATA:
    L_FILE  TYPE RLGRAP-FILENAME,
    LW_LINE TYPE REF TO DATA,
    LT_TAB  TYPE STANDARD TABLE OF ALSMEX_TABLINE,
    LW_TAB  TYPE ALSMEX_TABLINE.
  FIELD-SYMBOLS:
    <LINE>,
    <FS>.
  CREATE DATA LW_LINE LIKE LINE OF LT_E_DATA.
  ASSIGN LW_LINE->* TO <LINE>.
  L_FILE = P_FILENM.
  TRANSLATE L_FILE TO UPPER CASE.
  CALL FUNCTION 'ALSM_EXCEL_TO_INTERNAL_TABLE'
    EXPORTING
      FILENAME                = L_FILE
      I_BEGIN_COL             = 2
      I_BEGIN_ROW             = 1
      I_END_COL               = 250
      I_END_ROW               = 65536

      UPLOAD_OLE              = 2
      OTHERS                  = 3.

    MESSAGE E001(00) WITH '上传文件有误！' SPACE SPACE SPACE.
  ENDIF.

  DELETE LT_TAB WHERE ROW = 1.  "删除字段名行

  REFRESH:
    LT_E_DATA.
  LOOP AT LT_TAB INTO LW_TAB.

    ON CHANGE OF LW_TAB-ROW.
      IF SY-TABIX NE 1.
        APPEND  <LINE> TO LT_E_DATA.
        CLEAR <LINE>.
      ENDIF.
    ENDON.

    ASSIGN COMPONENT LW_TAB-COL OF STRUCTURE <LINE> TO <FS>.
    TRY.
        <FS> = LW_TAB-VALUE.
      CATCH CX_ROOT.
        MESSAGE E001(00) WITH '上传文件的列多于接收内表的列' SPACE SPACE SPACE.
    ENDTRY.

    AT LAST.
      APPEND <LINE> TO LT_E_DATA.
    ENDAT.


  ENDLOOP.


*   IF lt_e_data IS INITIAL.
*     MESSAGE E001(00) WITH '上传文件为空' SPACE SPACE SPACE.
*   ELSE.
  LOOP AT LT_E_DATA.

*   ENDIF.
