  FORM GET_DATA_MAIN USING ZRACCT.



*&---------------------------------------------------------------------*
*&数据取值
*&---------------------------------------------------------------------*
DATA: ZRACCT_ LIKE IT_RACCT. "定义科目
DATA: ZYEAR TYPE NUMC4,
      ZYEARMIN   TYPE NUMC4,
      ZYEARMAX   TYPE NUMC4.                                                                   "得到会计年度
      ZYEAR    = S_YEARMO-LOW+0(4).
      ZYEARMIN =  ZYEAR - 1.
      ZYEARMAX =  ZYEAR + 1.
      ZRACCT_ = ZRACCT.

    SELECT
      A~RACCT                                                                               "科目
      A~DRCRK                                                                               "借贷标识
      A~RYEAR                                                                               "会计年度
      B~TXT20                                                                               "收款方式
      A~HSL01                                                                               "期间01 发生额
      A~HSL02                                                                               "期间02 发生额
      A~HSL03                                                                               "期间03 发生额
      A~HSL04                                                                               "期间04 发生额
      A~HSL05                                                                               "期间05 发生额
      A~HSL06                                                                               "期间06 发生额
      A~HSL07                                                                               "期间07 发生额
      A~HSL08                                                                               "期间08 发生额
      A~HSL09                                                                               "期间09 发生额
      A~HSL10                                                                               "期间10 发生额
      A~HSL11                                                                               "期间11 发生额
      A~HSL12                                                                               "期间12 发生额
      A~HSL13                                                                               "期间13 发生额
      A~HSL14                                                                               "期间14 发生额
      INTO CORRESPONDING FIELDS OF TABLE IT_FAG
      FROM FAGLFLEXT AS A INNER JOIN SKAT AS B
      ON A~RACCT = B~SAKNR                                                                   "科目
      WHERE A~RACCT IN ZRACCT_                                                               "科目
      AND A~RBUKRS IN S_RBUKRS                                                               "公司代码
      AND A~RYEAR >= ZYEARMIN                                                                 "会计年度
      AND A~RYEAR < ZYEARMAX
      AND A~DRCRK = 'S'.

  ENDFORM.


*&---------------------------------------------------------------------*
*&数据处理
*&---------------------------------------------------------------------*
  FORM DO_DATA.
*

  FIELD-SYMBOLS <FS>.

DATA: L_INDEX TYPE NUMC2,
      L_FIELD       TYPE CHAR100,
      WA_TMP        TYPE TY_FAG,
      IT_ALV_SHOW_T TYPE STANDARD TABLE OF STUCT_AVL_SHOW,
      WA_ALV_SHOW_T TYPE                   STUCT_AVL_SHOW,
      ZMONTH        TYPE NUMC2,
      WA_FAG_       TYPE TY_FAG.

      ZMONTH  = S_YEARMO-LOW+4(2).

*----------------------------------------------*
*第一步：按科目汇总
*----------------------------------------------*
  SORT IT_FAG BY RACCT DRCRK RYEAR TXT20.

  CLEAR IT_ALV_SHOW.
  LOOP AT IT_FAG INTO WA_TMP.
*   防止*
    WA_FAG_ = WA_TMP.
*   按科目、功能范围汇总
    WA_FAG-HSL01 = WA_FAG-HSL01 + WA_FAG_-HSL01.
    WA_FAG-HSL02 = WA_FAG-HSL02 + WA_FAG_-HSL02.
    WA_FAG-HSL03 = WA_FAG-HSL03 + WA_FAG_-HSL03.
    WA_FAG-HSL04 = WA_FAG-HSL04 + WA_FAG_-HSL04.
    WA_FAG-HSL05 = WA_FAG-HSL05 + WA_FAG_-HSL05.
    WA_FAG-HSL06 = WA_FAG-HSL06 + WA_FAG_-HSL06.
    WA_FAG-HSL07 = WA_FAG-HSL07 + WA_FAG_-HSL07.
    WA_FAG-HSL08 = WA_FAG-HSL08 + WA_FAG_-HSL08.
    WA_FAG-HSL09 = WA_FAG-HSL09 + WA_FAG_-HSL09.
    WA_FAG-HSL10 = WA_FAG-HSL10 + WA_FAG_-HSL10.
    WA_FAG-HSL11 = WA_FAG-HSL11 + WA_FAG_-HSL11.
    WA_FAG-HSL12 = WA_FAG-HSL12 + WA_FAG_-HSL12 + WA_FAG_-HSL13 + WA_FAG_-HSL14.
    AT END OF RACCT.

      CLEAR WA_ALV_SHOW_T.
      DO 12 TIMES.
        L_INDEX = SY-INDEX.

        L_FIELD = 'WA_FAG-HSL' && L_INDEX.

        ASSIGN (L_FIELD) TO <FS>.
*       合计到当月
        IF L_INDEX EQ ZMONTH.
          IF WA_FAG_-RYEAR EQ  S_YEARMO-LOW+0(4).
            WA_ALV_SHOW_T-ZTMONTH = WA_ALV_SHOW_T-ZTMONTH + <FS>.                                             "计算当年本月金额
          ELSE.
            WA_ALV_SHOW_T-ZBMONTH = WA_ALV_SHOW_T-ZBMONTH + <FS>.                                             "计算上一年本月金额
          ENDIF.
        ENDIF.

*      计算当年累计值
        IF L_INDEX <= ZMONTH.
          IF WA_FAG_-RYEAR EQ  S_YEARMO-LOW+0(4).
            WA_ALV_SHOW_T-ZTYEAR = WA_ALV_SHOW_T-ZTYEAR + <FS>.                                               "计算当年金额
          ELSE.
            WA_ALV_SHOW_T-ZBYEAR = WA_ALV_SHOW_T-ZBYEAR + <FS>.                                               "计算前一年金额
          ENDIF.
        ENDIF.
      ENDDO.
*     计算当月和当年的发生额
      WA_ALV_SHOW_T-ZRACCT   = WA_FAG_-RACCT.
      WA_ALV_SHOW_T-ZRECTYPE = WA_FAG_-TXT20.                                                                    "科目
      APPEND WA_ALV_SHOW_T TO IT_ALV_SHOW_T.

*     处理完毕，清空工作区
      CLEAR:WA_FAG.
      CLEAR WA_ALV_SHOW_T.
    ENDAT.
  ENDLOOP.

*----------------------------------------------*
*第二步：按年度汇总
*----------------------------------------------*
  SORT IT_ALV_SHOW_T BY ZRACCT.                                                                               "按着科目汇总

  CLEAR: WA_ALV_SHOW_T,WA_ALV_SHOW,IT_ALV_SHOW[].

  LOOP AT IT_ALV_SHOW_T INTO WA_ALV_SHOW_T.
    READ TABLE IT_ALV_SHOW INTO WA_ALV_SHOW WITH KEY ZRACCT = WA_ALV_SHOW_T-ZRACCT.                           "都内表，判断是否有科目相同的记录
    IF SY-SUBRC <> 0.
      WA_ALV_SHOW-ZRACCT    = WA_ALV_SHOW_T-ZRACCT.
      WA_ALV_SHOW-ZRECTYPE  = WA_ALV_SHOW_T-ZRECTYPE.
      WA_ALV_SHOW-ZTMONTH   = WA_ALV_SHOW_T-ZTMONTH.
      WA_ALV_SHOW-ZBMONTH   = WA_ALV_SHOW_T-ZBMONTH.
      WA_ALV_SHOW-ZMRATE    = WA_ALV_SHOW_T-ZMRATE.
      WA_ALV_SHOW-ZTYEAR    = WA_ALV_SHOW_T-ZTYEAR.
      WA_ALV_SHOW-ZBYEAR    = WA_ALV_SHOW_T-ZBYEAR.
      WA_ALV_SHOW-ZYRATE    = WA_ALV_SHOW_T-ZYRATE.

      APPEND WA_ALV_SHOW TO IT_ALV_SHOW.
    ELSEIF SY-SUBRC = 0.
      WA_ALV_SHOW-ZRACCT    = WA_ALV_SHOW-ZRACCT + WA_ALV_SHOW_T-ZRACCT.
      WA_ALV_SHOW-ZRECTYPE  = WA_ALV_SHOW-ZRECTYPE + WA_ALV_SHOW_T-ZRECTYPE.
      WA_ALV_SHOW-ZTMONTH   = WA_ALV_SHOW-ZTMONTH + WA_ALV_SHOW_T-ZTMONTH.
      WA_ALV_SHOW-ZBMONTH   = WA_ALV_SHOW-ZBMONTH + WA_ALV_SHOW_T-ZBMONTH.
      WA_ALV_SHOW-ZMRATE    = WA_ALV_SHOW-ZMRATE + WA_ALV_SHOW_T-ZMRATE.
      WA_ALV_SHOW-ZTYEAR    = WA_ALV_SHOW-ZTYEAR + WA_ALV_SHOW_T-ZTYEAR.
      WA_ALV_SHOW-ZBYEAR    = WA_ALV_SHOW-ZBYEAR + WA_ALV_SHOW_T-ZBYEAR.
      WA_ALV_SHOW-ZYRATE    = WA_ALV_SHOW-ZYRATE + WA_ALV_SHOW_T-ZYRATE.

      MODIFY IT_ALV_SHOW FROM WA_ALV_SHOW INDEX SY-TABIX.
    ENDIF.
  ENDLOOP.

*----------------------------------------------*
*第三步：算比率
*----------------------------------------------*
  SORT IT_ALV_SHOW_T BY ZRACCT.

  CLEAR:WA_ALV_SHOW.
  LOOP AT IT_ALV_SHOW INTO WA_ALV_SHOW.
    IF WA_ALV_SHOW-ZBMONTH IS NOT INITIAL AND WA_ALV_SHOW-ZTMONTH IS NOT INITIAL.                                                                  "判断除数是否为0
      WA_ALV_SHOW-ZMRATE = ( WA_ALV_SHOW-ZTMONTH - WA_ALV_SHOW-ZBMONTH ) / WA_ALV_SHOW-ZBMONTH * 100.        "月比率计算
    ELSEIF  WA_ALV_SHOW-ZBMONTH IS NOT INITIAL AND  WA_ALV_SHOW-ZTMONTH IS INITIAL.
      WA_ALV_SHOW-ZMRATE = '-100'.
    ELSEIF WA_ALV_SHOW-ZBMONTH IS INITIAL AND  WA_ALV_SHOW-ZTMONTH IS NOT INITIAL.
      WA_ALV_SHOW-ZMRATE = '100'.
    ELSE.
      WA_ALV_SHOW-ZMRATE = 0.
    ENDIF.
    IF WA_ALV_SHOW-ZBYEAR IS NOT INITIAL AND WA_ALV_SHOW-ZTYEAR IS NOT INITIAL.                                                                  "判断除数是否为0
      WA_ALV_SHOW-ZYRATE = ( WA_ALV_SHOW-ZTYEAR - WA_ALV_SHOW-ZBYEAR ) / WA_ALV_SHOW-ZBYEAR * 100.          "月比率计算
    ELSEIF  WA_ALV_SHOW-ZBYEAR IS NOT INITIAL AND  WA_ALV_SHOW-ZTYEAR IS INITIAL.
      WA_ALV_SHOW-ZYRATE = '-100'.
    ELSEIF  WA_ALV_SHOW-ZBYEAR IS INITIAL AND  WA_ALV_SHOW-ZTYEAR IS NOT INITIAL.
      WA_ALV_SHOW-ZYRATE = '100'.
    ELSE.
      WA_ALV_SHOW-ZYRATE = 0.
    ENDIF.
    MODIFY IT_ALV_SHOW FROM WA_ALV_SHOW INDEX SY-TABIX TRANSPORTING ZMRATE ZYRATE.
  ENDLOOP.

*----------------------------------------------*
*第四步：改符号
*----------------------------------------------*
  CLEAR:WA_ALV_SHOW.

  LOOP AT IT_ALV_SHOW INTO WA_ALV_SHOW.

    WA_ALV_SHOW_CHA-ZRACCT     = WA_ALV_SHOW-ZRACCT.
    WA_ALV_SHOW_CHA-ZRECTYPE   = WA_ALV_SHOW-ZRECTYPE.
    WA_ALV_SHOW_CHA-ZTMONTH    = WA_ALV_SHOW-ZTMONTH.
    CALL FUNCTION 'CLOI_PUT_SIGN_IN_FRONT'
            CHANGING   VALUE = WA_ALV_SHOW_CHA-ZTMONTH.
    WA_ALV_SHOW_CHA-ZBMONTH    = WA_ALV_SHOW-ZBMONTH.
    CALL FUNCTION 'CLOI_PUT_SIGN_IN_FRONT'
            CHANGING   VALUE = WA_ALV_SHOW_CHA-ZBMONTH.
    WA_ALV_SHOW_CHA-ZMRATE    = WA_ALV_SHOW-ZMRATE.
    CALL FUNCTION 'CLOI_PUT_SIGN_IN_FRONT'
            CHANGING   VALUE = WA_ALV_SHOW_CHA-ZMRATE.
    WA_ALV_SHOW_CHA-ZTYEAR    = WA_ALV_SHOW-ZTYEAR.
    CALL FUNCTION 'CLOI_PUT_SIGN_IN_FRONT'
            CHANGING   VALUE = WA_ALV_SHOW_CHA-ZTYEAR.
    WA_ALV_SHOW_CHA-ZBYEAR    = WA_ALV_SHOW-ZBYEAR.
    CALL FUNCTION 'CLOI_PUT_SIGN_IN_FRONT'
            CHANGING   VALUE = WA_ALV_SHOW_CHA-ZBYEAR.
    WA_ALV_SHOW_CHA-ZYRATE    = WA_ALV_SHOW-ZYRATE.
    CALL FUNCTION 'CLOI_PUT_SIGN_IN_FRONT'
            CHANGING   VALUE = WA_ALV_SHOW_CHA-ZYRATE.

    APPEND WA_ALV_SHOW_CHA TO IT_ALV_SHOW_CHA.

    CLEAR: WA_ALV_SHOW_CHA.
  ENDLOOP.

*----------------------------------------------*
*第五步：转换类型
*----------------------------------------------*
  CLEAR:IT_ALV_SHOW_I[],WA_ALV_SHOW_I,WA_ALV_SHOW_CHA.

  LOOP AT IT_ALV_SHOW_CHA INTO WA_ALV_SHOW_CHA.
      WA_ALV_SHOW_I-ZRACCT    = WA_ALV_SHOW_CHA-ZRACCT.
      WA_ALV_SHOW_I-ZRECTYPE  = WA_ALV_SHOW_CHA-ZRECTYPE.
      WA_ALV_SHOW_I-ZTMONTH   = WA_ALV_SHOW_CHA-ZTMONTH.
      WA_ALV_SHOW_I-ZBMONTH   = WA_ALV_SHOW_CHA-ZBMONTH.
      WA_ALV_SHOW_I-ZMRATE    = WA_ALV_SHOW_CHA-ZMRATE && '%'.
      WA_ALV_SHOW_I-ZTYEAR    = WA_ALV_SHOW_CHA-ZTYEAR.
      WA_ALV_SHOW_I-ZBYEAR    = WA_ALV_SHOW_CHA-ZBYEAR.
      WA_ALV_SHOW_I-ZYRATE    = WA_ALV_SHOW_CHA-ZYRATE && '%'.

      APPEND WA_ALV_SHOW_I TO IT_ALV_SHOW_I.
      CLEAR:WA_ALV_SHOW_CHA,WA_ALV_SHOW_I.
  ENDLOOP.

 ENDFORM.
