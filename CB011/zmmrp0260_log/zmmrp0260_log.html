*&---------------------------------------------------------------------*
FORM FM_GET_DATA.

  TYPES: BEGIN OF STRU_MKPF,
           MBLNR    TYPE MKPF-MBLNR,
           BUDAT    TYPE MKPF-BUDAT,
           CPUTM    TYPE MKPF-CPUTM,
           USNAM    TYPE MKPF-USNAM,
           VBELN    TYPE MKPF-LE_VBELN,
           WERKS_SH TYPE MSEG-WERKS,     "收货地点
           LGORT    TYPE MSEG-LGORT,     "收货库存地点
         END OF STRU_MKPF.
  DATA: LT_MKPF TYPE STANDARD TABLE OF STRU_MKPF,
        LS_MKPF TYPE                   STRU_MKPF,
        LS_HEAD TYPE                   STRU_HEAD.

* 1.取POS单号,STO与交货单号
  SELECT  A~VERKF
          C~ERDAT
          A~EBELN
          B~VBELN
          A~BSART
          C~WADAT_IST
          D~WERKS
    INTO TABLE GT_HEAD
    FROM EKKO AS A
    INNER JOIN LIPS AS B ON A~EBELN = B~VGBEL
    INNER JOIN LIKP AS C ON C~VBELN = B~VBELN
    INNER JOIN EKPO AS D ON A~EBELN = D~EBELN
    WHERE C~WERKS IN S_WERKS[]
      AND C~ERDAT IN S_DATE[]
      AND A~VERKF IN S_VERKF[]
      AND A~BSART IN ('ZSTR','ZTR1','ZUBR')   "ZUBR、ZTR1或ZSTR订单
      AND C~VBELN IN S_VBELN[]
      .

  DELETE GT_HEAD WHERE VERKF IS INITIAL.

  IF GT_HEAD IS INITIAL.
    MESSAGE S000(ZCB) WITH '没有找到相关数据' DISPLAY LIKE 'E'.
    LEAVE LIST-PROCESSING.
  ENDIF.

  SORT GT_HEAD BY VERKF.
  DELETE ADJACENT DUPLICATES FROM GT_HEAD COMPARING VERKF.

* 2.通过交货单取对应商品凭证判断时候收货
  SELECT A~MBLNR
         A~BUDAT
         A~CPUTM
         A~USNAM
         A~LE_VBELN
         B~WERKS
         B~LGORT
    INTO TABLE LT_MKPF
    FROM MKPF AS A
    INNER JOIN MSEG AS B ON A~MBLNR = B~MBLNR AND A~MJAHR = B~MJAHR
    INNER JOIN LIKP AS C ON A~LE_VBELN = C~VBELN
    FOR ALL ENTRIES IN GT_HEAD
    WHERE A~LE_VBELN  = GT_HEAD-VBELN
      AND B~XAUTO EQ ABAP_FALSE
      AND C~WADAT_IST NE '00000000'
        .

  SORT LT_MKPF BY VBELN MBLNR DESCENDING.
  DELETE ADJACENT DUPLICATES FROM LT_MKPF COMPARING VBELN.

  LOOP AT GT_HEAD ASSIGNING <LS_HEAD>.
    CLEAR LS_MKPF.
    READ TABLE LT_MKPF INTO LS_MKPF WITH KEY VBELN = <LS_HEAD>-VBELN BINARY SEARCH.
    IF SY-SUBRC = 0.
      <LS_HEAD>-MBLNR = LS_MKPF-MBLNR.
      <LS_HEAD>-BUDAT = LS_MKPF-BUDAT.
      <LS_HEAD>-CPUTM = LS_MKPF-CPUTM.
      <LS_HEAD>-USNAM = LS_MKPF-USNAM.
      <LS_HEAD>-WERKS_SH = LS_MKPF-WERKS_SH.
      <LS_HEAD>-LGORT = LS_MKPF-LGORT.
      <LS_HEAD>-ZZT   = '3'.
    ELSE.
      <LS_HEAD>-ZZT   = '1'.
    ENDIF.
  ENDLOOP.

  LOOP AT GT_HEAD INTO LS_HEAD.
    LS_HEAD-ZSH = '@PG@'.
    MODIFY GT_HEAD FROM LS_HEAD TRANSPORTING ZSH.
  ENDLOOP.

  IF P_1 EQ ABAP_TRUE.
    DELETE GT_HEAD WHERE ZZT EQ '3'.
  ENDIF.

ENDFORM.

*&---------------------------------------------------------------------*
*&取ITEM数据
*&---------------------------------------------------------------------*
FORM FM_GET_ITEM USING E_ROW E_COLUMN.

  DATA: LT_CHARG       TYPE STANDARD TABLE OF ZMM_CHARG,
        LT_CHARACTERS  TYPE STANDARD TABLE OF ZMM_CHARACTERS,
        LS_CHARACTERS  TYPE                   ZMM_CHARACTERS,
        LT_BATCH       TYPE  ZCB01_TY_T_ZSD_BATCH_PRICE_KEY,
        LT_BATCH_PRICE TYPE  ZCB01_TY_T_ZSD_BATCH_PRICE,
        LS_BATCH_PRICE LIKE LINE OF LT_BATCH_PRICE,
        LT_ITEM        TYPE STANDARD TABLE OF STRU_ITEM
        .

  READ TABLE GT_HEAD ASSIGNING <LS_HEAD> INDEX E_ROW.
* 1.取商品基本信息
  SELECT A~VBELN
         A~POSNR
         A~MATNR
         A~CHARG
         A~WERKS
         A~LFIMG
         A~MEINS
*         B~TEMPB
    FROM LIPS AS A
*    INNER JOIN MARA AS B ON A~MATNR = B~MATNR
    INTO TABLE GT_ITEM
   WHERE A~VBELN = <LS_HEAD>-VBELN
      .

  LT_ITEM[] = GT_ITEM[].
  DELETE LT_ITEM WHERE CHARG IS INITIAL.

  MOVE-CORRESPONDING LT_ITEM TO LT_CHARG.
  MOVE-CORRESPONDING LT_ITEM TO LT_BATCH.

* 2.取商品批次信息
  call function 'ZSDRP0020_FM_01'
    TABLES
      T_CHARG      = LT_CHARG
      T_CHARACTERS = LT_CHARACTERS.

* 3.取商品零售总金额
  call function 'ZG_GET_SALES_PRICE_ITAB'
    TABLES
      IT_BATCH       = LT_BATCH
      ET_BATCH_PRICE = LT_BATCH_PRICE.

* 4.整合商品信息到GT_ITEM表
  SORT LT_CHARACTERS BY MATNR CHARG.
  SORT LT_BATCH_PRICE BY MATNR CHARG WERKS.
  LOOP AT GT_ITEM ASSIGNING <LS_ITEM>.
    READ TABLE LT_CHARACTERS INTO LS_CHARACTERS WITH KEY MATNR = <LS_ITEM>-MATNR
                                                         CHARG = <LS_ITEM>-CHARG
                                                         BINARY SEARCH.
    IF SY-SUBRC = 0.
      <LS_ITEM>-ZSPTM = LS_CHARACTERS-ZSPTM.
      <LS_ITEM>-ZGYS  = LS_CHARACTERS-ZGYSDM.
      <LS_ITEM>-ZBQMC = LS_CHARACTERS-ZBQMC.
      <LS_ITEM>-ZXSJGJLDW = LS_CHARACTERS-ZXSJGJLDW.
      <LS_ITEM>-ZJZ   = LS_CHARACTERS-ZJZ.

      IF <LS_ITEM>-MEINS EQ 'G'.
        <LS_ITEM>-ZJS   = <LS_ITEM>-LFIMG / LS_CHARACTERS-ZJZ.
        <LS_ITEM>-ZKZ   = <LS_ITEM>-LFIMG.
      ELSE.
        <LS_ITEM>-MEINS = 'PC'.
        <LS_ITEM>-ZJS   = <LS_ITEM>-LFIMG.
        <LS_ITEM>-ZKZ   = <LS_ITEM>-LFIMG * LS_CHARACTERS-ZJZ.
      ENDIF.
    ELSE.
      <LS_ITEM>-ZJS   = <LS_ITEM>-LFIMG.
    ENDIF.
    READ TABLE LT_BATCH_PRICE INTO LS_BATCH_PRICE WITH KEY MATNR = <LS_ITEM>-MATNR
                                                           CHARG = <LS_ITEM>-CHARG
                                                           WERKS = <LS_ITEM>-WERKS
                                                           BINARY SEARCH.
    IF SY-SUBRC = 0.
      <LS_ITEM>-DPLSJ = LS_BATCH_PRICE-KBETR_MJJG.
      <LS_ITEM>-LSZJE = <LS_ITEM>-DPLSJ * <LS_ITEM>-ZJS.
    ENDIF.
  ENDLOOP.

ENDFORM.
