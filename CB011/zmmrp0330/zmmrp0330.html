*         END OF STRU_HEAD,
*
*         BEGIN OF STRU_ITEM,
*           ZSCM0140_CGPZH TYPE ZSCMIF0140_ITEM-ZSCM0140_CGPZH,   "采购凭证号
*           MATNR          TYPE ZSCMIF0140_ITEM-MATNR,            "旧料编号

*           BTGEW_Y        TYPE ZSCMIF0140_ITEM-BTGEW_Y,          "总重量
*           SHKZ           TYPE ZSCMIF0140_ITEM-SHKZ,             "收货克重
*         END OF STRU_ITEM,
*
*         BEGIN OF STRU_EXMARA,
*           MATNR  TYPE ZMMEX0010_MARA-MATNR,           "旧料编号
*           EX_GJS TYPE ZMMEX0010_MARA-EX_GJS,          "贵金属材质
*         END OF STRU_EXMARA,
*
*         BEGIN OF STRU_GJSCZ,
*           MATNR TYPE ZMMDG0010_GJSCZ-MATNR,          "旧料编号(原料)
*           GJSCZ TYPE ZMMDG0010_GJSCZ-GJSCZ,          "贵金属材质
*           GJSBL TYPE ZMMDG0010_GJSCZ-GJSBL,          "贵金属比例
*           YLBL  TYPE ZMMDG0010_GJSCZ-YLBL,           "原料比例
*         END OF STRU_GJSCZ,
*
*         BEGIN OF STRU_MSLB,
*           MATNR TYPE MSLB-MATNR,                     "商品编号
*           LBLAB TYPE MSLB-LBLAB,                     "库存数量
*           WERKS TYPE MSLB-WERKS,                     "地点
*           CHARG TYPE MSLB-CHARG,                     "批次号
**           SOBKZ TYPE MSLB-SOBKZ,                     "特殊库存标识
*           LIFNR TYPE MSLB-LIFNR,                     "供应商
*         END OF STRU_MSLB.

  TYPES: BEGIN OF STRU_DDQL,
*           ZSCM0140_CGPZH TYPE ZSCMIF0140_HEAD-ZSCM0140_CGPZH,    "采购凭证号

           LIFNR    TYPE ZSCMIF0140_HEAD-LIFNR,             "供应商
           MATNR    TYPE ZSCMIF0140_ITEM-MATNR,

*           ZSPJG          TYPE ZSCMIF0140_HEAD-ZSPJG,             "审批结果
*           ZDDLB          TYPE ZSCMIF0140_HEAD-ZDDLB,             "订单类别

           "旧料编号
           PO_CLOSE TYPE ZSCMIF0140_ITEM-PO_CLOSE,         "订单关闭状态
           BTGEW_Y  TYPE ZSCMIF0140_ITEM-BTGEW_Y,          "总重量
           SHKZ     TYPE ZSCMIF0140_ITEM-SHKZ,             "收货克重

*           GJSCZ          TYPE ZMMDG0010_GJSCZ-GJSCZ,          "贵金属材质
           GJSBL    TYPE ZMMDG0010_GJSCZ-GJSBL,          "贵金属比例
           YLBL     TYPE ZMMDG0010_GJSCZ-YLBL,           "原料比例

         BEGIN OF STRU_MSLB,

           WERKS TYPE MSLB-WERKS,                     "地点
           CHARG TYPE MSLB-CHARG,                     "批次号
*           SOBKZ TYPE MSLB-SOBKZ,                     "特殊库存标识
           LIFNR TYPE MSLB-LIFNR,                     "供应商
         END OF STRU_MSLB,

           YLBL    TYPE ZMMDG0010_GJSCZ-YLBL,

         BEGIN OF STRU_NAME1,

           NAME1 TYPE LFA1-NAME1,

  DATA: IT_DDQL  TYPE STANDARD TABLE OF STRU_DDQL,
        WA_DDQL  TYPE                   STRU_DDQL,

        IT_DDHZ  TYPE STANDARD TABLE OF STRU_DDHZ,
        WA_DDHZ  TYPE                   STRU_DDHZ,

        IT_MSLB  TYPE STANDARD TABLE OF STRU_MSLB,
        WA_MSLB  TYPE                   STRU_MSLB,

        IT_NAME1 TYPE STANDARD TABLE OF STRU_NAME1,
        IT_LIFNR TYPE STANDARD TABLE OF STRU_NAME1,
        WA_NAME1 TYPE                   STRU_NAME1,

        IT_MAKT  TYPE STANDARD TABLE OF STRU_MAKT,
        IT_MATNR TYPE STANDARD TABLE OF STRU_MAKT,
        WA_MAKT  TYPE                   STRU_MAKT.

*--------------------1.取采购订单欠料相关------------------------
  SELECT B~LIFNR
         D~MATNR
         A~PO_CLOSE
         A~BTGEW_Y
         A~SHKZ
         D~GJSBL
         D~YLBL
    INTO TABLE IT_DDQL
    FROM ZSCMIF0140_ITEM AS A
    INNER JOIN ZSCMIF0140_HEAD AS B ON A~ZSCM0140_CGPZH = B~ZSCM0140_CGPZH
    INNER JOIN ZMMEX0010_MARA AS C ON A~MATNR = C~MATNR
    INNER JOIN ZMMDG0010_GJSCZ AS D ON C~EX_GJS = D~GJSCZ   " C~MATNR = D~MATNR AND
    WHERE B~LIFNR IN S_LIFNR
      AND A~MATNR IN S_MATNR
      AND B~ZSPJG EQ '通过'
      AND B~ZDDLB EQ '委外'
      .
*--------------------2.取MSLB,入库欠料库存-----------------
  SELECT MATNR
         LBLAB
         WERKS
         CHARG
         LIFNR
    FROM MSLB
    INTO TABLE IT_MSLB
    WHERE MATNR IN S_MATNR
      AND LIFNR IN S_LIFNR
      AND LBLAB NE 0
      .
*---------------------3.整合数据到ALV表---------------------
  LOOP AT IT_MSLB INTO WA_MSLB.
    CLEAR WA_ALV.
    WA_ALV-LIFNR  = WA_MSLB-LIFNR.
    WA_ALV-MATNR  = WA_MSLB-MATNR.
    WA_ALV-RKQLKZ = WA_MSLB-LBLAB.
    APPEND WA_ALV TO IT_ALV.
  ENDLOOP.

  DATA:  L_BTGEW     TYPE MSLB-LBLAB,
         L_SHKZ      TYPE MSLB-LBLAB,
         WA_TMP_DDQL TYPE STRU_DDQL.

  SORT IT_DDQL BY LIFNR MATNR.
  LOOP AT IT_DDQL INTO WA_TMP_DDQL.

    WA_DDQL = WA_TMP_DDQL.

    IF WA_DDQL-PO_CLOSE = 'X'.
      L_SHKZ  = L_SHKZ + WA_DDQL-SHKZ.
    ELSEIF WA_DDQL-PO_CLOSE IS INITIAL.
      L_BTGEW = L_BTGEW + WA_DDQL-BTGEW_Y.
    ENDIF.

    AT END OF MATNR.
      CLEAR WA_ALV.
      WA_ALV-MATNR = WA_DDQL-MATNR.
      WA_ALV-LIFNR = WA_DDQL-LIFNR.
      WA_ALV-DDQLKZ = ( L_SHKZ + L_BTGEW ) * WA_DDQL-YLBL / WA_DDQL-GJSBL.
      APPEND WA_ALV TO IT_ALV.
      CLEAR: L_SHKZ, L_BTGEW.
    ENDAT.

  ENDLOOP.

*----------------------4.取字段描述-------------------------
  SORT IT_ALV BY LIFNR.
  MOVE-CORRESPONDING IT_ALV TO IT_LIFNR.
  DELETE ADJACENT DUPLICATES FROM IT_LIFNR COMPARING LIFNR.
  SORT IT_ALV BY MATNR.
  MOVE-CORRESPONDING IT_ALV TO IT_MATNR.
  DELETE ADJACENT DUPLICATES FROM IT_MATNR COMPARING MATNR.

  SELECT LIFNR
         NAME1
    FROM LFA1
    INTO TABLE IT_NAME1
    FOR ALL ENTRIES IN IT_LIFNR
    WHERE LIFNR = IT_LIFNR-LIFNR.

  SELECT MATNR
         MAKTX
    FROM MAKT
    INTO TABLE IT_MAKT
    FOR ALL ENTRIES IN IT_MATNR
    WHERE MATNR = IT_MATNR-MATNR.

  SORT IT_NAME1 BY LIFNR.
  SORT IT_MAKT  BY MATNR.

  LOOP AT IT_ALV INTO WA_ALV.

    CLEAR WA_NAME1.
    READ TABLE IT_NAME1 INTO WA_NAME1 WITH KEY LIFNR = WA_ALV-LIFNR BINARY SEARCH.
    IF SY-SUBRC = 0.
      WA_ALV-NAME1 = WA_NAME1-NAME1.
    ENDIF.

    CLEAR WA_MAKT.
    READ TABLE IT_MAKT INTO WA_MAKT WITH KEY MATNR = WA_ALV-MATNR BINARY SEARCH.
    IF SY-SUBRC = 0.
      WA_ALV-MAKTX = WA_MAKT-MAKTX.
    ENDIF.

    MODIFY IT_ALV FROM WA_ALV TRANSPORTING NAME1 MAKTX.

  ENDLOOP.

****************20151103-根据供应商,物料号合并克重***********************
  DATA: L_DDKZ TYPE BSTMG,
        L_RKKZ TYPE BSTMG,
        IT_TMP TYPE STANDARD TABLE OF STRU_ALV,
        WA_TMP TYPE                   STRU_ALV.

  SORT IT_ALV BY LIFNR MATNR.
  LOOP AT IT_ALV INTO WA_TMP.

    CLEAR WA_ALV.
    WA_ALV = WA_TMP.

    L_DDKZ  = L_DDKZ + WA_ALV-DDQLKZ.
    L_RKKZ  = L_RKKZ + WA_ALV-RKQLKZ.

    AT END OF MATNR.

      CLEAR WA_TMP.
      WA_TMP-LIFNR  = WA_ALV-LIFNR.
      WA_TMP-MATNR  = WA_ALV-MATNR.
      WA_TMP-NAME1  = WA_ALV-NAME1.
      WA_TMP-MAKTX  = WA_ALV-MAKTX.
      WA_TMP-DDQLKZ = L_DDKZ.
      WA_TMP-RKQLKZ = L_RKKZ.
      APPEND WA_TMP TO IT_TMP.

      CLEAR: L_DDKZ,
             L_RKKZ.
    ENDAT.

  ENDLOOP.

  CLEAR IT_ALV[].
  IT_ALV[] = IT_TMP[].

ENDFORM.

FORM FM_MSEG.

  TYPES: BEGIN OF STRU_MSEG,

           MATNR TYPE MSEG-MATNR,
           LIFNR TYPE MSEG-LIFNR,
           BWART TYPE MSEG-BWART,
           MENGE TYPE MSEG-MENGE,

         END OF STRU_MSEG.

  DATA: IT_MSEG  TYPE STANDARD TABLE OF STRU_MSEG,
        WA_MSEG  TYPE                   STRU_MSEG,
        WA_TMP   TYPE                   STRU_MSEG,

        L_MENGE  TYPE                   MSEG-MENGE,

        IT_MATNR TYPE STANDARD TABLE OF STRU_MSEG,
        WA_MATNR TYPE                   STRU_MSEG.
*--------1.根据MATNR,LIFNR从MSEG取出所有移动类型为542,541的凭证---------
  MOVE-CORRESPONDING IT_ALV TO IT_MATNR.
  SORT IT_MATNR BY MATNR LIFNR.
  DELETE ADJACENT DUPLICATES FROM IT_MATNR COMPARING MATNR LIFNR.

  SELECT MATNR
         LIFNR
         BWART
         MENGE
    FROM MSEG
    INTO TABLE IT_MSEG
    FOR ALL ENTRIES IN IT_MATNR
    WHERE MATNR = IT_MATNR-MATNR
      AND LIFNR = IT_MATNR-LIFNR
      AND BWART IN ('541','542')
      AND XAUTO = ' '
      .

  IF SY-SUBRC <> 0.
    RETURN.
  ENDIF.

*--------2.汇总数量-----------
  CLEAR IT_MATNR.

  SORT IT_ALV BY MATNR LIFNR.
  SORT IT_MSEG BY MATNR LIFNR BWART.
  LOOP AT IT_MSEG INTO WA_TMP.
    CLEAR WA_MSEG.
    WA_MSEG = WA_TMP.

    L_MENGE = WA_MSEG-MENGE.

    AT END OF BWART.
      CLEAR WA_MATNR.
      WA_MATNR-MATNR = WA_MSEG-MATNR.
      WA_MATNR-LIFNR = WA_MSEG-LIFNR.
      WA_MATNR-BWART = WA_MSEG-BWART.
      WA_MATNR-MENGE = L_MENGE.
      APPEND WA_MATNR TO IT_MATNR.
      CLEAR L_MENGE.
    ENDAT.

  ENDLOOP.

  SORT IT_MATNR BY MATNR LIFNR.
  LOOP AT IT_ALV INTO WA_ALV .
    CLEAR WA_MATNR.
    READ TABLE IT_MATNR INTO WA_MATNR WITH KEY MATNR = WA_ALV-MATNR
