*&---------------------------------------------------------------------*

FORM FRM_GET_DATA .
  DATA:
    LT_DATA TYPE STANDARD TABLE OF TY_DATA,
    LS_DATA TYPE TY_DATA,
    LT_DOC  TYPE STANDARD TABLE OF ZFIRP0140_DOC,
    LS_DOC  TYPE ZFIRP0140_DOC.

  REFRESH GT_DATA.
  SELECT *
    INTO CORRESPONDING FIELDS OF TABLE GT_DATA
    FROM ZFIRP0140_HEADER
   WHERE ZJSD_NO IN S_NO
     AND LIFNR IN S_LIFNR
     AND ZSTART_DATE IN S_SDATE
     AND ZEND_DATE IN S_EDATE
     AND ZFAPIAO IN ('1','3')
     AND ZSH = 'X'.
  IF SY-SUBRC <> 0.
    MESSAGE '没有数据' TYPE 'S' DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.

  REFRESH LT_DATA.
  LOOP AT GT_DATA INTO GS_DATA WHERE ZFAPIAO = '3'.
    APPEND GS_DATA TO LT_DATA.
  ENDLOOP.

* 检查对应发票是否被冲销
  IF LT_DATA IS NOT INITIAL.
    SELECT *
      INTO TABLE LT_DOC
      FROM ZFIRP0140_DOC
       FOR ALL ENTRIES IN LT_DATA
     WHERE ZJSD_NO = LT_DATA-ZJSD_NO.
    LOOP AT GT_DATA INTO GS_DATA WHERE ZFAPIAO = '3'.
      READ TABLE LT_DOC INTO LS_DOC
        WITH KEY ZJSD_NO = GS_DATA-ZJSD_NO.
      IF SY-SUBRC = 0.
        SELECT COUNT(*)
          FROM BKPF
         WHERE BUKRS = LS_DOC-BUKRS
           AND BELNR = LS_DOC-BELNR
           AND GJAHR = LS_DOC-GJAHR
           AND STBLG = ''.
        IF SY-SUBRC <> 0.
          DELETE FROM ZFIRP0140_DOC WHERE ZJSD_NO = GS_DATA-ZJSD_NO.
          UPDATE ZFIRP0140_HEADER
             SET ZFAPIAO = '1'
           WHERE ZJSD_NO = GS_DATA-ZJSD_NO.
          IF SY-SUBRC = 0.
            COMMIT WORK AND WAIT.
            GS_DATA-ZFAPIAO = '1'.
            MODIFY GT_DATA FROM GS_DATA TRANSPORTING ZFAPIAO.
          ELSE.
            ROLLBACK WORK.
          ENDIF.
        ENDIF.
      ENDIF.
    ENDLOOP.
  ENDIF.
  DELETE GT_DATA WHERE ZFAPIAO = '3'.

  IF GT_DATA IS INITIAL.
    MESSAGE '没有数据' TYPE 'S' DISPLAY LIKE 'E'.
  ELSE.
    LOOP AT GT_DATA INTO GS_DATA.
      SELECT SINGLE NAME1
        INTO GS_DATA-NAME1
        FROM LFA1
       WHERE LIFNR = GS_DATA-LIFNR.
      IF SY-SUBRC = 0.
        MODIFY GT_DATA FROM GS_DATA TRANSPORTING NAME1.
      ENDIF.
    ENDLOOP.
    SORT GT_DATA BY LIFNR ZJSD_NO SDATE.
    CALL SCREEN 9001.
  ENDIF.
ENDFORM.

FORM FRM_TAX.
  DATA:
    L_KNUMH TYPE A003-KNUMH,
    L_KBETR TYPE KONP-KBETR.

  LOOP AT GT_DATA INTO GS_DATA.
*    IF GS_DATA-MWSKZ IS NOT INITIAL.
*      CLEAR:
*        L_KBETR,
*        L_KNUMH.
*      SELECT SINGLE KNUMH
*        INTO L_KNUMH
*        FROM A003
*       WHERE KAPPL = 'TX'
*         AND KSCHL = 'MWVS'
*         AND ALAND = 'CN'
*         AND MWSKZ = GS_DATA-MWSKZ.
*      SELECT KBETR
*        INTO L_KBETR
*        FROM KONP
*          UP TO 1 ROWS
*       WHERE KNUMH = L_KNUMH
*         AND KAPPL = 'TX'
*         AND KSCHL = 'MWVS'.
*      ENDSELECT.
*      L_KBETR = L_KBETR / 10.
*      GS_DATA-TAX = GS_DATA-ZZJE * L_KBETR / 100.
*    ENDIF.

    " 余额 = 结算金额 + 进项税金额 - 应付账款
    GS_DATA-BALANCE = GS_DATA-ZZJE + GS_DATA-TAX - GS_DATA-PAY.
    MODIFY GT_DATA FROM GS_DATA TRANSPORTING TAX BALANCE.
  ENDLOOP.

ENDFORM.

FORM FRM_CREATE_INV .

    LS_ITEM   TYPE BAPIACCR09,
    LT_GL     TYPE STANDARD TABLE OF BAPIACGL09,
    LS_GL     TYPE BAPIACGL09,
    LT_PAY    TYPE STANDARD TABLE OF BAPIACAP09,
    LS_PAY    TYPE BAPIACAP09,
    LT_RETURN TYPE STANDARD TABLE OF BAPIRET2,
    LS_RETURN TYPE BAPIRET2,
    LT_DOC    TYPE STANDARD TABLE OF ZFIRP0140_DOC,
    LS_DOC    TYPE ZFIRP0140_DOC,
    L_M1      TYPE ZFIRP0140_HEADER-ZZJE,     "结算金额
    L_M2      TYPE ZFIRP0140_HEADER-ZZJE,     "进项税金额

    L_M3      TYPE ZFIRP0140_HEADER-ZZJE,     "应付金额
    L_M4      TYPE ZFIRP0140_HEADER-ZZJE,     "余额

    L_NO      TYPE BAPIACHE09-OBJ_KEY,
    L_PRCTR   TYPE PRCTR,
    L_MESSAGE TYPE CHAR255.

  READ TABLE GT_DATA TRANSPORTING NO FIELDS
    WITH KEY BOX = 'X'
             DATUM = ''.
  IF SY-SUBRC = 0.
    MESSAGE '请输入发票日期' TYPE 'S' DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.

  READ TABLE GT_DATA TRANSPORTING NO FIELDS
    WITH KEY BOX = 'X'
             PAY = 0.
  IF SY-SUBRC = 0.
    MESSAGE '请输入应付账款' TYPE 'S' DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.

  REFRESH LT_DATA.
  LOOP AT GT_DATA INTO GS_DATA WHERE BOX = 'X'
                                 AND MESSAGE = ''.
    APPEND GS_DATA TO LT_DATA.
  ENDLOOP.
  IF SY-SUBRC <> 0.
    MESSAGE '请选中消息为空的行项目' TYPE 'S' DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.
  CLEAR:
    LS_HEAD,
    LS_ITEM,
    LS_GL,
    LS_PAY,
    L_M1,
    L_M2,
    L_M3,
    L_M4.

  REFRESH:
    LT_ITEM,
    LT_GL,
    LT_PAY,
    LT_RETURN.

  LS_HEAD-DOC_TYPE = 'Y2'.
  LS_HEAD-DOC_DATE = P_DATUM.

*   ls_head-pstng_date = sy-datum.
  LS_HEAD-PSTNG_DATE = P_DATUM2.
***End of Edit CAOSHICHUN.
  LS_HEAD-USERNAME   = SY-UNAME.
  LS_HEAD-HEADER_TXT = '联营商品发票校验'.

* 按供应商汇总应付金额

           LIFNR TYPE LFA1-LIFNR,
           PAY   TYPE ZFIRP0140_HEADER-ZZJE,
         END OF STRU_LIFNR.
  DATA: LT_LIFNR TYPE STANDARD TABLE OF STRU_LIFNR,
        LS_LIFNR TYPE                   STRU_LIFNR,
        LS_TMP   TYPE                   TY_DATA.

  SORT LT_DATA BY LIFNR.
  LOOP AT LT_DATA INTO LS_TMP.

    GS_DATA = LS_TMP.

    L_M1 = L_M1 + GS_DATA-ZZJE.     "结算金额
    L_M2 = L_M2 + GS_DATA-TAX.      "进项税金额
    L_M3 = L_M3 + GS_DATA-PAY.      "应付金额
    L_M4 = L_M4 + GS_DATA-BALANCE.  "余额
    LS_HEAD-COMP_CODE = GS_DATA-ZPO_BUKRS.

      LS_LIFNR-LIFNR = GS_DATA-LIFNR.
      LS_LIFNR-PAY   = L_M3.
      APPEND LS_LIFNR TO LT_LIFNR.

      CLEAR L_M3.
    ENDAT.

  ENDLOOP.

  CONCATENATE 'P'
              LS_HEAD-COMP_CODE
              '001'
         INTO L_PRCTR.

  CLEAR LS_ITEM.
  LS_ITEM-ITEMNO_ACC ='0010'.
  LS_ITEM-AMT_DOCCUR = L_M1.
  LS_ITEM-CURRENCY = 'CNY'.
  APPEND LS_ITEM TO LT_ITEM.
  CLEAR LS_GL.
  LS_GL-ITEMNO_ACC ='0010'.
  LS_GL-GL_ACCOUNT = '2202099002'.
  LS_GL-PROFIT_CTR = L_PRCTR.
  LS_GL-ITEM_TEXT = '联营商品发票校验'.

  LS_GL-GL_ACCOUNT = '6401009099'.
  LS_GL-ITEM_TEXT  = '联营商品发票校验'.

    ENDIF.
  ELSE.
    CALL FUNCTION 'BAPI_TRANSACTION_ROLLBACK'.

    CLEAR L_MESSAGE.
    LOOP AT LT_RETURN INTO LS_RETURN WHERE TYPE = 'E'.
      CONCATENATE L_MESSAGE
                  LS_RETURN-MESSAGE
             INTO L_MESSAGE.
    ENDLOOP.
    LOOP AT GT_DATA INTO LS_WB WHERE BOX = 'X'
                                 AND MESSAGE = ''.
      LS_WB-MESSAGE = L_MESSAGE.
      MODIFY GT_DATA FROM LS_WB TRANSPORTING MESSAGE.
    ENDLOOP.
  ENDIF.

ENDFORM.

*&---------------------------------------------------------------------*
*&      Form  FRM_FIND
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM FRM_FIND .
  DATA:
    LT_DATA TYPE STANDARD TABLE OF TY_DATA,
    LT_DOC  TYPE STANDARD TABLE OF ZFIRP0140_DOC,
    LS_DOC  TYPE ZFIRP0140_DOC.

  REFRESH:
    GT_DOC,
    LT_DATA.

  SELECT *
    INTO CORRESPONDING FIELDS OF TABLE LT_DATA
    FROM ZFIRP0140_HEADER
   WHERE ZJSD_NO IN S_NO
     AND LIFNR IN S_LIFNR
     AND ZSTART_DATE IN S_SDATE
     AND ZEND_DATE IN S_EDATE
     AND ZFAPIAO = '3'.
  IF SY-SUBRC <> 0.
    MESSAGE '没有数据' TYPE 'S' DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.

* 检查对应发票是否被冲销
  IF LT_DATA IS NOT INITIAL.
    SELECT *
      INTO TABLE LT_DOC
      FROM ZFIRP0140_DOC
       FOR ALL ENTRIES IN LT_DATA
     WHERE ZJSD_NO = LT_DATA-ZJSD_NO.
    LOOP AT LT_DATA INTO GS_DATA.
      READ TABLE LT_DOC INTO LS_DOC
        WITH KEY ZJSD_NO = GS_DATA-ZJSD_NO.
      IF SY-SUBRC = 0.
        SELECT COUNT(*)
          FROM BKPF
         WHERE BUKRS = LS_DOC-BUKRS
           AND BELNR = LS_DOC-BELNR
           AND GJAHR = LS_DOC-GJAHR
           AND STBLG = ''.
        IF SY-SUBRC <> 0.
          DELETE FROM ZFIRP0140_DOC WHERE ZJSD_NO = GS_DATA-ZJSD_NO.
          UPDATE ZFIRP0140_HEADER
             SET ZFAPIAO = '1'
           WHERE ZJSD_NO = GS_DATA-ZJSD_NO.
          IF SY-SUBRC = 0.
            COMMIT WORK AND WAIT.
            GS_DATA-ZFAPIAO = '1'.
            MODIFY LT_DATA FROM GS_DATA TRANSPORTING ZFAPIAO.
          ELSE.
            ROLLBACK WORK.
          ENDIF.
        ENDIF.
      ENDIF.
    ENDLOOP.
  ENDIF.
  DELETE LT_DATA WHERE ZFAPIAO = '1'.
  IF LT_DATA IS INITIAL.
    MESSAGE '没有数据' TYPE 'S' DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.

  SELECT *
    INTO TABLE GT_DOC
    FROM ZFIRP0140_DOC
     FOR ALL ENTRIES IN LT_DATA
   WHERE ZJSD_NO = LT_DATA-ZJSD_NO.

ENDFORM.
