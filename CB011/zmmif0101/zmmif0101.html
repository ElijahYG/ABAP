*&---------------------------------------------------------------------*

FORM FM_GET_DATA USING L_CHAR01.

  IF GRT_ZPDBH IS INITIAL.
    GRT_ZPDBH[] = S_ZPDBH[].
  ENDIF.

  CASE L_CHAR01.

    WHEN 'HEAD'.    "取盘点单号

      SELECT ZPDBH
             WERKS
             LGORT
             ZDJRQ
        FROM ZMMIF0150_PDHEAD
        INTO TABLE GT_OUT_TAB
        WHERE ZPDBH IN GRT_ZPDBH
          AND WERKS IN S_WERKS
          AND LGORT IN S_LGORT
          AND ZDJRQ IN S_ZDJRQ
          .

      IF GT_OUT_TAB IS INITIAL.
        MESSAGE S000(ZCB) WITH '没有找到符合条件的数据.' DISPLAY LIKE 'E'.
        LEAVE LIST-PROCESSING.
      ENDIF.

    WHEN 'SPITEM'. "取实盘数据

      SELECT A~ZPDBH
             A~ZSPTM
             A~ZPDSPS
             A~ZPDRQ
             A~ZPDSJ
             A~ZPDR
             A~ZPDFHR
             A~ZPDQBH
        FROM ZMMIF0150_SPITEM AS A
        INNER JOIN ZMMIF0150_PDHEAD AS B ON A~ZPDBH = B~ZPDBH
        INTO TABLE GT_SPITEM
        WHERE A~ZPDBH IN GRT_ZPDBH
          AND B~WERKS IN S_WERKS
          AND B~LGORT IN S_LGORT
          AND B~ZDJRQ IN S_ZDJRQ
          .

    WHEN 'PDITEM'.  "取得盘点表数

      SELECT *
        FROM ZMMIF0150_PDITEM
        INTO CORRESPONDING FIELDS OF TABLE GT_PDITEM
        FOR ALL ENTRIES IN GT_SPITEM
        WHERE ZPDBH = GT_SPITEM-ZPDBH
          AND ZSPTM = GT_SPITEM-ZSPTM
          .

  ENDCASE.

ENDFORM.

*&---------------------------------------------------------------------*
*& 计算逻辑
*&---------------------------------------------------------------------*
FORM FM_CAL_DATA USING L_CHAR02.

  DATA: LT_OUT_TAB TYPE STANDARD TABLE OF STRU_OUT.

  CLEAR GL_CHECK.

  CASE L_CHAR02.

    WHEN  'HEAD'.   "通过盘点单号从SPITEM中取盘点日期

      LOOP AT GT_OUT_TAB ASSIGNING <GS_OUT_TAB>.
        SELECT SINGLE ZPDRQ FROM ZMMIF0150_SPITEM INTO <GS_OUT_TAB>-ZPDRQ WHERE ZPDBH = <GS_OUT_TAB>-ZPDBH.
      ENDLOOP.

    WHEN  'JCRQ'.   "检查盘点日期判断是否已盘

      MOVE-CORRESPONDING GT_OUT_TAB TO LT_OUT_TAB.
      DELETE LT_OUT_TAB WHERE FLAG EQ ABAP_FALSE.
      DELETE LT_OUT_TAB WHERE ZPDRQ IS INITIAL.
      IF SY-SUBRC = 0.
        MESSAGE S000(ZCB) WITH '没有盘点日期的盘点单不能导入.' DISPLAY LIKE 'E'.
        GL_CHECK = ABAP_TRUE.
      ENDIF.

  ENDCASE.

ENDFORM.

    IMPORTING

  IF NOT L_GRID IS INITIAL.
    CALL METHOD L_GRID->REFRESH_TABLE_DISPLAY. "refresh alv

  CASE RS_UCOMM.       "捕捉用户动作

    WHEN 'BTN_BACK'.

      GL_NUM = 1.
*        PERFORM FM_ALVSHOW.

*&---------------------------------------------------------------------*
FORM FM_MODIFY_TAB.

  DATA: LT_PDITEM  TYPE STANDARD TABLE OF ZMMIF0150_PDITEM,
        LT_OUT_TAB TYPE STANDARD TABLE OF STRU_OUT,
        LS_OUT_TAB TYPE                   STRU_OUT,
        LRS_ZPDBH  LIKE LINE OF GRT_ZPDBH,
        L_ANSWER   TYPE C.


  FIELD-SYMBOLS: <LS_PDITEM> TYPE ZMMIF0150_PDITEM,
                 <LS_SPITEM> TYPE STRU_SPITEM.

  CLEAR GL_CHECK.

* 筛选选中的盘点单号
  MOVE-CORRESPONDING GT_OUT_TAB TO LT_OUT_TAB.
  DELETE LT_OUT_TAB WHERE FLAG EQ ABAP_FALSE.
  LOOP AT LT_OUT_TAB INTO LS_OUT_TAB.
    CLEAR LRS_ZPDBH.
    LRS_ZPDBH-SIGN   = 'I' .
    LRS_ZPDBH-OPTION = 'EQ'.
    LRS_ZPDBH-LOW    = LS_OUT_TAB-ZPDBH.
    APPEND LRS_ZPDBH TO GRT_ZPDBH.
  ENDLOOP.

* 取实盘数据
  PERFORM FM_GET_DATA USING 'SPITEM'.

* 取得计划数据
  PERFORM FM_GET_DATA USING 'PDITEM'.

* 全量更新还是增量更新?
  LT_PDITEM = GT_PDITEM.
  DELETE LT_PDITEM WHERE ZSPS IS NOT INITIAL. "删除实盘数不为空的项目
  IF SY-SUBRC = 0.   "如果有值则说明已经更新过
    PERFORM FM_WINDOWS USING L_ANSWER.
    IF L_ANSWER EQ 'N' OR  L_ANSWER EQ 'A' .
      MESSAGE S000(ZCB) WITH '操作已取消'.
      GL_CHECK = ABAP_TRUE.
      RETURN.
    ENDIF.
  ENDIF.

* 更新实盘数内表
  SORT GT_PDITEM BY ZPDBH ZSPTM.
  SORT GT_SPITEM BY ZPDBH ZSPTM.
  LOOP AT GT_PDITEM ASSIGNING <LS_PDITEM>.

    READ TABLE GT_SPITEM ASSIGNING <LS_SPITEM> WITH KEY ZPDBH = <LS_PDITEM>-ZPDBH
                                                        ZSPTM = <LS_PDITEM>-ZSPTM
                                                        BINARY SEARCH
                                                        .
    IF SY-SUBRC = 0.
      <LS_PDITEM>-ZSPS   = <LS_SPITEM>-ZPDSPS.
      <LS_PDITEM>-ZPDYGH = <LS_SPITEM>-ZPDR.
    ENDIF.

  ENDLOOP.

* 更新透明表实盘数
  MODIFY ZMMIF0150_PDITEM FROM TABLE GT_PDITEM.
  IF SY-SUBRC NE 0.
    MESSAGE S000(ZCB) WITH '数据库更新失败.' DISPLAY LIKE 'E'.
    GL_CHECK = ABAP_TRUE.
    RETURN.
  ENDIF.

* 更新盘点抬头表盘点日期
  DATA: LT_PDHEAD  TYPE STANDARD TABLE OF ZMMIF0150_PDHEAD.
*        LS_OUT_TAB TYPE                   STRU_OUT.

  FIELD-SYMBOLS: <LS_PDHEAD> TYPE ZMMIF0150_PDHEAD.
  SELECT *
    FROM ZMMIF0150_PDHEAD
    INTO CORRESPONDING FIELDS OF TABLE LT_PDHEAD
    FOR ALL ENTRIES IN GT_OUT_TAB
    WHERE ZPDBH = GT_OUT_TAB-ZPDBH
    .

  SORT GT_OUT_TAB BY ZPDBH.
  LOOP AT LT_PDHEAD ASSIGNING <LS_PDHEAD>.

    READ TABLE GT_OUT_TAB INTO LS_OUT_TAB WITH KEY ZPDBH = <LS_PDHEAD>-ZPDBH BINARY SEARCH.

    IF SY-SUBRC = 0.
      <LS_PDHEAD>-ZPDRQ = LS_OUT_TAB-ZPDRQ.
    ENDIF.

  ENDLOOP.

  MODIFY ZMMIF0150_PDHEAD FROM TABLE LT_PDHEAD.
  IF SY-SUBRC NE 0.
    MESSAGE S000(ZCB) WITH '数据库更新失败.' DISPLAY LIKE 'E'.
    GL_CHECK = ABAP_TRUE.
    RETURN.
  ENDIF.

ENDFORM.
