    ENDIF.

FORM FM_GET_NUM TABLES IT_TABLE.

  TYPES: BEGIN OF STRU_AUSP,
           OBJEK    TYPE AUSP-OBJEK,
           CUOBJ_BM TYPE MCH1-CUOBJ_BM,
         END OF STRU_AUSP.

  DATA: LT_AUSP  TYPE STANDARD TABLE OF STRU_AUSP,
        LS_AUSP  TYPE                   STRU_AUSP,
        RT_ATFLV TYPE RANGE OF AUSP-ATFLV,
        RW_ATFLV LIKE LINE OF RT_ATFLV,
        RT_OBJEK TYPE RANGE OF AUSP-OBJEK,
        RW_OBJEK LIKE LINE OF RT_OBJEK,
        LT_MCH1  TYPE STANDARD TABLE OF ZMM_CHARG,
        LS_MCH1  TYPE                   ZMM_CHARG.

  LOOP AT S_NUM.
    CLEAR RW_ATFLV.
    RW_ATFLV-SIGN   = 'I'.
    IF S_NUM-HIGH IS INITIAL.
      RW_ATFLV-OPTION = 'EQ'.
      RW_ATFLV-LOW    = S_NUM-LOW.
    ELSE.
      RW_ATFLV-OPTION = 'BT'.
      RW_ATFLV-LOW    = S_NUM-LOW.
      RW_ATFLV-HIGH   = S_NUM-HIGH.
    ENDIF.
    APPEND RW_ATFLV TO RT_ATFLV.
  ENDLOOP.

  SELECT OBJEK
    UP TO P_ROWS ROWS
    FROM AUSP AS A
    INNER JOIN CABN AS B ON A~ATINN = B~ATINN
    INTO CORRESPONDING FIELDS OF TABLE LT_AUSP
    WHERE ATFLV IN RT_ATFLV
      AND B~ATNAM = S_ATNAM-LOW
      .

  IF LT_AUSP IS INITIAL.
    MESSAGE S000(ZCB) WITH '没东西啊' DISPLAY LIKE 'E'.
    LEAVE LIST-PROCESSING.
  ELSE.
    LOOP AT LT_AUSP INTO LS_AUSP.
      LS_AUSP-CUOBJ_BM = LS_AUSP-OBJEK.
      MODIFY LT_AUSP FROM LS_AUSP TRANSPORTING CUOBJ_BM.
    ENDLOOP.
  ENDIF.

  SELECT MATNR
         CHARG
    FROM MCH1
    INTO TABLE IT_TABLE[]
    FOR ALL ENTRIES IN LT_AUSP
    WHERE CUOBJ_BM = LT_AUSP-CUOBJ_BM.

ENDFORM.

*&---------------------------------------------------------------------*
FORM FM_COUNT_ROWS.
  TYPES: BEGIN OF STRU_AUSP,
           OBJEK    TYPE AUSP-OBJEK,
           CUOBJ_BM TYPE MCH1-CUOBJ_BM,
         END OF STRU_AUSP.

  DATA: LT_AUSP TYPE STANDARD TABLE OF STRU_AUSP,
        LS_AUSP TYPE                   STRU_AUSP.
  DATA: L_COUNT  TYPE I,
        RT_ATFLV TYPE RANGE OF AUSP-ATFLV,
        RW_ATFLV LIKE LINE OF RT_ATFLV.

  IF P_1 EQ ABAP_FALSE.

    SELECT COUNT(*)
      INTO L_COUNT
      FROM ZV_CHARG
      WHERE ATWRT IN S_ATWRT[]
        AND ATNAM IN S_ATNAM[]
          .

    IF SY-SUBRC = 0.
      MESSAGE S000(ZCB) WITH '共有' L_COUNT '条数据.'.
    ELSE.
      MESSAGE S000(ZCB) WITH '没有相关数据' DISPLAY LIKE 'E'.
    ENDIF.

  ELSE.

    LOOP AT S_NUM.
      CLEAR RW_ATFLV.
      RW_ATFLV-SIGN   = 'I'.
      RW_ATFLV-OPTION = 'BT'.
      RW_ATFLV-LOW    = S_NUM-LOW.
      RW_ATFLV-HIGH   = S_NUM-HIGH.
      APPEND RW_ATFLV TO RT_ATFLV.
    ENDLOOP.

    SELECT OBJEK
      FROM AUSP AS A
      INNER JOIN CABN AS B ON A~ATINN = B~ATINN
      INTO CORRESPONDING FIELDS OF TABLE LT_AUSP
      WHERE ATFLV IN RT_ATFLV
        AND B~ATNAM = S_ATNAM-LOW
        .

    IF LT_AUSP IS INITIAL.
      MESSAGE S000(ZCB) WITH '没东西啊' DISPLAY LIKE 'E'.
      LEAVE LIST-PROCESSING.
    ELSE.
      LOOP AT LT_AUSP INTO LS_AUSP.
        LS_AUSP-CUOBJ_BM = LS_AUSP-OBJEK.
        MODIFY LT_AUSP FROM LS_AUSP TRANSPORTING CUOBJ_BM.
      ENDLOOP.
    ENDIF.

    SELECT COUNT(*)
      FROM MCH1
      INTO L_COUNT
      FOR ALL ENTRIES IN LT_AUSP
      WHERE CUOBJ_BM = LT_AUSP-CUOBJ_BM.

    IF L_COUNT IS NOT INITIAL.
      MESSAGE S000(ZCB) WITH '共有' L_COUNT '条数据.'.
    ELSE.
      MESSAGE S000(ZCB) WITH '没有相关数据' DISPLAY LIKE 'E'.
    ENDIF.
  ENDIF.

ENDFORM.

         END OF STRU_ATINN.

  DATA: LT_ATINN TYPE STANDARD TABLE OF STRU_ATINN,
        LS_ATINN TYPE                   STRU_ATINN,
        L_ATWRT  TYPE AUSP-ATWRT,
        RT_ATNAM TYPE RANGE OF CABN-ATNAM,
        RW_ATNAM LIKE LINE OF RT_ATNAM.

  SELECT ATINN
    FROM CABN
    INTO TABLE LT_ATINN.

  LOOP AT LT_ATINN INTO LS_ATINN.
    SELECT SINGLE ATWRT
      FROM AUSP
      INTO L_ATWRT
      WHERE ATINN = LS_ATINN-ATINN.

    IF L_ATWRT EQ ABAP_FALSE AND P_1 EQ ABAP_TRUE.
      CLEAR RW_ATNAM.
      RW_ATNAM-SIGN   = 'I'.
      RW_ATNAM-OPTION = 'EQ'.
      RW_ATNAM-LOW    = LS_ATINN-ATINN.
      APPEND RW_ATNAM TO RT_ATNAM.
    ELSEIF L_ATWRT NE ABAP_FALSE AND P_1 EQ ABAP_FALSE.
      CLEAR RW_ATNAM.
      RW_ATNAM-SIGN   = 'I'.
      RW_ATNAM-OPTION = 'EQ'.
      RW_ATNAM-LOW    = LS_ATINN-ATINN.
      APPEND RW_ATNAM TO RT_ATNAM.
    ENDIF.
  ENDLOOP.

  IT_TABLE = RT_ATNAM.

ENDFORM.
