FORM FM_GET_DATA USING I_YEAR.

  DATA: RT_BUDAT      TYPE RANGE OF FAGLFLEXA-BUDAT,
        RW_BUDAT      LIKE LINE OF RT_BUDAT,

        RT_YEAR       TYPE RANGE OF GJAHR,
        RW_YEAR       LIKE LINE OF RT_YEAR,

        L_YEARMO      TYPE CHAR6,
        L_LAST_YEARMO TYPE CHAR6,
        L_LAST_YEAR   TYPE GJAHR,
        L_MONTH       TYPE CHAR2,
        L_LOW         TYPE FAGLFLEXA-BUDAT,
        L_LAST_LOW    TYPE FAGLFLEXA-BUDAT,
        L_HIGH        TYPE FAGLFLEXA-BUDAT,
        L_LAST_HIGH   TYPE FAGLFLEXA-BUDAT.

  DEFINE ADD_YEAR.
    CLEAR RW_YEAR.
    RW_YEAR-SIGN    = 'I'.   "'E'
    RW_YEAR-OPTION  = 'BT'. "'CP','BT','EQ'
    RW_YEAR-LOW     = &1.
    RW_YEAR-HIGH    = &2.
    APPEND RW_YEAR TO RT_YEAR.
  END-OF-DEFINITION.

  L_LAST_YEAR = I_YEAR - 1.

  L_MONTH  = S_YEARMO-LOW+4(2).
  L_YEARMO = I_YEAR && L_MONTH.
  L_LAST_YEARMO = L_LAST_YEAR && L_MONTH.

  L_LOW    = L_YEARMO && '01'.
*   取当月最后一天
  CALL FUNCTION 'FIMA_DATE_CREATE'
    EXPORTING
      I_DATE                  = L_LOW
      I_SET_LAST_DAY_OF_MONTH = 'X'
    IMPORTING
      E_DATE                  = L_HIGH.

  L_LAST_LOW = L_LAST_YEARMO && '01'.

  CALL FUNCTION 'FIMA_DATE_CREATE'
    EXPORTING
      I_DATE                  = L_LAST_LOW
      I_SET_LAST_DAY_OF_MONTH = 'X'
    IMPORTING
      E_DATE                  = L_LAST_HIGH.

  L_LOW      = I_YEAR && '0101'.
  L_LAST_LOW = L_LAST_YEAR && '0101'.

  ADD_BUDAT:
    L_LOW      L_HIGH,
    L_LAST_LOW L_LAST_HIGH.

  ADD_YEAR:
    L_LAST_YEAR I_YEAR .

  SELECT
    A~RACCT
*    A~DRCRK
*    B~TXT20
*    A~RYEAR
    A~HSL
*    C~BLART
    A~BUDAT
    INTO TABLE IT_FAG
    FROM FAGLFLEXA AS A
*    INNER JOIN SKAT AS B ON A~RACCT = B~SAKNR       "科目
    INNER JOIN BKPF AS C ON A~BELNR = C~BELNR AND A~RYEAR = C~GJAHR AND C~BUKRS = A~RBUKRS   "财年
    WHERE A~RACCT  IN IT_RACCT                      "科目
      AND A~RBUKRS IN S_RBUKRS                      "公司代码
      AND A~RYEAR  IN RT_YEAR                       "会计年度
      AND A~BUDAT  IN RT_BUDAT                      "过账日期
      AND C~BLART  EQ 'RV'
      .

  IF IT_FAG IS INITIAL.
    GL_CHECK = '0'.
  ENDIF.

ENDFORM.

FORM FM_CAL_DATA.

  TYPES: BEGIN OF STRU_SUM,

           RACCT TYPE FAGLFLEXA-RACCT,


         END OF STRU_SUM.
*将内表分割为今年当月 今年全部 去年当月 去年全部 4张表
  DATA: IT_TMP_FAG  TYPE STANDARD TABLE OF STRU_FAG,
        WA_TMP_FAG  TYPE                   STRU_FAG,
*今年累计
        IT_YEAR_FAG TYPE STANDARD TABLE OF STRU_FAG,
        WA_YEAR_FAG TYPE                   STRU_FAG,
*去年累计
        IT_LAST_FAG TYPE STANDARD TABLE OF STRU_FAG,
        WA_LAST_FAG TYPE                   STRU_FAG,
*当月发生
        IT_MONT_FAG TYPE STANDARD TABLE OF STRU_FAG,
        WA_MONT_FAG TYPE                   STRU_FAG,
*去年同期
        IT_LAMO_FAG TYPE STANDARD TABLE OF STRU_FAG,
        WA_LAMO_FAG TYPE                   STRU_FAG.

  DATA: RT_BUDAT      TYPE RANGE OF FAGLFLEXA-BUDAT,
        RW_BUDAT      LIKE LINE OF RT_BUDAT,

        L_BL          TYPE P LENGTH 10 DECIMALS 2,

        L_YEARMO      TYPE CHAR6,
        L_LAST_YEARMO TYPE CHAR6,
        L_LAST_YEAR   TYPE GJAHR,
        L_MONTH       TYPE CHAR2,
        L_LOW         TYPE FAGLFLEXA-BUDAT,
        L_LAST_LOW    TYPE FAGLFLEXA-BUDAT,
        L_HIGH        TYPE FAGLFLEXA-BUDAT,
        L_LAST_HIGH   TYPE FAGLFLEXA-BUDAT.

  L_LAST_YEAR = SY-DATUM+0(4) - 1.

  L_MONTH  = S_YEARMO-LOW+4(2).
  L_YEARMO = SY-DATUM+0(4) && L_MONTH.
  L_LAST_YEARMO = L_LAST_YEAR && L_MONTH.

  L_LOW    = L_YEARMO && '01'.
*   取当月最后一天
  CALL FUNCTION 'FIMA_DATE_CREATE'
    EXPORTING
      I_DATE                  = L_LOW
      I_SET_LAST_DAY_OF_MONTH = 'X'
    IMPORTING
      E_DATE                  = L_HIGH.

  L_LAST_LOW = L_LAST_YEARMO && '01'.

  CALL FUNCTION 'FIMA_DATE_CREATE'
    EXPORTING
      I_DATE                  = L_LAST_LOW
      I_SET_LAST_DAY_OF_MONTH = 'X'
    IMPORTING
      E_DATE                  = L_LAST_HIGH.

*分割今年当月发生表
  CLEAR RT_BUDAT.
  ADD_BUDAT:
  L_LOW L_HIGH.
  IT_MONT_FAG = IT_FAG.
  DELETE IT_MONT_FAG WHERE BUDAT NOT IN RT_BUDAT.
*分割去年同期发生表
  CLEAR RT_BUDAT.
  ADD_BUDAT:
  L_LAST_LOW L_LAST_HIGH.
  IT_LAMO_FAG = IT_FAG.
  DELETE IT_LAMO_FAG WHERE BUDAT NOT IN RT_BUDAT.
*分割今年累计
  L_LOW      = SY-DATUM+0(4) && '0101'.
  CLEAR RT_BUDAT.
  ADD_BUDAT:
  L_LOW L_HIGH.
  IT_YEAR_FAG = IT_FAG.
  DELETE IT_YEAR_FAG WHERE BUDAT NOT IN RT_BUDAT.
*分割去年同期累计
  L_LAST_LOW = L_LAST_YEAR && '0101'.
  CLEAR RT_BUDAT.
  ADD_BUDAT:
  L_LAST_LOW L_LAST_HIGH.
  IT_LAST_FAG = IT_FAG.
  DELETE IT_LAST_FAG WHERE BUDAT NOT IN RT_BUDAT.

*----------------------------------------------*
*第一步：汇总
*----------------------------------------------*
*汇总今年当月
  PERFORM FM_HZ TABLES IT_MONT_FAG
                USING   '1'.
*汇总去年同期
  PERFORM FM_HZ TABLES IT_LAMO_FAG
                USING   '2'.
*汇总今年累计
  PERFORM FM_HZ TABLES IT_YEAR_FAG
                USING   '3'.
*汇总去年累计
  PERFORM FM_HZ TABLES IT_LAST_FAG
                USING   '4'.
*----------------------------------------------*
*第二步：算比率
*----------------------------------------------*
  IF IT_SAKNR IS INITIAL.
    SELECT SAKNR
           TXT20
      FROM SKAT
      INTO TABLE IT_SAKNR
      WHERE SAKNR IN IT_RACCT.
  ENDIF.

  LOOP AT IT_ALV INTO WA_ALV.
    CLEAR L_BL.
    IF WA_ALV-ZBMONTH IS NOT INITIAL OR WA_ALV-ZBMONTH NE 0.
      L_BL = WA_ALV-ZTMONTH / WA_ALV-ZBMONTH.
      WA_ALV-ZMRATE = L_BL && '%'.
    ELSEIF WA_ALV-ZBMONTH IS INITIAL OR WA_ALV-ZBMONTH EQ 0.
      WA_ALV-ZMRATE = 100 && '%'.
    ENDIF.
    CLEAR L_BL.
    IF WA_ALV-ZBYEAR IS NOT INITIAL OR WA_ALV-ZBYEAR NE 0.
      L_BL = WA_ALV-ZTYEAR / WA_ALV-ZBYEAR.
      WA_ALV-ZYRATE = L_BL && '%'.
    ELSEIF WA_ALV-ZBYEAR IS INITIAL OR WA_ALV-ZBYEAR EQ 0.
      WA_ALV-ZYRATE = 100 && '%'.
    ENDIF.

    READ TABLE IT_SAKNR INTO WA_SAKNR WITH KEY RACCT = WA_ALV-ZRACCT.
    IF SY-SUBRC = 0.
      WA_ALV-ZRECTYPE = WA_SAKNR-ZRECTYPE.
    ENDIF.
    MODIFY IT_ALV FROM WA_ALV TRANSPORTING ZMRATE ZYRATE ZRECTYPE.
  ENDLOOP.

ENDFORM.
*汇总处理
FORM FM_HZ TABLES IT_TABLE
            USING I_TYPE.

  DATA: IT_TMP_FAG TYPE STANDARD TABLE OF STRU_FAG,
        WA_TMP_FAG TYPE                   STRU_FAG,

        L_HSL      TYPE FAGLFLEXA-HSL.

  IT_TMP_FAG[] = IT_TABLE[].

  LOOP AT IT_TMP_FAG INTO WA_TMP_FAG.

    CLEAR WA_FAG.
    WA_FAG = WA_TMP_FAG.

    L_HSL = L_HSL + WA_FAG-HSL.

          IF SY-SUBRC <> 0.

            WA_ALV-ZTYEAR  = L_HSL.

* 数据填充
* I_PATH : 路径
* T_EXCEL : 下载的数据
*&---------------------------------------------------------------------*
FORM FM_FILL_DATA1 USING I_PATH
                        T_EXCEL TYPE TYT_EXCEL.
  DATA:
    LW_EXCEL   TYPE TY_EXCEL,
    L_ROW      TYPE I,
    L_PATH     TYPE STRING,
    L_EXCEL    TYPE OLE2_OBJECT,  "application
    L_WORKBOOK TYPE OLE2_OBJECT,  "workbook
    L_SHEET    TYPE OLE2_OBJECT,  "sheet
    L_CELL     TYPE OLE2_OBJECT.  "单元格
  L_PATH = I_PATH.
**********  STEP1:打开excel程序
  CREATE OBJECT L_EXCEL 'EXCEL.APPLICATION'.
* 0：不显示（后台运行），1：显示
  SET PROPERTY OF L_EXCEL 'VISIBLE' = 1.
* 不显示警告消息
  SET PROPERTY OF L_EXCEL 'DISPLAYALERTS' = 0.
**********  STEP2:读取workbook
* 创建具体excel文件对象
* 默认带有一个sheet
  CALL METHOD OF L_EXCEL 'WORKBOOKS' = L_WORKBOOK .
  CALL METHOD OF L_WORKBOOK 'OPEN'
    EXPORTING
      #1 = L_PATH.
**********  STEP3:创建sheet
* 获取当前激活的sheet，并设定sheet名
  CALL METHOD OF L_EXCEL 'SHEETS' = L_SHEET
    EXPORTING
      #1  = 1.
  CALL METHOD OF L_SHEET 'SELECT'.
  CALL METHOD OF L_SHEET 'ACTIVATE'.
* SET PROPERTY OF l_sheet 'NAME' = '应交税金明细表'. "设定sheet名称
**********  STEP4:数据填充

  LOOP AT T_EXCEL INTO LW_EXCEL.
    CALL METHOD OF L_EXCEL 'CELLS' = L_CELL
      EXPORTING
        #1      = LW_EXCEL-ROW
        #2      = LW_EXCEL-COL
        .
    SET PROPERTY OF L_CELL 'VALUE'       = LW_EXCEL-VALUE.
  ENDLOOP.
**********  STEP5:保存、关闭、退出
  GET PROPERTY OF L_EXCEL 'ACTIVEWORKBOOK' = L_WORKBOOK.  "将L_WORKBOOK设置为活动工作部,不然无法保存.ActiveWorkbook
* 保存文件
  CALL METHOD OF L_WORKBOOK 'SAVEAS'
    EXPORTING
      #1 = L_PATH
      #2 = 1.
* 关闭文件
  CALL METHOD OF L_WORKBOOK 'CLOSE'.
* 退出EXCEL
  CALL METHOD OF L_EXCEL 'QUIT'.

  FREE OBJECT:
    L_SHEET,
    L_WORKBOOK,
    L_EXCEL.
ENDFORM.
