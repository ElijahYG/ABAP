Description: 接口_盘点表生成传输
function zmmif0150_idoc_output_01.
*"----------------------------------------------------------------------
*"*"本地接口：
*" IMPORTING
*" VALUE(IM_ZPDBH) TYPE ZMMIF0150_PDHEAD-ZPDBH
*" VALUE(IS_WAIT) TYPE CHAR1 OPTIONAL
*"----------------------------------------------------------------------

*-----------------------------------------
* 判断接口状态
*-----------------------------------------
* Global data declarations

  data: wa_sys_config type zsys_config.

  call function 'ZSAP_GET_STATUS'
    exporting
      im_code       = 'PID'
    importing
      wa_sys_config = wa_sys_config.

  check wa_sys_config-rcvprt_status = '01' or wa_sys_config-rcvprt_status = '03'.
  .
*-----------------------------------------
* 判断是否等待
*-----------------------------------------
  if is_wait eq 'X'.
    wait up to 5 seconds.
  endif.

*-----------------------------------------
* 段定义的结构
*-----------------------------------------
  data:
    wa_head type zmmif0150_sgm_head,

    wa_item type zmmif0150_sgm_item.

*-----------------------------------------
* IDOC 定义的结构
*-----------------------------------------
  data:
    it_edidd   type standard table of edidd,           "IDOC 数据记录
    wa_edidd   type edidd,                             "IDOC 数据记录
    it_control type standard table of edidc,           "IDOC 通讯
    wa_control type edidc.                             "IDOC 控制记录

*-----------------------------------------
* 数据读取结构及内表
*-----------------------------------------
  types: begin of stru_name,
           name1 type  t001w-name1,
           lgobe type  t001l-lgobe,
         end of stru_name.

  data:
    it_pdhead type standard table of  zmmif0150_pdhead,
    wa_pdhead type                    zmmif0150_pdhead,
    wa_name   type                    stru_name,

    it_pditem type standard table of  zmmif0150_pditem,
    wa_pditem type                    zmmif0150_pditem
    .
*-----------------------------------------
* 读取数据
*-----------------------------------------
  clear wa_pdhead.
  select single *
    into corresponding fields of wa_pdhead
    from zmmif0150_pdhead
    where zpdbh eq im_zpdbh.

  clear wa_name.
  select single name1
    into wa_name-name1
    from t001w
    where werks eq wa_pdhead-werks.

  select single lgobe
    into wa_name-lgobe
    from t001l
    where werks eq wa_pdhead-werks
      and lgort eq wa_pdhead-lgort.

  select *
    into corresponding fields of table it_pditem
    from zmmif0150_pditem
    where zpdbh eq im_zpdbh
    .
*-----------------------------------------
* 整理数据
*-----------------------------------------

  if wa_pdhead is not initial.

    clear: wa_head.
    move-corresponding wa_pdhead to wa_head.
* WA_HEAD-ZMDKCD = WA_PDHEAD-WERKS && WA_PDHEAD-LGORT.
    wa_head-zmdkcdmc = wa_name-name1 && wa_name-lgobe.
*<-------------change wangzq 2015-12-29
    if wa_pdhead-zmdkcd is not initial .
      wa_head-zmdkcd = wa_pdhead-zmdkcd.
    else.
      wa_head-zmdkcd   = wa_pdhead-werks && wa_pdhead-lgort.
    endif.
*-------------->end

    wa_edidd-segnam = 'ZMMIF0150_SGM_HEAD'.
    wa_edidd-sdata  = wa_head.
    append wa_edidd to it_edidd.
  endif.

  if it_pditem[] is not initial.

    loop at it_pditem into wa_pditem.
      clear: wa_item.

      move-corresponding wa_pditem to wa_item.
      call function 'CONVERSION_EXIT_CUNIT_OUTPUT'
        exporting
          input          = wa_item-meins
        importing
* LONG_TEXT =
          output         = wa_item-meins
* SHORT_TEXT =
        exceptions
          unit_not_found = 1
          others         = 2.


      wa_edidd-segnam = 'ZMMIF0150_SGM_ITEM'.
      wa_edidd-sdata  = wa_item.
      append wa_edidd to it_edidd.
    endloop.
  endif.

****************设定idoc控制记录***************************
* 配置接收端信息
  wa_control-rcvprt = wa_sys_config-rcvprt.       "接收方伙伴类型
  wa_control-rcvprn = wa_sys_config-rcvprn.       "接收方的合作伙伴编号
  wa_control-rcvpor = wa_sys_config-rcvpor.       "接收者端口(SAP 系统，EDI 子系统)
* LW_CONTROL-SNDPRN = . "发送方的合作伙伴编号
* LW_CONTROL-SNDPRT = . "发送方伙伴类型

  wa_control-mestyp = 'ZMMIF0150_MSG_01'.       "信息类型
  wa_control-idoctp = 'ZMMIF0150_IDOC_01'.      "IDOC 类型

*----------------------------------------------------------------------
* 发送IDOC
*----------------------------------------------------------------------
  call function 'MASTER_IDOC_DISTRIBUTE'
    exporting
      master_idoc_control            = wa_control
    tables
      communication_idoc_control     = it_control
      master_idoc_data               = it_edidd
    exceptions
      error_in_idoc_control          = 1
      error_writing_idoc_status      = 2
      error_in_idoc_data             = 3
      sending_logical_system_unknown = 4
      others                         = 5.

  if sy-subrc <> 0.
    message id sy-msgid
          type sy-msgty
        number sy-msgno
          with sy-msgv1 sy-msgv2 sy-msgv3 sy-msgv4.
  else.
* 发送idoc时会锁住一些系统表，此过程为等待完全操作完毕
    call function 'DB_COMMIT'.
    call function 'DEQUEUE_ALL'.
    commit work and wait.
* 发送的IDOC
    call function 'EDI_DOCUMENT_DEQUEUE_LATER'
      exporting
        docnum                 = wa_control-docnum
      exceptions
        idoc_is_not_to_dequeue = 1
        others                 = 2.

  endif.


endfunction.
