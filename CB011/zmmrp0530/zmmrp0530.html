*&---------------------------------------------------------------------*

FORM FM_GET_DATA.

  DATA: LT_R_BWTAR TYPE RANGE OF BWTAR_D,
        LS_R_BWTAR LIKE LINE OF LT_R_BWTAR.

  CLEAR LS_R_BWTAR.
  LS_R_BWTAR-SIGN   = 'E'.
  LS_R_BWTAR-OPTION = 'EQ'.
  LS_R_BWTAR-LOW    = ABAP_FALSE.
  APPEND LS_R_BWTAR TO LT_R_BWTAR.

  SELECT A~MATNR    "商品款号
         A~BWTAR    "批次号
         A~BWKEY    "地点
         B~EX_JJJB  "款式克工费档
         C~ZXSKGFD  "批次克工费档
    INTO TABLE GT_OUT_TAB
    FROM MBEW AS A
    INNER JOIN ZMMEX0010_MARA AS B ON A~MATNR = B~MATNR
    INNER JOIN ZMMRP0210_BATCH AS C ON A~MATNR = C~MATNR AND A~BWTAR = C~CHARG
    WHERE A~LBKUM >= 0
      AND A~BWTAR IN LT_R_BWTAR
      .

ENDFORM.

*&---------------------------------------------------------------------*
*& 计算逻辑
*&---------------------------------------------------------------------*
FORM FM_CAL_DATA.

  DATA: LT_OUT_TAB TYPE STANDARD TABLE OF STRU_OUT.

  FIELD-SYMBOLS: <LS_OUT_TAB> TYPE STRU_OUT.

  LOOP AT GT_OUT_TAB ASSIGNING <LS_OUT_TAB>.
    IF <LS_OUT_TAB>-MA_KGFD NE <LS_OUT_TAB>-CH_KGFD.
      APPEND <LS_OUT_TAB> TO LT_OUT_TAB.
    ENDIF.
  ENDLOOP.

  CLEAR GT_OUT_TAB[].
  GT_OUT_TAB[] = LT_OUT_TAB[].

ENDFORM.

FORM FM_ALVSHOW.
  DATA:
    LS_FIELDCAT TYPE LVC_S_FCAT,
    LT_FIELDCAT TYPE LVC_T_FCAT,
    LS_LAYOUT   TYPE LVC_S_LAYO,
    LS_VARIANT  TYPE DISVARIANT,
    GV_LINES    TYPE I.

* ALV填充LT_FIELDCAT表
  DATA:
    L_TEXT     TYPE CHAR10,
    L_NUM      TYPE NUMC3,
    L_FLAG     TYPE CHAR1,
    CLASS      TYPE REF TO CL_ABAP_STRUCTDESCR,
    LS_OUT_TAB TYPE STRU_OUT
    .
  FIELD-SYMBOLS:
    <TEXT>,
    <COMP_WA> TYPE ABAP_COMPDESCR.

  CLEAR LT_FIELDCAT[].

  CLASS ?= CL_ABAP_TYPEDESCR=>DESCRIBE_BY_DATA( LS_OUT_TAB ).

  LOOP AT CLASS->COMPONENTS ASSIGNING <COMP_WA>.
    IF <COMP_WA>-NAME NE 'FLAG'.
      L_NUM = L_NUM + 1.
      L_TEXT = 'TEXT-' && L_NUM.
      ASSIGN (L_TEXT) TO <TEXT>.
      ADD_ALV:
        <COMP_WA>-NAME <TEXT>
        .
    ENDIF.
  ENDLOOP.
* 调整ALV显示布局
  CLEAR LS_LAYOUT.
  LS_LAYOUT-CWIDTH_OPT = ABAP_TRUE.     "优化列宽设置
  LS_LAYOUT-ZEBRA = 'X'.
  LS_LAYOUT-BOX_FNAME = 'FLAG'.           "选择行头按钮

* 管理布局结构
  CLEAR LS_VARIANT.
  LS_VARIANT-REPORT = SY-REPID.
*  LS_VARIANT-VARIANT = P_VARI .

  GV_LINES = LINES( GT_OUT_TAB ).
  MESSAGE S000(ZCB) WITH '共有' GV_LINES '数据！' .
* 调用ALV
  CALL FUNCTION 'REUSE_ALV_GRID_DISPLAY_LVC'
    EXPORTING
      I_CALLBACK_PROGRAM      = SY-REPID
*     I_CALLBACK_PF_STATUS_SET = 'FM_GUI_STATUS'
      I_CALLBACK_USER_COMMAND = 'FM_USER_COMMAND'
      IS_LAYOUT_LVC           = LS_LAYOUT
      IT_FIELDCAT_LVC         = LT_FIELDCAT
      IS_VARIANT              = LS_VARIANT
      I_SAVE                  = 'A'                     "保存布局
      I_DEFAULT               = ''                      "不能保存为缺省
    TABLES
      T_OUTTAB                = GT_OUT_TAB
    EXCEPTIONS
      PROGRAM_ERROR           = 1
      OTHERS                  = 2.
  IF SY-SUBRC <> 0.
    MESSAGE E001(00) WITH '系统错误' SPACE SPACE SPACE.
  ENDIF.
ENDFORM.
