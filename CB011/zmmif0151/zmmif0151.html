*&---------------------------------------------------------------------*
*& 取在库库存
*&---------------------------------------------------------------------*
FORM FM_GET_ZKKC.

  DATA:
    LT_TABLE          TYPE STANDARD TABLE OF STRU_OUT,
    LS_TABLE          TYPE                   STRU_OUT,

    LS_ZMMEX0010_MARA TYPE                   ZMMEX0010_MARA.

* 取无批次商品在库库存
  SELECT A~MATNR
         A~LABST
         A~WERKS
         A~LGORT
         B~MEINS
         B~EXTWG
         B~TEMPB
         B~MATKL
         B~XCHPF
         C~MAKTX
    INTO CORRESPONDING FIELDS OF TABLE GT_OUT_TAB
    FROM MARD AS A
   INNER JOIN MARA AS B
      ON B~MATNR = A~MATNR
    LEFT JOIN MAKT AS C
      ON C~MATNR EQ A~MATNR
   WHERE A~WERKS   IN S_WERKS
     AND A~LGORT   IN S_LGORT                  "以什么条件找
     AND B~EXTWG   IN S_EXTWG
     AND A~MATNR   IN S_MATNR
     AND B~MATKL   IN S_MATKL
            .

* 取批次商品在库库存
  SELECT A~MATNR
         A~WERKS
         A~LGORT

         B~MEINS
         B~EXTWG
         B~TEMPB
         B~XCHPF
         B~MATKL

         A~CHARG
         A~CLABS

         E~MAKTX
   APPENDING CORRESPONDING FIELDS OF TABLE GT_OUT_TAB
   FROM MCHB           AS A
   INNER JOIN  MARA           AS B
      ON B~MATNR = A~MATNR
    LEFT JOIN MAKT           AS E
      ON E~MATNR = A~MATNR
   WHERE A~WERKS   IN S_WERKS
     AND A~LGORT   IN S_LGORT                  "以什么条件找
     AND B~EXTWG   IN S_EXTWG
     AND A~MATNR   IN S_MATNR
     AND B~MATKL   IN S_MATKL.

  LT_TABLE[] = GT_OUT_TAB.
  SORT LT_TABLE BY MATNR.
  DELETE ADJACENT DUPLICATES FROM LT_TABLE COMPARING MATNR.
  IF LT_TABLE[] IS NOT INITIAL.
    SELECT MATNR
           EX_PZ01
      INTO CORRESPONDING FIELDS OF TABLE GT_ZMMEX0010_MARA[]
      FROM ZMMEX0010_MARA
       FOR ALL ENTRIES IN LT_TABLE
     WHERE MATNR = LT_TABLE-MATNR
       AND EX_PZ01 IN S_EX_PZ1
      .
  ENDIF.

  SORT GT_OUT_TAB        BY MATNR.
  SORT GT_ZMMEX0010_MARA BY MATNR.
  LOOP AT GT_OUT_TAB ASSIGNING <FS_OUT_TABLE>.
    READ TABLE GT_ZMMEX0010_MARA INTO LS_ZMMEX0010_MARA WITH KEY MATNR = <FS_OUT_TABLE>-MATNR BINARY SEARCH.
    IF SY-SUBRC = 0.
      <FS_OUT_TABLE>-EX_PZ01 = LS_ZMMEX0010_MARA-EX_PZ01.
    ENDIF.
    IF <FS_OUT_TABLE>-XCHPF EQ ABAP_TRUE.
      <FS_OUT_TABLE>-LABST = <FS_OUT_TABLE>-CLABS.
    ENDIF.
  ENDLOOP.

* 清除不符合输入条件的项目
  IF S_EX_PZ1[] IS NOT INITIAL.
    DELETE GT_OUT_TAB WHERE EX_PZ01 NOT IN S_EX_PZ1[].
  ENDIF.
* 清除零库存
  DELETE GT_OUT_TAB WHERE LABST IS INITIAL AND LABST = 0.

*------------------------------------------------------------------------------------------------------------------------------------
* 取数批次特性
*------------------------------------------------------------------------------------------------------------------------------------
  DATA:LT_CHARG      LIKE STANDARD TABLE OF ZMM_CHARG,
       LT_CHARACTERS LIKE STANDARD TABLE OF ZMM_CHARACTERS,
       LS_CHARACTERS LIKE                   ZMM_CHARACTERS.

  MOVE-CORRESPONDING GT_OUT_TAB[] TO LT_CHARG[].

  call function 'ZSDRP0020_FM_01'
    TABLES
      T_CHARG      = LT_CHARG
      T_CHARACTERS = LT_CHARACTERS.


  SORT GT_OUT_TAB      BY MATNR CHARG.
  SORT LT_CHARACTERS   BY MATNR CHARG.

  LOOP AT GT_OUT_TAB ASSIGNING <FS_OUT_TABLE>.
    CLEAR LS_CHARACTERS.
    READ TABLE LT_CHARACTERS INTO LS_CHARACTERS WITH KEY MATNR = <FS_OUT_TABLE>-MATNR
                                                         CHARG = <FS_OUT_TABLE>-CHARG
                                                         BINARY SEARCH.
* 如果有则是批次管理商品
    IF SY-SUBRC = 0.
      <FS_OUT_TABLE>-ZJZ    = LS_CHARACTERS-ZJZ.         "取单品克重

      <FS_OUT_TABLE>-ZBQMC  = LS_CHARACTERS-ZBQMC.       "标签名称
      <FS_OUT_TABLE>-ZCKLSJ = LS_CHARACTERS-ZCKLSJ.      "参考零售价
      <FS_OUT_TABLE>-ZSPTM  = LS_CHARACTERS-ZSPTM.       "商品条码
      <FS_OUT_TABLE>-ZSPBM  = LS_CHARACTERS-ZSPBM.       "商品包码
      <FS_OUT_TABLE>-ZFPH   = LS_CHARACTERS-ZFPH.        "分配号
      <FS_OUT_TABLE>-ZJSZBM = LS_CHARACTERS-ZJSZBM.      "检测站编码
      <FS_OUT_TABLE>-ZCT1   = LS_CHARACTERS-ZCT1.        "抽屉号1
      <FS_OUT_TABLE>-ZCT2   = LS_CHARACTERS-ZCT2.        "抽屉号2

      IF <FS_OUT_TABLE>-MEINS = 'ST'.
        <FS_OUT_TABLE>-ZDPJS   = <FS_OUT_TABLE>-CLABS.
        <FS_OUT_TABLE>-ZZKZ    = <FS_OUT_TABLE>-ZJZ * <FS_OUT_TABLE>-CLABS.
      ELSEIF <FS_OUT_TABLE>-MEINS = 'G'.
        IF <FS_OUT_TABLE>-ZJZ IS NOT INITIAL.
          <FS_OUT_TABLE>-ZDPJS = <FS_OUT_TABLE>-CLABS / <FS_OUT_TABLE>-ZJZ.
        ENDIF.
        <FS_OUT_TABLE>-ZZKZ = <FS_OUT_TABLE>-CLABS.
      ENDIF.
* 如果没有并且不是批次管理的商品
    ELSEIF SY-SUBRC <> 0 AND <FS_OUT_TABLE>-XCHPF = ABAP_FALSE.
      <FS_OUT_TABLE>-CLABS = <FS_OUT_TABLE>-LABST.
      <FS_OUT_TABLE>-ZBQMC = <FS_OUT_TABLE>-MAKTX.
      IF <FS_OUT_TABLE>-MEINS = 'ST'.
        <FS_OUT_TABLE>-ZDPJS = <FS_OUT_TABLE>-CLABS.
      ELSEIF <FS_OUT_TABLE>-MEINS = 'G'.
        <FS_OUT_TABLE>-ZZKZ = <FS_OUT_TABLE>-CLABS.
      ENDIF.
    ENDIF.
  ENDLOOP.

ENDFORM.

  DATA: IT_TMP         TYPE STANDARD TABLE OF STRU_OUT,
*        IT_TMP1        TYPE STANDARD TABLE OF STRU_0010,
*取参考单价
        IT_BATCH       TYPE  ZCB01_TY_T_ZSD_BATCH_PRICE_KEY,
