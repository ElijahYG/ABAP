FORM FM_GET_PODATA.
*定义存放EKBE内容的结构
  TYPES: BEGIN OF STRU_EKBE,

           LFBNR TYPE EKBE-LFBNR,      "参考凭证号(判断冲销用)
           LFPOS TYPE EKBE-LFPOS,      "参考行项目
           MATNR TYPE EKBE-MATNR,      "商品号
           CHARG TYPE EKBE-CHARG,      "批次号
           DMBTR TYPE EKBE-DMBTR,
           BRTWR TYPE EKPO-BRTWR,
           NETWR TYPE EKPO-NETWR,
           ELIKZ TYPE EKPO-ELIKZ,      "完成交货标识
           MENGE TYPE EKBE-MENGE,      "数量
*           BEDAT TYPE EKKO-BEDAT,      "采购凭证日期
         END OF STRU_EKBE,

         BEGIN OF STRU_PO,
           EBELN   TYPE ZFIRP0140_ITEM3-EBELN,
           ZJSD_NO TYPE ZFIRP0140_JXHEAD-ZJSD_NO,
         END OF STRU_PO.

  DATA: IT_PO TYPE STANDARD TABLE OF STRU_PO,
        WA_PO TYPE                   STRU_PO.
  DATA: IT_R_PO TYPE RANGE OF EBELN,
        WA_R_PO LIKE LINE OF IT_R_PO.
  DATA: IT_EKBE TYPE STANDARD TABLE OF STRU_EKBE,
        WA_EKBE TYPE                   STRU_EKBE.

  IF P_1 EQ ABAP_TRUE. "生成结算单时
*1.取采购订单号
    SELECT EBELN BSART
      FROM EKKO
      INTO CORRESPONDING FIELDS OF TABLE IT_EBELN
      WHERE EKKO~BSART IN ('Z001','Z002','Z004')
        AND EKKO~BEDAT IN S_DATE
        AND EKKO~LIFNR IN S_LIFNR
        AND EKKO~BUKRS IN S_BUKRS
        AND EKKO~EBELN IN S_EBELN.
*2.校验自建表中以生成的采购订单号
    IF IT_EBELN IS NOT INITIAL.
      SELECT A~EBELN A~ZJSD_NO
        FROM ZFIRP0140_ITEM3 AS A
        INNER JOIN ZFIRP0140_JXHEAD AS B ON A~ZJSD_NO EQ B~ZJSD_NO
        INTO CORRESPONDING FIELDS OF TABLE IT_PO
        FOR ALL ENTRIES IN IT_EBELN
        WHERE EBELN EQ IT_EBELN-EBELN
*        AND B~ZSTART_DATE > S_DATE-LOW
*        AND B~ZEND_DATE < S_DATE-HIGH
          .

      IF IT_PO IS NOT INITIAL.   "如果不为空则说明有重复PO号
        LOOP AT IT_PO INTO WA_PO.
          CLEAR WA_R_PO.
          WA_R_PO-SIGN   = 'I'.
          WA_R_PO-OPTION = 'EQ'.
          WA_R_PO-LOW    = WA_PO-EBELN.
          APPEND WA_R_PO TO IT_R_PO.
        ENDLOOP.
        DELETE IT_EBELN WHERE EBELN IN IT_R_PO.  "删除所有重复的PO单
        IF IT_EBELN IS INITIAL.
          GL_CHECK = '0'.
          EXIT.
        ENDIF.
      ENDIF.
*3.检查PO单是否被冲销
      SELECT EKBE~EBELN
             EKBE~EBELP
             EKBE~BELNR
             EKBE~BUZEI
             EKBE~GJAHR
             EKBE~DMBTR
             EKBE~LFBNR
             EKBE~LFPOS
             EKBE~MATNR
             EKBE~CHARG
             EKPO~BRTWR
             EKPO~NETWR
             EKBE~ELIKZ
             EKBE~MENGE
        FROM EKBE
        INNER JOIN EKPO ON EKBE~EBELN EQ EKPO~EBELN AND EKBE~EBELP EQ EKPO~EBELP
        INTO CORRESPONDING FIELDS OF TABLE IT_EKBE
        FOR ALL ENTRIES IN IT_EBELN
        WHERE EKBE~VGABE EQ '1'
          AND EKBE~EBELN EQ IT_EBELN-EBELN
          AND EKPO~ELIKZ EQ 'X'
          .

      IF IT_EKBE IS NOT INITIAL.    "如果不为空则说明有入库凭证

        CLEAR IT_R_PO.
        SORT IT_EKBE BY EBELN EBELP BELNR BUZEI.
*        DELETE ADJACENT DUPLICATES FROM IT_EKBE COMPARING EBELN EBELP.    "每个PO单行项目只取最新的入库单凭证
        LOOP AT IT_EKBE INTO WA_EKBE.
          IF WA_EKBE-BELNR NE WA_EKBE-LFBNR.  "判断是否是冲销
            CLEAR WA_R_PO.
            WA_R_PO-SIGN   = 'I'.
            WA_R_PO-OPTION = 'EQ'.
            WA_R_PO-LOW    = WA_EKBE-LFBNR.
            APPEND WA_R_PO TO IT_R_PO.
          ENDIF.
        ENDLOOP.
        IF IT_R_PO IS NOT INITIAL.

FORM FM_GET_JSDNO.
  DATA: L_NUM  TYPE NUMC4.
*        L_CHAR TYPE CHAR4.

  SELECT *
    FROM ZFIRP0140_JXHEAD

    SORT IT_TMP_JXHEAD BY ZJSD_NO DESCENDING.
