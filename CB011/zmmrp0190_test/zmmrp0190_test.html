REPORT ZMMRP0190_TEST.

*&---------------------------------------------------------------------*

*&---------------------------------------------------------------------*
TYPE-POOLS: SLIS. "ALV显示使用到的共享池

DATA:
  WA_FIELDCAT TYPE LVC_S_FCAT,
  IT_FIELDCAT TYPE LVC_T_FCAT,
  WA_LAYOUT   TYPE LVC_S_LAYO,
  L_GRID      TYPE REF TO CL_GUI_ALV_GRID.

*&---------------------------------------------------------------------*
*& 引用表
*&---------------------------------------------------------------------*
TABLES: MKPF,
        MSEG
        .

*&---------------------------------------------------------------------*
*&定义结构体
*&---------------------------------------------------------------------*
TYPES: BEGIN OF STRU_ALV,

         MBLNR TYPE MKPF-MBLNR,       "退货单号
         LIFNR TYPE MSEG-LIFNR,       "供应商
         WERKS TYPE MSEG-WERKS,       "退货部门
         BUDAT TYPE MKPF-BUDAT,       "退货日期

         SEL   TYPE CHAR1,

       END OF STRU_ALV.

*&---------------------------------------------------------------------*
*& 定义常量,定义宏
*&---------------------------------------------------------------------*
*定义ALV显示宏
DEFINE ADD_ALV.
  CLEAR WA_FIELDCAT.
  WA_FIELDCAT-COL_POS      = &1.      "列位置
  WA_FIELDCAT-FIELDNAME    = &2.      "要显示的字段
  WA_FIELDCAT-COLTEXT      = &3.      "中文描述字段名
   IF WA_FIELDCAT-FIELDNAME = 'SEL'.
        WA_FIELDCAT-CHECKBOX = ABAP_TRUE.
        WA_FIELDCAT-EDIT     = ABAP_TRUE.
   ENDIF.

*    WA_FIELDCAT-JUST         =  .      "对齐方式
*    WA_FIELDCAT-EDIT         =  .      "可编辑
*    WA_FIELDCAT-REF_TABLE    =  .      "引用表
*    WA_FIELDCAT-REF_FIELD    =  .      "引用字段
*    WA_FIELDCAT-INTLEN       =  .      "内部长度
*    WA_FIELDCAT-INTTYPE      =  .
*    WA_FIELDCAT-CHECKBOX     =  .      "勾选框
*    WA_FIELDCAT-FIX_COLUMN  = ABAP_TRUE. "固定列

  APPEND WA_FIELDCAT TO IT_FIELDCAT.
END-OF-DEFINITION.

*&---------------------------------------------------------------------*
*& 定义全局数据
*&---------------------------------------------------------------------*
*定义变量
DATA: GL_CHECK TYPE CHAR1.

*定义内表
DATA: IT_ALV TYPE STANDARD TABLE OF STRU_ALV,
      WA_ALV TYPE                   STRU_ALV.
*&---------------------------------------------------------------------*
*& 选择屏幕
*&---------------------------------------------------------------------*
SELECTION-SCREEN BEGIN OF BLOCK BK1 WITH FRAME TITLE TEXT-001.

SELECT-OPTIONS:
   S_MBLNR    FOR MKPF-MBLNR,      "退货单号
   S_LIFNR    FOR MSEG-LIFNR,      "供应商
   S_WERKS    FOR MSEG-WERKS,      "退货部门
   S_BUDAT    FOR MKPF-BUDAT.      "退货日期

SELECTION-SCREEN END OF BLOCK BK1.

*&---------------------------------------------------------------------*
*& 初始化
*&---------------------------------------------------------------------*
INITIALIZATION.
  PERFORM FM_INIT.

*&---------------------------------------------------------------------*
*& PBO处理
*&---------------------------------------------------------------------*
AT SELECTION-SCREEN OUTPUT.

*&---------------------------------------------------------------------*
*&F8
*&---------------------------------------------------------------------*
START-OF-SELECTION.
  PERFORM FM_GET_DATA.
  IF GL_CHECK = '0'.
    MESSAGE S001(00) WITH '该商品没有相应的库存' DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.
*&---------------------------------------------------------------------*
*&选择屏幕结束
*&---------------------------------------------------------------------*
END-OF-SELECTION.
  PERFORM FM_ALVSHOW.

*&---------------------------------------------------------------------*
*& 初始化：清空全局变量
*&---------------------------------------------------------------------*
FORM FM_INIT.
  CLEAR: GL_CHECK,
         IT_ALV
         .
ENDFORM.

*&---------------------------------------------------------------------*
*& 取数逻辑
*&---------------------------------------------------------------------*
FORM FM_GET_DATA.

  SELECT MBLNR
         LIFNR
         WERKS
         BUDAT_MKPF
    FROM MSEG
    INTO TABLE IT_ALV
    WHERE MBLNR      IN S_MBLNR
      AND LIFNR      IN S_LIFNR
      AND WERKS      IN S_WERKS
      AND BUDAT_MKPF IN S_BUDAT
      AND BWART      EQ '161'
      .

ENDFORM.
*&---------------------------------------------------------------------*
*& ALV显示
*&---------------------------------------------------------------------*
FORM FM_ALVSHOW.

  CLEAR IT_FIELDCAT[].

  ADD_ALV:
    1  'SEL'    '是否打印',
    1  'MBLNR'  '退货单号',
    2  'LIFNR'  '供应商',
    3  'WERKS'  '退货部门',
    4  'BUDAT'  '退货日期'
       .

  CLEAR WA_LAYOUT.
  WA_LAYOUT-CWIDTH_OPT = ABAP_TRUE.     "优化列宽设置
  WA_LAYOUT-ZEBRA = 'X'.
*  WA_LAYOUT-BOX_FNAME = 'FLAG'.           "选择行头按钮

* 调用ALV
  CALL FUNCTION 'REUSE_ALV_GRID_DISPLAY_LVC'
    EXPORTING
      I_CALLBACK_PROGRAM       = SY-REPID
      I_CALLBACK_PF_STATUS_SET = 'FM_GUI_STATUS'
      I_CALLBACK_USER_COMMAND  = 'FM_USER_COMMAND'
      IS_LAYOUT_LVC            = WA_LAYOUT
      IT_FIELDCAT_LVC          = IT_FIELDCAT
    TABLES
      T_OUTTAB                 = IT_ALV
    EXCEPTIONS
      PROGRAM_ERROR            = 1
      OTHERS                   = 2.
  IF SY-SUBRC <> 0.
    MESSAGE E001(00) WITH '系统错误' SPACE SPACE SPACE.
  ENDIF.

ENDFORM.
