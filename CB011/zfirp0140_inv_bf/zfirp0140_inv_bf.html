FORM frm_get_data .
  DATA:
    lt_data TYPE STANDARD TABLE OF ty_data,
    ls_data TYPE ty_data,
    lt_doc  TYPE STANDARD TABLE OF zfirp0140_doc,
    ls_doc  TYPE zfirp0140_doc.

  REFRESH gt_data.
  SELECT *
    INTO CORRESPONDING FIELDS OF TABLE gt_data
    FROM zfirp0140_header
   WHERE zjsd_no IN s_no
     AND lifnr IN s_lifnr
     AND zstart_date IN s_sdate
     AND zend_date IN s_edate
     AND zfapiao IN ('1','3')
     AND ZSH = 'X'.
  IF sy-subrc <> 0.
    MESSAGE '没有数据' TYPE 'S' DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.

  REFRESH lt_data.
  LOOP AT gt_data INTO gs_data WHERE zfapiao = '3'.
    APPEND gs_data TO lt_data.
  ENDLOOP.

* 检查对应发票是否被冲销
  IF lt_data IS NOT INITIAL.
    SELECT *
      INTO TABLE lt_doc
      FROM zfirp0140_doc
       FOR ALL ENTRIES IN lt_data
     WHERE zjsd_no = lt_data-zjsd_no.
    LOOP AT gt_data INTO gs_data WHERE zfapiao = '3'.
      READ TABLE lt_doc INTO ls_doc
        WITH KEY zjsd_no = gs_data-zjsd_no.
      IF sy-subrc = 0.
        SELECT COUNT(*)
          FROM bkpf
         WHERE bukrs = ls_doc-bukrs
           AND belnr = ls_doc-belnr
           AND gjahr = ls_doc-gjahr
           AND stblg = ''.
        IF sy-subrc <> 0.
          DELETE FROM zfirp0140_doc WHERE zjsd_no = gs_data-zjsd_no.
          UPDATE zfirp0140_header
             SET zfapiao = '1'
           WHERE zjsd_no = gs_data-zjsd_no.
          IF sy-subrc = 0.
            COMMIT WORK AND WAIT.
            gs_data-zfapiao = '1'.
            MODIFY gt_data FROM gs_data TRANSPORTING zfapiao.
          ELSE.
            ROLLBACK WORK.
          ENDIF.
        ENDIF.
      ENDIF.
    ENDLOOP.
  ENDIF.
  DELETE gt_data WHERE zfapiao = '3'.

  IF gt_data IS INITIAL.
    MESSAGE '没有数据' TYPE 'S' DISPLAY LIKE 'E'.
  ELSE.
    LOOP AT gt_data INTO gs_data.
      SELECT SINGLE name1
        INTO gs_data-name1
        FROM lfa1
       WHERE lifnr = gs_data-lifnr.
      IF sy-subrc = 0.
        MODIFY gt_data FROM gs_data TRANSPORTING name1.
      ENDIF.
    ENDLOOP.
    SORT gt_data BY lifnr zjsd_no sdate.
    CALL SCREEN 9001.
  ENDIF.
ENDFORM.

FORM frm_create_inv .

    lt_data   TYPE STANDARD TABLE OF ty_data,
    ls_wb     TYPE ty_data,
    ls_head   TYPE bapiache09,
    lt_item   TYPE STANDARD TABLE OF bapiaccr09,
    ls_item   TYPE bapiaccr09,
    lt_gl     TYPE STANDARD TABLE OF bapiacgl09,
    ls_gl     TYPE bapiacgl09,
    lt_pay    TYPE STANDARD TABLE OF bapiacap09,
    ls_pay    TYPE bapiacap09,
    lt_return TYPE STANDARD TABLE OF bapiret2,
    ls_return TYPE bapiret2,
    lt_doc    TYPE STANDARD TABLE OF zfirp0140_doc,
    ls_doc    TYPE zfirp0140_doc,
    l_m1      TYPE zfirp0140_header-zzje,
    l_m2      TYPE zfirp0140_header-zzje,
    l_m3      TYPE zfirp0140_header-zzje,
    l_m4      TYPE zfirp0140_header-zzje,
    l_no      TYPE bapiache09-obj_key,
    l_prctr   TYPE prctr,
    l_message TYPE char255.

  READ TABLE gt_data TRANSPORTING NO FIELDS
    WITH KEY box = 'X'
             datum = ''.
  IF sy-subrc = 0.
    MESSAGE '请输入发票日期' TYPE 'S' DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.

  READ TABLE gt_data TRANSPORTING NO FIELDS
    WITH KEY box = 'X'
             pay = 0.
  IF sy-subrc = 0.
    MESSAGE '请输入应付账款' TYPE 'S' DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.

  REFRESH lt_data.
  LOOP AT gt_data INTO gs_data WHERE box = 'X'
                                 AND message = ''.
    APPEND gs_data TO lt_data.
  ENDLOOP.
  IF sy-subrc <> 0.
    MESSAGE '请选中消息为空的行项目' TYPE 'S' DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.
  CLEAR:
    ls_head,
    ls_item,
    ls_gl,
    ls_pay,
    l_m1,
    l_m2,
    l_m3,
    l_m4.

  REFRESH:
    lt_item,
    lt_gl,
    lt_pay,
    lt_return.

  ls_head-doc_type = 'Y2'.
  ls_head-doc_date = p_datum.
***Edit by CAOSHICHUN 2016.02.02
*   ls_head-pstng_date = sy-datum.

    l_m2 = l_m2 + gs_data-tax.
    l_m3 = l_m3 + gs_data-pay.
    l_m4 = l_m4 + gs_data-balance.
    ls_head-comp_code = gs_data-zpo_bukrs.
  ENDLOOP.

  CONCATENATE 'P'
              ls_head-comp_code
              '001'
         INTO l_prctr.

  CLEAR ls_item.
  ls_item-itemno_acc ='0010'.
  ls_item-amt_doccur = l_m1.
  ls_item-currency = 'CNY'.
  APPEND ls_item TO lt_item.
  CLEAR ls_gl.
  ls_gl-itemno_acc ='0010'.
  ls_gl-gl_account = '2202099002'.
  ls_gl-profit_ctr = l_prctr.
  ls_gl-item_text = '联营商品发票校验'.
  APPEND ls_gl TO lt_gl.

  CLEAR ls_item.
  ls_item-itemno_acc ='0020'.
  ls_item-amt_doccur = l_m2.
  ls_item-currency = 'CNY'.
  APPEND ls_item TO lt_item.
  CLEAR ls_gl.
  ls_gl-itemno_acc ='0020'.
  ls_gl-gl_account = '2221001001'.
  ls_gl-item_text = '联营商品发票校验'.
  ls_gl-profit_ctr = l_prctr.
  APPEND ls_gl TO lt_gl.

  CLEAR ls_item.
  ls_item-itemno_acc ='0030'.
  ls_item-amt_doccur = 0 - l_m3.
  ls_item-currency = 'CNY'.
  APPEND ls_item TO lt_item.
  CLEAR ls_pay.
  ls_pay-itemno_acc ='0030'.
  ls_pay-vendor_no = gs_data-lifnr.
  ls_pay-item_text = '联营商品发票校验'.
  APPEND ls_pay TO lt_pay.

  CLEAR ls_item.
  ls_item-itemno_acc ='0040'.
  ls_item-amt_doccur = 0 - l_m4.
  ls_item-currency = 'CNY'.
  APPEND ls_item TO lt_item.
  CLEAR ls_gl.
  ls_gl-itemno_acc ='0040'.
  ls_gl-gl_account = '6401009099'.
  ls_gl-item_text = '联营商品发票校验'.
  APPEND ls_gl TO lt_gl.

  CLEAR:
    l_no.

    EXPORTING
      documentheader = ls_head
    IMPORTING
      obj_key        = l_no
    TABLES
      currencyamount = lt_item
      accountgl      = lt_gl
      accountpayable = lt_pay
      return         = lt_return.
  READ TABLE lt_return TRANSPORTING NO FIELDS
    WITH KEY type = 'E'.
  IF sy-subrc <> 0.
    REFRESH lt_doc.
    LOOP AT gt_data INTO ls_wb WHERE box = 'X'
                                 AND message = ''.
      CONCATENATE '凭证编号'
                  l_no+0(10)
                  '公司代码'
                  l_no+10(4)
                  '会计年度'
                  l_no+14(4)
             INTO ls_wb-message.
      MODIFY gt_data FROM ls_wb TRANSPORTING message.
      UPDATE zfirp0140_header
         SET zfapiao = '3'
       WHERE zjsd_no = gs_data-zjsd_no.
      CLEAR ls_doc.
      ls_doc-zjsd_no = ls_wb-zjsd_no.
      ls_doc-bukrs = l_no+10(4).
      ls_doc-belnr = l_no+0(10).
      ls_doc-gjahr = l_no+14(4).
      APPEND ls_doc TO lt_doc.
    ENDLOOP.

    MODIFY zfirp0140_doc FROM TABLE lt_doc.
    IF sy-subrc = 0.
      MESSAGE '凭证创建成功 请查看消息' TYPE 'S'.
      CALL FUNCTION 'BAPI_TRANSACTION_COMMIT'
        EXPORTING
          wait = 'X'.
    ELSE.
      MESSAGE '凭证记录失败,请重新创建' TYPE 'S' DISPLAY LIKE 'E'.
      CALL FUNCTION 'BAPI_TRANSACTION_ROLLBACK'.
    ENDIF.
  ELSE.
    CALL FUNCTION 'BAPI_TRANSACTION_ROLLBACK'.

    CLEAR l_message.
    LOOP AT lt_return INTO ls_return WHERE type = 'E'.
      CONCATENATE l_message
                  ls_return-message
             INTO l_message.
    ENDLOOP.
    LOOP AT gt_data INTO ls_wb WHERE box = 'X'
                                 AND message = ''.
      ls_wb-message = l_message.
      MODIFY gt_data FROM ls_wb TRANSPORTING message.
    ENDLOOP.
  ENDIF.

ENDFORM.

FORM frm_user_command USING r_ucomm LIKE sy-ucomm
                             rs_selfield TYPE slis_selfield.

  DATA g_grid TYPE REF TO cl_gui_alv_grid. " 刷新数据
  CALL FUNCTION 'GET_GLOBALS_FROM_SLVC_FULLSCR'
    IMPORTING
      e_grid = g_grid.

  IF NOT g_grid IS INITIAL.
    CALL METHOD g_grid->refresh_table_display. "refresh alv
  ENDIF.

  CASE r_ucomm.

        ENDIF.
      ENDIF.
  ENDCASE.
