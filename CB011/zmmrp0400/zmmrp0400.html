*      WA_MSEG-LIFNR = WA_CHARACTERS-ZGYSDM.

*--------------------5.汇总凭证与物料数据----------------

    WA_ALV-ZTHJE   = WA_MBLNR-ZJE.

  ENDLOOP.
*--------------------6.按条件汇总克重,件数----------------
  SORT IT_GROUP BY LIFNR BWART RAUBE EX_PZ01 GRUND.
  LOOP AT IT_GROUP INTO WA_ALV.
    CLEAR WA_GROUP.
    WA_GROUP = WA_ALV.

    IF WA_GROUP-MEINS = 'G'.
      L_THKZ = L_THKZ + WA_GROUP-MENGE.
      L_THJE = L_THJE + WA_GROUP-ZTHJE.
      IF WA_GROUP-ZJZ NE 0.
        L_THJS = L_THJS + WA_GROUP-MENGE / WA_GROUP-ZJZ.
      ENDIF.
    ELSEIF WA_GROUP-MEINS = 'ST'.
      IF WA_GROUP-ZJZ NE 0.
        L_THKZ = L_THKZ + WA_GROUP-MENGE * WA_GROUP-ZJZ.
      ENDIF.
      L_THJS = L_THJS + WA_GROUP-MENGE.
      L_THJE = L_THJE + WA_GROUP-ZTHJE.
    ENDIF.

    AT END OF GRUND.

            L_THJE.

        L_JHJS,
        L_THJE.

      ENDIF.

      IF WA_HZ-ZJZ NE 0.
        L_JHKZ = L_JHKZ + WA_HZ-MENGE * WA_HZ-ZJZ.
      ENDIF.
      L_JHJS = L_JHJS + WA_HZ-MENGE.
    ENDIF.
    AT END OF LIFNR.

      CLEAR WA_ALV.
      WA_ALV-LIFNR = WA_HZ-LIFNR.
      WA_ALV-ZJHKZ = L_JHKZ.
      WA_ALV-ZJHJS = L_JHJS.
      APPEND WA_ALV TO IT_GROUP1.

      CLEAR:L_JHKZ,
            L_JHJS.
    ENDAT.
  ENDLOOP.


*--------------------7.分析占比----------------
  CLEAR IT_GROUP.
  IT_GROUP = IT_ALV.
*  IT_GROUP1 = IT_ALV.
*  DELETE IT_GROUP  WHERE BWART EQ '101'.
*  DELETE IT_GROUP1 WHERE BWART NE '101'.
  CLEAR IT_ALV.

  LOOP AT IT_GROUP INTO WA_GROUP.
    CLEAR WA_GROUP1.
    READ TABLE IT_GROUP1 INTO WA_GROUP1 WITH KEY LIFNR = WA_GROUP-LIFNR BINARY SEARCH.

    IF SY-SUBRC = 0.

      CLEAR WA_ALV.
      WA_ALV = WA_GROUP.

      WA_ALV-ZJHKZ = WA_GROUP1-ZJHKZ.
      WA_ALV-ZJHJS = WA_GROUP1-ZJHJS.

      IF WA_ALV-ZJHKZ NE 0.
        WA_ALV-ZKZZB = WA_ALV-ZTHKZ / WA_ALV-ZJHKZ.
      ENDIF.

      IF WA_ALV-ZJHJS NE 0.
        WA_ALV-ZJSZB = WA_ALV-ZTHJS / WA_ALV-ZJHJS.
      ENDIF.

      WA_ALV-ZKZZB = WA_ALV-ZKZZB * 100.
      WA_ALV-ZJSZB = WA_ALV-ZJSZB * 100.

      WA_ALV-KZZB_O = WA_ALV-ZKZZB && '%'.
      WA_ALV-JSZB_O = WA_ALV-ZJSZB && '%'.

      APPEND WA_ALV TO IT_ALV.

    ENDIF.

  ENDLOOP.

  IF IT_ALV IS INITIAL .

  ENDIF.

ENDFORM.

FORM FM_BWART_161 TABLES IT_TAB.

           MJAHR TYPE MSEG-MJAHR,      "凭证年度
           ZEILE TYPE MSEG-ZEILE,      "凭证行项目

           EBELN TYPE EKKO-EBELN,      "采购订单号
           ABSGR TYPE EKKO-ABSGR,      "退货原因

         A~EBELN
