*&---------------------------------------------------------------------*
*&读取EXCEL内容到任意内表
* I_PATH  : Excel文档路径
* O_EXCEL : 从EXCEL中获取到的内表数据
*&---------------------------------------------------------------------*
FORM FM_READ_EXCEL USING I_PATH O_EXCEL TYPE STANDARD TABLE.
  DATA:
    L_FILE  TYPE RLGRAP-FILENAME,
    LW_LINE TYPE REF TO DATA,
    LT_TAB  TYPE STANDARD TABLE OF ALSMEX_TABLINE,
    LW_TAB  TYPE ALSMEX_TABLINE.
  FIELD-SYMBOLS:
    <LINE>,
    <FS>.
  CREATE DATA LW_LINE LIKE LINE OF O_EXCEL.
  ASSIGN LW_LINE->* TO <LINE>.
  L_FILE = I_PATH.
  TRANSLATE L_FILE TO UPPER CASE. "字符转换成大写
  CALL FUNCTION 'ALSM_EXCEL_TO_INTERNAL_TABLE'
    EXPORTING
      FILENAME                = L_FILE
      I_BEGIN_COL             = 1
      I_BEGIN_ROW             = 1
      I_END_COL               = 250
      I_END_ROW               = 65536
    TABLES
      INTERN                  = LT_TAB
    EXCEPTIONS
      INCONSISTENT_PARAMETERS = 1
      UPLOAD_OLE              = 2
      OTHERS                  = 3.
  IF SY-SUBRC <> 0 .
    MESSAGE E001(00) WITH '上传文件有误！' SPACE SPACE SPACE.
  ENDIF.

  DELETE LT_TAB WHERE ROW = 1.  "删除字段名行

  REFRESH:
    O_EXCEL.
  LOOP AT LT_TAB INTO LW_TAB.

    ON CHANGE OF LW_TAB-ROW.
      IF SY-TABIX NE 1.
        APPEND  <LINE> TO O_EXCEL.
        CLEAR <LINE>.
      ENDIF.
    ENDON.

    ASSIGN COMPONENT LW_TAB-COL OF STRUCTURE <LINE> TO <FS>.
    TRY.
        <FS> = LW_TAB-VALUE.
      CATCH CX_ROOT.
        MESSAGE E001(00) WITH '上传文件的列多于接收内表的列' SPACE SPACE SPACE.
    ENDTRY.

    AT LAST.
      APPEND <LINE> TO O_EXCEL.
    ENDAT.

  ENDLOOP.

  IF O_EXCEL IS INITIAL.
    MESSAGE E001(00) WITH '上传文件为空' SPACE SPACE SPACE.
  ELSE.
    IT_TAB_01[] = O_EXCEL[].
  ENDIF.

ENDFORM.

*& 创建采购,入库,调拨用的内表
*&---------------------------------------------------------------------*
FORM FM_CREATE_TAB.
*检查商品款号是否正确
  TYPES: BEGIN OF STRU_MAKT,
           MATNR TYPE MAKT-MATNR,
           MAKTX TYPE MAKT-MAKTX,
         END OF STRU_MAKT.

  DATA: LT_MAKT   TYPE STANDARD TABLE OF STRU_MAKT,
        LS_MAKT   TYPE                   STRU_MAKT,
        LS_TAB_01 TYPE                   STRU_EXCEL.

  SELECT *
    FROM MAKT
    INTO CORRESPONDING FIELDS OF TABLE LT_MAKT
    WHERE MAKTX LIKE '%腕表%'.

  SORT LT_MAKT BY MATNR.
  LOOP AT IT_TAB_01 INTO LS_TAB_01.
    CLEAR LS_MAKT.
    READ TABLE LT_MAKT INTO LS_MAKT WITH KEY MATNR = LS_TAB_01-MATNR BINARY SEARCH.
    IF SY-SUBRC <> 0.
      MESSAGE S000(ZCB) WITH '输入的商品款号有错误,请检查' DISPLAY LIKE 'E'.
      LEAVE LIST-PROCESSING.
    ENDIF.
  ENDLOOP.

* 将EXCEL表转换成行项目表
  LOOP AT IT_TAB_01 INTO LS_TAB_01.
    CLEAR WA_ITEM.
    WA_ITEM-MATNR = LS_TAB_01-MATNR.
    WA_ITEM-MENGE = LS_TAB_01-MENGE.
    WA_ITEM-MWSKZ = S_MWSKZ-LOW.
    WA_ITEM-WERKS = S_WERKS-LOW.
    WA_ITEM-LGORT = S_LGORT-LOW.
    WA_ITEM-MEINS = 'ST'.
    APPEND WA_ITEM TO IT_ITEM.
  ENDLOOP.

ENDFORM.
