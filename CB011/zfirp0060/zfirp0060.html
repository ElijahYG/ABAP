FORM FM_CHECK_DATA.
  DATA:
    L_COUNT   TYPE I.    "计数器

  ENDIF.

    L_COUNT = L_COUNT + 1.
  ENDIF.

FORM FM_CAL_DATA.

  DATA:
    BEGIN OF LW_SUM,
      RACCT      TYPE   FAGLFLEXT-RACCT, "科目
      MONTH      TYPE   FAGLFLEXT-HSL01,  "月份
      VALUE01    TYPE   FAGLFLEXT-HSL01,  "当月贷方发生额
    END OF LW_SUM,
    L_MAX       TYPE I,
    LT_SUM      LIKE LW_SUM  OCCURS 0,
    LW_FAG      TYPE TY_FAG,

    LW_FSUM     TYPE TY_FAG.

* STEP1：按科目汇总数据
  SORT GT_FAG BY RACCT.

   CLEAR LW_FSUM.
  LOOP AT GT_FAG INTO LW_TMP.
*   防止*
    LW_FAG = LW_TMP.
*   按科目、期间汇总
    LW_FSUM-HSL01 = LW_FSUM-HSL01 + LW_FAG-HSL01.
    LW_FSUM-HSL02 = LW_FSUM-HSL02 + LW_FAG-HSL02.
    LW_FSUM-HSL03 = LW_FSUM-HSL03 + LW_FAG-HSL03.
    LW_FSUM-HSL04 = LW_FSUM-HSL04 + LW_FAG-HSL04.
    LW_FSUM-HSL05 = LW_FSUM-HSL05 + LW_FAG-HSL05.
    LW_FSUM-HSL06 = LW_FSUM-HSL06 + LW_FAG-HSL06.
    LW_FSUM-HSL07 = LW_FSUM-HSL07 + LW_FAG-HSL07.
    LW_FSUM-HSL08 = LW_FSUM-HSL08 + LW_FAG-HSL08.
    LW_FSUM-HSL09 = LW_FSUM-HSL09 + LW_FAG-HSL09.
    LW_FSUM-HSL10 = LW_FSUM-HSL10 + LW_FAG-HSL10.
    LW_FSUM-HSL11 = LW_FSUM-HSL11 + LW_FAG-HSL11.
    LW_FSUM-HSL12 = LW_FSUM-HSL12 + LW_FAG-HSL12 + LW_FAG-HSL13 + LW_FAG-HSL14.
    AT END OF RACCT.

      DO 12 TIMES.
       L_INDEX = SY-INDEX.
        L_FIELD = 'LW_FSUM-HSL' && L_INDEX.
        ASSIGN (L_FIELD) TO <FS>.
*       合计到当月
       IF L_INDEX <= S_MONAT-LOW.
         CLEAR LW_SUM.
         LW_SUM-MONTH = L_INDEX.
         LW_SUM-VALUE01 =  <FS>.
         LW_SUM-RACCT = LW_FAG-RACCT.
         APPEND LW_SUM TO LT_SUM.
         ELSEIF L_INDEX > S_MONAT-LOW.
           EXIT.
       ENDIF.
       ENDDO.
*     处理完毕，清空工作区
CLEAR LW_FSUM.
  ENDAT.
  ENDLOOP.

* STEP2：数据匹配到输出内表中
  SORT LT_SUM BY RACCT.
  LOOP AT LT_SUM INTO LW_SUM.
    CASE LW_SUM-RACCT.
        WHEN '2221001005'.
          CLEAR LW_OUT_TAB.
         READ TABLE GT_OUT_TAB INTO LW_OUT_TAB WITH KEY LINENO = LW_SUM-MONTH.
         IF  SY-SUBRC <> 0.
          LW_OUT_TAB-LINENO = LW_SUM-MONTH.
          LW_OUT_TAB-VALUE01 = LW_SUM-VALUE01 * -1.
          APPEND LW_OUT_TAB TO GT_OUT_TAB.
         ELSEIF  SY-SUBRC = 0.
          LW_OUT_TAB-VALUE01 = LW_SUM-VALUE01 * -1.
          MODIFY GT_OUT_TAB FROM LW_OUT_TAB INDEX SY-TABIX TRANSPORTING VALUE01.
          ENDIF.

        WHEN '2221001009'.
         CLEAR LW_OUT_TAB.
         READ TABLE GT_OUT_TAB INTO LW_OUT_TAB WITH KEY LINENO = LW_SUM-MONTH.
         IF  SY-SUBRC <> 0.

         ELSEIF  SY-SUBRC = 0.
          LW_OUT_TAB-VALUE02 = LW_SUM-VALUE01 * -1.
          MODIFY GT_OUT_TAB FROM LW_OUT_TAB INDEX SY-TABIX TRANSPORTING VALUE02.

          ELSEIF SY-SUBRC = 0.
          LW_OUT_TAB-VALUE09 = LW_SUM-VALUE01 * -1.
          MODIFY GT_OUT_TAB FROM LW_OUT_TAB INDEX SY-TABIX TRANSPORTING VALUE09.
          ENDIF.

         WHEN '2221001013'.
         CLEAR LW_OUT_TAB.
         READ TABLE GT_OUT_TAB INTO LW_OUT_TAB WITH KEY LINENO = LW_SUM-MONTH.
          IF  SY-SUBRC <> 0.
          LW_OUT_TAB-LINENO = LW_SUM-MONTH.
          LW_OUT_TAB-VALUE10 = LW_SUM-VALUE01 * -1.
          APPEND LW_OUT_TAB TO GT_OUT_TAB.
         ELSEIF SY-SUBRC = 0.
          LW_OUT_TAB-VALUE10 = LW_SUM-VALUE01 * -1.
          MODIFY GT_OUT_TAB FROM LW_OUT_TAB INDEX SY-TABIX TRANSPORTING VALUE10.
          ENDIF.
          ENDCASE.

          IF LW_SUM-RACCT BETWEEN '6000000000' AND '6799999999'.
          CLEAR LW_OUT_TAB.
          READ TABLE GT_OUT_TAB INTO LW_OUT_TAB WITH KEY LINENO = LW_SUM-MONTH.
          IF  SY-SUBRC <> 0.
          LW_OUT_TAB-LINENO = LW_SUM-MONTH.
          LW_OUT_TAB-VALUE11 = LW_SUM-VALUE01 * -1.
          APPEND LW_OUT_TAB TO GT_OUT_TAB.
          ELSEIF  SY-SUBRC = 0.
          LW_OUT_TAB-VALUE11 =  LW_OUT_TAB-VALUE11 + LW_SUM-VALUE01  * -1.
          MODIFY GT_OUT_TAB FROM LW_OUT_TAB INDEX SY-TABIX TRANSPORTING VALUE11.
          ENDIF.
          ENDIF.
ENDLOOP.
* 汇总值
*行汇总
 CLEAR LW_OUT_TAB.
DO 12 TIMES.
 LOOP AT GT_OUT_TAB INTO LW_OUT_TMP.
      CASE LW_OUT_TMP-LINENO.
       WHEN  SY-INDEX.
          LW_OUT_TAB-VALUE12 = LW_OUT_TMP-VALUE01 + LW_OUT_TMP-VALUE02 + LW_OUT_TMP-VALUE03 + LW_OUT_TMP-VALUE04 +
                               LW_OUT_TMP-VALUE05 + LW_OUT_TMP-VALUE06 + LW_OUT_TMP-VALUE07 + LW_OUT_TMP-VALUE08 +

      MODIFY GT_OUT_TAB FROM LW_OUT_TAB INDEX SY-TABIX TRANSPORTING VALUE12.
     ENDCASE.
    ENDLOOP.
 ENDDO.
*列汇总
  CLEAR LW_OUT_TAB.
    DO 12 TIMES.
  LOOP AT GT_OUT_TAB INTO LW_OUT_TMP.
    CASE LW_OUT_TMP-LINENO.
         WHEN SY-INDEX.
        LW_OUT_TAB-VALUE01 = LW_OUT_TAB-VALUE01 + LW_OUT_TMP-VALUE01.
        LW_OUT_TAB-VALUE02 = LW_OUT_TAB-VALUE02 + LW_OUT_TMP-VALUE02.

        LW_OUT_TAB-VALUE11 = LW_OUT_TAB-VALUE11 + LW_OUT_TMP-VALUE11.
        LW_OUT_TAB-VALUE12 = LW_OUT_TAB-VALUE12 + LW_OUT_TMP-VALUE12.
        ENDCASE.
  ENDLOOP.
   ENDDO.
         LW_OUT_TAB-LINENO = 13.

ENDFORM.

*&---------------------------------------------------------------------*
*&下载
*&---------------------------------------------------------------------*
FORM FM_DOWNLOAD.
  DATA:
    L_TEXT     TYPE CHAR100,
    L_DATE     TYPE SY-DATUM,
    LW_OUT_TAB TYPE TY_OUT_TAB,
    LW_EXCEL   TYPE TY_EXCEL,
    LT_EXCEL   TYPE TYT_EXCEL,
    L_PATH     TYPE STRING.
* 获取路径
  PERFORM FM_GET_FILEPATH USING L_PATH '利税表'.

  IF L_PATH IS NOT INITIAL.
*   下载模板到指定路径
    PERFORM FM_GET_TEMPLATE USING CNS_OBJID L_PATH.
*   数据编辑
*   1.编辑抬头
    CLEAR LW_EXCEL.
    LW_EXCEL-ROW = 1.
    LW_EXCEL-COL = 1.
    LW_EXCEL-VALUE = S_GJAHR-LOW && '年' && '利税表'.
    APPEND LW_EXCEL TO LT_EXCEL.
    IF S_MONAT-HIGH IS NOT INITIAL.
      L_DATE = S_GJAHR-LOW && S_MONAT-HIGH+1(2) && '01'.
    ELSE.
      L_DATE = S_GJAHR-LOW && S_MONAT-LOW+1(2) && '01'.
    ENDIF.
*   取当月最后一天
    CALL FUNCTION 'FIMA_DATE_CREATE'
      EXPORTING
        I_DATE                        = L_DATE
        I_SET_LAST_DAY_OF_MONTH       = 'X'
      IMPORTING
        E_DATE                        = L_DATE
              .
    CLEAR LW_EXCEL.
    LW_EXCEL-ROW = 2.
    LW_EXCEL-COL = 1.
    LW_EXCEL-VALUE = L_DATE+0(4) && '年' && L_DATE+4(2) && '月' && L_DATE+6(2) && '日'.
    APPEND LW_EXCEL TO LT_EXCEL.
*   2.编辑编制单位
    PERFORM FM_GET_TEXT USING L_TEXT.
    CLEAR LW_EXCEL.
    LW_EXCEL-ROW = 3.
    LW_EXCEL-COL = 1.
    LW_EXCEL-VALUE = '编制单位：'&& L_TEXT.
    APPEND LW_EXCEL TO LT_EXCEL.
*   3.编辑数据行
    CLEAR <FS>.
    DO 12 TIMES.
    LOOP AT GT_OUT_TAB1 INTO  LW_OUT_TAB1.
        L_INDEX = SY-INDEX.
        L_FIELD = 'LW_OUT_TAB1-VALUE' && L_INDEX.
        ASSIGN (L_FIELD) TO <FS>.
        IF <FS> IS NOT INITIAL.
        CLEAR LW_EXCEL.
        LW_EXCEL-ROW = LW_OUT_TAB1-LINENO + 4.
        LW_EXCEL-COL = SY-INDEX + 1.
        LW_EXCEL-VALUE = <FS>.
        APPEND LW_EXCEL TO LT_EXCEL.
      ENDIF.
    ENDLOOP.
    ENDDO.
*   数据填充
    PERFORM FM_FILL_DATA USING L_PATH LT_EXCEL.
  ELSE.
    MESSAGE S001(00) WITH '取消下载'.
    LEAVE LIST-PROCESSING.
  ENDIF.
ENDFORM.
