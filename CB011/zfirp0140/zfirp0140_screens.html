FORM FM_CHECK_OB.

  CLEAR GL_CHECK.

  CASE NUM1.

    WHEN 700.
      IF P_1 = 'X' AND S_DATE IS INITIAL.
        GL_CHECK = 'X'.
      ELSEIF P_2 = 'X' AND ( P_GJAHR IS INITIAL OR S_MONTH IS INITIAL ).
        GL_CHECK = 'X'.
      ENDIF.

      IF S_LIFNR IS INITIAL OR S_BUKRS IS INITIAL OR S_EXTWG IS INITIAL.
        GL_CHECK = 'X'.
*      ELSEIF S_EXTWG IS INITIAL AND S_RAUBE IS NOT INITIAL.
*        MESSAGE E001(00) WITH '请填写商品组对应的商品大类' DISPLAY LIKE 'E'.
*        LEAVE LIST-PROCESSING.
      ENDIF.

    WHEN 600.
      IF S_EXTWG3 IS INITIAL AND S_RAUBE3 IS NOT INITIAL.
        GL_CHECK = 'X'.
      ENDIF.
      IF P_3 = ABAP_TRUE AND S_ZJSDH3 IS INITIAL.
        GL_CHECK = ABAP_TRUE.
      ENDIF.
*检查日期是否全填
      IF S_DATE3 IS NOT INITIAL.
        IF S_DATE3-HIGH IS INITIAL OR S_DATE3-LOW IS INITIAL.
          GL_CHECK = 'D'.
          MESSAGE S001(00) WITH '请将日期区间填写完整' DISPLAY LIKE 'E'.
          EXIT.
        ENDIF.
      ENDIF.
    WHEN 800.
      IF S_EXTWG1 IS INITIAL AND S_RAUBE1 IS NOT INITIAL.
        GL_CHECK = 'X'.
      ENDIF.
      IF P_4 = ABAP_TRUE AND S_ZJSDH1 IS INITIAL.
        GL_CHECK = ABAP_TRUE.
      ENDIF.
*检查日期是否全填
      IF S_DATE1 IS NOT INITIAL.
        IF S_DATE1-HIGH IS INITIAL OR S_DATE1-LOW IS INITIAL.
          GL_CHECK = 'D'.
          MESSAGE S001(00) WITH '请将日期区间填写完整' DISPLAY LIKE 'E'.
          EXIT.
        ENDIF.
      ENDIF.
    WHEN 900.
*      IF S_LIFNR2 IS INITIAL OR S_BUKRS2 IS INITIAL.
*        GL_CHECK = 'X'.
*      ENDIF.

  ENDCASE.

ENDFORM.


*切屏清楚输入值,隐藏输入框
FORM FM_CHECK_SCREEN9000.
  LOOP AT SCREEN.
    CASE NUM1.
      WHEN 700.
        CLEAR:S_LIFNR2[], S_LIFNR1[], S_EXTWG1[], S_ZJSDH1[],S_LIFNR3[], S_EXTWG3[], S_ZJSDH3[],
              S_BUKRS2[], S_BUKRS1[], S_RAUBE1[],S_BUKRS3[], S_RAUBE3[],S_GJAHR3[],
              S_ZJSDH2[], S_GJAHR1[].
        P_3 = ABAP_FALSE.P_4 = ABAP_FALSE.
        IF P_1 = 'X' AND SCREEN-GROUP1 = 'BL2'.   "自由结算
          SCREEN-ACTIVE = '0'.
          CLEAR:S_MONTH[]."P_GJAHR.
        ENDIF.
        IF P_2 = 'X' AND SCREEN-GROUP1 = 'BL1'.   "自然月结算
*      SCREEN-INPUT = '0'.
          SCREEN-ACTIVE = '0'.
          CLEAR S_DATE[].
        ENDIF.

      WHEN 600.
        CLEAR: S_LIFNR[],S_BUKRS[],S_EXTWG[],S_RAUBE[],S_DATE[],S_LIFNR1[], S_EXTWG1[], S_ZJSDH1[],
                S_MONTH[],S_LIFNR2[],S_BUKRS2[],S_ZJSDH2[],S_BUKRS1[], S_RAUBE1[],S_GJAHR1[],P_2.
        P_1 = 'X'.
        P_4 = ABAP_FALSE.
        IF P_3 = 'X' AND SCREEN-GROUP1 = '601'.   "按单号查询
          SCREEN-ACTIVE = '0'.
          CLEAR: S_BUKRS3[], S_RAUBE3[],S_GJAHR3[],S_LIFNR3[], S_EXTWG3[].
        ELSEIF P_3 = ABAP_FALSE AND SCREEN-GROUP1 = '602'.
          SCREEN-ACTIVE = '0'.
          CLEAR: S_ZJSDH3[].
        ENDIF.

      WHEN 800.
        CLEAR: S_LIFNR[],S_BUKRS[],S_EXTWG[],S_RAUBE[],S_DATE[],
               S_BUKRS3[], S_RAUBE3[],S_GJAHR3[],S_LIFNR3[], S_EXTWG3[], S_ZJSDH3[],
               S_MONTH[],S_LIFNR2[],S_BUKRS2[],S_ZJSDH2[],P_2.
        P_1 = 'X'.
        P_3 = ABAP_FALSE.
        IF P_4 = 'X' AND SCREEN-GROUP1 = '801'.   "按单号查询
          SCREEN-ACTIVE = '0'.
          CLEAR: S_BUKRS1[], S_RAUBE1[],S_GJAHR1[],S_LIFNR1[], S_EXTWG1[].
        ELSEIF P_4 = ABAP_FALSE AND SCREEN-GROUP1 = '802'.
          SCREEN-ACTIVE = '0'.
          CLEAR: S_ZJSDH1[].
        ENDIF.

      WHEN 900.
        CLEAR:S_LIFNR1[], S_EXTWG1[], S_ZJSDH1[],S_LIFNR[],S_BUKRS[],S_EXTWG[],S_RAUBE[],
              S_BUKRS3[], S_RAUBE3[],S_GJAHR3[],S_LIFNR3[], S_EXTWG3[], S_ZJSDH3[],
              S_DATE[],S_MONTH[],S_BUKRS1[], S_RAUBE1[],S_GJAHR1[],P_2.
        P_1 = 'X'.P_3 = ABAP_FALSE.P_4 = ABAP_FALSE.
    ENDCASE.
    MODIFY SCREEN.
  ENDLOOP.


ENDFORM.
