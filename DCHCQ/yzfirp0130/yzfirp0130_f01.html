FORM FRM_GET_BANKCOPY.
  REFRESH: GT_YZFIRP0130A,
           GT_YZFIRP0130B,
           GT_YZFIRP0130C.
  REFRESH: GT_BANKCOPY,
           GT_BCRESULT.

*>分组
  SELECT * INTO TABLE GT_YZFIRP0130A
    FROM YZFIRP0130A
   WHERE PRCTR  EQ YZFIRP0130A-PRCTR
     AND HKONT  EQ YZFIRP0130A-HKONT
     AND ZBVSTA IN (GC_UNPROCESSED, GC_PROCESSING, GC_PROCESSED).
  CHECK SY-SUBRC EQ 0 AND NOT GT_YZFIRP0130A IS INITIAL.

*>对帐单
  SELECT * INTO TABLE GT_YZFIRP0130B
    FROM YZFIRP0130B
     FOR ALL ENTRIES IN GT_YZFIRP0130A
   WHERE HKONT  EQ GT_YZFIRP0130A-HKONT
     AND ZGRPID EQ GT_YZFIRP0130A-ZGRPID
     AND ZBVSTA IN (GC_UNPROCESSED, GC_PROCESSED).
  CHECK SY-SUBRC EQ 0 AND NOT GT_YZFIRP0130B IS INITIAL.

*>对帐结果
  SELECT * INTO TABLE GT_YZFIRP0130C
    FROM YZFIRP0130C
     FOR ALL ENTRIES IN GT_YZFIRP0130B
   WHERE HKONT  EQ GT_YZFIRP0130B-HKONT
     AND ZGRPID EQ GT_YZFIRP0130B-ZGRPID
     AND ZBCKNO EQ GT_YZFIRP0130B-ZBCKNO.

  LOOP AT GT_YZFIRP0130B INTO WA_YZFIRP0130B.
    CLEAR WA_BANKCOPY.
    READ TABLE GT_YZFIRP0130A INTO WA_YZFIRP0130A WITH KEY HKONT  = WA_YZFIRP0130B-HKONT
                                                 ZGRPID = WA_YZFIRP0130B-ZGRPID.
    CHECK SY-SUBRC EQ 0.
    MOVE-CORRESPONDING: WA_YZFIRP0130A TO WA_BANKCOPY,
                        WA_YZFIRP0130B TO WA_BANKCOPY.
    APPEND WA_BANKCOPY TO GT_BANKCOPY.
  ENDLOOP.
  SORT GT_BANKCOPY BY HKONT ZGRPID ZBCKNO.

  LOOP AT GT_YZFIRP0130C INTO WA_YZFIRP0130C.
    CLEAR WA_BCRESULT.
    MOVE-CORRESPONDING WA_YZFIRP0130C TO WA_BCRESULT.
    APPEND WA_BCRESULT TO GT_BCRESULT.
  ENDLOOP.
  CLEAR GV_CKNUM.
  LOOP AT GT_BCRESULT INTO WA_BCRESULT.
    CHECK WA_BCRESULT-CKNUM EQ 0.
    ADD 1 TO GV_CKNUM.
    MOVE: GV_CKNUM     TO WA_BCRESULT-CKNUM.
    MODIFY GT_BCRESULT FROM WA_BCRESULT TRANSPORTING CKNUM WHERE HKONT  EQ WA_BCRESULT-HKONT
                                                             AND ZGRPID EQ WA_BCRESULT-ZGRPID
                                                             AND ZBCKNO EQ WA_BCRESULT-ZBCKNO.
    MODIFY GT_BCRESULT FROM WA_BCRESULT TRANSPORTING CKNUM WHERE BUKRS EQ WA_BCRESULT-BUKRS
                                                             AND BELNR EQ WA_BCRESULT-BELNR
                                                             AND GJAHR EQ WA_BCRESULT-GJAHR
                                                             AND BUZEI EQ WA_BCRESULT-BUZEI.
  ENDLOOP.
ENDFORM. "FRM_GET_BANKCOPY

FORM FRM_SET_STATUS_ICON USING UV_ZBVSTA
                         CHANGING CV_ICON_STATUS.
  CASE UV_ZBVSTA.
    WHEN GC_UNPROCESSED.
      PERFORM FRM_CREATE_ICON USING    ICON_LED_YELLOW '' '未处理'
                              CHANGING CV_ICON_STATUS.

    WHEN GC_PROCESSING.
      PERFORM FRM_CREATE_ICON USING    ICON_LED_YELLOW '' '未处理'
                              CHANGING CV_ICON_STATUS.

    WHEN GC_PROCESSED.
      PERFORM FRM_CREATE_ICON USING    ICON_LED_GREEN '' '处理完成'
                              CHANGING CV_ICON_STATUS.

    WHEN GC_DELETED.
      PERFORM FRM_CREATE_ICON USING    ICON_DELETE    '' '处理完成'
                              CHANGING CV_ICON_STATUS.

    WHEN OTHERS.
  ENDCASE.
ENDFORM. "frm_set_status_icon
