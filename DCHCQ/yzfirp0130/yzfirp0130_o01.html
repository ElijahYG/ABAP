MODULE SET_DYNP_FIELDS_0002 OUTPUT.
  LOOP AT SCREEN.
    CASE SCREEN-GROUP1.
      WHEN 'KEY'.
        CASE GV_OPERATION.
          WHEN GC_SELECT_GROUP.
            SCREEN-INPUT = 1.
            SCREEN-REQUIRED = 1.
          WHEN GC_CREATE_GROUP.
            IF SCREEN-GROUP2 EQ 'INP'.
              SCREEN-INPUT = 1.
              SCREEN-REQUIRED = 1.
            ELSE.
              SCREEN-INPUT = 0.
              SCREEN-REQUIRED = 0.
            ENDIF.
          WHEN GC_CHANGE_GROUP.
            SCREEN-INPUT = 0.
            SCREEN-REQUIRED = 0.
          WHEN GC_DISPLAY_GROUP.
            SCREEN-INPUT = 1.
            SCREEN-REQUIRED = 1.
          WHEN GC_SELECT_UPLOAD.
            SCREEN-INPUT = 1.
            SCREEN-REQUIRED = 1.
          WHEN OTHERS.
        ENDCASE.

      WHEN 'INP'.
        CASE GV_OPERATION.
          WHEN GC_SELECT_GROUP.
            SCREEN-INPUT    = 0.
            SCREEN-REQUIRED = 0.
          WHEN GC_CREATE_GROUP.
            SCREEN-INPUT    = 1.
            SCREEN-REQUIRED = 1.
          WHEN GC_CHANGE_GROUP.
            SCREEN-INPUT = 1.
            SCREEN-REQUIRED = 1.
          WHEN GC_DISPLAY_GROUP.
            SCREEN-INPUT = 0.
            SCREEN-REQUIRED = 0.
          WHEN GC_SELECT_UPLOAD.
            SCREEN-INPUT = 0.
            SCREEN-REQUIRED = 0.
*            screen-active = 0.
          WHEN OTHERS.
        ENDCASE.

      WHEN 'DIS'.
        CASE GV_OPERATION.
          WHEN GC_SELECT_UPLOAD.
            SCREEN-ACTIVE = 0.
          WHEN OTHERS.
        ENDCASE.

      WHEN 'TXT'.

      WHEN 'FIL'.
        CASE GV_OPERATION.
          WHEN GC_SELECT_UPLOAD.
            SCREEN-INPUT = 1.
            SCREEN-REQUIRED = 1.
          WHEN OTHERS.
            SCREEN-ACTIVE = 0.
        ENDCASE.

      WHEN OTHERS.
        CONTINUE.
    ENDCASE.
    MODIFY SCREEN.
  ENDLOOP.
ENDMODULE. " SET_DYNP_FIELDS_0002 OUTPUT
