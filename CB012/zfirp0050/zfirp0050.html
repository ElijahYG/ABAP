*&---------------------------------------------------------------------*
*&计算
*&---------------------------------------------------------------------*
FORM FM_CAL_DATA.
  FIELD-SYMBOLS <FS>.

  DATA:
    BEGIN OF LW_SUM,
      RACCT     TYPE   FAGLFLEXT-RACCT, "科目
      DRCRK     TYPE   FAGLFLEXT-DRCRK, "借贷标识
*      VALUE0    TYPE   FAGLFLEXT-HSL01, "期初余额
      VALUE1    TYPE   FAGLFLEXT-HSL01, "当期汇总值
*      VALUE2    TYPE   FAGLFLEXT-HSL01, "当年汇总值
    END OF LW_SUM,
    L_INDEX     TYPE NUMC2,
    L_MAX       TYPE I,
    LT_SUM      LIKE LW_SUM  OCCURS 0,
    LW_FAG      TYPE TY_FAG,
    LW_TMP      TYPE TY_FAG,
    LW_FSUM     TYPE TY_FAG,
    L_FIELD     TYPE CHAR100.
* STEP1：按科目、借贷标识汇总数据
  SORT GT_FAG BY RACCT.
* 确定合计到哪个期间
  IF S_MONAT-HIGH IS NOT INITIAL.
    L_MAX = S_MONAT-HIGH.
  ELSE.
    L_MAX = S_MONAT-LOW.
  ENDIF.

  CLEAR LW_FSUM.
  LOOP AT GT_FAG INTO LW_TMP.
*   防止*
    LW_FAG = LW_TMP.
*   按科目、借贷标识、期间汇总
    LW_FSUM-HSLVT = LW_FSUM-HSLVT + LW_FAG-HSLVT.
    LW_FSUM-HSL01 = LW_FSUM-HSL01 + LW_FAG-HSL01.
    LW_FSUM-HSL02 = LW_FSUM-HSL02 + LW_FAG-HSL02.
    LW_FSUM-HSL03 = LW_FSUM-HSL03 + LW_FAG-HSL03.
    LW_FSUM-HSL04 = LW_FSUM-HSL04 + LW_FAG-HSL04.
    LW_FSUM-HSL05 = LW_FSUM-HSL05 + LW_FAG-HSL05.
    LW_FSUM-HSL06 = LW_FSUM-HSL06 + LW_FAG-HSL06.
    LW_FSUM-HSL07 = LW_FSUM-HSL07 + LW_FAG-HSL07.
    LW_FSUM-HSL08 = LW_FSUM-HSL08 + LW_FAG-HSL08.
    LW_FSUM-HSL09 = LW_FSUM-HSL09 + LW_FAG-HSL09.
    LW_FSUM-HSL10 = LW_FSUM-HSL10 + LW_FAG-HSL10.
    LW_FSUM-HSL11 = LW_FSUM-HSL11 + LW_FAG-HSL11.
    LW_FSUM-HSL12 = LW_FSUM-HSL12 + LW_FAG-HSL12 + LW_FAG-HSL13 + LW_FAG-HSL14.
    AT END OF RACCT.
*
      CLEAR LW_SUM.
      DO 12 TIMES.
        L_INDEX = SY-INDEX.

        L_FIELD = 'LW_FSUM-HSL' && L_INDEX.
        ASSIGN (L_FIELD) TO <FS>.
*       合计到当月
        IF L_INDEX IN S_MONAT.
          LW_SUM-VALUE1 = LW_SUM-VALUE1 + <FS>.
        ENDIF.

*       计算当年累计值
*        IF L_INDEX <= L_MAX.
*          LW_SUM-VALUE2 = LW_SUM-VALUE2 + <FS>.
*        ENDIF.
      ENDDO.
*     期初
*      LW_SUM-VALUE0 = LW_FSUM-HSLVT.
*     计算当月和当年的发生额
      LW_SUM-RACCT = LW_FAG-RACCT.  "科目
*      LW_SUM-DRCRK = LW_FAG-DRCRK.  "借贷标识
      APPEND LW_SUM TO LT_SUM.

*     处理完毕，清空工作区
      CLEAR LW_FSUM.
    ENDAT.
  ENDLOOP.

* STEP2：数据匹配到输出内表中
  SORT LT_SUM BY RACCT.
  LOOP AT LT_SUM INTO LW_SUM.
    READ TABLE GT_OUT_TAB INTO LW_OUT_TAB WITH KEY LINENO = 1.
        IF SY-SUBRC <> 0.
      CLEAR LW_OUT_TAB.
           LW_OUT_TAB-LINENO = 1.                   "行号
           LW_OUT_TAB-VALUE1 = LW_SUM-VALUE1 * -1.
      APPEND LW_OUT_TAB TO GT_OUT_TAB.
      ELSE.
          LW_OUT_TAB-VALUE1 = LW_SUM-VALUE1 * -1 + LW_OUT_TAB-VALUE1.
      MODIFY GT_OUT_TAB FROM LW_OUT_TAB INDEX SY-TABIX TRANSPORTING VALUE1.
      ENDIF.
  ENDLOOP.
* 汇总值

*2行

 CLEAR LW_OUT_TAB.
   LOOP AT GT_OUT_TAB INTO LW_OUT_TMP.
     IF LW_OUT_TMP-LINENO = 1.
         LW_OUT_TAB-VALUE1 = LW_OUT_TMP-VALUE1 * 25 / 100.
     ENDIF.
     ENDLOOP.
           LW_OUT_TAB-LINENO = 2.
   APPEND LW_OUT_TAB TO GT_OUT_TAB.

*3
    CLEAR LW_OUT_TAB.
    LOOP AT GT_OUT_TAB INTO LW_OUT_TMP.
      CASE LW_OUT_TMP-LINENO.
      WHEN 1.
        LW_OUT_TAB-VALUE1 = LW_OUT_TAB-VALUE1 + LW_OUT_TMP-VALUE1.
      WHEN 2.
        LW_OUT_TAB-VALUE1 = LW_OUT_TAB-VALUE1 - LW_OUT_TMP-VALUE1.
    ENDCASE.
    ENDLOOP.
        LW_OUT_TAB-LINENO = 3.
   APPEND LW_OUT_TAB TO GT_OUT_TAB.

*4
   CLEAR LW_OUT_TAB.
   LOOP AT GT_OUT_TAB INTO LW_OUT_TMP.
     IF LW_OUT_TMP-LINENO = 3.
         LW_OUT_TAB-VALUE1 = LW_OUT_TMP-VALUE1 * 10 / 100.
     ENDIF.
     ENDLOOP.
           LW_OUT_TAB-LINENO = 4.
   APPEND LW_OUT_TAB TO GT_OUT_TAB.
*5
    CLEAR LW_OUT_TAB.
    LOOP AT GT_OUT_TAB INTO LW_OUT_TMP.
      CASE LW_OUT_TMP-LINENO.
      WHEN 3.
        LW_OUT_TAB-VALUE1 = LW_OUT_TAB-VALUE1 + LW_OUT_TMP-VALUE1 * 10 / 100.
      WHEN 4.
        LW_OUT_TAB-VALUE1 = LW_OUT_TAB-VALUE1 - LW_OUT_TMP-VALUE1 * 10 / 100.
    ENDCASE.
    ENDLOOP.
        LW_OUT_TAB-LINENO = 5.
   APPEND LW_OUT_TAB TO GT_OUT_TAB.
*6
    CLEAR LW_OUT_TAB.
    LOOP AT GT_OUT_TAB INTO LW_OUT_TMP.
      CASE LW_OUT_TMP-LINENO.
      WHEN 3.
        LW_OUT_TAB-VALUE1 = LW_OUT_TAB-VALUE1 + LW_OUT_TMP-VALUE1.
      WHEN 4.
        LW_OUT_TAB-VALUE1 = LW_OUT_TAB-VALUE1 - LW_OUT_TMP-VALUE1.
      WHEN 5.
        LW_OUT_TAB-VALUE1 = LW_OUT_TAB-VALUE1 - LW_OUT_TMP-VALUE1.
    ENDCASE.
    ENDLOOP.
        LW_OUT_TAB-LINENO = 6.
   APPEND LW_OUT_TAB TO GT_OUT_TAB.

ENDFORM.

FORM FM_MINUS.

  LOOP AT  GT_OUT_TAB INTO LW_OUT_TMP.
       CLEAR LW_OUT_TAB.
       LW_OUT_TAB-VALUE2 = LW_OUT_TMP-VALUE1.
       MODIFY GT_OUT_TAB FROM LW_OUT_TAB INDEX SY-TABIX TRANSPORTING VALUE2.
  ENDLOOP.


    LOOP AT GT_OUT_TAB INTO LW_OUT_TAB.
    IF LW_OUT_TAB-VALUE2 = 0.
      LW_OUT_TAB-VALUE2 = ' '.
    ENDIF.
    CALL FUNCTION 'CLOI_PUT_SIGN_IN_FRONT'
      CHANGING
       VALUE = LW_OUT_TAB-VALUE2.
   MODIFY GT_OUT_TAB FROM LW_OUT_TAB INDEX SY-TABIX TRANSPORTING VALUE2.
  ENDLOOP.



ENDFORM.

FORM FM_DOWNLOAD.
  DATA:
    L_TEXT     TYPE CHAR100,
    L_DATE     TYPE SY-DATUM,
    LW_OUT_TAB TYPE TY_OUT_TAB,
    LW_EXCEL   TYPE TY_EXCEL,
    LT_EXCEL   TYPE TYT_EXCEL,
    L_PATH     TYPE STRING.
* 获取路径
  PERFORM FM_GET_FILEPATH USING L_PATH '计提所得税'.

  IF L_PATH IS NOT INITIAL.
*   下载模板到指定路径
    PERFORM FM_GET_TEMPLATE USING CNS_OBJID L_PATH.
*   数据编辑
*   1.编辑抬头
    IF S_MONAT-HIGH IS NOT INITIAL.
      L_DATE = S_GJAHR-LOW && S_MONAT-HIGH+1(2) && '01'.
    ELSE.
      L_DATE = S_GJAHR-LOW && S_MONAT-LOW+1(2) && '01'.
    ENDIF.
*   取当月最后一天
    CALL FUNCTION 'FIMA_DATE_CREATE'
      EXPORTING
        I_DATE                        = L_DATE
        I_SET_LAST_DAY_OF_MONTH       = 'X'
      IMPORTING
        E_DATE                        = L_DATE
              .

    CLEAR LW_EXCEL.
    LW_EXCEL-ROW = 2.
    LW_EXCEL-COL = 1.
    LW_EXCEL-VALUE = L_DATE+0(4) && '年' && L_DATE+4(2) && '月' && L_DATE+6(2) && '日'.
    APPEND LW_EXCEL TO LT_EXCEL.
*   2.编辑编制单位
    PERFORM FM_GET_TEXT USING L_TEXT.
    CLEAR LW_EXCEL.
    LW_EXCEL-ROW = 3.
    LW_EXCEL-COL = 1.
    LW_EXCEL-VALUE = '编制单位：'&& L_TEXT && '                              单位: 元'.
    APPEND LW_EXCEL TO LT_EXCEL.
*   3.编辑数据行
     LOOP AT GT_OUT_TAB INTO LW_OUT_TAB.
    IF LW_OUT_TAB-VALUE1 IS NOT INITIAL.
      CLEAR LW_EXCEL.
      LW_EXCEL-ROW = LW_OUT_TAB-LINENO + 3.
      LW_EXCEL-COL = 2.
      LW_EXCEL-VALUE = LW_OUT_TAB-VALUE2.
      APPEND LW_EXCEL TO LT_EXCEL.
      ENDIF.
     ENDLOOP.
*   数据填充
    PERFORM FM_FILL_DATA USING L_PATH LT_EXCEL.
  ELSE.
    MESSAGE S001(00) WITH '取消下载'.
    LEAVE LIST-PROCESSING.
  ENDIF.
ENDFORM.

*&---------------------------------------------------------------------*
*&取公司文本、利润中心文本或者利润中心组文本
*&---------------------------------------------------------------------*
FORM FM_GET_TEXT USING O_TEXT.
  DATA:
    L_TEXT TYPE CHAR100.
* 取公司代码文本
  IF S_BUKRS-LOW IS NOT INITIAL.
    SELECT SINGLE BUTXT
      FROM T001
      INTO L_TEXT
     WHERE BUKRS = S_BUKRS-LOW  "公司代码
      .
  ENDIF.
* 取利润中心文本
  IF S_PRCTR-LOW IS NOT INITIAL.
    SELECT SINGLE LTEXT
      FROM CEPCT
      INTO L_TEXT
     WHERE PRCTR = S_PRCTR-LOW "利润中心
       AND SPRAS = SY-LANGU    "语言代码
       AND DATBI >= SY-DATUM   "有效日期

         .
  ENDIF.
* 取利润中心组文本
  IF S_SNAME-LOW IS NOT INITIAL.
    SELECT SINGLE DESCRIPT
      FROM SETHEADERT
      INTO L_TEXT
     WHERE SETCLASS = '0106'  "固定值标识利润中心
       AND SETNAME  = S_SNAME-LOW
       AND LANGU    = SY-LANGU
      .
  ENDIF.

  O_TEXT = L_TEXT.
ENDFORM.
