  ZMMRP0050_SCREEN-BLDAT  = S_BLDAT-LOW.
  ZMMRP0050_SCREEN-BUDAT  = S_BUDAT-LOW.
  ZMMRP0050_SCREEN-EKGRP  = '110'.

* 1：判断发货类型
  DATA:
    L_BUKRS1 TYPE BUKRS,
    L_BUKRS2 TYPE BUKRS.
* 取发货地点对应的公司
  SELECT SINGLE BUKRS
    INTO L_BUKRS1
    FROM T001K AS A
    INNER JOIN T001W AS B
      ON A~BWKEY = B~BWKEY
   WHERE B~WERKS = S_WERKS-LOW
    .
* 取收货地点对应的公司
  SELECT SINGLE BUKRS
    INTO L_BUKRS2
    FROM T001K AS A
    INNER JOIN T001W AS B
      ON A~BWKEY = B~BWKEY
   WHERE B~WERKS = S_WERKST-LOW
    .
  IF L_BUKRS1 = L_BUKRS2.
    G_BSART = 'ZSUB'.
  ELSE.
    G_BSART = 'ZSTO'.
  ENDIF.

* 2.取采购组织
  SELECT SINGLE EKORG
    INTO G_EKORG                      "采购组织
    FROM T001W
   WHERE WERKS = S_WERKST-LOW
      .
  CALL SCREEN 8000.

FORM FM_START_CHECK.
  DATA:
    L_WERKS TYPE WERKS_D.
* 要货单
* 检查要货单状态
  IF S_YHDH-LOW IS NOT INITIAL.
    SELECT COUNT(*)
      FROM ZSDIF0050_HEAD
     WHERE DOC_NUMBER = S_YHDH-LOW
       AND ORD_TYP <> 'LX01'
       AND DOC_IND = '0'
      .
    IF SY-SUBRC <> 0.
      MESSAGE S009(ZCB) WITH S_YHDH-LOW DISPLAY LIKE 'E'.
      LEAVE LIST-PROCESSING.
    ELSE.
      SELECT COUNT(*)
        FROM ZSDIF0050_ITEM
       WHERE DOC_NUMBER = S_YHDH-LOW
         AND ITEM_IND = '0'
        .
      IF SY-SUBRC <> 0.
        MESSAGE S009(ZCB) WITH S_YHDH-LOW DISPLAY LIKE 'E'.
        LEAVE LIST-PROCESSING.
      ENDIF.
    ENDIF.
*   要货门店和收货门店一致性检查
    SELECT SINGLE WERKS
      INTO L_WERKS
      FROM ZSDIF0050_HEAD
     WHERE DOC_NUMBER = S_YHDH-LOW
      .
    IF L_WERKS <> S_WERKST-LOW .
      MESSAGE S024(ZCB) WITH L_WERKS S_WERKST-LOW DISPLAY LIKE 'E'.
      LEAVE LIST-PROCESSING.
    ENDIF.
  ENDIF.
ENDFORM.
