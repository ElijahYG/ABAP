Description: test
REPORT ZLQTEST5.

CONTROLS TC_MAIN TYPE TABSTRIP.
DATA DYNNR TYPE SY-DYNNR.
DATA OK_CODE TYPE SY-UCOMM.

CALL SCREEN 8000.
*&---------------------------------------------------------------------*
*&      Module  STATUS_8000  OUTPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE STATUS_8000 OUTPUT.
  CASE OK_CODE.
    WHEN 'TC1'.
      DYNNR = '8001'.
      TC_MAIN-ACTIVETAB = 'TC1'.
    WHEN 'TC2'.
      DYNNR = '8002'.
      TC_MAIN-ACTIVETAB = 'TC2'.
    WHEN ''.
      DYNNR = '8001'.
      TC_MAIN-ACTIVETAB = 'TC1'.
  ENDCASE.
ENDMODULE.
*&---------------------------------------------------------------------*
*&      Module  USER_COMMAND_8000  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE USER_COMMAND_8000 INPUT.
  CASE OK_CODE.
    WHEN 'TC1'.
      DYNNR = '8001'.
      TC_MAIN-ACTIVETAB = 'TC1'.
    WHEN 'TC2'.
      DYNNR = '8002'.
      TC_MAIN-ACTIVETAB = 'TC2'.
    WHEN OTHERS.
      LEAVE TO SCREEN 0.
  ENDCASE.
ENDMODULE.
