FORM FRM_GET_EXCEL_DATA.
*倒入EXCEL数据到内表
  CALL FUNCTION 'ALSM_EXCEL_TO_INTERNAL_TABLE'
    EXPORTING
      FILENAME    = P_PATH
      I_BEGIN_COL = '1'
      I_BEGIN_ROW = '2'
      I_END_COL   = '12'
      I_END_ROW   = '65000'
    TABLES
      INTERN      = GT_DATA.

  SORT GT_DATA BY ROW COL VALUE.

  LOOP AT GT_DATA.
    CASE GT_DATA-COL.
      WHEN 1.  IT_ITAB-BUDAT = GT_DATA-VALUE.
      WHEN 2.  IT_ITAB-CPUDT = GT_DATA-VALUE.
      WHEN 3.  IT_ITAB-HEADER = GT_DATA-VALUE.
      WHEN 4.  IT_ITAB-MATNR = GT_DATA-VALUE.
      WHEN 5.  IT_ITAB-PLANT = GT_DATA-VALUE.
      WHEN 6.  IT_ITAB-STGE_LOC = GT_DATA-VALUE.
      WHEN 7.  IT_ITAB-MENGE = GT_DATA-VALUE.
      WHEN 8.


      WHEN 9.  IT_ITAB-BUALT = GT_DATA-VALUE.   "价格
      WHEN 10.  IT_ITAB-BWART = GT_DATA-VALUE.
      WHEN 11.
        CALL FUNCTION 'CONVERSION_EXIT_ALPHA_INPUT'
          EXPORTING
            INPUT         = GT_DATA-VALUE
          IMPORTING
            OUTPUT        = IT_ITAB-LIFNR
                   .
      WHEN 12.
        IT_ITAB-CHARG = GT_DATA-VALUE.

    ENDCASE.

    SELECT SINGLE MEINS INTO IT_ITAB-MEINS
            FROM MARA
           WHERE MATNR = IT_ITAB-MATNR.


    AT END OF ROW.
      APPEND IT_ITAB.
      CLEAR: IT_ITAB.
    ENDAT.

  ENDLOOP.
ENDFORM.






FORM FRM_DOWNDLOAD_TEMPLATE USING P_CLNM TYPE SEOCLSNAME
                                    P_CLTYP TYPE BDS_CLSTYP
                                    P_KEY   TYPE BDS_TYPEID.

  DATA: L_FILE TYPE STRING,
        L_PATH TYPE STRING,
        L_NAME TYPE STRING.

  DATA: BDS_INSTANCE          TYPE REF TO CL_BDS_DOCUMENT_SET,

*&---------------------------------------------------------------------*
*&      Form  FRM_CALL_BAPI_FOR_CREATE
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM FRM_CALL_BAPI_FOR_CREATE .

  DATA: L_COUNT   TYPE I,
        L_TABIX   LIKE SY-TABIX,
        L_V_TABIX LIKE SY-TABIX.

  DATA: LS_ITAB LIKE IT_ITAB.
  DATA: L_FLAG TYPE C.

*&*****根据工厂 库存地点 拆分凭证，但是当行项目数量大于450条时，也许要分拆
  SORT IT_ITAB BY PLANT STGE_LOC.
  LOOP AT IT_ITAB .
    L_COUNT = L_COUNT + 1.
    L_TABIX = SY-TABIX.


    CLEAR: LS_ITAB.
    L_V_TABIX = L_TABIX - 1.
    READ TABLE IT_ITAB INTO LS_ITAB INDEX L_V_TABIX.
    IF IT_ITAB-PLANT <> LS_ITAB-PLANT OR IT_ITAB-STGE_LOC <> LS_ITAB-STGE_LOC  OR L_FLAG = 'N'.
      CLEAR: L_FLAG.
      PERFORM FRM_SET_BAPI_HEADER USING IT_ITAB.
    ENDIF.


    PERFORM FRM_SET_BAPI_ITEM USING IT_ITAB.

    IF L_COUNT >= 450 .
      L_COUNT = 0.
      L_FLAG = 'N'.
      PERFORM FRM_CALL_BAPI_FOR_CREATE_PZ USING IT_ITAB.
      MODIFY IT_ITAB INDEX L_TABIX TRANSPORTING MBLNR MESSAGE.
    ELSE.
      CLEAR: LS_ITAB.
      L_V_TABIX = L_TABIX + 1.
      READ TABLE IT_ITAB INTO LS_ITAB INDEX L_V_TABIX.
      IF IT_ITAB-PLANT <> LS_ITAB-PLANT OR IT_ITAB-STGE_LOC <> LS_ITAB-STGE_LOC.
        L_COUNT = 0.
        L_FLAG = 'N'.
        PERFORM FRM_CALL_BAPI_FOR_CREATE_PZ USING IT_ITAB.
        MODIFY IT_ITAB INDEX L_TABIX TRANSPORTING MBLNR MESSAGE.
      ENDIF.
    ENDIF.

  ENDLOOP.

ENDFORM. " FRM_CALL_BAPI_FOR_CREATE

    EXPORTING
      GOODSMVT_HEADER       = I_HEADER
      GOODSMVT_CODE         = I_GOODSMVT_CODE
      TESTRUN               = P_CK
