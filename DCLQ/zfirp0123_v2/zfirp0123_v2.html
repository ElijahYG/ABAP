FORM FM_GET_FAGA.
  FIELD-SYMBOLS <FS> TYPE TY_OUT_TAB.
  DATA:
    BEGIN OF LW_FAGA,
      BUKRS    TYPE BSEG-BUKRS,
      BELNR    TYPE BSEG-BELNR,
      GJAHR    TYPE BSEG-GJAHR,
      BUZEI    TYPE FAGLFLEXA-DOCLN,

      RCNTR    TYPE FAGLFLEXA-RCNTR,
      PRCTR    TYPE BSEG-PRCTR,
      SAKNR    TYPE BSEG-SAKNR,
      DMBTR    TYPE FAGLFLEXA-HSL,
    END OF LW_FAGA,
    LT_OUT_TAB LIKE GT_OUT_TAB,
    LT_FAGA    LIKE LW_FAGA OCCURS 0.
* 取符合条件的bkpf数据
  LOOP AT GT_OUT_TAB ASSIGNING <FS>.
    <FS>-DOCLN = '000' && <FS>-BUZEI.
  ENDLOOP.
  LT_OUT_TAB   = GT_OUT_TAB.

  SORT LT_OUT_TAB BY BUKRS BELNR GJAHR DOCLN.
  DELETE ADJACENT DUPLICATES FROM LT_OUT_TAB COMPARING BUKRS BELNR GJAHR DOCLN.
  IF LT_OUT_TAB IS NOT INITIAL.
    SELECT RBUKRS
           DOCNR
           RYEAR
           DOCLN

           RCNTR
           PRCTR
           RACCT
           HSL
      FROM FAGLFLEXA
      INTO TABLE LT_FAGA
       FOR ALL ENTRIES IN LT_OUT_TAB
     WHERE RBUKRS = LT_OUT_TAB-BUKRS
       AND DOCNR  = LT_OUT_TAB-BELNR
       AND RYEAR  = LT_OUT_TAB-GJAHR
       AND DOCLN  = LT_OUT_TAB-DOCLN
      .
  ENDIF.
* 将数据填充到OUT表
  SORT LT_FAGA BY BUKRS BELNR GJAHR BUZEI.
  LOOP AT GT_OUT_TAB ASSIGNING <FS>.
    CLEAR LW_FAGA.
    READ TABLE LT_FAGA INTO LW_FAGA
                   WITH KEY BUKRS = <FS>-BUKRS
                            BELNR = <FS>-BELNR
                            GJAHR = <FS>-GJAHR
                            BUZEI = <FS>-DOCLN
                     BINARY SEARCH.
    IF SY-SUBRC = 0.
      <FS>-PRCTR  = LW_FAGA-PRCTR.
      <FS>-KOSTL  = LW_FAGA-RCNTR.
      <FS>-SAKNR  = LW_FAGA-SAKNR.
      <FS>-DMBTR  = ABS( LW_FAGA-DMBTR ).
    ENDIF.
  ENDLOOP.
ENDFORM.

FORM FM_CAL_DEL_DOC.
  DATA:
    BEGIN OF LW_BKPF,
      BUKRS        TYPE BKPF-BUKRS,
      BELNR        TYPE BKPF-BELNR,
      GJAHR        TYPE BKPF-GJAHR,
      MONAT        TYPE BKPF-MONAT,
      STBLG        TYPE BKPF-STBLG,
      STJAH        TYPE BKPF-STJAH,
    END OF LW_BKPF,
    LT_DOC      TYPE ZFIRP0121_DOC OCCURS 0,
    LT_OUT_TAB  LIKE GT_OUT_TAB,
    LW_OUT_TAB  LIKE LINE OF GT_OUT_TAB,
    LT_BKPF_CX  LIKE LW_BKPF OCCURS 0,
    LW_BKPF_CX  LIKE LW_BKPF ,
    LT_TMP      LIKE LW_BKPF OCCURS 0,
    LT_BKPF     LIKE LW_BKPF OCCURS 0.
* 删除重复凭证
  LT_OUT_TAB = GT_OUT_TAB.
  SORT LT_OUT_TAB BY BUKRS BELNR GJAHR .
  DELETE ADJACENT DUPLICATES FROM LT_OUT_TAB COMPARING BUKRS BELNR GJAHR.
* 取凭证抬头
  IF LT_OUT_TAB IS NOT INITIAL.
    SELECT BUKRS
           BELNR
           GJAHR
           MONAT
           STBLG
           STJAH
      FROM BKPF
      INTO TABLE LT_BKPF
       FOR ALL ENTRIES IN LT_OUT_TAB
     WHERE BUKRS = LT_OUT_TAB-BUKRS
       AND BELNR = LT_OUT_TAB-BELNR
       AND GJAHR = LT_OUT_TAB-GJAHR
       AND STBLG <> ''
      .
  ENDIF.
* 取冲销凭证
  IF LT_BKPF IS NOT INITIAL.
    SELECT BUKRS
           BELNR
           GJAHR
           MONAT
           STBLG
           STJAH
      FROM BKPF
      INTO TABLE LT_BKPF_CX
       FOR ALL ENTRIES IN LT_BKPF
     WHERE BUKRS = LT_BKPF-BUKRS
       AND BELNR = LT_BKPF-STBLG
       AND GJAHR = LT_BKPF-STJAH

      .
  ENDIF.
* 确定冲销凭证:LT_TMP
  SORT LT_BKPF_CX BY BUKRS BELNR GJAHR .
  LOOP AT LT_BKPF INTO LW_BKPF.
    CLEAR LW_BKPF_CX.
    READ TABLE LT_BKPF_CX INTO LW_BKPF_CX
                     WITH KEY BUKRS = LW_BKPF-BUKRS
                              BELNR = LW_BKPF-STBLG
                              GJAHR = LW_BKPF-STJAH
                       BINARY SEARCH.
    CHECK SY-SUBRC = 0.
    IF LW_BKPF-MONAT BETWEEN 12 AND 16.
      LW_BKPF-MONAT = 12.
    ENDIF.
    IF LW_BKPF_CX-MONAT BETWEEN 12 AND 16.
      LW_BKPF_CX-MONAT = 12.
    ENDIF.
*   冲销和被冲销凭证期间相同，则不需要归档
    IF LW_BKPF-MONAT = LW_BKPF_CX-MONAT.
      APPEND LW_BKPF TO LT_TMP.
    ENDIF.
  ENDLOOP.
* 确定已冲销的归档凭证
  CLEAR LW_OUT_TAB.
  CLEAR LT_OUT_TAB.
  IF LT_TMP IS NOT INITIAL.
*   确定已冲销凭证
    SORT GT_OUT_TAB BY BUKRS BELNR GJAHR.
    LOOP AT LT_TMP INTO LW_BKPF.
      DO .
        READ TABLE GT_OUT_TAB INTO LW_OUT_TAB
                         WITH KEY BUKRS = LW_BKPF-BUKRS
                                  BELNR = LW_BKPF-BELNR
                                  GJAHR = LW_BKPF-GJAHR
                         BINARY SEARCH .
        IF SY-SUBRC = 0.
          DELETE GT_OUT_TAB INDEX SY-TABIX.
          APPEND LW_OUT_TAB TO LT_OUT_TAB.
        ELSE.
          EXIT.
        ENDIF.
      ENDDO.
    ENDLOOP.
  ENDIF.

  GT_OUT_TAB = LT_OUT_TAB.
ENDFORM.

  FIELD-SYMBOLS <FS_OUT_TAB> TYPE TY_OUT_TAB.
  DATA:
    LT_OUT_TAB TYPE TYT_OUT_TAB,
    LW_OUT_TAB TYPE TY_OUT_TAB,

      SAKNR  TYPE SKAT-SAKNR,
      TXT20  TYPE SKAT-TXT20,
    END OF LW_SKAT,
    LT_SKAT LIKE LW_SKAT OCCURS 0.

* 取描述
  LT_OUT_TAB = GT_OUT_TAB.
  SORT LT_OUT_TAB BY SAKNR.
  DELETE ADJACENT DUPLICATES FROM LT_OUT_TAB COMPARING SAKNR.
  IF LT_OUT_TAB IS NOT INITIAL.
    SELECT SAKNR
           TXT20
      INTO TABLE LT_SKAT
      FROM SKAT
       FOR ALL ENTRIES IN LT_OUT_TAB
     WHERE SAKNR = LT_OUT_TAB-SAKNR
       AND KTOPL = 'CBAI'
       AND SPRAS = SY-LANGU.
  ENDIF.
* 将描述追加到输出表中
  SORT LT_SKAT BY SAKNR.
  LOOP AT GT_OUT_TAB ASSIGNING <FS_OUT_TAB>.
    CLEAR LW_SKAT.
    READ TABLE LT_SKAT INTO LW_SKAT
                   WITH KEY SAKNR = <FS_OUT_TAB>-SAKNR
                     BINARY SEARCH .
    <FS_OUT_TAB>-TXT20 = LW_SKAT-TXT20.
  ENDLOOP.
ENDFORM.
