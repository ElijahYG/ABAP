FORM FM_CHECK_DATA.
* 制表日期检查
  IF S_DATE[] IS INITIAL.
    SET CURSOR FIELD 'S_DATE-LOW' .
    MESSAGE E000(ZCB) WITH '制表日期为必填项' DISPLAY LIKE 'E'.
  ENDIF.

  IF S_DATUM-LOW IS INITIAL.
    SET CURSOR FIELD 'S_DATUM-LOW' .
    MESSAGE E000(ZCB) WITH '日期(起)为必填项' DISPLAY LIKE 'E'.
  ENDIF.

  IF S_DATUM-HIGH IS INITIAL.
    SET CURSOR FIELD 'S_DATUM-HIGH' .
    MESSAGE E000(ZCB) WITH '日期(止)为必填项' DISPLAY LIKE 'E'.
  ENDIF.
* 商品大类检查
  IF S_EXTWG[] IS INITIAL.
    SET CURSOR FIELD 'S_EXTWG-LOW' .
    MESSAGE E000(ZCB) WITH '商品大类为必填项' DISPLAY LIKE 'E'.
  ENDIF.

    MESSAGE E000(ZCB) WITH '地点为必填项' DISPLAY LIKE 'E'.
  ENDIF.

  ENDIF.

FORM FM_CAL_OUT_TAB.
  DATA:
    L_TABIX       TYPE I,
    L_LINE        TYPE I,
    LW_NEXT       TYPE TY_MCHB,
    L_XKCKZ       TYPE P LENGTH 10 DECIMALS 3,
    L_END         TYPE CHAR1,
    LT_OUT_TAB    TYPE TYT_OUT_TAB,
    LW_OUT_TAB    TYPE TY_OUT_TAB,
    LW_MCHB       TYPE TY_MCHB.

* step1：将现库存克重追加到输出表
  SORT GT_MCHB BY EXTWG CZ EX_PZ01.
  L_LINE = LINES( GT_MCHB ).
  LOOP AT GT_MCHB INTO LW_MCHB.
    L_TABIX   = SY-TABIX.
    CLEAR L_END.

    IF L_TABIX = L_LINE.
      L_END = 'X'.
    ELSE.
      L_TABIX = L_TABIX + 1.
      READ TABLE GT_MCHB INTO LW_NEXT INDEX L_TABIX.
      IF LW_NEXT-EXTWG <> LW_MCHB-EXTWG
      OR LW_NEXT-CZ    <> LW_MCHB-CZ
      OR LW_NEXT-EX_PZ01 <> LW_MCHB-EX_PZ01.
        L_END = 'X'.
      ENDIF.
    ENDIF.

    L_XKCKZ  = L_XKCKZ + LW_MCHB-XKCZL.

*   如果是最后一条
    IF L_END = 'X'.
      CLEAR LW_OUT_TAB.
      LW_OUT_TAB-XKCKZ    = L_XKCKZ.         "现库存克重
      LW_OUT_TAB-EXTWG    = LW_MCHB-EXTWG.   "商品大类
      LW_OUT_TAB-CZ       = LW_MCHB-CZ.      "贵金属材质
      LW_OUT_TAB-EX_PZ01  = LW_MCHB-EX_PZ01. "品种一级
      APPEND LW_OUT_TAB TO LT_OUT_TAB.
      CLEAR L_XKCKZ.
    ENDIF.
  ENDLOOP.

* LT_OUT_TAB 和 GT_OUT_TAB 合并
  FIELD-SYMBOLS <FS_OUT_TAB>  TYPE TY_OUT_TAB.
  DATA:
    LT_OUT_TMP TYPE TYT_OUT_TAB.
  SORT GT_OUT_TAB BY EXTWG CZ EX_PZ01.
  LOOP AT LT_OUT_TAB INTO LW_OUT_TAB.

    READ TABLE GT_OUT_TAB ASSIGNING <FS_OUT_TAB>
                           WITH KEY EXTWG   = LW_OUT_TAB-EXTWG
                                    CZ      = LW_OUT_TAB-CZ
                                    EX_PZ01 = LW_OUT_TAB-EX_PZ01
                             BINARY SEARCH.
    IF SY-SUBRC = 0.
      <FS_OUT_TAB>-XKCKZ = <FS_OUT_TAB>-XKCKZ + LW_OUT_TAB-XKCKZ.
    ELSE.
      APPEND LW_OUT_TAB TO LT_OUT_TMP.
    ENDIF.
  ENDLOOP.

  APPEND LINES OF LT_OUT_TMP TO GT_OUT_TAB.


* STEP2:未到货克重增加到OUT表
  DATA:
    LW_MARA        TYPE TY_MARA,
    L_WDHKZ        TYPE P LENGTH 10 DECIMALS 3,
    LW_MARA_NEXT   TYPE TY_MARA.
  CLEAR LT_OUT_TAB.
  SORT GT_MARA BY EXTWG EX_GJS EX_PZ01.
  L_LINE = LINES( GT_MARA ).
  LOOP AT GT_MARA INTO LW_MARA.
    L_TABIX   = SY-TABIX.
    CLEAR L_END.

    IF L_TABIX = L_LINE.
      L_END = 'X'.
    ELSE.
      L_TABIX = L_TABIX + 1.
      READ TABLE GT_MARA INTO LW_MARA_NEXT INDEX L_TABIX.
      IF LW_MARA_NEXT-EXTWG   <> LW_MARA-EXTWG
      OR LW_MARA_NEXT-EX_GJS  <> LW_MARA-EX_GJS
      OR LW_MARA_NEXT-EX_PZ01 <> LW_MARA-EX_PZ01.
        L_END = 'X'.
      ENDIF.
    ENDIF.

    L_WDHKZ  = L_WDHKZ + LW_MARA-WDHKZ.

*   最后一条
    IF L_END = 'X'.
      CLEAR LW_OUT_TAB.
      LW_OUT_TAB-WDHKZ    = L_WDHKZ.         "未到货克重
      LW_OUT_TAB-EXTWG    = LW_MCHB-EXTWG.   "商品大类
      LW_OUT_TAB-CZ       = LW_MCHB-CZ.      "贵金属材质
      LW_OUT_TAB-EX_PZ01  = LW_MCHB-EX_PZ01. "品种一级
      APPEND LW_OUT_TAB TO LT_OUT_TAB.
      CLEAR L_WDHKZ.
    ENDIF.
  ENDLOOP.
* LT_OUT_TAB 和 GT_OUT_TAB 合并
  SORT GT_OUT_TAB BY EXTWG CZ EX_PZ01.
  CLEAR LT_OUT_TMP.
  LOOP AT LT_OUT_TAB INTO LW_OUT_TAB.

    READ TABLE GT_OUT_TAB ASSIGNING <FS_OUT_TAB>
                           WITH KEY EXTWG   = LW_OUT_TAB-EXTWG
                                    CZ      = LW_OUT_TAB-CZ
                                    EX_PZ01 = LW_OUT_TAB-EX_PZ01
                             BINARY SEARCH.
    IF SY-SUBRC = 0.
      <FS_OUT_TAB>-WDHKZ = <FS_OUT_TAB>-WDHKZ + LW_OUT_TAB-WDHKZ.
    ELSE.
      APPEND LW_OUT_TAB TO LT_OUT_TMP.
    ENDIF.
  ENDLOOP.

  APPEND LINES OF LT_OUT_TMP TO GT_OUT_TAB.
  DATA:
    L_DATE TYPE I.
* STEP3:编辑所有字段
  LOOP AT GT_OUT_TAB ASSIGNING <FS_OUT_TAB>.
    <FS_OUT_TAB>-DATE     = S_DATE-LOW.      "制表日期
    <FS_OUT_TAB>-DATUM_L  = S_DATUM-LOW.     "采购开始日期
    <FS_OUT_TAB>-DATUM_H  = S_DATUM-HIGH.    "采购结束日期
    <FS_OUT_TAB>-XKCKZ    = <FS_OUT_TAB>-XKCKZ / 1000.  "现库存克重
    <FS_OUT_TAB>-WDHKZ    = <FS_OUT_TAB>-WDHKZ / 1000.  "未到货克重
    <FS_OUT_TAB>-HJKCKZ   = <FS_OUT_TAB>-XKCKZ + <FS_OUT_TAB>-WDHKZ. "合计库存克重
    L_DATE = S_DAT2-HIGH - S_DAT2-LOW + 1.
    <FS_OUT_TAB>-HBRJXSKZ = <FS_OUT_TAB>-HBRJXSKZ / 1000 / L_DATE. "环比日均销售克重

    <FS_OUT_TAB>-XSZQ     = S_DATUM-HIGH - S_DATE-LOW + 1."销售周期
    <FS_OUT_TAB>-YJXSKZ   = <FS_OUT_TAB>-HBRJXSKZ * <FS_OUT_TAB>-XSZQ. "预计销售克重
    <FS_OUT_TAB>-YJSYKCKZ = <FS_OUT_TAB>-HJKCKZ - <FS_OUT_TAB>-YJXSKZ.

    <FS_OUT_TAB>-TJXS     = S_XS-LOW.        "调节系数
    <FS_OUT_TAB>-YJJHKZ   = <FS_OUT_TAB>-YJSYKCKZ + <FS_OUT_TAB>-TJXS.
    <FS_OUT_TAB>-TQJHKZ   = <FS_OUT_TAB>-TQJHKZ / 1000.
    <FS_OUT_TAB>-HBJHKZ   = <FS_OUT_TAB>-HBJHKZ / 1000.
  ENDLOOP.

ENDFORM.
*&---------------------------------------------------------------------*
*&取符合条件的商品
*&---------------------------------------------------------------------*
FORM FM_GET_MARA.
  CLEAR GT_MARA.
* 取商品主数据
  SELECT A~MATNR    "物料号
         A~EXTWG    "商品大类
         A~MEINS    "基本单位
         B~EX_PZ01  "品种一级
         B~EX_GJS   "贵金属材质
    INTO TABLE GT_MARA
    FROM MARA AS A
   INNER JOIN ZMMEX0010_MARA AS B
      ON A~MATNR = B~MATNR
   WHERE A~EXTWG   IN S_EXTWG
     AND B~EX_PZ01 IN S_PZ01
     AND B~EX_PZ02 IN S_PZ02
     AND B~EX_PZ03 IN S_PZ03
     AND A~MTART   IN ('Z002','Z009')
        .
ENDFORM.

*&---------------------------------------------------------------------*
FORM FM_GET_MCHB.
  CLEAR GT_MCHB.
* 取符合条件的批次
  IF GT_MARA IS NOT INITIAL.
    SELECT MATNR
           WERKS
           LGORT
           CHARG

           CLABS
      FROM MCHB
      INTO TABLE GT_MCHB
       FOR ALL ENTRIES IN GT_MARA
