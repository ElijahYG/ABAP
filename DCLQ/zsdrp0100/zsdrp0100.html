FORM FM_CONTROLL_SCREEN.
*  DATA:
*    L_FKBER   TYPE I,
*    L_RCNTR   TYPE I,
*    L_SNAME   TYPE I.
** 修改显示方式
*  LOOP AT SCREEN.
*    L_RCNTR   = 1.
*    L_SNAME   = 1.
**   按成本中心
*    IF P_1 IS NOT INITIAL.
*      L_RCNTR   = 1.
*      L_SNAME   = 0.
*      L_FKBER   = 1.
*    ENDIF.
**   按成本中心组
*    IF P_2 IS NOT INITIAL.
*      L_RCNTR   = 0.
*      L_SNAME   = 1.
*      L_FKBER   = 0.
*    ENDIF.
*
**   激活
*    IF SCREEN-NAME CP '*S_RCNTR*'.
*      SCREEN-ACTIVE = L_RCNTR.
*    ENDIF.
*    IF SCREEN-NAME CP '*S_SNAME*'.
*      SCREEN-ACTIVE = L_SNAME.
*    ENDIF.
*    IF SCREEN-NAME CP '*S_FKBER*'.
*      SCREEN-ACTIVE = L_FKBER.
*    ENDIF.
*    MODIFY SCREEN.
*  ENDLOOP.
ENDFORM.

FORM FM_GET_DATA.
  DATA:
    BEGIN OF LW_EDIDC,
      DOCNUM    TYPE EDI_DOCNUM,
    END OF LW_EDIDC,
    BEGIN OF LW_EDIDS,
      DOCNUM    TYPE EDI_DOCNUM,
      STAPA1    TYPE EDIDS-STAPA1,
      STAMID    TYPE EDIDS-STAMID,
      STAMNO    TYPE EDIDS-STAMNO,
    END OF LW_EDIDS,
    BEGIN OF LW_EDID4,
      DOCNUM    TYPE EDI_DOCNUM,
      SEGNAM    TYPE EDID4-SEGNAM,
      SDATA     TYPE EDID4-SDATA,
    END OF LW_EDID4,
    LW_OUT_TAB  TYPE TY_OUT_TAB,
    LW_SEG_HEAD TYPE ZSDIF0101_SGM_HEAD,
    LW_SEG_ITEM TYPE ZSDIF0101_SGM_ITEM,
    BEGIN OF LW_HEAD,
      DOCNUM    TYPE EDI_DOCNUM.
      include structure ZSDIF0101_SGM_HEAD.
  DATA:
    END OF LW_HEAD,
    BEGIN OF LW_ITEM,
      DOCNUM    TYPE EDI_DOCNUM.
      include structure ZSDIF0101_SGM_ITEM.
  DATA:
    END OF LW_ITEM,
    BEGIN OF LW_MKPF,
      MBLNR   TYPE MKPF-MBLNR,
      BKTXT   TYPE MKPF-BKTXT,
    END OF LW_MKPF,
    BEGIN OF LW_MSEG,
      MBLNR   TYPE MSEG-MBLNR,
      ZEILE   TYPE MSEG-ZEILE,
      MATNR   TYPE MSEG-MATNR,
      CHARG   TYPE MSEG-CHARG,
      DMBTR   TYPE MSEG-DMBTR,
    END OF LW_MSEG,
    L_MENGE_I           TYPE MENGE_D,
    L_MENGE             TYPE MENGE_D,
    LW_CHARG            TYPE ZMM_CHARG,
    LT_CHARG            TYPE ZMM_CHARG OCCURS 0,
    LT_CHARACTERS       TYPE ZMM_CHARACTERS OCCURS 0,
    LW_CHARACTERS       TYPE ZMM_CHARACTERS,
    LW_CHARG_CHARACTERS TYPE ZMM_CHARG_CHARACTERS,
    LT_CHARG_CHARACTERS TYPE ZMM_CHARG_CHARACTERS OCCURS 0,
    LW_ZSDRP0020_CAL    TYPE ZSDRP0020_CAL,
    LT_ZSDRP0020_CAL    TYPE ZSDRP0020_CAL OCCURS 0,
    L_MBLNR     TYPE MSEG-MBLNR,
    L_MJAHR     TYPE MSEG-MJAHR,
    LW_OBJECT   TYPE BORIDENT,
    LW_RS       TYPE NEIGHBOR,
    LT_RS       TYPE NEIGHBOR OCCURS 0,
    LT_MKPF_TMP LIKE LW_MKPF  OCCURS 0,
    LT_MKPF     LIKE LW_MKPF  OCCURS 0,
    LT_MSEG     LIKE LW_MSEG  OCCURS 0,
    LW_PRE      LIKE LW_ITEM,
    L_ZEILE     LIKE LW_OUT_TAB-ZEILE,
    LT_HEAD     LIKE LW_HEAD  OCCURS 0,
    LT_ITEM     LIKE LW_ITEM  OCCURS 0,
    LT_EDIDS    LIKE LW_EDIDS OCCURS 0,
    LT_EDID4    LIKE LW_EDID4 OCCURS 0,
    LT_EDIDC    LIKE LW_EDIDC OCCURS 0.

* 取指定时间内的IDOC
* 状态控制记录
  SELECT DOCNUM
    INTO TABLE LT_EDIDC
    FROM EDIDC
   WHERE CREDAT  IN S_DATE
     AND STATUS  = '53'
     AND MESTYP  = 'ZSDIF0101_MSG'
     AND IDOCTP  = 'ZSDIF0101_IDOC'
    .

  IF LT_EDIDC IS NOT INITIAL.
*   取状态记录
    SELECT DOCNUM
           STAPA1
           STAMID
           STAMNO
      INTO TABLE LT_EDIDS
      FROM EDIDS
       FOR ALL ENTRIES IN LT_EDIDC
     WHERE DOCNUM = LT_EDIDC-DOCNUM
      .
*   取数据记录
    SELECT *
      INTO CORRESPONDING FIELDS OF TABLE LT_EDID4
      FROM EDID4
       FOR ALL ENTRIES IN LT_EDIDC
     WHERE DOCNUM = LT_EDIDC-DOCNUM
      .
  ENDIF.

  LOOP AT LT_EDID4 INTO LW_EDID4.
    CASE LW_EDID4-SEGNAM.
      WHEN 'ZSDIF0101_SGM_HEAD'.
        LW_SEG_HEAD = LW_EDID4-SDATA.
        LW_HEAD-DOCNUM  = LW_EDID4-DOCNUM.
        MOVE-CORRESPONDING LW_SEG_HEAD TO LW_HEAD.
        APPEND LW_HEAD TO LT_HEAD.
      WHEN 'ZSDIF0101_SGM_ITEM'.
        LW_SEG_ITEM = LW_EDID4-SDATA.
        LW_ITEM-DOCNUM = LW_EDID4-DOCNUM.
        MOVE-CORRESPONDING LW_SEG_ITEM TO LW_ITEM.
        APPEND LW_ITEM TO LT_ITEM.
    ENDCASE.
  ENDLOOP.
** 取销售商品物料凭证
*  IF LT_HEAD IS NOT INITIAL.
**   赋值
*    LOOP AT LT_HEAD INTO LW_HEAD.
*      LW_MKPF-BKTXT = LW_HEAD-ZPOSN.
*      APPEND LW_MKPF TO LT_MKPF_TMP.
*    ENDLOOP.
**   取抬头凭证
*    SELECT MBLNR
*           BKTXT
*      FROM MKPF
*      INTO TABLE LT_MKPF
*       FOR ALL ENTRIES IN LT_MKPF_TMP
*     WHERE BKTXT = LT_MKPF_TMP-BKTXT
*      .
*    IF LT_MKPF IS NOT INITIAL.
*      SELECT MBLNR
*             ZEILE
*             MATNR
*             CHARG
*             DMBTR
*        FROM MSEG
*        INTO TABLE LT_MSEG
*         FOR ALL ENTRIES IN LT_MKPF
*       WHERE MBLNR = LT_MKPF-MBLNR
*        .
*    ENDIF.
*  ENDIF.
* 数据赋值
  SORT LT_HEAD BY DOCNUM.
  SORT LT_EDIDS BY DOCNUM STAMID STAMNO.
  SORT LT_MKPF BY BKTXT.
  SORT LT_MSEG BY MBLNR MATNR CHARG.
  LOOP AT LT_ITEM INTO LW_ITEM WHERE ZHXMLX = '1'.
    IF LW_ITEM-DOCNUM <> LW_PRE-DOCNUM.
      CLEAR L_ZEILE.
    ENDIF.
    LW_PRE = LW_ITEM.
    CLEAR LW_OUT_TAB.
    CLEAR LW_HEAD.
    READ TABLE LT_HEAD INTO LW_HEAD WITH KEY DOCNUM = LW_ITEM-DOCNUM
                                      BINARY SEARCH.
    CALL FUNCTION 'CONVERSION_EXIT_ALPHA_INPUT'
      EXPORTING
        INPUT         = LW_HEAD-WERKS
      IMPORTING
        OUTPUT        = LW_OUT_TAB-KUNNR"门店编码
              .
    LW_OUT_TAB-DOCNUM_1 = LW_ITEM-DOCNUM.
    LW_OUT_TAB-ZPOSN   = LW_HEAD-ZPOSN. "POS单号
    LW_OUT_TAB-ZPLX    = LW_HEAD-ZPLX.  "业务类型
    LW_OUT_TAB-ZDATE   = LW_HEAD-ZDATE. "销售日期
*   行项目号
    L_ZEILE  = L_ZEILE + 10.
    LW_OUT_TAB-ZEILE   = L_ZEILE.
*   取小票IDOC号
    CLEAR LW_EDIDS.
    READ TABLE LT_EDIDS INTO LW_EDIDS
                    WITH KEY DOCNUM = LW_ITEM-DOCNUM
                             STAMID = 'ZCB'
                             STAMNO = '018'
                      BINARY SEARCH.
    LW_OUT_TAB-DOCNUM = LW_EDIDS-STAPA1. "小票IDOC号
*   物料凭证号
    CLEAR LW_EDIDS.
    READ TABLE LT_EDIDS INTO LW_EDIDS
                    WITH KEY DOCNUM = LW_ITEM-DOCNUM
                             STAMID = 'ZCB'
                             STAMNO = '020'
                      BINARY SEARCH.
    LW_OUT_TAB-MBLNR = LW_EDIDS-STAPA1. "小票IDOC号
*   商品号
    CALL FUNCTION 'CONVERSION_EXIT_ALPHA_INPUT'
      EXPORTING
        INPUT         = LW_ITEM-ZSPKH
      IMPORTING
        OUTPUT        = LW_OUT_TAB-MATNR
              .

*   批次
    LW_OUT_TAB-CHARG = LW_ITEM-ZSPPC.
    LW_OUT_TAB-MENGE = LW_ITEM-ZSPJH.  "数量
    LW_OUT_TAB-MEINS = 'ST'.           "单位
    LW_OUT_TAB-ZXSJE = LW_ITEM-ZXSJE.  "销售金额
*   抵值金额 = （ 销售金额 / 销售总金额 ）*抵制总金额
    IF LW_HEAD-ZHJJE IS NOT INITIAL.
      LW_OUT_TAB-ZDZJE = ( LW_ITEM-ZXSJE / LW_HEAD-ZHJJE ) * LW_HEAD-ZHJJZ.
    ENDIF.
*   超重金额
    LW_OUT_TAB-ZCZJE = LW_OUT_TAB-ZXSJE - LW_OUT_TAB-ZDZJE.
*   实际成交金额 = ( 销售金额 / 合计销售金额) * 收款金额
    IF LW_HEAD-ZHJJE IS NOT INITIAL.
      LW_OUT_TAB-ZCJJE = ( LW_ITEM-ZXSJE / LW_HEAD-ZHJJE ) * LW_HEAD-ZHJJS.
    ENDIF.
*   商品成本
*    CLEAR LW_MKPF.
*    READ TABLE LT_MKPF INTO LW_MKPF
*                   WITH KEY BKTXT = LW_HEAD-ZPOSN
*                     BINARY SEARCH.
*    CLEAR LW_MSEG.
*    READ TABLE LT_MSEG INTO LW_MSEG
*                   WITH KEY MBLNR = LW_MKPF-MBLNR
*                            MATNR = LW_OUT_TAB-MATNR
*                            CHARG = LW_OUT_TAB-CHARG
*                     BINARY SEARCH.
*    LW_OUT_TAB-VERPR = LW_MSEG-DMBTR.
*   获取小票产生的商品凭证
    LW_OBJECT-OBJKEY  = LW_OUT_TAB-DOCNUM.
    LW_OBJECT-OBJTYPE = 'IDOC'.
    CLEAR:
      LT_MSEG,
      LT_RS.
    CALL FUNCTION 'SREL_GET_NEXT_NEIGHBORS'
      EXPORTING
        OBJECT               = LW_OBJECT
      TABLES
        NEIGHBORS            = LT_RS
      EXCEPTIONS
        INTERNAL_ERROR       = 1
        NO_LOGSYS            = 2
        OTHERS               = 3
              .
    READ TABLE LT_RS INTO LW_RS WITH KEY OBJTYPE = 'BUS2017'
                                        .
*   取成本
    IF SY-SUBRC = 0.
      L_MBLNR = LW_RS-OBJKEY+0(10).
      L_MJAHR = LW_RS-OBJKEY+10(4).
      SELECT MBLNR
             ZEILE
             MATNR
             CHARG
             DMBTR
        FROM MSEG
        INTO TABLE LT_MSEG
       WHERE MBLNR = L_MBLNR
         AND MJAHR = L_MJAHR
        .
    ENDIF.
    CLEAR LW_MSEG.
    READ TABLE LT_MSEG INTO LW_MSEG
                   WITH KEY
                            MATNR = LW_OUT_TAB-MATNR
                            CHARG = LW_OUT_TAB-CHARG
                          .
    LW_OUT_TAB-VERPR = LW_MSEG-DMBTR.
*   联营商品成本
    CLEAR:
      LW_CHARG,
      LW_CHARACTERS,
      LT_CHARG,
      LT_CHARACTERS.
    LW_CHARG-MATNR    = LW_OUT_TAB-MATNR.
    LW_CHARG-CHARG    = LW_OUT_TAB-CHARG.
    APPEND LW_CHARG TO LT_CHARG.
    call function 'ZSDRP0020_FM_01'
      TABLES
        T_CHARG            = LT_CHARG
        T_CHARACTERS       = LT_CHARACTERS
              .
    READ TABLE LT_CHARACTERS INTO LW_CHARACTERS
                         WITH KEY MATNR = LW_OUT_TAB-MATNR
                                  CHARG = LW_OUT_TAB-CHARG.
    IF LW_CHARACTERS-SFLY = 'X'.
      CLEAR:
        LT_ZSDRP0020_CAL,
        LW_ZSDRP0020_CAL.
      LW_ZSDRP0020_CAL-MATNR = LW_OUT_TAB-MATNR .      "物料
      LW_ZSDRP0020_CAL-CHARG = LW_OUT_TAB-CHARG.      "批次
      LW_ZSDRP0020_CAL-WERKS = LW_HEAD-WERKS . "地点
      LW_ZSDRP0020_CAL-FKDAT = LW_HEAD-ZDATE . "日期
      L_MENGE_I = LW_ITEM-ZSPJH.
      call function 'ZSD_GET_MENGE'
        EXPORTING
          I_MATNR       = LW_OUT_TAB-MATNR
          I_CHARG       = LW_OUT_TAB-CHARG
          I_MENGE       = L_MENGE_I
        IMPORTING
          E_MENGE       = L_MENGE
                .

      LW_ZSDRP0020_CAL-FKIMG     = L_MENGE.   "数量（sap基本单位的数量）
      IF LW_HEAD-ZPLX = '20'.
        LW_ZSDRP0020_CAL-SHKZG   = 'X'.       "销售为空， 退货为X （业务类型=20)
      ENDIF.
      LW_ZSDRP0020_CAL-ZCKLSJ_IN    = LW_HEAD-ZHJJE." 参考零售价（金额，批次特性上参考零售价 ）

      APPEND LW_ZSDRP0020_CAL TO LT_ZSDRP0020_CAL .
      call function 'ZSDRP0020_FM'
        TABLES
          CT_TABLE       = LT_ZSDRP0020_CAL  .
      READ TABLE LT_ZSDRP0020_CAL INTO LW_ZSDRP0020_CAL INDEX 1 .
      IF SY-SUBRC = 0 .
        LW_OUT_TAB-VERPR = ABS( LW_ZSDRP0020_CAL-ZSAL_COST ).
      ENDIF.
    ENDIF.
