FORM FM_CAL_DATA.
  FIELD-SYMBOLS <FS>.

  DATA:
    BEGIN OF LW_SUM,
      RACCT  TYPE   FAGLFLEXT-RACCT, "科目
      DRCRK  TYPE   FAGLFLEXT-DRCRK, "借贷标识
      VALUE0 TYPE   FAGLFLEXT-HSL01, "期初余额
      VALUE1 TYPE   FAGLFLEXT-HSL01, "当年汇总值

    END OF LW_SUM,
    L_INDEX    TYPE NUMC2,
    L_MAX      TYPE I,
    LW_OUT_TAB TYPE TY_OUT_TAB,
    LW_OUT_TMP TYPE TY_OUT_TAB,
    LT_SUM     LIKE LW_SUM  OCCURS 0,
    LW_FAG     TYPE TY_FAG,
    LW_TMP     TYPE TY_FAG,
    LW_FSUM    TYPE TY_FAG,
    L_FIELD    TYPE CHAR100.

* STEP1：按科目、借贷标识汇总数据
  SORT GT_FAG BY RACCT DRCRK.
* 确定合计到哪个期间
  IF S_MONAT-HIGH IS NOT INITIAL.
    L_MAX = S_MONAT-HIGH.
  ELSE.
    L_MAX = S_MONAT-LOW.
  ENDIF.

  CLEAR LW_FSUM.
  LOOP AT GT_FAG INTO LW_TMP.
*   防止*
    LW_FAG = LW_TMP.
*   按科目、借贷标识、期间汇总
    LW_FSUM-HSLVT = LW_FSUM-HSLVT + LW_FAG-HSLVT.
    LW_FSUM-HSL01 = LW_FSUM-HSL01 + LW_FAG-HSL01.
    LW_FSUM-HSL02 = LW_FSUM-HSL02 + LW_FAG-HSL02.
    LW_FSUM-HSL03 = LW_FSUM-HSL03 + LW_FAG-HSL03.
    LW_FSUM-HSL04 = LW_FSUM-HSL04 + LW_FAG-HSL04.
    LW_FSUM-HSL05 = LW_FSUM-HSL05 + LW_FAG-HSL05.
    LW_FSUM-HSL06 = LW_FSUM-HSL06 + LW_FAG-HSL06.
    LW_FSUM-HSL07 = LW_FSUM-HSL07 + LW_FAG-HSL07.
    LW_FSUM-HSL08 = LW_FSUM-HSL08 + LW_FAG-HSL08.
    LW_FSUM-HSL09 = LW_FSUM-HSL09 + LW_FAG-HSL09.
    LW_FSUM-HSL10 = LW_FSUM-HSL10 + LW_FAG-HSL10.
    LW_FSUM-HSL11 = LW_FSUM-HSL11 + LW_FAG-HSL11.
    LW_FSUM-HSL12 = LW_FSUM-HSL12 + LW_FAG-HSL12 + LW_FAG-HSL13 + LW_FAG-HSL14.
    AT END OF DRCRK.
*
      CLEAR LW_SUM.
      DO 12 TIMES.
        L_INDEX = SY-INDEX.
        L_FIELD = 'LW_FSUM-HSL' && L_INDEX.
        ASSIGN (L_FIELD) TO <FS>.

*       计算当年累计值
        IF L_INDEX <= L_MAX.
          LW_SUM-VALUE1 = LW_SUM-VALUE1 + <FS>.
        ENDIF.
      ENDDO.
*     期初
      LW_SUM-VALUE0 = LW_FSUM-HSLVT.
*     计算当月和当年的发生额
      LW_SUM-RACCT = LW_FAG-RACCT.  "科目
      LW_SUM-DRCRK = LW_FAG-DRCRK.  "借贷标识
      APPEND LW_SUM TO LT_SUM.

*     处理完毕，清空工作区
      CLEAR LW_FSUM.
    ENDAT.
  ENDLOOP.

* STEP2：数据匹配到输出内表中
  SORT LT_SUM BY RACCT.
  LOOP AT LT_SUM INTO LW_SUM.
    CASE LW_SUM-RACCT.

      WHEN '4104001004'.
        IF LW_SUM-DRCRK = 'H'.
          CLEAR LW_OUT_TAB.
          LW_OUT_TAB-LINENO = 1.                   "行号
          LW_OUT_TAB-VALUE9 = LW_SUM-VALUE0 * -1.
          APPEND LW_OUT_TAB TO GT_OUT_TAB.
        ENDIF.

      WHEN '4101001001'.
        IF LW_SUM-DRCRK = 'H'.
          CLEAR LW_OUT_TAB.
          LW_OUT_TAB-LINENO = 18.                   "行号
          LW_OUT_TAB-VALUE7 = LW_SUM-VALUE1 * -1.
          LW_OUT_TAB-VALUE9 = LW_SUM-VALUE1.
          APPEND LW_OUT_TAB TO GT_OUT_TAB.
        ENDIF.

      WHEN '4101002001'.
        IF LW_SUM-DRCRK = 'H'.
          CLEAR LW_OUT_TAB.
          LW_OUT_TAB-LINENO = 19.                   "行号
          LW_OUT_TAB-VALUE7 = LW_SUM-VALUE1 * -1.
          LW_OUT_TAB-VALUE9 = LW_SUM-VALUE1.
          APPEND LW_OUT_TAB TO GT_OUT_TAB.
        ENDIF.

      WHEN '4104001003'.
        IF LW_SUM-DRCRK = 'S'.
          CLEAR LW_OUT_TAB.
          LW_OUT_TAB-LINENO = 24.                   "行号
          LW_OUT_TAB-VALUE9 = LW_SUM-VALUE1 * -1.
          APPEND LW_OUT_TAB TO GT_OUT_TAB.
        ENDIF.


    ENDCASE.
  ENDLOOP.

*取第七行
  CLEAR LW_OUT_TAB..
  LOOP AT LT_SUM INTO LW_SUM WHERE RACCT+0(2) BETWEEN 60 AND 68.
    LW_OUT_TAB-LINENO = 7.
    LW_OUT_TAB-VALUE9 = LW_OUT_TAB-VALUE9 + LW_SUM-VALUE1 * -1.
  ENDLOOP.

*取第一行
  CLEAR LW_OUT_TAB.
  LOOP AT LT_SUM INTO LW_SUM WHERE RACCT+0(4) BETWEEN 4001 AND 4001.
    LW_OUT_TAB-LINENO = 1.
    LW_OUT_TAB-VALUE1 = LW_OUT_TAB-VALUE1 + LW_SUM-VALUE0 * -1.
  ENDLOOP.

  APPEND LW_OUT_TAB TO GT_OUT_TAB.

*取第九行
  CLEAR LW_OUT_TAB.
  LOOP AT LT_SUM INTO LW_SUM WHERE RACCT+0(4) BETWEEN 4001 AND 4001.
    IF LW_SUM-DRCRK = 'H'.
      LW_OUT_TAB-LINENO = 9.
      LW_OUT_TAB-VALUE1 = LW_OUT_TAB-VALUE1 + LW_SUM-VALUE1 * -1.
    ENDIF.
  ENDLOOP.
  APPEND LW_OUT_TAB TO GT_OUT_TAB.


*取第一行


  CLEAR LW_OUT_TAB.
  LOOP AT LT_SUM INTO LW_SUM WHERE RACCT+0(4) BETWEEN 4002 AND 4002.
    LW_OUT_TAB-LINENO = 1.
    LW_OUT_TAB-VALUE3 = LW_OUT_TAB-VALUE3 + LW_SUM-VALUE0 * -1.
  ENDLOOP.
  APPEND LW_OUT_TAB TO GT_OUT_TAB.


*取第一行
  CLEAR LW_OUT_TAB.
  LOOP AT LT_SUM INTO LW_SUM WHERE RACCT+0(4) BETWEEN 4101 AND 4101.
    LW_OUT_TAB-LINENO = 1.
    LW_OUT_TAB-VALUE7 = LW_OUT_TAB-VALUE7  + LW_SUM-VALUE0 * -1.
  ENDLOOP.
  APPEND LW_OUT_TAB TO GT_OUT_TAB.


*取第17行
  CLEAR LW_OUT_TAB.
  LOOP AT LT_SUM INTO LW_SUM WHERE RACCT = '4101001001' OR RACCT = '4101002001'.
    IF LW_SUM-DRCRK = 'H'.
      LW_OUT_TAB-LINENO = 17.
      LW_OUT_TAB-VALUE7 = LW_OUT_TAB-VALUE7 + LW_SUM-VALUE1 * -1.
      LW_OUT_TAB-VALUE9 = LW_OUT_TAB-VALUE9 + LW_SUM-VALUE1.
    ENDIF.
  ENDLOOP.

  CLEAR LW_OUT_TAB.
  LOOP AT GT_OUT_TAB INTO LW_OUT_TMP.
    CASE LW_OUT_TMP-LINENO.
      WHEN 1.
        LW_OUT_TAB-VALUE1 = LW_OUT_TAB-VALUE1 + LW_OUT_TMP-VALUE1.

        LW_OUT_TAB-VALUE9 = LW_OUT_TAB-VALUE8 + LW_OUT_TMP-VALUE9.
        LW_OUT_TAB-VALUE12 = LW_OUT_TAB-VALUE12 + LW_OUT_TMP-VALUE12.

        LW_OUT_TAB-VALUE1 = LW_OUT_TAB-VALUE1 + LW_OUT_TMP-VALUE1.
        LW_OUT_TAB-VALUE3 = LW_OUT_TAB-VALUE3 + LW_OUT_TMP-VALUE3.
        LW_OUT_TAB-VALUE7 = LW_OUT_TAB-VALUE7 + LW_OUT_TMP-VALUE7.
        LW_OUT_TAB-VALUE9 = LW_OUT_TAB-VALUE8 + LW_OUT_TMP-VALUE9.
        LW_OUT_TAB-VALUE12 = LW_OUT_TAB-VALUE12 + LW_OUT_TMP-VALUE12.
      WHEN 3.
        LW_OUT_TAB-VALUE1 = LW_OUT_TAB-VALUE1 + LW_OUT_TMP-VALUE1.
