FORM FM_CHECK_VERKF.
  DATA:
    L_USER   TYPE SY-MSGV1.

  CLEAR:
    GW_EKKO,
    GT_EKPO,
    G_BELNR.
* 取采购抬头
  SELECT SINGLE *
    INTO GW_EKKO
    FROM EKKO
   WHERE VERKF IN S_VERKF    "返库单号
    .
  IF SY-SUBRC <> 0.
    MESSAGE S000(ZCB) WITH '无此返库单，请与返库门店确认' DISPLAY LIKE 'E'.
    LEAVE LIST-PROCESSING.
  ENDIF.

* 判断订单类型
  IF GW_EKKO-BSART <> 'ZUBR'
 AND GW_EKKO-BSART <> 'ZSTR'
 AND GW_EKKO-BSART <> 'ZTR1'.
    MESSAGE S000(ZCB) WITH '返库单' GW_EKKO-VERKF '不属于ZUBR、ZTR1或ZSTR订单' DISPLAY LIKE 'E'.
    LEAVE LIST-PROCESSING.
  ENDIF.
* 判断是否有交货单
  SELECT SINGLE BELNR
    INTO G_BELNR
    FROM EKBE
   WHERE EBELN = GW_EKKO-EBELN
     AND VGABE = '8'      "8表示交货单
    .
  IF SY-SUBRC <> 0.
    MESSAGE S000(ZCB) WITH '返库单' GW_EKKO-VERKF '的交货单未创建' DISPLAY LIKE 'E'.
    LEAVE LIST-PROCESSING.
  ENDIF.
* 判断交货单是否已过账
  SELECT COUNT(*)
    FROM VBUK
   WHERE VBELN = G_BELNR
     AND WBSTK = 'C'
    .
  IF SY-SUBRC = 0.
    MESSAGE S000(ZCB) WITH '返库单' GW_EKKO-VERKF '已收货' DISPLAY LIKE 'E'.
    LEAVE LIST-PROCESSING.
  ENDIF.
* 判断是否收货
  SELECT COUNT(*)
    FROM EKBE
   WHERE EBELN = GW_EKKO-EBELN
     AND VGABE = '1'      "1表示收货
    .
  IF SY-SUBRC <> 0.
    MESSAGE S000(ZCB) WITH '返库单' GW_EKKO-VERKF '未发货' DISPLAY LIKE 'E'.
    LEAVE LIST-PROCESSING.
  ENDIF.
* 锁交货单
  CALL FUNCTION 'ENQUEUE_EVVBLKE'
    EXPORTING
      VBELN                = G_BELNR
    EXCEPTIONS
      FOREIGN_LOCK         = 1
      SYSTEM_FAILURE       = 2
      OTHERS               = 3
             .
  CASE SY-SUBRC .
    WHEN 0.
    WHEN 1.
      L_USER = SY-MSGV1.
      MESSAGE S046(VL) WITH G_BELNR L_USER DISPLAY LIKE 'E'.
      LEAVE LIST-PROCESSING.
    WHEN OTHERS.
      MESSAGE S000(ZCB) WITH '锁定交货单时' G_BELNR '出错' DISPLAY LIKE 'E'.
      LEAVE LIST-PROCESSING.
  ENDCASE.
* 取采购明细
  SELECT *
    FROM EKPO
    INTO TABLE GT_EKPO
   WHERE EBELN = GW_EKKO-EBELN
    .

ENDFORM.

FORM FM_INIT_SCREEN.
  DATA:
    BEGIN OF LW_MAKT,
      MATNR    TYPE MAKT-MATNR,
      MAKTX    TYPE MAKT-MAKTX,
    END OF LW_MAKT,
    BEGIN OF LW_MARA,
      MATNR    TYPE MARA-MATNR,
      EXTWG    TYPE MARA-EXTWG,
    END OF LW_MARA,

    LT_MARA      LIKE LW_MARA OCCURS 0,
    LT_LIPS      TYPE LIPS OCCURS 0,
    LW_LIPS      TYPE LIPS,
    LT_MAKT LIKE LW_MAKT OCCURS 0,
    LW_ITEM TYPE TY_ITEM,
    LT_EKET TYPE EKET OCCURS 0,
    LW_EKET TYPE EKET,
    LW_EKPO TYPE EKPO.
  ZMMRP0260_SCREEN-VERKF = GW_EKKO-VERKF.  "返库单号
  ZMMRP0260_SCREEN-ABSGR = GW_EKKO-ABSGR.  "返库类型
  ZMMRP0260_SCREEN-WERKST = GW_EKKO-RESWK. "收货门店
* ZMMRP0260_SCREEN-LGORTT = GW_EKKO-LGORT. "收货库存地
  READ TABLE GT_EKPO INTO LW_EKPO INDEX 1.
  ZMMRP0260_SCREEN-WERKS = LW_EKPO-WERKS.  "返库门店
  ZMMRP0260_SCREEN-BLDAT = SY-DATUM. "凭证日期
  ZMMRP0260_SCREEN-BUDAT = SY-DATUM. "记账日期
  ZMMRP0260_SCREEN-EBELN = GW_EKKO-EBELN. "STO单号
* 取移动原因描述
  CLEAR ZMMRP0260_SCREEN-ABSGR_TXT.
  IF ZMMRP0260_SCREEN-ABSGR IS NOT INITIAL.
    SELECT SINGLE ABSGR_TXT
      INTO ZMMRP0260_SCREEN-ABSGR_TXT
      FROM T165M
     WHERE ABSGR = ZMMRP0260_SCREEN-ABSGR
       AND SPRAS = SY-LANGU
     .
  ENDIF.
* 取批次信息
  SELECT *
    INTO TABLE LT_EKET
    FROM EKET
   WHERE EBELN = GW_EKKO-EBELN
    .
  SORT LT_EKET BY EBELN EBELP.
* 取物料信息
  IF GT_EKPO IS NOT INITIAL.
    SELECT MATNR
           MAKTX
      INTO TABLE LT_MAKT
      FROM MAKT
       FOR ALL ENTRIES IN GT_EKPO
     WHERE MATNR = GT_EKPO-MATNR
       AND SPRAS = SY-LANGU
      .
    SELECT MATNR
           EXTWG
      INTO TABLE LT_MARA
      FROM MARA
       FOR ALL ENTRIES IN GT_EKPO
     WHERE MATNR = GT_EKPO-MATNR
       .
    SORT LT_MARA BY MATNR.
    SORT LT_MAKT BY MATNR.
  ENDIF.
* 取交货单
  SELECT *
    INTO TABLE LT_LIPS
    FROM LIPS
   WHERE VBELN = G_BELNR.
  SORT LT_LIPS BY VBELN   POSNR.


* 数据整合
  CLEAR GT_ITEM.
  LOOP AT GT_EKPO INTO LW_EKPO.
    CLEAR:
      LW_LIPS,
      LW_MAKT,
      LW_MARA,
      LW_ITEM,
      LW_EKET.
    READ TABLE LT_MARA INTO LW_MARA
                   WITH KEY MATNR = LW_EKPO-MATNR
                     BINARY SEARCH.
    READ TABLE LT_EKET INTO LW_EKET
                   WITH KEY EBELN = LW_EKPO-EBELN
                            EBELP = LW_EKPO-EBELP
                     BINARY SEARCH .
    READ TABLE LT_MAKT INTO LW_MAKT
                   WITH KEY MATNR = LW_EKPO-MATNR
                     BINARY SEARCH.
    READ TABLE LT_LIPS INTO LW_LIPS
                   WITH KEY VGBEL = LW_EKPO-EBELN
                            VGPOS = LW_EKPO-EBELP
                     BINARY SEARCH.
    LW_ITEM-EXTWG = LW_MARA-EXTWG.
    LW_ITEM-GRUND = LW_LIPS-GRUND.
    LW_ITEM-EBELP = LW_LIPS-POSNR.
    LW_ITEM-MATNR = LW_EKPO-MATNR.
    LW_ITEM-MAKTX = LW_MAKT-MAKTX.
    LW_ITEM-CHARG = LW_EKET-CHARG.
    LW_ITEM-MENGE = LW_EKPO-MENGE.
    LW_ITEM-MEINS = LW_EKPO-MEINS.
    LW_ITEM-BWART = '671'.
    APPEND LW_ITEM TO GT_ITEM.
  ENDLOOP.
* 更新批次相关信息
  PERFORM FM_GET_CHARACTERS.
* 设定光标位置
  G_CURSOR = 'ZMMRP0260_SCREEN-BARCODE'.
ENDFORM.

FORM FM_GET_CHARACTERS.
  FIELD-SYMBOLS <FS> TYPE TY_ITEM.
  DATA:
    LT_CHARG             TYPE ZMM_CHARG            OCCURS 0,
    LW_CHARG_CHARACTERS  TYPE ZMM_CHARG_CHARACTERS ,
    LT_CHARG_CHARACTERS  TYPE ZMM_CHARG_CHARACTERS OCCURS 0.

  MOVE-CORRESPONDING GT_EKPO TO LT_CHARG.
* 取批次信息
  call function 'ZMM_GET_CHARG_CHARACTERS'
    TABLES
      T_CHARG                  = LT_CHARG
      T_CHARG_CHARACTERS       = LT_CHARG_CHARACTERS
            .
  SORT LT_CHARG_CHARACTERS BY MATNR CHARG ATNAM.
  LOOP AT GT_ITEM ASSIGNING <FS>.
*   标签名称
    CLEAR LW_CHARG_CHARACTERS.
    READ TABLE LT_CHARG_CHARACTERS INTO LW_CHARG_CHARACTERS
                               WITH KEY MATNR = <FS>-MATNR
                                        CHARG = <FS>-CHARG
                                        ATNAM = 'ZBQMC'
                                BINARY SEARCH.
    <FS>-BQMC = LW_CHARG_CHARACTERS-ATWRT.
**   件数
*    CLEAR LW_CHARG_CHARACTERS.
*    READ TABLE LT_CHARG_CHARACTERS INTO LW_CHARG_CHARACTERS
*                               WITH KEY MATNR = <FS>-MATNR
*                                        CHARG = <FS>-CHARG
*                                        ATNAM = 'ZDPJS'
*                                BINARY SEARCH.
*    <FS>-FKJS = LW_CHARG_CHARACTERS-ATFLV.
*   单品克重
    CLEAR LW_CHARG_CHARACTERS.
    READ TABLE LT_CHARG_CHARACTERS INTO LW_CHARG_CHARACTERS
                               WITH KEY MATNR = <FS>-MATNR
                                        CHARG = <FS>-CHARG
                                        ATNAM = 'ZJZ'
                                BINARY SEARCH.
    <FS>-DPKZ = LW_CHARG_CHARACTERS-ATFLV.

    IF <FS>-MEINS = 'ST'.
      <FS>-FKJS = <FS>-MENGE.
    ELSE.
      CLEAR <FS>-FKJS.
      IF <FS>-DPKZ IS NOT INITIAL.
        <FS>-FKJS = <FS>-MENGE / <FS>-DPKZ.
      ELSE.
        <FS>-FKJS = 1.
      ENDIF.
    ENDIF.
*   零售价
    CLEAR LW_CHARG_CHARACTERS.
    READ TABLE LT_CHARG_CHARACTERS INTO LW_CHARG_CHARACTERS
                               WITH KEY MATNR = <FS>-MATNR
                                        CHARG = <FS>-CHARG
                                        ATNAM = 'ZCKLSJ'
                                BINARY SEARCH.
    <FS>-LSJ = LW_CHARG_CHARACTERS-ATFLV.
*   总克重
    IF <FS>-MEINS = 'ST'.
      <FS>-ZKZ  = <FS>-FKJS * <FS>-DPKZ.
    ELSE.
      <FS>-ZKZ  = <FS>-MENGE.
    ENDIF.
*   总金额
    <FS>-ZJE = <FS>-FKJS * <FS>-LSJ.
  ENDLOOP.
ENDFORM.

MODULE USER_COMMAND_8000 INPUT.
  DATA:
    L_CODE LIKE OK_CODE.

  L_CODE = OK_CODE.
  CLEAR:
    OK_CODE.

  IF GO_ALV_ITEM IS NOT INITIAL.
    GO_ALV_ITEM->CHECK_CHANGED_DATA( ).
  ENDIF.

  CASE L_CODE.
    WHEN '&F03'.
      LEAVE TO SCREEN 0.
    WHEN '&F15'.
      LEAVE PROGRAM.
    WHEN '&SAVE'.
      PERFORM FM_POST.
    WHEN 'CHECK'.
      PERFORM FM_CHECK.
    WHEN 'SET_VAL'.
      PERFORM FM_SET_VAL.
  ENDCASE.
ENDMODULE.

*&---------------------------------------------------------------------*
*&      Module  M_ENTER  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE M_ENTER INPUT.
  CLEAR G_ERROR.

* 不是回车键，不处理
  CHECK OK_CODE IS INITIAL.
  CHECK ZMMRP0260_SCREEN-BARCODE IS NOT INITIAL.

* 设定光标位置
  G_CURSOR = 'ZMMRP0260_SCREEN-BARCODE'.

* 回车处理
  PERFORM FM_BARCODE_ENTER.
  CHECK G_ERROR IS INITIAL.
* 刷新屏幕
  PERFORM FM_REFRESH_ALV.

ENDMODULE.

*&---------------------------------------------------------------------*
*& BARCODE 回车处理
*&---------------------------------------------------------------------*
FORM FM_BARCODE_ENTER.
  DATA:
    L_COUNT      TYPE I,
    LW_ITEM      TYPE TY_ITEM,
    LW_INFO      TYPE ZMM_BARCODE_INFO,
    LT_INFO      TYPE ZMM_BARCODE_INFO OCCURS 0.

* 根据barcode取批次信息
  call function 'ZMM_GET_BARCODE_INFO'
    EXPORTING
      I_BARCODE       = ZMMRP0260_SCREEN-BARCODE
    TABLES
      T_INFO          = LT_INFO
            .
* 判断
  IF LT_INFO IS INITIAL.
    MESSAGE S000(ZCB) WITH '条码/包码不存在' DISPLAY LIKE 'E'.
    G_ERROR = 'X'.
    RETURN.
  ENDIF.
  SORT LT_INFO BY MATNR CHARG.
  CLEAR L_COUNT.
  LOOP AT GT_ITEM INTO LW_ITEM.
    CLEAR LW_INFO.
    READ TABLE LT_INFO INTO LW_INFO
                   WITH KEY MATNR = LW_ITEM-MATNR
                            CHARG = LW_ITEM-CHARG
                     BINARY SEARCH.
    IF SY-SUBRC = 0.
      L_COUNT = L_COUNT + 1.
      LW_ITEM-SSJS = LW_ITEM-SSJS + 1.
    ENDIF.
    MODIFY GT_ITEM FROM LW_ITEM TRANSPORTING SSJS.
  ENDLOOP.
* 没有匹配的批次
  IF L_COUNT = 0.
    MESSAGE S000(ZCB) WITH '扫描的商品不属于返库单商品' DISPLAY LIKE 'E'.
    G_ERROR = 'X'.
    RETURN.
  ENDIF.
  CLEAR ZMMRP0260_SCREEN-BARCODE .
* 刷新数量
  PERFORM FM_CAL_ITEM.
ENDFORM.

*----------------------------------------------------------------------*
FORM FM_HANDLE_ENTER USING E_MODIFIED TYPE CHAR1
                           ET_GOOD_CELLS TYPE LVC_T_MODI.
* 如果修改件数，自动更新数量
  READ TABLE ET_GOOD_CELLS WITH KEY FIELDNAME = 'SSJS' TRANSPORTING NO FIELDS.
  IF SY-SUBRC = 0.
*   重新计算数量
    PERFORM FM_CAL_ITEM.
*   刷新
    PERFORM FM_REFRESH_ALV.
  ENDIF.

* 如果是修改件数也刷新
  READ TABLE ET_GOOD_CELLS WITH KEY FIELDNAME = 'SSSL' TRANSPORTING NO FIELDS.
  IF SY-SUBRC = 0.
*   刷新
    PERFORM FM_REFRESH_ALV.
  ENDIF.
ENDFORM.
*----------------------------------------------------------------------*
* 重新计算数量
*----------------------------------------------------------------------*
FORM FM_CAL_ITEM.
  FIELD-SYMBOLS <FS> TYPE TY_ITEM.

  LOOP AT GT_ITEM ASSIGNING <FS>.
    IF <FS>-MEINS = 'ST'.
      <FS>-SSSL = <FS>-SSJS.
    ELSE.
*     单品克重为空，一件 = 1g
      IF <FS>-DPKZ IS NOT INITIAL.
        <FS>-SSSL = <FS>-SSJS * <FS>-DPKZ.
      ENDIF.
    ENDIF.
  ENDLOOP.
ENDFORM.

*----------------------------------------------------------------------*
* 数量检查
*----------------------------------------------------------------------*
FORM FM_CHECK.
  DATA:
    LW_MSG      TYPE  ESP1_MESSAGE_WA_TYPE,
    L_LINENO    TYPE  I,
    LT_MSG      TYPE  ESP1_MESSAGE_TAB_TYPE,
    LW_ITEM     TYPE  TY_ITEM.
  CLEAR G_ERROR.
  DEFINE DF_ADD_MSG .
    CLEAR LW_MSG.
    LW_MSG-MSGID  = 'ZCB'.
    LW_MSG-MSGTY  = 'E'.
    LW_MSG-MSGNO  = &1.
    L_LINENO  = L_LINENO + 1.
    LW_MSG-LINENO = L_LINENO.
    LW_MSG-MSGV1  = &2.
    LW_MSG-MSGV2  = &3.
    APPEND LW_MSG TO LT_MSG.
  END-OF-DEFINITION.
  LOOP AT GT_ITEM INTO LW_ITEM.
    IF LW_ITEM-MENGE <> LW_ITEM-SSSL.
      DF_ADD_MSG '000' '实收数量不等于返库数量' ''.
    ENDIF.

    IF LW_ITEM-FKJS <> LW_ITEM-SSJS.
      DF_ADD_MSG '000' '实收件数不等于返库件数' ''.
    ENDIF.
  ENDLOOP.
* 弹出窗口
  IF LT_MSG IS NOT INITIAL.
    CALL FUNCTION 'C14Z_MESSAGES_SHOW_AS_POPUP'
      TABLES
        I_MESSAGE_TAB       = LT_MSG
              .
    G_ERROR = 'X'.
  ENDIF.
  MESSAGE S000(ZCB) WITH '匹配成功'.
ENDFORM.

FORM FM_POST.
  DATA:
    L_VBELN      TYPE VBELN_VA,
    LT_BDCDATA   TYPE BDCDATA OCCURS 0,
    L_MSG        TYPE CHAR100,
    LW_BDCDATA   TYPE BDCDATA,
    LW_OPTION    TYPE CTU_PARAMS,
    LW_MSG       TYPE BDCMSGCOLL,
    LT_MSG       TYPE BDCMSGCOLL OCCURS 0,
    L_MESSAGE    TYPE CHAR200,
    L_TEXT       TYPE CHAR200,
    L_CHAR14     TYPE CHAR14,
    LW_LIKP      TYPE LIKP,
    LT_LIPS      TYPE LIPS OCCURS 0,
    LW_LIPS      TYPE LIPS,
    LW_VBKOK     TYPE VBKOK,
    LT_PROT      TYPE PROTT OCCURS 0,
    LW_PROT      TYPE PROTT,
    LW_PORT      TYPE PROTT,
    LW_VBPOK     TYPE VBPOK,
    LT_VBPOK     TYPE VBPOK OCCURS 0,
    L_DELIVERY   TYPE LIKP-VBELN,
    LW_T_ITEM    TYPE TY_ITEM,
    L_DOCU       TYPE BAPI2017_GM_HEAD_RET-MAT_DOC,
    LW_HEADER    TYPE BAPI2017_GM_HEAD_01 ,

    LW_RETURN    TYPE BAPIRET2,
    LT_RETURN    TYPE BAPIRET2 OCCURS 0.
* 检查数量
  PERFORM FM_CHECK.

  CHECK G_ERROR IS INITIAL.

***************抬头信息*******************
  L_DELIVERY = G_BELNR.       "交货单号

  DEFINE DF_ADD_BDCDATA.

    APPEND LW_BDCDATA TO LT_BDCDATA.
  END-OF-DEFINITION.

  SELECT SINGLE *
    INTO LW_LIKP
    FROM LIKP
   WHERE VBELN = L_DELIVERY.

***************行项目信息*****************
  SELECT *
    INTO TABLE LT_LIPS
    FROM LIPS
   WHERE VBELN = L_DELIVERY.
* 使用bdc发货过账
  DF_ADD_BDCDATA:
    '        ' '    ' 'T' 'VL02N'             'BS AA X   F',
    'SAPMV50A' '4004' 'X' '               '   '              ',
    '        ' '    ' ' ' 'BDC_OKCODE'        '/00',
    '        ' '    ' ' ' 'LIKP-VBELN'        L_DELIVERY,
    'SAPMV50A' '1000' 'X' '               '   '              ',
    '        ' '    ' ' ' 'BDC_OKCODE     '   '=T\01          ',
    'SAPMV50A' '1000' 'X' '               '   '              ',
    '        ' '    ' ' ' 'BDC_OKCODE     '   '=T\02          ',
    '        ' '    ' ' ' 'LIKP-WADAT_IST'    ZMMRP0260_SCREEN-BUDAT,
    '        ' '    ' ' ' 'LIKP-BLDAT     '   ZMMRP0260_SCREEN-BLDAT.
*   'SAPMV50A' '1000' 'X' '               '   '              '.

  LOOP AT LT_LIPS INTO LW_LIPS.
*   读取移动原因
    READ TABLE GT_ITEM INTO LW_T_ITEM
                   WITH KEY EBELP = LW_LIPS-POSNR.

    DF_ADD_BDCDATA:
    'SAPMV50A' '1000' 'X' '               '   '              ',
    '        ' '    ' ' ' 'BDC_OKCODE'        '=POPO_T',
    'SAPMV50A' '0111' 'X' '               '   '              ',
    '        ' '    ' ' ' 'BDC_OKCODE'        '=WEIT',
    '        ' '    ' ' ' 'RV50A-POSNR'       LW_LIPS-POSNR,
    'SAPMV50A' '1000' 'X' '               '   '              ',
    '        ' '    ' ' ' 'BDC_OKCODE'        '=IDET_T',

    '        ' '    ' ' ' 'LIPS-LGORT(01)'    ZMMRP0260_SCREEN-LGORTT ,
    'SAPMV50A' '3000' 'X' '               '   '              ',
    '        ' '    ' ' ' 'BDC_OKCODE'        '=T\11',
    'SAPMV50A' '3000' 'X' '               '   '              ',
    '        ' '    ' ' ' 'BDC_OKCODE'        '=BACK_T'.
    IF LW_T_ITEM-GRUND IS NOT INITIAL .
      DF_ADD_BDCDATA:
        '        ' '    ' ' ' 'LIPS-GRUND'        LW_T_ITEM-GRUND.
    ELSE.
      DF_ADD_BDCDATA:
        '        ' '    ' ' ' 'LIPS-GRUND'        ''.
    ENDIF.
  ENDLOOP.
  DF_ADD_BDCDATA:
    '        ' '    ' ' ' 'BDC_OKCODE'        '=WABU_T'.
  LW_OPTION-UPDMODE  = 'S'.  "更新模式
  LW_OPTION-DISMODE  = 'N'.  "显示模式
* 解锁
  CALL FUNCTION 'DEQUEUE_EVVBLKE'
    EXPORTING
      VBELN           = G_BELNR
            .

* 过账
  CALL TRANSACTION 'VL02N' USING LT_BDCDATA
                    OPTIONS FROM LW_OPTION
                   MESSAGES INTO LT_MSG.

* 状态判定
  CLEAR L_TEXT.
  READ TABLE LT_MSG INTO LW_MSG
            WITH KEY MSGID = 'VL'
                     MSGNR = '311'
                     .

    L_TEXT = '收货失败：'.
  ENDIF.

  LOOP AT LT_MSG INTO LW_MSG WHERE MSGTYP = 'A' OR MSGTYP = 'E'.
    CLEAR L_MSG.
    MESSAGE ID LW_MSG-MSGID
          TYPE LW_MSG-MSGTYP
        NUMBER LW_MSG-MSGNR
          WITH LW_MSG-MSGV1
               LW_MSG-MSGV2
               LW_MSG-MSGV3
               LW_MSG-MSGV4
          INTO L_MSG.

    IF L_TEXT IS INITIAL.
      L_TEXT = L_MSG.
    ELSE.
      CONCATENATE L_TEXT L_MSG INTO L_TEXT SEPARATED BY '//'.
    ENDIF.
  ENDLOOP.
* 结果处理
  IF L_TEXT IS NOT INITIAL.

    CALL FUNCTION 'BAPI_TRANSACTION_ROLLBACK'
             .
    MESSAGE S000(ZCB) WITH L_TEXT DISPLAY LIKE 'E'.
    G_ERROR = 'X'.
*   锁表
*   锁交货单
    CALL FUNCTION 'ENQUEUE_EVVBLKE'
      EXPORTING
        VBELN                = G_BELNR
      EXCEPTIONS
        FOREIGN_LOCK         = 1
        SYSTEM_FAILURE       = 2
        OTHERS               = 3
               .
    RETURN.
  ELSE.
    CALL FUNCTION 'BAPI_TRANSACTION_COMMIT'
      EXPORTING
        WAIT          = 'X'
              .
    SELECT SINGLE MAX( VBELN )
      INTO L_VBELN
      FROM VBFA
     WHERE VBELV = L_DELIVERY
       AND VBTYP_N = 'R'
      .

    LEAVE TO SCREEN 0.
  ENDIF.
ENDFORM.
