FORM FRM_GET_DATA .
  IF RB_1 = 'X' .
    SELECT *
      INTO TABLE GT_QC_PUR_INFO
      FROM ZQC_PUR_INFO
      WHERE SEQ IN S_SEQ
        AND CHKFLG = 'S'
        AND NEED_IMPORT = 'Y'
        AND ZZHAS_IMPORT IN ( '' , 'E' ) .
  ELSEIF RB_2 = 'X' .
    SELECT *
      INTO TABLE GT_QC_PUR_INFO
      FROM ZQC_PUR_INFO
      WHERE SEQ IN S_SEQ
        AND CHKFLG = 'S'
        AND NEED_IMPORT = 'Y'
        AND ZZHAS_IMPORT = '' .
  ENDIF.
ENDFORM.

FORM FRM_CREATE_PUR_INFO .
  DATA: LV_ERR TYPE C LENGTH 1 .

  MOVE-CORRESPONDING GT_QC_PUR_INFO TO GT_ZMMDG0010_02 .

  CLEAR: GS_ZMMDG0010_02 .
  GS_ZMMDG0010_02-ZYHID = SY-UNAME .
  GS_ZMMDG0010_02-ZRQ   = SY-DATUM .
  MODIFY GT_ZMMDG0010_02 FROM GS_ZMMDG0010_02 TRANSPORTING ZYHID ZRQ WHERE ZGYSDM <> '' .

  " 更新数据库
  MODIFY ZMMDG0010_02 FROM TABLE GT_ZMMDG0010_02 .
  IF SY-SUBRC = 0 .
    COMMIT WORK .
    LV_ERR = '' .
    GV_NUM_ALL = LINES( GT_ZMMDG0010_02 ) .
    MESSAGE S000(ZCB) WITH '本次共导入采购信息记录'  GV_NUM_ALL '条！' .
  ELSE.
    ROLLBACK WORK .
    LV_ERR = 'X' .
    MESSAGE S000(ZCB) WITH '更新数据库表ZMMDG0010_02失败！' .
  ENDIF.

  IF LV_ERR = '' .
    CLEAR: GS_QC_PUR_INFO .
    GS_QC_PUR_INFO-ZZHAS_IMPORT = 'S' .
    GS_QC_PUR_INFO-ERR_MSG = '' .
    MODIFY GT_QC_PUR_INFO FROM GS_QC_PUR_INFO TRANSPORTING ZZHAS_IMPORT ERR_MSG WHERE SEQ <> '' .
  ELSE.
    CLEAR: GS_QC_PUR_INFO .
    GS_QC_PUR_INFO-ZZHAS_IMPORT = 'E' .
    GS_QC_PUR_INFO-ERR_MSG = '更新数据库表ZMMDG0010_02失败！' .
    MODIFY GT_QC_PUR_INFO FROM GS_QC_PUR_INFO TRANSPORTING ZZHAS_IMPORT ERR_MSG WHERE SEQ <> '' .
  ENDIF.

  MODIFY ZQC_PUR_INFO FROM TABLE GT_QC_PUR_INFO .
  IF SY-SUBRC = 0 .
    COMMIT WORK .
  ELSE.
    ROLLBACK WORK .
  ENDIF.

ENDFORM.
