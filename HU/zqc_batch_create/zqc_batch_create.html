Code listing for: ZQC_BATCH_CREATE

Description: 期初批次导入 - 创建批次
REPORT ZQC_BATCH_CREATE.

include ZQC_BATCH_CREATE_TOP.
include ZQC_BATCH_CREATE_SCR.
include ZQC_BATCH_CREATE_F01.

START-OF-SELECTION .
  PERFORM FRM_GET_DATA .
  IF GT_BATCH[] IS INITIAL .
    MESSAGE S000(ZCB) WITH '没有找到需要创建的批次信息！' .
    STOP .
  ENDIF.

  PERFORM FRM_CREATE_BATCH TABLES GT_BATCH USING GV_NUM_ERR.


  GV_NUM_ALL = LINES( GT_BATCH ) .
  IF GV_NUM_ERR > 0 .
    MESSAGE S000(ZCB) WITH '本次需要创建' GV_NUM_ALL '条批次信息，其中创建失败的有' GV_NUM_ERR .
  ELSE.
    MESSAGE S000(ZCB) WITH '本次需要创建' GV_NUM_ALL '条批次信息，全部创建成功！' .
  ENDIF.
