FORM FRM_CHECK_AUTH .
* 报表的权限检查，如果权限对象是单一字段，处理方法：
* 1 首先根据用户输入的查询条件，取得所有符合用户输入查询条件的对象；
* 2 对每个对象检查该用户是否有权限；
* 3 将有权限的对象添加到一个Range表，作为后续取业务数据的查询条件
* 4 如果所有对象都没有权限，则提示错误信息，程序结束

  " 取得所有符合查询条件的地点
  DATA: BEGIN OF LS_WERKS,
        WERKS TYPE T001W-WERKS ,
        END OF LS_WERKS ,
        LT_WERKS LIKE TABLE OF LS_WERKS,
        LRS_WERKS LIKE LINE OF GRT_WERKS .
  SELECT WERKS
    INTO TABLE LT_WERKS
    FROM T001W
    WHERE WERKS IN S_WERKS .
  IF SY-SUBRC <> 0 .
    MESSAGE S000(ZCB) WITH '没有找到符合条件的地点，请检查您输入的查询条件！' .
    STOP .
  ENDIF.

  LOOP AT LT_WERKS INTO LS_WERKS .
    " 对每个地点检查用户是否有权限
    AUTHORITY-CHECK OBJECT 'M_MSEG_WMB'
             ID 'ACTVT' FIELD '03'
             ID 'WERKS' FIELD LS_WERKS-WERKS .
    IF SY-SUBRC = 0.
      " 对于有权限的地点添加到Range表，作为后续取数的查询条件
      CLEAR: LRS_WERKS .
      LRS_WERKS-SIGN = 'I' .
      LRS_WERKS-OPTION = 'EQ' .
      LRS_WERKS-LOW = LS_WERKS-WERKS .
      LRS_WERKS-HIGH = '' .
      APPEND LRS_WERKS TO GRT_WERKS .
    ENDIF.
  ENDLOOP.
  " 没有所有地点都没有权限，则提示错误信息，程序结束
  IF GRT_WERKS[] IS INITIAL .
    MESSAGE S000(ZCB) WITH '您没有查询输入的地点的权限，请检查您输入的查询条件！' DISPLAY LIKE 'E'.
    STOP .
  ENDIF.

* 报表的权限检查，如果权限对象是组合字段，处理方法：
* 1 首先根据用户输入的查询条件，取得所有符合用户输入查询条件的字段组合；
* 2 对每个对象检查该用户是否有权限；
* 3 将有权限的对象添加到一个内表，作为后续取业务数据的查询条件
* 4 如果所有对象都没有权限，则提示错误信息，程序结束
  FIELD-SYMBOLS <FS_VBAK_VKO> LIKE LINE OF GT_VBAK_VKO .

  " 对于组合字段的权限对象，一般在后台都有一个配置表维护了这些字段组合
  " 例如SD很多功能都检查销售范围（销售组织+分销渠道+产品组）的权限对象，
  " 在后台TVTA表维护了允许的销售组织+分销渠道+产品组的组合
  SELECT VKORG VTWEG SPART
    INTO TABLE GT_VBAK_VKO
    FROM TVTA
    WHERE VKORG IN S_VKORG
      AND VTWEG IN S_VTWEG
      AND SPART IN S_SPART .
  IF SY-SUBRC <> 0 .
    MESSAGE S000(ZCB) WITH '没有找到符合条件的销售组织+分销渠道+产品组，请检查您输入的查询条件！' .
    STOP .
  ENDIF.

  LOOP AT GT_VBAK_VKO ASSIGNING <FS_VBAK_VKO> .
    " 对每个组合，检查用户是否有权限
    AUTHORITY-CHECK OBJECT 'V_VBAK_VKO'
             ID 'VKORG' FIELD <FS_VBAK_VKO>-VKORG
             ID 'VTWEG' FIELD <FS_VBAK_VKO>-VTWEG
             ID 'SPART' FIELD <FS_VBAK_VKO>-SPART
             ID 'ACTVT' FIELD '03'.
    IF SY-SUBRC <> 0.
      <FS_VBAK_VKO>-VKORG = '' .
    ENDIF.
  ENDLOOP.
  DELETE GT_VBAK_VKO WHERE VKORG = '' . " 从内表中删除用户没有权限的组合
  IF GT_VBAK_VKO[] IS INITIAL .
    MESSAGE S000(ZCB) WITH '您没有查询输入的销售组织+分销渠道+产品组的权限，请检查您输入的查询条件！' DISPLAY LIKE 'E'.
    STOP .
  ENDIF .

ENDFORM.

FORM FRM_PREPARE_CRITERIA .
* 有时，有些屏幕的查询条件，在后续查询业务数据时不能直接使用，或者要通过较复杂的表关联
* 来查询。对于这种情况，可以将这种查询条件转换为另一个可直接使用的查询条件。
* 这种处理称为查询条件预处理。实际上对权限的检查和处理，也是一种查询条件预处理。

* 例如本例当中的查询条件“是否联营”，在查询时要通过VBRP的批次，关联ZMMRP0210_BATCH，然后
* 根据ZMMRP0210_BATCH的ZLYJSFS(结算方式) 关联ZMMDG0010_JSFS才能取得是否联营。因此可以
* 根据查询条件“是否联营”，先取得对应的结算方式，在后续的业务数据查询时使用结算方式。
  DATA: BEGIN OF LS_JSFS,
          BM TYPE ZMMDG0010_JSFS-BM ,
        END OF LS_JSFS,
        LT_JSFS LIKE TABLE OF LS_JSFS,
        LRS_JSFS LIKE LINE OF GRT_JSFS .

  SELECT BM
    INTO TABLE LT_JSFS
    FROM ZMMDG0010_JSFS
    WHERE SFLY IN S_SFLY .
  IF SY-SUBRC <> 0 .
    MESSAGE S000(ZCB) WITH '根据您输入的是否联营没有找到对应的结算方式，请检查您的输入！' .
    STOP .
  ENDIF.

  LOOP AT LT_JSFS INTO LS_JSFS .
    CLEAR: LRS_JSFS .
    LRS_JSFS-SIGN = 'I' .
    LRS_JSFS-OPTION = 'EQ' .
    LRS_JSFS-LOW = LS_JSFS-BM .
    LRS_JSFS-HIGH = '' .
    APPEND LRS_JSFS TO GRT_JSFS .
  ENDLOOP.

ENDFORM .

FORM FRM_GET_MAIN_DATA .

* 因为原始数据和报表结果是多对1，所以先取得明细的原始数据，即每个Billing Item
  SELECT
         " 每个Billing Item的明细字段
         P~VBELN P~POSNR P~SHKZG P~CHARG
         " 关键字
         K~FKDAT P~WERKS P~MATNR
         " 直接指标
         P~FKIMG P~VRKME P~KZWI3 P~KZWI2
         " 部分属性
         A~EXTWG                     " 因为该属性也是查询条件之一，所以在这里就关联表顺便取得
    INTO CORRESPONDING FIELDS OF TABLE GT_BILLING
    FROM VBRK AS K
    INNER JOIN VBRP AS P ON P~VBELN = K~VBELN
    INNER JOIN MARA AS A ON P~MATNR = A~MATNR
    INNER JOIN ZMMRP0210_BATCH AS Z ON  Z~MATNR = P~MATNR
                                    AND Z~CHARG = P~CHARG
    FOR ALL ENTRIES IN GT_VBAK_VKO    " 使用权限检查后的销售组织+分销渠道+产品组，而不是用户输入的S_VKORG, S_VTWEG, S_SPART
                                      " 对于这种组合字段的权限对象，使用  FOR ALL ENTRIES IN 的查询方式
    WHERE K~FKDAT IN S_FKDAT
      AND K~VKORG = GT_VBAK_VKO-VKORG
      AND K~VTWEG = GT_VBAK_VKO-VTWEG
      AND P~SPART = GT_VBAK_VKO-SPART
      AND P~WERKS IN GRT_WERKS           " 使用权限检查后的地点，而不是用户输入的地点S_WERKS
                                         " 对于这种单一字段的权限对象，使用Range表作为查询条件
      AND P~MATNR IN S_MATNR
      AND P~CHARG IN S_CHARG
      AND A~EXTWG IN S_EXTWG
      AND Z~ZLYJSFS IN GRT_JSFS .        " 预处理的查询条件
  IF SY-SUBRC <> 0 .
    MESSAGE S000(ZCB) WITH '没有查到任何系统发票！' .
    STOP .
  ENDIF.
ENDFORM.

FORM FRM_GET_ATTRIBUTE .
* 取得关键字属性的方法：
* 1. 如果该关键字是权限检查对象，可以根据权限检查过滤后的表作为查询条件取得关键字属性
* 2. 如果该关键字是查询条件，那可以根据查询条件取得其属性
*    该方法不适用数据量很大的关键字，例如商品，因为用户可能在查询屏幕中没有输入该查询条件，
*    导致取出很大数据量
* 3. 如果该关键字的数据量本身就很小，干脆直接取得全部
* 4. 根据第1步取得的数据中的关键字，取得其属性
* 按以上顺序优先选择方法。


* 在本程序中，要取得地点的属性：是否扣点店，如果报表要显示关键字的描述，在这里可以顺便一起取得

* 方法1示例： 在本程序中地点本身也是权限检查对象，那这里可以使用权限检查过滤后的表来取数
  SELECT W~WERKS C~SORT1 Z~EX_POS
    INTO CORRESPONDING FIELDS OF TABLE GT_WERKS
    FROM T001W AS W
    INNER JOIN ADRC AS C ON C~ADDRNUMBER = W~ADRNR
    INNER JOIN ZT001W AS Z ON Z~WERKS = W~WERKS
    WHERE W~WERKS IN GRT_WERKS . "GRT_WERKS 是权限检查过滤后的表

* 方法2示例： 因为地点本身就是查询条件，可以根据查询条件取得其属性。
  SELECT W~WERKS C~SORT1 Z~EX_POS
    INTO CORRESPONDING FIELDS OF TABLE GT_WERKS
    FROM T001W AS W
    INNER JOIN ADRC AS C ON C~ADDRNUMBER = W~ADRNR
    INNER JOIN ZT001W AS Z ON Z~WERKS = W~WERKS
    WHERE W~WERKS IN S_WERKS .  "S_WERKS 是屏幕输入查询条件


* 方法3示例： 因为地点本身数据量小，干脆取得全部数据
  SELECT W~WERKS C~SORT1 Z~EX_POS
    INTO CORRESPONDING FIELDS OF TABLE GT_WERKS
    FROM T001W AS W
    INNER JOIN ADRC AS C ON C~ADDRNUMBER = W~ADRNR
    INNER JOIN ZT001W AS Z ON Z~WERKS = W~WERKS .

* 方法4示例：根据第1步取得的数据中的地点，取得其属性
  MOVE-CORRESPONDING GT_BILLING TO GT_WERKS . " 这是MOVE-CORRESPONDING语法的新应用，
                                              " 可以将A内表中相同字段复制到B内表中，
                                              " 过去只能从结构复制到结构
  SORT GT_WERKS BY WERKS .
  DELETE ADJACENT DUPLICATES FROM GT_WERKS COMPARING WERKS .
  IF GT_WERKS[] IS NOT INITIAL .               " 用于FOR ALL ENTRIES IN的内表，一
                                               " 定要确认其不能为空，否则会取出所有数据
    SELECT W~WERKS C~SORT1 Z~EX_POS
      INTO CORRESPONDING FIELDS OF TABLE GT_WERKS
      FROM T001W AS W
      INNER JOIN ADRC AS C ON C~ADDRNUMBER = W~ADRNR
      INNER JOIN ZT001W AS Z ON Z~WERKS = W~WERKS
      FOR ALL ENTRIES IN GT_WERKS             " 为什么这里不直接使用GT_BILLING，而是
                                              " 使用排序去重后的GT_WERKS。因为这样可以
                                              " 减少向DB传输的查询语句的大小，提高查询效率
      WHERE W~WERKS = GT_WERKS-WERKS .
  ENDIF.
ENDFORM.

FORM FRM_GET_DESCRIPTION .
* 取得关键字和属性的描述的方法：
* 1. 如果关键字的描述和属性在同一个表，可以在前面取得关键字属性的同时取得
* 2. 如果该关键字的数据量本身就很小，干脆直接取得全部
* 3. 如果是域描述，一次取得全部
* 4. 根据前面取得的关键字或属性，取得其描述
* 按以上顺序优先选择方法。
* 注意：很多SAP标准的Code描述表是多语言的，在取数时要确定取哪个语言的条件，
*       一般取SY-LANGU

* 方法2示例：商品大类名称，商品大类数据量很少，一次全部取得
  SELECT EXTWG EWBEZ
    INTO CORRESPONDING FIELDS OF TABLE GT_TWEWT
    FROM TWEWT
    WHERE SPRAS = SY-LANGU .

* 方法3示例：取得域描述，这里以域ZDE_RATE_TYPE举例
  DATA: LT_TABLE LIKE GT_ZDE_RATE_TYPE .

  CALL FUNCTION 'DD_DOMA_GET'
    EXPORTING
      DOMAIN_NAME         = 'ZDE_RATE_TYPE'
      LANGU               = SY-LANGU
    TABLES
      DD07V_TAB_A         = GT_ZDE_RATE_TYPE
      DD07V_TAB_N         = LT_TABLE
    EXCEPTIONS
      ILLEGAL_VALUE       = 1
      OP_FAILURE          = 2
      OTHERS              = 3 .
  IF SY-SUBRC <> 0.
    " Nothing to do
  ENDIF.

* 方法4示例：商品名称，根据前面取得业务数据中的商品，取得其描述
  MOVE-CORRESPONDING GT_BILLING TO GT_MAKT .
  SORT GT_MAKT BY MATNR .
  DELETE ADJACENT DUPLICATES FROM GT_MAKT COMPARING MATNR .
  SELECT MATNR MAKTX
    INTO CORRESPONDING FIELDS OF TABLE GT_MAKT
    FROM MAKT
    FOR ALL ENTRIES IN GT_MAKT
    WHERE MATNR = GT_MAKT-MATNR .

ENDFORM.

* 2. 再对半成品最终报表每一条记录进行处理，取得关键字属性、描述、附加指标，并处理计算指标
  DATA: LS_BILLING LIKE LINE OF GT_BILLING,
        LS_ALV LIKE LINE OF GT_ALV .
  FIELD-SYMBOLS <FS_ALV> LIKE LINE OF GT_ALV .

  LOOP AT GT_BILLING INTO LS_BILLING .

  " 先对属性、描述、附加指标表进行排序，在后面读取时使用BINARY SEARCH，提高效率
