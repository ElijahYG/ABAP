* 处理逻辑
START-OF-SELECTION .
  SELECT A~EXTWG, G~BUKRS, G~LIFNR, SUM( DMBTR * CASE  G~SHKZG
                                        WHEN 'S' THEN 1
                                        WHEN 'H' THEN -1
                                      END ) AS DMBTR
    INTO TABLE @GT_ALV
    FROM MSEG AS G
    INNER JOIN MARA AS A ON G~MATNR = A~MATNR
    WHERE G~BUDAT_MKPF IN @S_BUDAT
      AND G~WERKS IN @S_WERKS
      AND G~BWART IN ( '101','102','161','162' )
      AND A~EXTWG IN @S_EXTWG
    GROUP BY A~EXTWG, G~BUKRS, G~LIFNR .

  IF GT_ALV[] IS NOT INITIAL .
    SORT GT_ALV BY BUKRS LIFNR EXTWG .
    PERFORM FRM_DIS_ALV .
  ELSE.
    MESSAGE S000(ZCB) WITH '没有查询结果！' DISPLAY LIKE 'E' .
  ENDIF.
