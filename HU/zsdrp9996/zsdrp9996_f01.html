FORM FRM_GET_IDOC .
  SELECT *
    INTO TABLE GT_EDIDC
    FROM EDIDC
    WHERE MESTYP = 'ZSDIF0191_MSG'
      AND DOCNUM IN S_DOCNUM
      AND STATUS IN S_STATUS
      AND RCVPRN IN S_RCVPRN
      AND CREDAT IN S_CREDAT
      AND CRETIM IN S_CRETIM .
  IF SY-SUBRC = 0 .
    SELECT *
      INTO TABLE GT_EDIDD
      FROM EDID4
      FOR ALL ENTRIES IN GT_EDIDC
      WHERE DOCNUM = GT_EDIDC-DOCNUM .

    SELECT WERKS NAME1
      INTO TABLE GT_SITE
      FROM T001W .
  ENDIF.
ENDFORM.

FORM FRM_READ_IDOC.
  DATA: LS_ALV_TEMP LIKE GS_ALV,
        LS_ALV      LIKE GS_ALV,
        LT_ALV      LIKE GT_ALV.

  SORT GT_EDIDC BY DOCNUM .
  SORT GT_EDIDD BY DOCNUM COUNTER SEGNUM .

  LOOP AT GT_EDIDD INTO GS_EDIDD .
    AT NEW DOCNUM .
      IF LT_ALV[] IS NOT INITIAL.
        PERFORM FRM_ADD_ALV TABLES LT_ALV .
      ENDIF.

      CLEAR: LS_ALV_TEMP,LS_ALV .
      REFRESH: LT_ALV .
      READ TABLE GT_EDIDC INTO GS_EDIDC WITH KEY DOCNUM = GS_EDIDD-DOCNUM .
      IF SY-SUBRC <> 0 .
        CONTINUE .
      ELSE.
        MOVE-CORRESPONDING GS_EDIDC TO LS_ALV_TEMP .
      ENDIF.
    ENDAT .

    IF GS_EDIDD-SEGNAM = 'ZSDIF0191_SGM_HEAD'.
      GS_SEG_0191_HEAD = GS_EDIDD-SDATA .
      MOVE-CORRESPONDING GS_SEG_0191_HEAD TO LS_ALV_TEMP .
      IF GS_SEG_0191_HEAD-EX_MDBM IS NOT INITIAL .
        READ TABLE GT_SITE INTO GS_SITE WITH KEY WERKS = GS_SEG_0191_HEAD-EX_MDBM .
        IF SY-SUBRC = 0 .
          LS_ALV_TEMP-NAME1 = GS_SITE-NAME1 .
        ENDIF.
      ENDIF.
    ELSEIF GS_EDIDD-SEGNAM = 'ZSDIF0191_SGM_ITEM'.
      CLEAR: LS_ALV .
      GS_SEG_0191_ITEM = GS_EDIDD-SDATA .
      MOVE-CORRESPONDING LS_ALV_TEMP TO LS_ALV .
      MOVE-CORRESPONDING GS_SEG_0191_ITEM TO LS_ALV .
      APPEND LS_ALV TO LT_ALV .

      LS_ALV_TEMP-NUM_OF_ITEM = LS_ALV_TEMP-NUM_OF_ITEM + 1 .
      LS_ALV_TEMP-VAL_OF_ITEM = LS_ALV_TEMP-VAL_OF_ITEM + GS_SEG_0191_ITEM-EX_DJJE . " 结算金额
      MODIFY LT_ALV FROM LS_ALV_TEMP TRANSPORTING NUM_OF_ITEM VAL_OF_ITEM
                                     WHERE DOCNUM = LS_ALV_TEMP-DOCNUM .
    ELSEIF GS_EDIDD-SEGNAM = 'ZSDIF0191_SGM_POST' .
      GS_SEG_0191_POST = GS_EDIDD-SDATA .
      LS_ALV_TEMP-NUM_OF_POST = LS_ALV_TEMP-NUM_OF_POST + 1 .
      LS_ALV_TEMP-VAL_OF_POST = LS_ALV_TEMP-VAL_OF_POST + GS_SEG_0191_POST-EX_SFJE . " 实付金额
      MODIFY LT_ALV FROM LS_ALV_TEMP TRANSPORTING NUM_OF_POST VAL_OF_POST
                                     WHERE DOCNUM = LS_ALV_TEMP-DOCNUM .
    ELSEIF GS_EDIDD-SEGNAM = 'ZSDIF0191_SGM_HUMAN' .
      GS_SEG_0191_HUMAN = GS_EDIDD-SDATA .
      LS_ALV_TEMP-NUM_OF_HUMAN = LS_ALV_TEMP-NUM_OF_HUMAN + 1 .
      LS_ALV_TEMP-VAL_OF_HUMAN = LS_ALV_TEMP-VAL_OF_HUMAN + GS_SEG_0191_HUMAN-ASSISTANT_AMOUNT . " 分摊金额
      LS_ALV_TEMP-RAT_OF_HUMAN = LS_ALV_TEMP-RAT_OF_HUMAN + GS_SEG_0191_HUMAN-ASSISTANT_PERCENT . " 分摊比例
      MODIFY LT_ALV FROM LS_ALV_TEMP TRANSPORTING NUM_OF_HUMAN VAL_OF_HUMAN RAT_OF_HUMAN
                                     WHERE DOCNUM = LS_ALV_TEMP-DOCNUM .
    ENDIF.
  ENDLOOP.

  IF LT_ALV[] IS NOT INITIAL.
    PERFORM FRM_ADD_ALV TABLES LT_ALV .
  ENDIF.
ENDFORM.
