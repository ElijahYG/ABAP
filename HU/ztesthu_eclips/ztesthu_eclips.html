SELECTION-SCREEN BEGIN OF BLOCK B1 WITH FRAME TITLE TEXT-T01 .
PARAMETERS: P_MAX TYPE N LENGTH 6 .
SELECTION-SCREEN END OF BLOCK B1 .

START-OF-SELECTION .
  DO P_MAX TIMES.
    GS_1-VBELN = 'A' && SY-INDEX .
    APPEND GS_1 TO GT_1 .
    APPEND GS_1 TO GT_2 .
  ENDDO.

  GET TIME.
  WRITE: / 'Begin of 1st loop:', sy-datum, sy-uzeit .
  LOOP  AT GT_1 INTO GS_1 .
    READ TABLE GT_2 INTO GS_2 WITH KEY VBELN = GS_1-VBELN .
  ENDLOOP.
  GET TIME.
  WRITE: / 'End of 1st loop:', sy-datum, sy-uzeit .

  GET TIME.
  WRITE: / 'Begin of 2nd loop:', sy-datum, sy-uzeit .
  SORT GT_2 BY VBELN .
  LOOP  AT GT_1 INTO GS_1 .
    READ TABLE GT_2 INTO GS_2 WITH KEY VBELN = GS_1-VBELN BINARY SEARCH.
  ENDLOOP.
  GET TIME.
  WRITE: / 'End of 2nd loop:', sy-datum, sy-uzeit .
