FORM FRM_GET_ZB00 .

  SELECT A~EX_SITE A~EXTWG A~RAUBE A~EX_GJS P~KBETR
    INTO TABLE GT_ZB00
    FROM A909 AS A
    INNER JOIN KONP AS P ON P~KNUMH = A~KNUMH
    WHERE A~KAPPL = 'V'
      AND A~KSCHL = 'ZB00'
      AND A~DATAB <= GT_DATUM
      AND A~DATBI >= GT_DATUM .

ENDFORM.

*       取得工费
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM FRM_GET_ZKSG .

  DATA: LT_ZKG1 LIKE GT_ZKSG .

  SELECT A~ZXSKGFD P~KBETR
    INTO TABLE GT_ZKSG
    FROM A911 AS A
    INNER JOIN KONP AS P ON P~KNUMH = A~KNUMH
    WHERE A~KAPPL = 'V'
      AND A~KSCHL  = 'ZKSG'
      AND A~DATAB <= GT_DATUM
      AND A~DATBI >= GT_DATUM .

  " 取投资银产品加价百分比
  SELECT A~ZXSKGFD P~KBETR
    INTO TABLE LT_ZKG1
    FROM A911 AS A
    INNER JOIN KONP AS P ON P~KNUMH = A~KNUMH
    WHERE A~KAPPL = 'V'
      AND A~KSCHL  = 'ZKG1'
      AND A~DATAB <= GT_DATUM
      AND A~DATBI >= GT_DATUM .
  IF SY-SUBRC = 0 .
    " 将加价百分比转换为加价金额
    SORT GT_ZTJJ BY EXTWG RAUBE EX_GJS .
    LOOP AT LT_ZKG1 INTO DATA(LS_ZKG1) .
      READ TABLE GT_ZTJJ INTO GS_ZTJJ WITH KEY EXTWG = 'D04'
                                               RAUBE = '19'
                                               EX_GJS = '403'
                                               BINARY SEARCH .
      IF SY-SUBRC = 0 .
        CLEAR: GS_ZKSG .
        GS_ZKSG-ZXSKGFD = LS_ZKG1-ZXSKGFD .
        GS_ZKSG-KBETR = LS_ZKG1-KBETR * GS_ZTJJ-KBETR / 1000 .  "将百分比乘以基础银价计算出克工费加价金额
        APPEND GS_ZKSG TO GT_ZKSG .
      ENDIF.
    ENDLOOP.
  ENDIF.
ENDFORM.
