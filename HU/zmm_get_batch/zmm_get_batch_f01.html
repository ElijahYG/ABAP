FORM FRM_GET_MATNR .

  MOVE-CORRESPONDING GT_ZMM_CHARG TO GT_MATNR .
  SORT GT_MATNR BY MATNR .
  DELETE ADJACENT DUPLICATES FROM GT_MATNR COMPARING MATNR .

  SELECT MATNR EXTWG RAUBE TEMPB
    INTO TABLE GT_MATNR
    FROM MARA
    FOR ALL ENTRIES IN GT_MATNR
    WHERE MATNR = GT_MATNR-MATNR .

ENDFORM.

FORM FRM_GET_PRICE .
  DATA: LT_BATCH TYPE ZCB01_TY_T_ZSD_BATCH_PRICE_KEY,
        LS_BATCH LIKE LINE OF LT_BATCH ,
        LT_BATCH_PRICE TYPE ZCB01_TY_T_ZSD_BATCH_PRICE,
        LS_BATCH_PRICE LIKE LINE OF LT_BATCH_PRICE  .
  DATA: BEGIN OF LS_MATNR,
          MATNR TYPE MARA-MATNR,
          TEMPB TYPE MARA-TEMPB,
        END OF LS_MATNR,
        LT_MATNR LIKE TABLE OF LS_MATNR .

  IF GT_ALV[] IS INITIAL .
    RETURN .
  ENDIF.
* 取得商品的条码类型
  MOVE-CORRESPONDING GT_ALV[] TO LT_MATNR .
  SORT LT_MATNR BY MATNR .
  DELETE ADJACENT DUPLICATES FROM LT_MATNR COMPARING MATNR .
  SELECT MATNR TEMPB
    INTO TABLE LT_MATNR
    FROM MARA
    FOR ALL ENTRIES IN LT_MATNR
    WHERE MATNR = LT_MATNR-MATNR .

* 取得批次的价格
  MOVE-CORRESPONDING GT_ALV[] TO LT_BATCH[] .
  LS_BATCH-WERKS = P_WERKS .
  MODIFY LT_BATCH FROM LS_BATCH TRANSPORTING WERKS WHERE WERKS = '' .
  call function 'ZG_GET_SALES_PRICE_ITAB'
    TABLES
      IT_BATCH                             = LT_BATCH
      ET_BATCH_PRICE                       = LT_BATCH_PRICE
    EXCEPTIONS
      ITEM_WITHOUT_WERKS                   = 1
      CANNOT_FIND_IN_ZMMRP0210_BATCH       = 2
      OTHERS                               = 3 .
  IF SY-SUBRC = 0.
    SORT LT_BATCH_PRICE BY MATNR CHARG .
    SORT LT_MATNR BY MATNR .
    LOOP AT GT_ALV ASSIGNING FIELD-SYMBOL(<FS_ALV>) .
      READ TABLE LT_BATCH_PRICE INTO LS_BATCH_PRICE WITH KEY MATNR = <FS_ALV>-MATNR
                                                             CHARG = <FS_ALV>-CHARG
                                                             BINARY SEARCH .
      IF SY-SUBRC = 0 .
        <FS_ALV>-EX_SITE  = LS_BATCH_PRICE-WERKS .
        <FS_ALV>-EXTWG    = LS_BATCH_PRICE-EXTWG .
        <FS_ALV>-RAUBE    = LS_BATCH_PRICE-RAUBE .
        <FS_ALV>-KBETR_JJ = LS_BATCH_PRICE-KBETR_JJ .
        <FS_ALV>-KBETR_GF = LS_BATCH_PRICE-KBETR_GF .
        <FS_ALV>-KBETR    = LS_BATCH_PRICE-KBETR_KDJ .
        <FS_ALV>-PRICE    = LS_BATCH_PRICE-KBETR_MJJG .
      ENDIF.

      READ TABLE LT_MATNR INTO LS_MATNR WITH KEY MATNR = <FS_ALV>-MATNR BINARY SEARCH .
      IF SY-SUBRC = 0 .
        <FS_ALV>-TEMPB    = LS_MATNR-TEMPB .
      ENDIF.
    ENDLOOP.
  ENDIF.
ENDFORM.
