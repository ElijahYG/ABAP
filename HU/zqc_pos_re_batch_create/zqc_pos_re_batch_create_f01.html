FORM FRM_CREATE_BATCH TABLES IT_BATCH STRUCTURE ZQC_BATCH

  SELECT MATNR CHARG

  LOOP AT IT_BATCH ASSIGNING <LS_BATCH> .

      EXPORTING
        INPUT  = <LS_BATCH>-ZGYSDM
      IMPORTING
        OUTPUT = <LS_BATCH>-ZGYSDM.

    READ TABLE LT_BATCH_EXIST INTO LS_BATCH_EXIST WITH KEY MATNR = <LS_BATCH>-MATNR
                                                           CHARG = <LS_BATCH>-CHARG BINARY SEARCH.
    IF SY-SUBRC = 0 .
      APPEND <LS_BATCH> TO LT_BATCH_UPDATE .  " 已经存在，
    ELSE.
      APPEND <LS_BATCH> TO LT_BATCH_CREATE .  " 还不存在
    ENDIF.
  ENDLOOP.

  REFRESH IT_BATCH .
  CLEAR LV_NUM_ERR .

  IF LT_BATCH_UPDATE[] IS NOT INITIAL .
*    PERFORM FRM_UPD_BATCH TABLES LT_BATCH_UPDATE USING LV_NUM_ERR .

FORM FRM_UPD_BATCH TABLES IT_BATCH STRUCTURE ZQC_BATCH

  CLEAR: LV_FROM, LV_TO,P_NUM_ERR .

  DO LV_TIMES TIMES.
    LV_FROM = LV_MAX * ( SY-INDEX - 1 ) + 1 . " 本次循环从内表IT_BATCH的哪一行开始

    IF SY-INDEX = LV_TIMES . " 最后一次
      LV_TO = LV_LINES . " 本次循环从内表IT_BATCH的哪一行结束
    ELSE.
      LV_TO = LV_MAX * SY-INDEX . " 本次循环从内表IT_BATCH的哪一行结束
    ENDIF.

    REFRESH: LT_BATCH_CHANGE,LT_RETURN .
    LOOP AT IT_BATCH INTO LS_BATCH FROM LV_FROM TO LV_TO .
      CLEAR LS_BATCH_CHANGE .
      MOVE-CORRESPONDING LS_BATCH TO LS_BATCH_CHANGE .
*      LS_BATCH_CHANGE-ZCGGFJF = LS_BATCH-ZCGKFJF . " 采购克附加费
      LS_BATCH_CHANGE-ZCGGFJF = LS_BATCH-ZCGGFJF . " 采购克附加费
      LS_BATCH_CHANGE-ZJSKDKZ = LS_BATCH-ZJSDKZ . " 结算倒扣值
      LS_BATCH_CHANGE-ZHXM    = SY-TABIX .
      APPEND LS_BATCH_CHANGE TO LT_BATCH_CHANGE .
    ENDLOOP.

    IF LT_BATCH_CHANGE[] IS NOT INITIAL .
      call function 'ZMM_BATCH_CHANGE'
        EXPORTING
*         IM_NUMBER      =
          IM_CREATE_IDOC = P_IDOC
*       IMPORTING
*         O_RETURN       =
        TABLES
          T_DATA         = LT_BATCH_CHANGE
          T_RETURN       = LT_RETURN.

      LOOP AT IT_BATCH ASSIGNING <LS_BATCH> FROM LV_FROM TO LV_TO .
        LV_TABIX = SY-TABIX .
        READ TABLE LT_RETURN INTO LS_RETURN WITH KEY ROW  = LV_TABIX
                                                     TYPE = 'E'.
        IF SY-SUBRC = 0.
          LOOP AT LT_RETURN INTO LS_RETURN WHERE ROW  = LV_TABIX
                                             AND TYPE       = 'E' .
            <LS_BATCH>-ERR_MSG = <LS_BATCH>-ERR_MSG && LS_RETURN-MESSAGE && ';' .
          ENDLOOP.
          <LS_BATCH>-ZZHAS_IMPORT = 'E' .
          P_NUM_ERR = P_NUM_ERR + 1 .
        ELSE.
          <LS_BATCH>-ZZHAS_IMPORT = 'S' .
        ENDIF.
      ENDLOOP.
    ENDIF.
  ENDDO.
ENDFORM.

FORM FRM_CRE_BATCH TABLES IT_BATCH STRUCTURE ZQC_BATCH
                    USING P_NUM_ERR TYPE I .
  DATA: LS_RETURN       TYPE BAPIRET2,
        LT_RETURN       LIKE TABLE OF LS_RETURN,
        LV_NUM_ERR      TYPE I,
        LT_BATCH_UPDATE LIKE TABLE OF ZQC_BATCH.
  DATA: LV_BAPIBATCHCTRL TYPE BAPIBATCHCTRL,
        LV_STR           TYPE STRING.
  FIELD-SYMBOLS <LS_BATCH> LIKE LINE OF IT_BATCH .

  DATA: LV_OBJECTKEY   TYPE  BAPI1003_KEY-OBJECT,
        LV_OBJECTTABLE TYPE  BAPI1003_KEY-OBJECTTABLE,
        LV_CLASSNUM    TYPE  BAPI1003_KEY-CLASSNUM,
        LV_CLASSTYPE   TYPE  BAPI1003_KEY-CLASSTYPE.

  IF IT_BATCH[] IS INITIAL .
    RETURN .
  ENDIF.

  CLEAR: LV_NUM_ERR .
  REFRESH: LT_BATCH_UPDATE .
  LV_BAPIBATCHCTRL-CLASS_NUM = 'Z_CLASS_BATCH_XQ' .
  LV_BAPIBATCHCTRL-DOCLASSIFY = 'X' .
  LOOP AT IT_BATCH ASSIGNING <LS_BATCH> .
    REFRESH: LT_RETURN .
    CLEAR: LS_RETURN .
    CALL FUNCTION 'BAPI_BATCH_CREATE'
      EXPORTING
        MATERIAL           = <LS_BATCH>-MATNR
        BATCH              = <LS_BATCH>-CHARG
*       PLANT              =
*       BATCHATTRIBUTES    =
        BATCHCONTROLFIELDS = LV_BAPIBATCHCTRL
*       BATCHSTORAGELOCATION       =
*       INTERNALNUMBERCOM  =
*       EXTENSION1         =
*       MATERIAL_EVG       =
*     IMPORTING
*       BATCH              =
*       BATCHATTRIBUTES    =
      TABLES
        RETURN             = LT_RETURN.
    READ TABLE LT_RETURN INTO LS_RETURN WITH KEY TYPE = 'E' .
    IF SY-SUBRC = 0 .
      CALL FUNCTION 'BAPI_TRANSACTION_ROLLBACK' .
      LV_NUM_ERR = LV_NUM_ERR + 1 .
      <LS_BATCH>-ZZHAS_IMPORT = 'E' .
      LOOP AT LT_RETURN INTO LS_RETURN WHERE TYPE = 'E' .
        CLEAR LV_STR .
        MESSAGE ID LS_RETURN-ID TYPE LS_RETURN-TYPE NUMBER LS_RETURN-NUMBER
                INTO LV_STR
                WITH LS_RETURN-MESSAGE_V1
                     LS_RETURN-MESSAGE_V2
                     LS_RETURN-MESSAGE_V3
                     LS_RETURN-MESSAGE_V4 .
        <LS_BATCH>-ERR_MSG = <LS_BATCH>-ERR_MSG && LV_STR && ';' .
      ENDLOOP.
    ELSE.
      CALL FUNCTION 'BAPI_TRANSACTION_COMMIT' .
      APPEND <LS_BATCH> TO LT_BATCH_UPDATE .
      <LS_BATCH>-MATNR = '' .
    ENDIF .
  ENDLOOP.

  DELETE IT_BATCH WHERE MATNR = '' .
  P_NUM_ERR = LV_NUM_ERR .

  IF LT_BATCH_UPDATE[] IS NOT INITIAL .
    WAIT UP TO 3 SECONDS .
    CLEAR LV_NUM_ERR .
    PERFORM FRM_UPD_BATCH TABLES LT_BATCH_UPDATE USING LV_NUM_ERR .
    P_NUM_ERR = P_NUM_ERR + LV_NUM_ERR .
    APPEND LINES OF LT_BATCH_UPDATE TO IT_BATCH .
  ENDIF.

ENDFORM.
