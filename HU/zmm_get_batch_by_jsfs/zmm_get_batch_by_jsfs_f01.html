*----------------------------------------------------------------------*
FORM FRM_GET_BATCH .
  DATA: LV_ATINN_ZSPTM TYPE ATINN .

  IF S_ZSPTM[] IS INITIAL .
    SELECT A~MATNR, A~CHARG
      INTO TABLE @GT_ZMM_CHARG
      FROM ZMMRP0210_BATCH AS A
      LEFT OUTER JOIN ZMMDG0010_JSFS AS B ON B~BM = A~ZLYJSFS
      WHERE A~MATNR IN @S_MATNR
        AND A~CHARG IN @S_CHARG
        AND A~ZLYJSFS IN @S_ZJSFS
        AND A~ZXSJGJLDW IN @S_ZXSDW
        AND A~ZXSKGFD  IN @S_ZXSGF
        AND A~ZGYSDM IN @S_ZGYSD
        AND B~SFLY IN @S_ZSFLY .
  ELSE.
    CALL FUNCTION 'CONVERSION_EXIT_ATINN_INPUT'
      EXPORTING
        INPUT  = 'ZSPTM'
      IMPORTING
        OUTPUT = LV_ATINN_ZSPTM.

    SELECT A~MATNR, A~CHARG
      INTO TABLE @GT_ZMM_CHARG
      FROM ZMMRP0210_BATCH AS A
      INNER JOIN MCH1 AS C ON  C~MATNR = A~MATNR
                           AND C~CHARG = A~CHARG
      INNER JOIN AUSP AS P ON  P~OBJEK = C~CUOBJ_BM
                           AND P~ATINN = @LV_ATINN_ZSPTM
      LEFT OUTER JOIN ZMMDG0010_JSFS AS B ON B~BM = A~ZLYJSFS
      WHERE A~MATNR IN @S_MATNR
        AND A~CHARG IN @S_CHARG
        AND A~ZLYJSFS IN @S_ZJSFS
        AND A~ZXSJGJLDW IN @S_ZXSDW
        AND A~ZXSKGFD  IN @S_ZXSGF
        AND A~ZGYSDM IN @S_ZGYSD
        AND P~ATWRT  IN @S_ZSPTM
        AND B~SFLY IN @S_ZSFLY .
  ENDIF.

  IF GT_ZMM_CHARG[] IS NOT INITIAL .
    PERFORM FRM_GET_MATNR .
    PERFORM FRM_GET_ZB00 .
    PERFORM FRM_GET_ZTJJ .
    PERFORM FRM_GET_ZKSG . " 注意在这个之前要运行PERFORM FRM_GET_ZTJJ
  ENDIF.

ENDFORM.

FORM FRM_GET_MATNR .

  MOVE-CORRESPONDING GT_ZMM_CHARG TO GT_MATNR .
  SORT GT_MATNR BY MATNR .
  DELETE ADJACENT DUPLICATES FROM GT_MATNR COMPARING MATNR .

  SELECT MATNR EXTWG RAUBE TEMPB
    INTO TABLE GT_MATNR
    FROM MARA
    FOR ALL ENTRIES IN GT_MATNR
    WHERE MATNR = GT_MATNR-MATNR .

ENDFORM.

*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM FRM_GET_ZKSG .

  DATA: LT_ZKG1 LIKE GT_ZKSG .

  SELECT A~ZXSKGFD P~KBETR
    INTO TABLE GT_ZKSG
    FROM A911 AS A
    INNER JOIN KONP AS P ON P~KNUMH = A~KNUMH
    WHERE A~KAPPL = 'V'
      AND A~KSCHL  = 'ZKSG'
      AND A~DATAB <= SY-DATUM
      AND A~DATBI >= SY-DATUM .

  " 取投资银产品加价百分比
  SELECT A~ZXSKGFD P~KBETR
    INTO TABLE LT_ZKG1
    FROM A911 AS A
    INNER JOIN KONP AS P ON P~KNUMH = A~KNUMH
    WHERE A~KAPPL = 'V'
      AND A~KSCHL  = 'ZKG1'
      AND A~DATAB <= SY-DATUM
      AND A~DATBI >= SY-DATUM .
  IF SY-SUBRC = 0 .
    " 将加价百分比转换为加价金额
    SORT GT_ZTJJ BY EXTWG RAUBE EX_GJS .
    LOOP AT LT_ZKG1 INTO DATA(LS_ZKG1) .
      READ TABLE GT_ZTJJ INTO GS_ZTJJ WITH KEY EXTWG = 'D04'
                                               RAUBE = '19'
                                               EX_GJS = '403'
                                               BINARY SEARCH .
      IF SY-SUBRC = 0 .
        CLEAR: GS_ZKSG .
        GS_ZKSG-ZXSKGFD = LS_ZKG1-ZXSKGFD .
        GS_ZKSG-KBETR = LS_ZKG1-KBETR * GS_ZTJJ-KBETR / 1000 .  "将百分比乘以基础银价计算出克工费加价金额
        APPEND GS_ZKSG TO GT_ZKSG .
      ENDIF.
    ENDLOOP.
  ENDIF.
ENDFORM.

FORM FRM_GET_CHARACTERS .

  DATA: LV_LINE_TOTAL TYPE I,
        LV_LINE_FROM  TYPE I,
        LV_LINE_TO    TYPE I,
        LV_LINE_MAX   TYPE I,
        LV_TIMES      TYPE I.

  DATA: LT_ZMM_CHARACTERS TYPE TABLE OF ZMM_CHARACTERS,
        LT_ZMM_CHARG      TYPE TABLE OF ZMM_CHARG,
        LT_ALV            LIKE GT_ALV.

  SORT GT_ZMM_CHARG BY MATNR CHARG .
  LV_LINE_TOTAL = LINES( GT_ZMM_CHARG ) .
  LV_LINE_MAX = 10000 .
  LV_TIMES = LV_LINE_TOTAL DIV LV_LINE_MAX + 1 .
  DO LV_TIMES TIMES .
    REFRESH: LT_ZMM_CHARG, LT_ZMM_CHARACTERS, LT_ALV .
    LV_LINE_FROM = LV_LINE_MAX * ( SY-INDEX - 1 ) + 1 . " 本次循环从内表的哪一行开始
    IF SY-INDEX = LV_TIMES . " 最后一次
      LV_LINE_TO = LV_LINE_TOTAL . " 本次循环到内表的哪一行结束
    ELSE.
      LV_LINE_TO = LV_LINE_MAX * SY-INDEX . " 本次循环到内表的哪一行结束
    ENDIF.

    APPEND LINES OF GT_ZMM_CHARG FROM LV_LINE_FROM TO LV_LINE_TO TO LT_ZMM_CHARG .

    call function 'ZSDRP0020_FM_01'
      TABLES
        T_CHARG      = LT_ZMM_CHARG
        T_CHARACTERS = LT_ZMM_CHARACTERS.
    MOVE-CORRESPONDING LT_ZMM_CHARACTERS TO LT_ALV .
    APPEND LINES OF LT_ALV TO GT_ALV .
  ENDDO.
  REFRESH: LT_ZMM_CHARG, LT_ZMM_CHARACTERS, LT_ALV .

  SORT GT_ALV BY MATNR CHARG .

ENDFORM.

*&---------------------------------------------------------------------*
*&      Form  FRM_PROCESS_DATA
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM FRM_PROCESS_DATA .
  SORT GT_MATNR BY MATNR .
  SORT GT_ZKSG BY ZXSKGFD .
  SORT GT_ZB00 BY EX_SITE EXTWG RAUBE EX_GJS .
  SORT GT_ZTJJ BY EXTWG RAUBE EX_GJS .
  LOOP AT GT_ALV ASSIGNING FIELD-SYMBOL(<GS_ALV>) .
    READ TABLE GT_MATNR INTO GS_MATNR WITH KEY MATNR = <GS_ALV>-MATNR BINARY SEARCH .
    IF SY-SUBRC = 0 .
      <GS_ALV>-EXTWG = GS_MATNR-EXTWG .
      <GS_ALV>-RAUBE = GS_MATNR-RAUBE .
      <GS_ALV>-TEMPB = GS_MATNR-TEMPB .
    ENDIF.

    IF <GS_ALV>-ZXSJGJLDW = 'G' .
      " 按克销售的商品
      " 取投资商品的金价
      IF <GS_ALV>-EXTWG = 'D04'.
        READ TABLE GT_ZTJJ INTO GS_ZTJJ WITH KEY EXTWG = <GS_ALV>-EXTWG
                                                 RAUBE = <GS_ALV>-RAUBE
                                                 EX_GJS = <GS_ALV>-ZGJSCZ
                                                 BINARY SEARCH .
        IF SY-SUBRC = 0 .
          <GS_ALV>-KBETR_JJ = GS_ZTJJ-KBETR .
        ENDIF.
      ELSE.
        " 取一般贵金属商品的金价
        READ TABLE GT_ZB00 INTO GS_ZB00 WITH KEY EX_SITE = P_WERKS
                                                 EXTWG   = <GS_ALV>-EXTWG
                                                 RAUBE   = <GS_ALV>-RAUBE
                                                 EX_GJS  = <GS_ALV>-ZGJSCZ
                                                 BINARY SEARCH .
        IF SY-SUBRC = 0 .
          <GS_ALV>-KBETR_JJ = GS_ZB00-KBETR .
        ENDIF.
      ENDIF.

      READ TABLE GT_ZKSG INTO GS_ZKSG WITH KEY ZXSKGFD = <GS_ALV>-ZXSKGFD BINARY SEARCH .
      IF SY-SUBRC = 0 .
        <GS_ALV>-KBETR_GF = GS_ZKSG-KBETR .
      ENDIF.
      " 克单价 = 克金价 + 克工费 + 克附加费
      <GS_ALV>-KBETR = <GS_ALV>-KBETR_JJ + <GS_ALV>-KBETR_GF + <GS_ALV>-ZXSKFJGF .
      " 每件售价 = 克单价 * 克金重 ，并舍分
      <GS_ALV>-PRICE = <GS_ALV>-KBETR * <GS_ALV>-ZJZ .
      <GS_ALV>-PRICE = <GS_ALV>-PRICE * 10 DIV 1 / 10 . " 舍分
    ELSEIF <GS_ALV>-ZXSJGJLDW = 'PC' .
      <GS_ALV>-PRICE = <GS_ALV>-ZCKLSJ .
    ENDIF .

  ENDLOOP.
ENDFORM.
