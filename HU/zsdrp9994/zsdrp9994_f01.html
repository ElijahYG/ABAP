FORM FRM_GET_DATA .
  DATA: BEGIN OF LS_VBRP_YJHX ,
          FKDAT       TYPE VBRK-FKDAT,
          EX_POSNUM   TYPE VBRK-EX_POSNUM,
          EX_POSTYP   TYPE VBRK-EX_POSTYP,
          VBELN       TYPE VBRP-VBELN,
          WERKS       TYPE VBRP-WERKS,
          ZLINES      TYPE I,
          ZXSJS       TYPE ZDE_ZXSJS,
          ZXSKZ       TYPE ZDE_ZXSKZ,
          KZWI1       TYPE VBRP-KZWI1,  " 超重收款
          KZWI3       TYPE VBRP-KZWI3,  " 超重收款（扣除商场扣点）
          KZWI2       TYPE VBRP-KZWI2,  " 销售成本
          KZWI6       TYPE VBRP-KZWI6,  " 商品售价
          KZWI1_NET   TYPE ATFLV ,  " 超重收款(不含税)
          KZWI3_NET   TYPE ATFLV,  " 超重收款（扣除商场扣点，不含税）
          KZWI6_NET   TYPE ATFLV,  " 商品售价 (不含税)
          VAL_SXF     TYPE VBRP-KZWI3, " 以旧换新手续费
          VAL_SXF_NET TYPE ATFLV, " 以旧换新手续费
        END OF LS_VBRP_YJHX,
        LT_VBRP_YJHX LIKE TABLE OF LS_VBRP_YJHX.
  DATA: LS_VBRP_SXF LIKE LS_VBRP_YJHX,
        LT_VBRP_SXF LIKE TABLE OF LS_VBRP_SXF.
  DATA: LV_MATNR_SXF TYPE MATNR VALUE 'Y59600004' .
  DATA: BEGIN OF LS_JLRK,
          EX_POSNUM TYPE MKPF-BKTXT,
          FKDAT     TYPE MKPF-BUDAT,
          MBLNR     TYPE MKPF-MBLNR,
          MJAHR     TYPE MKPF-MJAHR,
          WERKS     TYPE MSEG-WERKS,
          MENGE     TYPE MSEG-MENGE,
          DMBTR     TYPE MSEG-DMBTR,
        END OF LS_JLRK,
        LT_JLRK LIKE TABLE OF LS_JLRK.
  DATA: BEGIN OF LS_WERKS,
          WERKS  TYPE T001W-WERKS,
          NAME1  TYPE T001W-NAME1,
          EX_POS TYPE ZT001W-EX_POS,
        END OF LS_WERKS,
        LT_WERKS LIKE TABLE OF LS_WERKS.

  " 取以旧换新销售小票
*        KZWI6 TYPE VBRP-KZWI6,  " 商品售价
*        KZWI1 TYPE VBRP-KZWI1,  " 超重收款
*        KZWI3 TYPE VBRP-KZWI3,  " 超重收款（扣除商场扣点）
  SELECT K~FKDAT, K~EX_POSNUM, K~EX_POSTYP,
         P~VBELN, P~WERKS,
         SUM( CASE WHEN SHKZG = 'X'  THEN P~ZXSJS * -1
                   WHEN SHKZG <> 'X' THEN P~ZXSJS END ) AS ZXSJS,
         SUM( CASE WHEN SHKZG = 'X'  THEN P~ZXSKZ * -1
                   WHEN SHKZG <> 'X' THEN P~ZXSKZ END ) AS ZXSKZ,
         COUNT( DISTINCT P~POSNR ) AS ZLINES,
         SUM( CASE WHEN SHKZG = 'X'  THEN P~KZWI1 * -1
                   WHEN SHKZG <> 'X' THEN P~KZWI1 END ) AS KZWI1,
         SUM( CASE WHEN SHKZG = 'X'  THEN P~KZWI2 * -1
                   WHEN SHKZG <> 'X' THEN P~KZWI2 END ) AS KZWI2,
         SUM( CASE WHEN SHKZG = 'X'  THEN P~KZWI3 * -1
                   WHEN SHKZG <> 'X' THEN P~KZWI3 END ) AS KZWI3,
         SUM( CASE WHEN SHKZG = 'X'  THEN P~KZWI6 * -1
                   WHEN SHKZG <> 'X' THEN P~KZWI6 END ) AS KZWI6,
         SUM( CASE WHEN SHKZG = 'X'  AND P~NETWR <> 0 THEN CAST( P~KZWI1 AS FLTP ) / ( CAST( 1 AS FLTP ) + CAST( P~MWSBP AS FLTP )  / CAST( P~NETWR AS FLTP ) ) * CAST( -1 AS FLTP )
                   WHEN SHKZG = 'X'  AND P~NETWR = 0  THEN CAST( P~KZWI1 AS FLTP )  * CAST( -1 AS FLTP )
                   WHEN SHKZG <> 'X' AND P~NETWR <> 0 THEN CAST( P~KZWI1 AS FLTP ) / ( CAST( 1 AS FLTP ) + CAST( P~MWSBP AS FLTP )  / CAST( P~NETWR AS FLTP ) )
                   WHEN SHKZG <> 'X' AND P~NETWR = 0  THEN CAST( P~KZWI1 AS FLTP ) END ) AS KZWI1_NET,
         SUM( CASE WHEN SHKZG = 'X'  AND P~NETWR <> 0 THEN CAST( P~KZWI3 AS FLTP ) / ( CAST( 1 AS FLTP ) + CAST( P~MWSBP AS FLTP )  / CAST( P~NETWR AS FLTP ) ) * CAST( -1 AS FLTP )
                   WHEN SHKZG = 'X'  AND P~NETWR = 0  THEN CAST( P~KZWI3 AS FLTP )  * CAST( -1 AS FLTP )
                   WHEN SHKZG <> 'X' AND P~NETWR <> 0 THEN CAST( P~KZWI3 AS FLTP ) / ( CAST( 1 AS FLTP ) + CAST( P~MWSBP AS FLTP )  / CAST( P~NETWR AS FLTP ) )
                   WHEN SHKZG <> 'X' AND P~NETWR = 0  THEN CAST( P~KZWI3 AS FLTP ) END ) AS KZWI3_NET,
         SUM( CASE WHEN SHKZG = 'X'  AND P~NETWR <> 0 THEN CAST( P~KZWI6 AS FLTP ) / ( CAST( 1 AS FLTP ) + CAST( P~MWSBP AS FLTP )  / CAST( P~NETWR AS FLTP ) ) * CAST( -1 AS FLTP )
                   WHEN SHKZG = 'X'  AND P~NETWR = 0  THEN CAST( P~KZWI6 AS FLTP ) * CAST( -1 AS FLTP )
                   WHEN SHKZG <> 'X' AND P~NETWR <> 0 THEN CAST( P~KZWI6 AS FLTP ) / ( CAST( 1 AS FLTP ) + CAST( P~MWSBP AS FLTP )  / CAST( P~NETWR AS FLTP ) )
                   WHEN SHKZG <> 'X' AND P~NETWR =  0 THEN CAST( P~KZWI6 AS FLTP ) END ) AS KZWI6_NET
    INTO CORRESPONDING FIELDS OF TABLE @LT_VBRP_YJHX
    FROM VBRK AS K
    INNER JOIN VBRP AS P ON P~VBELN = K~VBELN
    WHERE K~FKDAT IN @S_FKDAT
      AND K~EX_POSNUM IN @S_P_YJHX
      AND K~FKART = 'FP'
      AND K~EX_POSTYP IN ('20', '21')
      AND K~VBELN IN @S_B_YJHX
      AND P~WERKS IN @S_WERKS
    GROUP BY K~FKDAT, K~EX_POSNUM, K~EX_POSTYP,P~VBELN,P~WERKS .

  "取旧饰入库商品凭证
  SELECT K~BKTXT AS EX_POSNUM, K~BUDAT AS FKDAT, K~MBLNR AS BMLNR, K~MJAHR AS MJAHR, G~WERKS,
         SUM( CASE WHEN G~SHKZG = 'H' THEN G~MENGE * -1
                   WHEN G~SHKZG = 'S' THEN G~MENGE END ) AS MENGE,
         SUM( CASE WHEN G~SHKZG = 'H' THEN G~DMBTR * -1
                   WHEN G~SHKZG = 'S' THEN G~DMBTR END ) AS DMBTR
    INTO TABLE @LT_JLRK
    FROM MKPF AS K
    INNER JOIN MSEG AS G ON  G~MBLNR = K~MBLNR
                         AND G~MJAHR = K~MJAHR
    WHERE K~BUDAT IN @S_FKDAT
      AND G~WERKS IN @S_WERKS
      AND G~BWART IN ('Z56','Z57')
    GROUP BY K~BKTXT, K~BUDAT, K~MBLNR, K~MJAHR,G~WERKS .

  "取以旧换新手续费POS小票
  SELECT K~FKDAT, K~EX_POSNUM, K~EX_POSTYP,
         P~VBELN, P~WERKS,
         SUM( CASE WHEN SHKZG = 'X' THEN P~KZWI3 * -1
                   WHEN SHKZG <> 'X' THEN P~KZWI3 END ) AS VAL_SXF,
         SUM( CASE WHEN SHKZG = 'X'  THEN CAST( P~KZWI3 AS FLTP ) / ( CAST( 1 AS FLTP ) + CAST( P~MWSBP AS FLTP )  / CAST( P~NETWR AS FLTP ) ) * CAST( -1 AS FLTP )
                   WHEN SHKZG <> 'X' THEN CAST( P~KZWI3 AS FLTP ) / ( CAST( 1 AS FLTP ) + CAST( P~MWSBP AS FLTP )  / CAST( P~NETWR AS FLTP ) )  END ) AS VAL_SXF_NET
   INTO CORRESPONDING FIELDS OF TABLE @LT_VBRP_SXF
   FROM VBRK AS K
   INNER JOIN VBRP AS P ON P~VBELN = K~VBELN
   WHERE K~FKDAT IN @S_FKDAT
     AND K~EX_POSNUM IN @S_P_SXF
     AND K~FKART = 'FP'
     AND K~EX_POSTYP IN ('11', '12')
     AND K~VBELN IN @S_B_SXF
     AND P~MATNR = @LV_MATNR_SXF
     AND P~WERKS IN @S_WERKS
   GROUP BY K~FKDAT, K~EX_POSNUM, K~EX_POSTYP,P~VBELN,P~WERKS .

* 取得地点名称
  SELECT W~WERKS, W~NAME1,Z~EX_POS
    INTO TABLE @LT_WERKS
    FROM T001W AS W
    INNER JOIN ZT001W AS Z  ON Z~WERKS = W~WERKS .

  SORT LT_VBRP_YJHX BY VBELN .
  SORT LT_JLRK BY EX_POSNUM WERKS FKDAT .
  SORT LT_WERKS BY WERKS .
  LOOP AT LT_VBRP_YJHX INTO LS_VBRP_YJHX.
    CLEAR: GS_ALV .

    MOVE-CORRESPONDING LS_VBRP_YJHX TO GS_ALV .
    IF LS_VBRP_YJHX-EX_POSTYP = '20' .
      GS_ALV-ZYWLX = '2-以旧换新退货' .
    ELSE.
      GS_ALV-ZYWLX = '1-以旧换新销售' .
    ENDIF.

    READ TABLE LT_JLRK INTO LS_JLRK WITH KEY EX_POSNUM = LS_VBRP_YJHX-EX_POSNUM
                                             WERKS     = LS_VBRP_YJHX-WERKS
                                             FKDAT     = LS_VBRP_YJHX-FKDAT
                                             BINARY SEARCH .
    IF SY-SUBRC = 0 .
      GS_ALV-MBLNR = LS_JLRK-MBLNR .
      GS_ALV-MJAHR = LS_JLRK-MJAHR .
      GS_ALV-MENGE = LS_JLRK-MENGE .
      GS_ALV-DMBTR = LS_JLRK-DMBTR .
    ENDIF.

    "原毛利额和毛利率
    GS_ALV-GP_VAL_ORIG = ABS( GS_ALV-KZWI6_NET ) - ABS( GS_ALV-KZWI2 ) .
    IF ABS( GS_ALV-KZWI6_NET ) > 0 .
*      GS_ALV-GP_PER_ORIG = GS_ALV-GP_VAL_ORIG / ABS( GS_ALV-KZWI2 ) * 100.
       GS_ALV-GP_PER_ORIG = GS_ALV-GP_VAL_ORIG / ABS( GS_ALV-KZWI6_NET ) * 100.
    ELSE.
      IF GS_ALV-GP_VAL_ORIG = 0 .
        GS_ALV-GP_PER_ORIG = 0 .
      ELSE.
        GS_ALV-GP_PER_ORIG = '99999.99' .
      ENDIF.
    ENDIF.
    "现毛利额和毛利率
    GS_ALV-GP_VAL_CURR = ABS( GS_ALV-KZWI1_NET ) - ABS( GS_ALV-KZWI2 ) + ABS( GS_ALV-DMBTR ) .
    IF ABS( GS_ALV-KZWI1_NET ) > 0 .
*      GS_ALV-GP_PER_CURR = GS_ALV-GP_VAL_CURR / ( ABS( GS_ALV-KZWI2 ) - ABS( GS_ALV-DMBTR ) )  * 100 .
      GS_ALV-GP_PER_CURR = GS_ALV-GP_VAL_CURR /  ABS( GS_ALV-KZWI1_NET )   * 100 .
    ELSE.
      IF GS_ALV-GP_VAL_CURR = 0 .
        GS_ALV-GP_PER_CURR = 0 .
      ELSE.
        GS_ALV-GP_PER_CURR = '99999.99' .
      ENDIF.
    ENDIF.
    "毛利率差
    " 如果现毛利率不等于0（等于0是收款完全低值），也不等于99999.99（等于99999.99是成本完全低值），
    " 且原、现毛利率差在0.05%的，将毛利率差填写到GP_PER_DIFF字段
    IF GS_ALV-GP_PER_CURR <> 0 AND GS_ALV-GP_PER_CURR <> '99999.99' .
      GS_ALV-GP_PER_DIFF = ABS( GS_ALV-GP_PER_ORIG - GS_ALV-GP_PER_CURR ) .
      IF GS_ALV-GP_PER_DIFF < '0.05' .
        GS_ALV-GP_PER_DIFF = 0 .
      ENDIF.
    ENDIF.
    "现毛利额和毛利率，使用超重收款（扣除商场扣点，不含税）
    GS_ALV-GP_VAL_CURR_KD = ABS( GS_ALV-KZWI3_NET ) - ABS( GS_ALV-KZWI2 ) + ABS( GS_ALV-DMBTR ) .
    IF ABS( GS_ALV-KZWI3_NET ) > 0 .
      GS_ALV-GP_PER_CURR_KD = GS_ALV-GP_VAL_CURR_KD / ABS( GS_ALV-KZWI3_NET )   * 100 .
    ELSE.
      IF GS_ALV-GP_VAL_CURR_KD = 0 .
        GS_ALV-GP_PER_CURR_KD = 0 .
      ELSE.
        GS_ALV-GP_PER_CURR_KD = '99999.99' .
      ENDIF.
    ENDIF.

    READ TABLE LT_WERKS INTO LS_WERKS WITH KEY WERKS = GS_ALV-WERKS BINARY SEARCH .
    IF SY-SUBRC = 0 .
      GS_ALV-NAME1 = LS_WERKS-NAME1 .
      IF LS_WERKS-EX_POS = '10' .
        GS_ALV-ZKD = '是' .
      ENDIF.
    ENDIF.

    APPEND GS_ALV TO GT_ALV .
  ENDLOOP.

  LOOP AT LT_VBRP_SXF INTO LS_VBRP_SXF .
    CLEAR: GS_ALV .
    MOVE-CORRESPONDING LS_VBRP_SXF TO GS_ALV .
    GS_ALV-ZYWLX = '3-以旧换新手续费' .
    READ TABLE LT_WERKS INTO LS_WERKS WITH KEY WERKS = GS_ALV-WERKS BINARY SEARCH .
    IF SY-SUBRC = 0 .
      GS_ALV-NAME1 = LS_WERKS-NAME1 .
      IF LS_WERKS-EX_POS = '10' .
        GS_ALV-ZKD = '是' .
      ENDIF.
    ENDIF.
    APPEND GS_ALV TO GT_ALV .
  ENDLOOP.

  SORT GT_ALV BY FKDAT WERKS ZYWLX EX_POSNUM VBELN .

ENDFORM.
