  GV_ATZHL = '1' .
  GV_MAFID = 'O' .
  GV_KLART = '023' .
  GV_ADZHL = '' .

  CALL FUNCTION 'CONVERSION_EXIT_ATINN_INPUT'
    EXPORTING
      INPUT  = 'ZSPTM'
    IMPORTING
      OUTPUT = GV_ATINN.

  SELECT M~MATNR M~CHARG
    INTO TABLE GT_CHARG
    FROM AUSP AS A
    INNER JOIN MCH1 AS M ON A~OBJEK = M~CUOBJ_BM
    WHERE A~ATINN = GV_ATINN
      AND A~ATZHL = GV_ATZHL
      AND A~MAFID = GV_MAFID
      AND A~KLART = GV_KLART
      AND A~ADZHL = GV_ADZHL
      AND A~ATWRT IN S_BCODE .
  IF SY-SUBRC <> 0 .
    MESSAGE S000(ZCB) WITH '没有找到对应的SAP批次！' DISPLAY LIKE 'E' .
    STOP .
  ELSE.
    call function 'ZSDRP0020_FM_01'
      TABLES
        T_CHARG      = GT_CHARG
        T_CHARACTERS = GT_BATCH.
  ENDIF.

  IF GT_BATCH[] IS NOT INITIAL .
    SORT GT_BATCH BY MATNR CHARG .
    GV_LINES = LINES( GT_BATCH ).
    MESSAGE S000(ZCB) WITH '找到' GV_LINES '条SAP批次！' .
    PERFORM FRM_DIS_ALV .
  ENDIF.
