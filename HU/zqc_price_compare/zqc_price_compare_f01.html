FORM FRM_GET_DATA_POS .
  SELECT ZSPTM AS ZSPTM,
         MATNR AS MATNR_POS,
         CHARG AS CHARG_POS,
         EXTWG AS EXTWG_POS,
         RAUBE AS RAUBE_POS,
         EX_GJS AS EX_GJS_POS,
         ZXSKGFD AS ZXSKGFD_POS,
         ZXSJGJLDW AS ZXSJGJLDW_POS,
         ZJZ AS ZJZ_POS,
         KBETR_GF AS KBETR_GF_POS,
         PRICE AS PRICE_POS
    INTO TABLE @GT_DATA_POS
    FROM ZQC_PRICE_COMPAR
    WHERE ZSYS = 'POS'
      AND ZSPTM IN @S_ZSPTM.
ENDFORM.

*&---------------------------------------------------------------------*
*&      Form  FRM_GET_DATA_SAP
*&---------------------------------------------------------------------*
*       取得SAP的数据
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM FRM_GET_DATA_SAP .
  DATA: LV_ATINN_ZSPTM  TYPE ATINN,
        LV_ATINN_ZCKLSJ TYPE ATINN.
  DATA: BEGIN OF LS_MATNR,
          MATNR TYPE MATNR,
        END OF LS_MATNR,
        LT_MATNR LIKE TABLE OF LS_MATNR.

  CALL FUNCTION 'CONVERSION_EXIT_ATINN_INPUT'
    EXPORTING
      INPUT  = 'ZSPTM'
    IMPORTING
      OUTPUT = LV_ATINN_ZSPTM.
  CALL FUNCTION 'CONVERSION_EXIT_ATINN_INPUT'
    EXPORTING
      INPUT  = 'ZCKLSJ'
    IMPORTING
      OUTPUT = LV_ATINN_ZCKLSJ.

  "取得批次的条码
  SELECT C~MATNR AS MATNR,
         C~CHARG AS CHARG,
         P~ATWRT AS ZSPTM
    INTO TABLE @DATA(LT_BATCH_ZSPTM)
    FROM MCH1 AS C
    INNER JOIN AUSP AS P ON P~OBJEK = C~CUOBJ_BM
    WHERE ATINN = @LV_ATINN_ZSPTM
      AND ATWRT IN @S_ZSPTM .
  IF SY-SUBRC = 0 .
    SORT LT_BATCH_ZSPTM BY MATNR CHARG .
  ELSE.
    RETURN .
  ENDIF.

  MOVE-CORRESPONDING LT_BATCH_ZSPTM TO LT_MATNR .
  SORT LT_MATNR BY MATNR .
  DELETE ADJACENT DUPLICATES FROM LT_MATNR COMPARING MATNR .

  SELECT Z~MATNR AS MATNR_SAP,
         Z~CHARG AS CHARG_SAP,
         A~EXTWG AS EXTWG_SAP,
         A~RAUBE AS RAUBE_SAP,
         Z~ZGJSCZ AS EX_GJS_SAP,
         Z~ZXSKGFD AS ZXSKGFD_SAP,
         Z~ZXSJGJLDW AS ZXSJGJLDW_SAP,
         Z~ZJZ AS ZJZ_SAP,
         Z~ZXSKFJGF AS ZXSKFJGF_SAP
    INTO CORRESPONDING FIELDS OF TABLE @GT_DATA_SAP
    FROM ZMMRP0210_BATCH AS Z
    INNER JOIN MARA AS A ON A~MATNR = Z~MATNR
    FOR ALL ENTRIES IN @LT_MATNR
    WHERE Z~MATNR = @LT_MATNR-MATNR .
  IF SY-SUBRC = 0 .
    SORT GT_DATA_SAP BY MATNR_SAP CHARG_SAP .
  ELSE.
    RETURN .
  ENDIF.

  " 取得批次的参考零售价
  SELECT C~MATNR AS MATNR,
         C~CHARG AS CHARG,
         P~ATFLV AS ZCKLSJ
    INTO TABLE @DATA(LT_BATCH_ZCKLSJ)
    FROM MCH1 AS C
    INNER JOIN AUSP AS P ON P~OBJEK = C~CUOBJ_BM
    FOR ALL ENTRIES IN @LT_MATNR
    WHERE C~MATNR = @LT_MATNR-MATNR
      AND P~ATINN = @LV_ATINN_ZCKLSJ .
  IF SY-SUBRC = 0 .
    SORT LT_BATCH_ZCKLSJ BY MATNR CHARG .
  ENDIF.

  " 取基础金价
  SELECT A~EXTWG, A~RAUBE, A~EX_GJS, P~KBETR
    INTO TABLE @DATA(LT_ZB00)
    FROM A909 AS A
    INNER JOIN KONP AS P ON P~KNUMH = A~KNUMH
    WHERE A~KAPPL = 'V'
      AND A~KSCHL = 'ZB00'
      AND A~EX_SITE = @P_WERKS
      AND A~DATAB <= @SY-DATUM
      AND A~DATBI >= @SY-DATUM .
  IF SY-SUBRC = 0 .
    SORT LT_ZB00 BY EXTWG RAUBE EX_GJS .
  ENDIF.

  " 取投资金价
  SELECT A~EXTWG, A~RAUBE, A~EX_GJS, P~KBETR
    INTO TABLE @DATA(LT_ZTJJ)
    FROM A908 AS A
    INNER JOIN KONP AS P ON P~KNUMH = A~KNUMH
    WHERE A~KAPPL = 'V'
      AND A~KSCHL = 'ZTJJ'
      AND A~DATAB <= @SY-DATUM
      AND A~DATBI >= @SY-DATUM .
  IF SY-SUBRC = 0 .
    SORT LT_ZTJJ BY EXTWG RAUBE EX_GJS .
  ENDIF.

  " 取克工费
  SELECT A~ZXSKGFD, P~KBETR
    INTO TABLE @DATA(LT_ZKSG)
    FROM A911 AS A
    INNER JOIN KONP AS P ON P~KNUMH = A~KNUMH
    WHERE A~KAPPL = 'V'
      AND A~KSCHL = 'ZKSG'
      AND A~DATAB <= @SY-DATUM
      AND A~DATBI >= @SY-DATUM .
  IF SY-SUBRC = 0 .
    SORT LT_ZKSG BY ZXSKGFD .
  ENDIF.

  " 取投资银加价率
  SELECT A~ZXSKGFD, P~KBETR
    INTO TABLE @DATA(LT_ZKG1)
    FROM A911 AS A
    INNER JOIN KONP AS P ON P~KNUMH = A~KNUMH
    WHERE A~KAPPL = 'V'
      AND A~KSCHL = 'ZKG1'
      AND A~DATAB <= @SY-DATUM
      AND A~DATBI >= @SY-DATUM .
  IF SY-SUBRC = 0 .
    " 将加价百分比转换成克加价金额
    DATA: LS_ZKSG LIKE LINE OF LT_ZKSG .
    LOOP AT LT_ZKG1 INTO DATA(LS_ZKG1) .
      READ TABLE LT_ZTJJ INTO DATA(LS_ZTJJ) WITH KEY EXTWG = 'D04'
                                                     RAUBE = '19'
                                                     EX_GJS = '403'
                                                     BINARY SEARCH .
      IF SY-SUBRC = 0 .
        CLEAR: LS_ZKSG .
        LS_ZKSG-ZXSKGFD = LS_ZKG1-ZXSKGFD .
        LS_ZKSG-KBETR = LS_ZKG1-KBETR * LS_ZTJJ-KBETR / 1000 .  "将百分比乘以基础银价计算出克工费加价金额
        APPEND LS_ZKSG TO LT_ZKSG .
      ENDIF.
    ENDLOOP.
    SORT LT_ZKSG BY ZXSKGFD .
  ENDIF.

  " 处理GT_DATA_SAP中各批次的金额
  LOOP AT GT_DATA_SAP ASSIGNING FIELD-SYMBOL(<DATA_SAP>) .

    READ TABLE LT_BATCH_ZSPTM INTO DATA(LS_BATCH_ZSPTM) WITH KEY MATNR = <DATA_SAP>-MATNR_SAP
                                                                 CHARG = <DATA_SAP>-CHARG_SAP
                                                                 BINARY SEARCH .
    IF SY-SUBRC = 0 .
      <DATA_SAP>-ZSPTM = LS_BATCH_ZSPTM-ZSPTM .
    ELSE.
      CONTINUE .
    ENDIF.

    IF <DATA_SAP>-ZXSJGJLDW_SAP = 'G' .
      " 按克销售的商品
      " 取投资商品的金价
      IF <DATA_SAP>-EXTWG_SAP = 'D04'.
        READ TABLE LT_ZTJJ INTO LS_ZTJJ WITH KEY EXTWG = <DATA_SAP>-EXTWG_SAP
                                                       RAUBE = <DATA_SAP>-RAUBE_SAP
                                                       EX_GJS = <DATA_SAP>-EX_GJS_SAP
                                                       BINARY SEARCH .
        IF SY-SUBRC = 0 .
          <DATA_SAP>-KBETR_JJ_SAP = LS_ZTJJ-KBETR .
        ENDIF.
      ELSE.
        " 取一般贵金属商品的金价
        READ TABLE LT_ZB00 INTO DATA(LS_ZB00) WITH KEY EXTWG = <DATA_SAP>-EXTWG_SAP
                                                       RAUBE = <DATA_SAP>-RAUBE_SAP
                                                       EX_GJS = <DATA_SAP>-EX_GJS_SAP
                                                       BINARY SEARCH .
        IF SY-SUBRC = 0 .
          <DATA_SAP>-KBETR_JJ_SAP = LS_ZB00-KBETR .
        ENDIF.
      ENDIF.
      " 取工费
      READ TABLE LT_ZKSG INTO LS_ZKSG WITH KEY ZXSKGFD = <DATA_SAP>-ZXSKGFD_SAP BINARY SEARCH .
      IF SY-SUBRC = 0 .
        <DATA_SAP>-KBETR_GF_SAP = LS_ZKSG-KBETR .
      ENDIF.
      " 克单价 = 克金价 + 克工费 + 克附加费
      <DATA_SAP>-KBETR_G_SAP = <DATA_SAP>-KBETR_JJ_SAP + <DATA_SAP>-KBETR_GF_SAP + <DATA_SAP>-ZXSKFJGF_SAP .
      " 每件售价 = 克单价 * 克金重 ，并舍分
      <DATA_SAP>-PRICE_SAP = <DATA_SAP>-KBETR_G_SAP * <DATA_SAP>-ZJZ_SAP .
      <DATA_SAP>-PRICE_SAP = <DATA_SAP>-PRICE_SAP * 10 DIV 1 / 10 .
    ELSEIF <DATA_SAP>-ZXSJGJLDW_SAP = 'PC' .
      " 按件销售的商品
      READ TABLE LT_BATCH_ZCKLSJ INTO DATA(LS_BATCH_ZCKLSJ) WITH KEY MATNR = <DATA_SAP>-MATNR_SAP
                                                               CHARG = <DATA_SAP>-CHARG_SAP
                                                               BINARY SEARCH .
      IF SY-SUBRC = 0 .
        <DATA_SAP>-PRICE_SAP = LS_BATCH_ZCKLSJ-ZCKLSJ .
      ENDIF.
    ENDIF.
    CLEAR: LS_BATCH_ZSPTM,LS_ZTJJ,LS_ZB00,LS_ZKSG,LS_BATCH_ZCKLSJ .
  ENDLOOP.

  DELETE GT_DATA_SAP WHERE ZSPTM = '' .
ENDFORM.

*&---------------------------------------------------------------------*
*&      Form  FRM_COMPARE_PRICE
*&---------------------------------------------------------------------*
*       对SAP、POS、商友系统的商品价格数据进行比较
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM FRM_COMPARE_PRICE .
  DATA: BEGIN OF LS_ZSPTM ,
          ZSPTM TYPE ZQC_PRICE_COMPAR-ZSPTM,
        END OF LS_ZSPTM ,
        LT_ZSPTM     LIKE TABLE OF LS_ZSPTM,
        LT_ZSPTM_TMP LIKE LT_ZSPTM.
  MOVE-CORRESPONDING GT_DATA_POS TO LT_ZSPTM_TMP .
  APPEND LINES OF LT_ZSPTM_TMP TO LT_ZSPTM .
  REFRESH: LT_ZSPTM_TMP .
  MOVE-CORRESPONDING GT_DATA_SAY TO LT_ZSPTM_TMP .
  APPEND LINES OF LT_ZSPTM_TMP TO LT_ZSPTM .
  REFRESH: LT_ZSPTM_TMP .
  MOVE-CORRESPONDING GT_DATA_SAP TO LT_ZSPTM_TMP .
  APPEND LINES OF LT_ZSPTM_TMP TO LT_ZSPTM .
  REFRESH: LT_ZSPTM_TMP .

  SORT LT_ZSPTM BY ZSPTM .
  DELETE ADJACENT DUPLICATES FROM LT_ZSPTM COMPARING ZSPTM .
  GV_ZSPTM_TOTAL = LINES( LT_ZSPTM ) .
  IF GV_ZSPTM_TOTAL = 0 .
    RETURN .
  ENDIF.

  SORT GT_DATA_POS BY ZSPTM .
  SORT GT_DATA_SAY BY ZSPTM .
  SORT GT_DATA_SAP BY ZSPTM .
  LOOP AT LT_ZSPTM INTO LS_ZSPTM .
    CLEAR: GS_ALV .
    GS_ALV-ZSPTM = LS_ZSPTM-ZSPTM .
    READ TABLE GT_DATA_POS INTO GS_DATA_POS WITH KEY ZSPTM = GS_ALV-ZSPTM BINARY SEARCH .
    IF SY-SUBRC = 0 .
      MOVE-CORRESPONDING GS_DATA_POS TO GS_ALV .
      GS_ALV-ZSPTM_POS = GS_DATA_POS-ZSPTM .
    ENDIF.

    READ TABLE GT_DATA_SAY INTO GS_DATA_SAY WITH KEY ZSPTM = GS_ALV-ZSPTM BINARY SEARCH .
    IF SY-SUBRC = 0 .
      MOVE-CORRESPONDING GS_DATA_SAY TO GS_ALV .
      GS_ALV-ZSPTM_SAY = GS_DATA_SAY-ZSPTM .
    ENDIF.

    READ TABLE GT_DATA_SAP INTO GS_DATA_SAP WITH KEY ZSPTM = GS_ALV-ZSPTM BINARY SEARCH .
    IF SY-SUBRC = 0 .
      MOVE-CORRESPONDING GS_DATA_SAP TO GS_ALV .
      GS_ALV-ZSPTM_SAP = GS_DATA_SAP-ZSPTM .
    ENDIF.
* 开始比较
    " 比价条形码
    IF RB_1 = 'X' . " 比较SAP和POS
      IF GS_ALV-ZSPTM_SAP <> GS_ALV-ZSPTM_POS .
        GS_ALV-ZSPTM_DIFF = 1 .
        GS_ALV-DIFF_ALL = GS_ALV-DIFF_ALL + 1 .
      ENDIF.
    ELSEIF RB_2 = 'X' . " 比较SAP和商友
      IF GS_ALV-ZSPTM_SAP <> GS_ALV-ZSPTM_SAY .
        GS_ALV-ZSPTM_DIFF = 1 .
        GS_ALV-DIFF_ALL = GS_ALV-DIFF_ALL + 1 .
      ENDIF.
    ELSEIF RB_3 = 'X' . " 全部比较
      IF GS_ALV-ZSPTM_SAP <> GS_ALV-ZSPTM_POS OR GS_ALV-ZSPTM_SAP <> GS_ALV-ZSPTM_SAY .
        GS_ALV-ZSPTM_DIFF = 1 .
        GS_ALV-DIFF_ALL = GS_ALV-DIFF_ALL + 1 .
      ENDIF.
    ENDIF.

    " 比较商品款号
    IF RB_1 = 'X' . " 比较SAP和POS
      " 比较商品款号
      IF GS_ALV-MATNR_SAP <> GS_ALV-MATNR_POS .
        GS_ALV-MATNR_DIFF = 1 .
        GS_ALV-DIFF_ALL = GS_ALV-DIFF_ALL + 1 .
      ENDIF.
      " 比较批次号
      IF GS_ALV-CHARG_SAP <> GS_ALV-CHARG_POS .
        GS_ALV-CHARG_DIFF = 1 .
        GS_ALV-DIFF_ALL = GS_ALV-DIFF_ALL + 1 .
      ENDIF.
      " 比较商品大类
      IF GS_ALV-EXTWG_SAP <> GS_ALV-EXTWG_POS .
        GS_ALV-EXTWG_DIFF = 1 .
        GS_ALV-DIFF_ALL = GS_ALV-DIFF_ALL + 1 .
      ENDIF.
      " 比较商品小类
      IF GS_ALV-RAUBE_SAP <> GS_ALV-RAUBE_POS .
        GS_ALV-RAUBE_DIFF = 1 .
        GS_ALV-DIFF_ALL = GS_ALV-DIFF_ALL + 1 .
      ENDIF.
      " 比较贵金属材质（批次属性）
      IF GS_ALV-EX_GJS_SAP <> GS_ALV-EX_GJS_POS .
        GS_ALV-EX_GJS_DIFF = 1 .
        GS_ALV-DIFF_ALL = GS_ALV-DIFF_ALL + 1 .
      ENDIF.
      " 比较销售克工费档
      IF GS_ALV-ZXSKGFD_SAP <> GS_ALV-ZXSKGFD_POS .
        GS_ALV-ZXSKGFD_DIFF = 1 .
        GS_ALV-DIFF_ALL = GS_ALV-DIFF_ALL + 1 .
      ENDIF.
    ELSEIF RB_2 = 'X' . " 比较SAP和商友
    ELSEIF RB_3 = 'X' . " 全部比较
    ENDIF.

    " 比较销售计价单位
    IF RB_1 = 'X' . " 比较SAP和POS
      IF GS_ALV-ZXSJGJLDW_SAP <> GS_ALV-ZXSJGJLDW_POS .
        GS_ALV-ZXSJGJLDW_DIFF = 1 .
        GS_ALV-DIFF_ALL = GS_ALV-DIFF_ALL + 1 .
      ENDIF.
    ELSEIF RB_2 = 'X' . " 比较SAP和商友
      IF GS_ALV-ZXSJGJLDW_SAP <> GS_ALV-ZXSJGJLDW_SAY .
        GS_ALV-ZXSJGJLDW_DIFF = 1 .
        GS_ALV-DIFF_ALL = GS_ALV-DIFF_ALL + 1 .
      ENDIF.
    ELSEIF RB_3 = 'X' . " 全部比较
      IF GS_ALV-ZXSJGJLDW_SAP <> GS_ALV-ZXSJGJLDW_POS OR GS_ALV-ZXSJGJLDW_SAP <> GS_ALV-ZXSJGJLDW_SAY .
        GS_ALV-ZXSJGJLDW_DIFF = 1 .
        GS_ALV-DIFF_ALL = GS_ALV-DIFF_ALL + 1 .
      ENDIF.
    ENDIF.

    " 比较单品克重
    IF RB_1 = 'X' . " 比较SAP和POS
      IF GS_ALV-ZJZ_SAP <> GS_ALV-ZJZ_POS .
        GS_ALV-ZJZ_DIFF = 1 .
        GS_ALV-DIFF_ALL = GS_ALV-DIFF_ALL + 1 .
      ENDIF.
    ELSEIF RB_2 = 'X' . " 比较SAP和商友
      IF GS_ALV-ZJZ_SAP <> GS_ALV-ZJZ_SAY .
        GS_ALV-ZJZ_DIFF = 1 .
        GS_ALV-DIFF_ALL = GS_ALV-DIFF_ALL + 1 .
      ENDIF.
    ELSEIF RB_3 = 'X' . " 全部比较
      IF GS_ALV-ZJZ_SAP <> GS_ALV-ZJZ_POS OR GS_ALV-ZJZ_SAP <> GS_ALV-ZJZ_SAY .
        GS_ALV-ZJZ_DIFF = 1 .
        GS_ALV-DIFF_ALL = GS_ALV-DIFF_ALL + 1 .
      ENDIF.
    ENDIF.

*    IF GS_ALV-ZXSJGJLDW_DIFF = 0 AND GS_ALV-ZXSJGJLDW_SAP = 'G' .
*      IF GS_ALV-KBETR_GF_SAP <> GS_ALV-KBETR_GF_POS .
*        GS_ALV-KBETR_GF_DIFF = 1 .
*        GS_ALV-DIFF_ALL = GS_ALV-DIFF_ALL + 1 .
*      ENDIF.
*
*      IF GS_ALV-KBETR_G_SAP <> GS_ALV-KBETR_G_SAY .
*        GS_ALV-KBETR_G_DIFF = 1 .
*        GS_ALV-DIFF_ALL = GS_ALV-DIFF_ALL + 1 .
*      ENDIF.
*    ENDIF .

    " 在POS和SAP的销售计价单位和克工费档相同的情况下，比较克工费
    IF GS_ALV-ZXSJGJLDW_SAP = GS_ALV-ZXSJGJLDW_POS AND
       GS_ALV-ZXSKGFD_SAP = GS_ALV-ZXSKGFD_POS AND
       GS_ALV-ZXSJGJLDW_POS = 'G' .

      IF RB_1 = 'X' . " 比较SAP和POS
        IF GS_ALV-KBETR_GF_SAP <> GS_ALV-KBETR_GF_POS .
          GS_ALV-KBETR_GF_DIFF = 1 .
          GS_ALV-DIFF_ALL = GS_ALV-DIFF_ALL + 1 .
        ENDIF.
      ELSEIF RB_2 = 'X' . " 比较SAP和商友
      ELSEIF RB_3 = 'X' . " 全部比较
      ENDIF.

    ENDIF.

    " 在商友和SAP的销售计价单位相同且为G的情况下，比较克单价和件售价
    " 因为当单位为G时POS无法提供件售价和克单价，在PRICE_POS中是单位为PC时的参考零售价
    IF GS_ALV-ZXSJGJLDW_SAP = GS_ALV-ZXSJGJLDW_SAY AND GS_ALV-ZXSJGJLDW_SAP = 'G' .

      IF RB_1 = 'X' . " 比较SAP和POS
        IF GS_ALV-KBETR_G_SAP <> GS_ALV-KBETR_G_SAY .
          GS_ALV-KBETR_G_DIFF = 1 .
          GS_ALV-DIFF_ALL = GS_ALV-DIFF_ALL + 1 .
        ENDIF.
        IF GS_ALV-PRICE_SAP <> GS_ALV-PRICE_SAY .
          GS_ALV-PRICE_DIFF = 1 .
          GS_ALV-DIFF_ALL = GS_ALV-DIFF_ALL + 1 .
        ENDIF.
      ELSEIF RB_2 = 'X' . " 比较SAP和商友
      ELSEIF RB_3 = 'X' . " 全部比较
      ENDIF.

    ENDIF.

    " 如果销售计价单位相同，且为PC，则比较SAP和商友的件售价，因为当单位为G时POS无法提供件售价和克单价
    IF GS_ALV-ZXSJGJLDW_DIFF = 0 AND GS_ALV-ZXSJGJLDW_SAP = 'PC' .

      IF RB_1 = 'X' . " 比较SAP和POS
        IF GS_ALV-PRICE_SAP <> GS_ALV-PRICE_POS .
          GS_ALV-PRICE_DIFF = 1 .
          GS_ALV-DIFF_ALL = GS_ALV-DIFF_ALL + 1 .
        ENDIF.
      ELSEIF RB_2 = 'X' . " 比较SAP和商友
        IF GS_ALV-PRICE_SAP <> GS_ALV-PRICE_SAY.
          GS_ALV-PRICE_DIFF = 1 .
          GS_ALV-DIFF_ALL = GS_ALV-DIFF_ALL + 1 .
        ENDIF.
      ELSEIF RB_3 = 'X' . " 全部比较
        IF GS_ALV-PRICE_SAP <> GS_ALV-PRICE_POS OR GS_ALV-PRICE_SAP <> GS_ALV-PRICE_SAY.
          GS_ALV-PRICE_DIFF = 1 .
          GS_ALV-DIFF_ALL = GS_ALV-DIFF_ALL + 1 .
        ENDIF.
      ENDIF.

    ENDIF.
    " 如果有错误，将该条记录添加到输出ALV；或者如果选择的是显示所有记录，无论是否有错误都将该条记录添加到输出ALV中
    IF GS_ALV-DIFF_ALL > 0 OR RB_5 ='X'.
      APPEND GS_ALV TO GT_ALV .
    ENDIF.
  ENDLOOP.

  GV_ZSPTM_ERR = LINES( GT_ALV ).
ENDFORM .
