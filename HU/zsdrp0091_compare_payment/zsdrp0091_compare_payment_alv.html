*----------------------------------------------------------------------*

*----------------------------------------------------------------------*
FORM FRM_DISPLAY_DOC_FLOW_BILLING USING IV_EX_POSNUM TYPE VBRK-EX_POSNUM
                                        IV_EX_YWLX TYPE VBRK-EX_POSTYP
                                        IV_FKDAT TYPE VBRK-FKDAT
                                        IV_WERKS TYPE T001W-WERKS.
  DATA: LS_DOC_FLOW_BILLING LIKE LINE OF GT_DOC_FLOW_BILLING .
  REFRESH: GT_DOC_FLOW_BILLING .

  " 取发票和关联的会计凭证
  SELECT VBRK~VBELN, BKPF~BUKRS, BKPF~BELNR, BKPF~GJAHR, VBRK~ERDAT, VBRK~ERZET,
         SUM( VBRK~NETWR + VBRK~MWSBK ) AS ZSALES_BILLING,
         SUM( CASE WHEN BSEG~SHKZG = 'S' THEN BSEG~DMBTR
                   WHEN BSEG~SHKZG = 'H' THEN BSEG~DMBTR * -1 END ) AS ZAR_BILLING
    INTO CORRESPONDING FIELDS OF TABLE @GT_DOC_FLOW_BILLING
    FROM VBRK AS VBRK
    INNER JOIN T001W AS T001W ON T001W~KUNNR = VBRK~KUNAG
    LEFT OUTER JOIN ( BKPF AS BKPF
                      INNER JOIN BSEG AS BSEG ON  BSEG~BUKRS = BKPF~BUKRS
                                              AND BSEG~BELNR = BKPF~BELNR
                                              AND BSEG~GJAHR = BKPF~GJAHR
                                              AND BSEG~BSCHL IN ( '01' , '11' ) )
                      ON  BKPF~XBLNR = VBRK~XBLNR
                      AND BKPF~BUKRS = VBRK~BUKRS
                      AND BKPF~BUDAT = VBRK~FKDAT
                      AND BKPF~BLART = 'RV'
    WHERE VBRK~FKDAT = @IV_FKDAT
      AND VBRK~EX_POSNUM = @IV_EX_POSNUM
      AND T001W~WERKS = @IV_WERKS
    GROUP BY VBRK~VBELN, BKPF~BUKRS, BKPF~BELNR, BKPF~GJAHR,VBRK~ERDAT, VBRK~ERZET .

  " 取付款会计凭证
  SELECT BKPF~BUKRS, BKPF~BELNR, BKPF~GJAHR, BKPF~CPUDT AS ERDAT, BKPF~CPUTM AS ERZET, BSEG~BUZEI, BSEG~HKONT,
         T~TXT50 AS HKONT_T,
         CASE WHEN BSEG~SHKZG = 'S' THEN BSEG~DMBTR
              WHEN BSEG~SHKZG = 'H' THEN BSEG~DMBTR * -1 END AS ZPAYMENT
    APPENDING CORRESPONDING FIELDS OF TABLE @GT_DOC_FLOW_BILLING
    FROM BKPF
    INNER JOIN BSEG AS BSEG ON  BSEG~BUKRS = BKPF~BUKRS
                            AND BSEG~BELNR = BKPF~BELNR
                            AND BSEG~GJAHR = BKPF~GJAHR
                            AND BSEG~BSCHL NOT IN ( '01' , '11' )
    INNER JOIN GL_ACCT_CA_TEXT AS T ON  T~SPRAS = @SY-LANGU
                                    AND T~KTOPL = 'CBAI'
                                    AND T~SAKNR = BSEG~HKONT
    WHERE BKPF~BUDAT = @IV_FKDAT
      AND BKPF~BKTXT = @IV_EX_POSNUM
      AND BKPF~BLART = 'RV'
      AND BKPF~AWTYP = 'BKPFF'
      AND BSEG~HKONT NOT IN ( '1405001003' , '6401004002' ) . " 这两个是以旧换新的科目

  LS_DOC_FLOW_BILLING-FKDAT = IV_FKDAT .
  LS_DOC_FLOW_BILLING-EX_POSNUM = IV_EX_POSNUM .
  LS_DOC_FLOW_BILLING-WERKS = IV_WERKS .
  LS_DOC_FLOW_BILLING-EX_YWLX = IV_EX_YWLX.
  MODIFY GT_DOC_FLOW_BILLING FROM LS_DOC_FLOW_BILLING
                             TRANSPORTING FKDAT EX_POSNUM WERKS EX_YWLX
                             WHERE EX_POSNUM = '' .
  SORT GT_DOC_FLOW_BILLING BY ERDAT ERZET BUKRS BELNR GJAHR BUZEI VBELN.
ENDFORM.
