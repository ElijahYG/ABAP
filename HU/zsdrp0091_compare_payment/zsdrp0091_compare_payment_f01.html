*&      Form  FRM_GET_MAIN_DATA
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM FRM_GET_MAIN_DATA .
** 取得Billing销售金额
*  SELECT K~BUKRS AS BUKRS, K~VBELN, K~FKDAT, P~WERKS, K~EX_POSNUM, K~EX_POSTYP AS EX_YWLX,
*         SUM( CASE WHEN P~SHKZG = 'X' THEN ABS( P~KZWI3 ) * -1
*              WHEN P~SHKZG <> 'X' THEN ABS( P~KZWI3 )     END  ) AS ZSALES
*    INTO CORRESPONDING FIELDS OF TABLE @GT_BILLING
*    FROM VBRK AS K
*    INNER JOIN VBRP AS P ON  P~VBELN = K~VBELN
*    WHERE K~FKDAT IN @S_FKDAT
*      AND P~WERKS IN @S_WERKS
*      AND K~EX_POSNUM IN @S_POSNUM
*    GROUP BY K~BUKRS, K~VBELN, K~FKDAT, P~WERKS, K~EX_POSNUM, K~EX_POSTYP.
*
** 取得Billing对应的会计凭证AR金额
*  SELECT VBRK~VBELN,
*         SUM( CASE WHEN BSEG~SHKZG = 'S' THEN BSEG~DMBTR
*                   WHEN BSEG~SHKZG = 'H' THEN BSEG~DMBTR * -1 END ) AS ZAR
*    INTO TABLE @GT_FI_AR
*    FROM BKPF AS BKPF
*    INNER JOIN BSEG AS BSEG ON  BSEG~BUKRS = BKPF~BUKRS
*                            AND BSEG~BELNR = BKPF~BELNR
*                            AND BSEG~GJAHR = BKPF~GJAHR
*                            AND BSEG~BSCHL = '01'
*    INNER JOIN VBRK AS VBRK ON  VBRK~XBLNR = BKPF~XBLNR
*                            AND VBRK~BUKRS = BKPF~BUKRS
*                            AND VBRK~FKDAT = BKPF~BUDAT
*    INNER JOIN VBRP AS VBRP ON VBRP~VBELN = VBRK~VBELN
*    WHERE BKPF~BUDAT IN @S_FKDAT
*      AND BKPF~BLART = 'RV'
*      AND VBRP~WERKS IN @S_WERKS
*      AND VBRK~EX_POSNUM IN @S_POSNUM
*    GROUP BY VBRK~VBELN .

* 取得Billing销售金额和对应会计凭证的AR金额
  SELECT VBRK~FKDAT, VBRK~EX_POSNUM, VBRK~EX_POSTYP AS EX_YWLX, T001W~WERKS,
         SUM( VBRK~NETWR + VBRK~MWSBK ) AS ZSALES_BILLING,
         SUM( CASE WHEN BSEG~SHKZG = 'S' THEN BSEG~DMBTR
                   WHEN BSEG~SHKZG = 'H' THEN BSEG~DMBTR * -1 END ) AS ZAR_BILLING
    INTO CORRESPONDING FIELDS OF TABLE @GT_BILLING
    FROM VBRK AS VBRK
    INNER JOIN T001W AS T001W ON T001W~KUNNR = VBRK~KUNAG
    LEFT OUTER JOIN ( BKPF AS BKPF
                      INNER JOIN BSEG AS BSEG ON  BSEG~BUKRS = BKPF~BUKRS
                                              AND BSEG~BELNR = BKPF~BELNR
                                              AND BSEG~GJAHR = BKPF~GJAHR
                                              AND BSEG~BSCHL IN ( '01' , '11' ) )
                      ON  BKPF~XBLNR = VBRK~XBLNR
                      AND BKPF~BUKRS = VBRK~BUKRS
                      AND BKPF~BUDAT = VBRK~FKDAT
                      AND BKPF~BLART = 'RV'
    WHERE VBRK~FKDAT IN @S_FKDAT
      AND VBRK~EX_POSNUM IN @S_POSNUM
      AND T001W~WERKS IN @S_WERKS
    GROUP BY VBRK~FKDAT, VBRK~EX_POSNUM, VBRK~EX_POSTYP, T001W~WERKS.

* 取得POS销售付款会计凭证的金额，付款会计凭证是冲抵AR的，所以在这里统计时正好与上面判断借贷项相反
  " 如果发票是销售，其会计凭证是借 AR，过账码是01，那么在付款会计凭证是对冲AR，也就是贷AR，过账码是11
  " 如果发票是退货，其会计凭证是贷 AR，过账码是11，那么在付款会计凭证是对冲AR，也就是借AR，过账码是01
  " 在本程序中为了对比方便，销售时，销售金额为正，AR为正，付款金额也为正
  " 所以在这里判断借贷取正负的逻辑刚好与上面相反
  SELECT BKPF~BUDAT, BKPF~BKTXT,
          SUM( CASE WHEN BSEG~SHKZG = 'S' THEN BSEG~DMBTR * -1
                    WHEN BSEG~SHKZG = 'H' THEN BSEG~DMBTR END ) AS ZPAYMENT_BILLING
    INTO TABLE @GT_PAYMENT
    FROM BKPF
    INNER JOIN BSEG AS BSEG ON  BSEG~BUKRS = BKPF~BUKRS
                            AND BSEG~BELNR = BKPF~BELNR
                            AND BSEG~GJAHR = BKPF~GJAHR
                            AND BSEG~BSCHL IN ( '01' , '11' )
    WHERE BKPF~BUDAT IN @S_FKDAT
      AND BKPF~BKTXT IN @S_POSNUM
      AND BKPF~BLART = 'RV'
      AND BKPF~AWTYP = 'BKPFF'
    GROUP BY BKPF~BUDAT, BKPF~BKTXT .

* 取得投资预定和其会计凭证金额
  SELECT EX_YERQ AS FKDAT, EX_PBH AS EX_POSNUM, EX_MDBM AS WERKS, EX_YWDJLX AS EX_YWLX,
         SUM( CASE WHEN EX_STA = 'X1' THEN ABS( EX_ZJE )
                   WHEN EX_STA = 'X2' THEN ABS( EX_ZJE ) * -1 END ) AS ZSALES_YD,
         SUM( CASE WHEN BSEG~SHKZG = 'S' THEN BSEG~DMBTR * -1
                   WHEN BSEG~SHKZG = 'H' THEN BSEG~DMBTR  END ) AS ZPAYMENT_YD
    INTO CORRESPONDING FIELDS OF TABLE @GT_YD
    FROM ZYDHEAD AS ZYDHEAD
    LEFT OUTER JOIN ( BKPF AS BKPF
                      INNER JOIN BSEG AS BSEG ON  BSEG~BUKRS = BKPF~BUKRS
                                              AND BSEG~BELNR = BKPF~BELNR
                                              AND BSEG~GJAHR = BKPF~GJAHR
                                              " 这两个科目是在ZSDIF0191_FM_INPUT中Hard Code的
                                              AND BSEG~HKONT IN ( '2203003005' , '2203003007' ) )
                      ON  BKPF~BELNR = ZYDHEAD~BELNR
                      AND BKPF~BUKRS = ZYDHEAD~BUKRS
                      AND BKPF~GJAHR = ZYDHEAD~GJAHR
    WHERE EX_YERQ IN @S_FKDAT
      AND EX_MDBM IN @S_WERKS
      AND EX_PBH  IN @S_POSNUM
    GROUP BY EX_YERQ, EX_PBH, EX_MDBM, EX_YWDJLX.

  IF GT_BILLING[] IS INITIAL AND GT_YD[] IS INITIAL .
    MESSAGE S000(ZCB) WITH '没有找到销售发票和投资预定数据！' .
    STOP .
  ENDIF.

ENDFORM.

*  <--  p2        text

FORM FRM_PROCESS_DATA .
  DATA: LS_ALV_1 LIKE LINE OF GT_ALV_1,
        LS_ALV_2 LIKE LINE OF GT_ALV_2,
        LS_WERKS LIKE LINE OF GT_WERKS,
        LS_PAYMENT LIKE LINE OF GT_PAYMENT .

  MOVE-CORRESPONDING GT_BILLING TO GT_ALV_2 .
  LS_ALV_2-ZSOURCE = '1' .
  MODIFY GT_ALV_2 FROM LS_ALV_2 TRANSPORTING ZSOURCE WHERE ZSOURCE = '' .

  MOVE-CORRESPONDING GT_YD TO GT_ALV_2_TEMP .
  LS_ALV_2-ZSOURCE = '2' .
  MODIFY GT_ALV_2_TEMP FROM LS_ALV_2 TRANSPORTING ZSOURCE WHERE ZSOURCE = '' .
  APPEND LINES OF GT_ALV_2_TEMP TO GT_ALV_2 .
  REFRESH: GT_ALV_2_TEMP .

  SORT GT_WERKS BY WERKS .
  SORT GT_PAYMENT BY BUDAT BKTXT .
  LOOP AT GT_ALV_2 ASSIGNING FIELD-SYMBOL(<FS_ALV_2>) .
    " 读取地点简称和是否扣点店
    READ TABLE GT_WERKS INTO LS_WERKS WITH KEY WERKS = <FS_ALV_2>-WERKS BINARY SEARCH.
    IF SY-SUBRC = 0 .
      <FS_ALV_2>-SORT1 = LS_WERKS-SORT1 .
      IF LS_WERKS-EX_POS = '00' .
        <FS_ALV_2>-EX_POS = '' .
      ELSEIF LS_WERKS-EX_POS = '10' .
        <FS_ALV_2>-EX_POS = 'X' .
      ENDIF.
    ENDIF.

    IF <FS_ALV_2>-ZSOURCE = '1' . " 来源于系统Billing
      " 读取付款金额
      READ TABLE GT_PAYMENT INTO LS_PAYMENT WITH KEY BUDAT = <FS_ALV_2>-FKDAT
                                                     BKTXT = <FS_ALV_2>-EX_POSNUM BINARY SEARCH .
      IF SY-SUBRC = 0 .
        <FS_ALV_2>-ZPAYMENT_BILLING = LS_PAYMENT-ZPAYMENT_BILLING .
      ENDIF.
    " 判断是否金额一致
      IF <FS_ALV_2>-EX_POS = '' . " 非扣点店
        IF <FS_ALV_2>-ZSALES_BILLING = <FS_ALV_2>-ZAR_BILLING AND <FS_ALV_2>-ZAR_BILLING = <FS_ALV_2>-ZPAYMENT_BILLING .
          <FS_ALV_2>-ZDIFF_BILLING = '' . " 正确
        ELSE.
          <FS_ALV_2>-ZDIFF_BILLING = 'X' . " 错误
        ENDIF.
      ELSEIF <FS_ALV_2>-EX_POS = 'X' . " 扣点店，没有付款的会计凭证
        IF <FS_ALV_2>-ZSALES_BILLING = <FS_ALV_2>-ZAR_BILLING AND <FS_ALV_2>-ZPAYMENT_BILLING = 0 .
          <FS_ALV_2>-ZDIFF_BILLING = '' . " 正确
        ELSE.
          <FS_ALV_2>-ZDIFF_BILLING = 'X' . " 错误
        ENDIF.
      ENDIF.
    ELSEIF <FS_ALV_2>-ZSOURCE = '2' . " 来源于投资预定
      IF <FS_ALV_2>-ZSALES_YD = <FS_ALV_2>-ZPAYMENT_YD .
        <FS_ALV_2>-ZDIFF_YD = '' . " 正确
      ELSE.
        <FS_ALV_2>-ZDIFF_YD = 'X' . " 错误
      ENDIF.
    ENDIF.

    IF <FS_ALV_2>-ZDIFF_BILLING = '' AND <FS_ALV_2>-ZDIFF_YD = '' .
      <FS_ALV_2>-ZDIFF = '3' . " Billing和投资预定都正确才算该行正确
    ELSE.
      <FS_ALV_2>-ZDIFF = '1' .
    ENDIF.

    <FS_ALV_2>-ZLEVEL = '2' .

    MOVE-CORRESPONDING <FS_ALV_2> TO LS_ALV_1 .
    CLEAR: LS_ALV_1-EX_POSNUM, LS_ALV_1-ZSOURCE, LS_ALV_1-EX_YWLX,LS_ALV_1-ZLEVEL,
           LS_ALV_1-ZDIFF,LS_ALV_1-ZDIFF_BILLING,LS_ALV_1-ZDIFF_YD  .
    COLLECT LS_ALV_1 INTO GT_ALV_1 .
  ENDLOOP.

  LOOP AT GT_ALV_1 ASSIGNING FIELD-SYMBOL(<FS_ALV_1>) .
    " 判断该日期+地点的Billing金额是否一致
    IF <FS_ALV_1>-EX_POS = '' . " 非扣点店
      IF <FS_ALV_1>-ZSALES_BILLING = <FS_ALV_1>-ZAR_BILLING AND <FS_ALV_1>-ZAR_BILLING = <FS_ALV_1>-ZPAYMENT_BILLING .
        <FS_ALV_1>-ZDIFF_BILLING = '' . " 正确
      ELSE.
        <FS_ALV_1>-ZDIFF_BILLING = 'X' . " 错误
      ENDIF.
    ELSE.
      " 扣点店不应该产生付款会计凭证
      IF <FS_ALV_1>-ZSALES_BILLING = <FS_ALV_1>-ZAR_BILLING AND <FS_ALV_1>-ZPAYMENT_BILLING = 0  .
        <FS_ALV_1>-ZDIFF_BILLING = '' . " 正确
      ELSE.
        <FS_ALV_1>-ZDIFF_BILLING = 'X' . " 错误
      ENDIF.
    ENDIF.

    " 判断该日期+地点的投资预定金额是否一致
    IF <FS_ALV_1>-ZSALES_YD = <FS_ALV_1>-ZPAYMENT_YD .
      <FS_ALV_1>-ZDIFF_YD = '' . " 正确
    ELSE.
      <FS_ALV_1>-ZDIFF_YD = 'X' . " 错误
    ENDIF.

    IF <FS_ALV_1>-ZDIFF_BILLING = '' AND <FS_ALV_1>-ZDIFF_YD = '' .
      <FS_ALV_1>-ZDIFF = '3' . " Billing和投资预定都正确才算该行正确
    ELSE.
      <FS_ALV_1>-ZDIFF = '1' .
    ENDIF.

    <FS_ALV_1>-ZLEVEL = '1' .
  ENDLOOP.

  SORT GT_ALV_2 BY FKDAT WERKS EX_POSNUM .
  SORT GT_ALV_1 BY FKDAT WERKS .
ENDFORM.
