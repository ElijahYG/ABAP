*       根据查询条件，取得在指定价格有效日期前后有变化的条件类型ZB00（基础金价）

*----------------------------------------------------------------------*

*  <--  p2        text
*----------------------------------------------------------------------*
FORM FRM_GET_CHANGED_PRICE .
  DATA: LV_DATBI TYPE A909-DATBI .
  DATA: LV_KSCHL TYPE A911-KSCHL .

* LV_DATBI说明 *
* 假设一个条件记录是从2015.11.01~9999.12.31，在2015.12.01发生调价，因此会有
* 两个条件记录，一个是2015.11.01~2015.11.31，另一个是2015.12.01~9999.12.31
* 对这个程序来说，查询条件指定日期是2015.12.01，因此需要把2015.12.01发生调价
* 的前后记录都找到。所以查询条件是开始日期<= 2015.12.01，结束日期>= 2015.11.31
  LV_DATBI = P_DATAB - 1 .

  " 取得条件类型ZB00
  SELECT A~EX_SITE AS EX_SITE, A~EXTWG AS EXTWG, A~RAUBE AS RAUBE, A~EX_GJS AS EX_GJS,
         A~DATBI AS DATBI, A~DATAB AS DATAB,
         P~KBETR AS KBETR
    INTO TABLE @GT_ZB00
    FROM A909 AS A
    INNER JOIN KONP AS P ON P~KNUMH = A~KNUMH
    WHERE A~KAPPL = 'V'
      AND A~KSCHL = 'ZB00'
      AND A~EX_SITE = @P_WERKS
      AND A~EXTWG IN @S_EXTWG
      AND A~RAUBE IN @S_RAUBE
      AND A~EX_GJS IN @S_GJS
      AND A~DATAB <= @P_DATAB
      AND A~DATBI >= @LV_DATBI .
  IF SY-SUBRC = 0 .
    LOOP AT GT_ZB00 INTO GS_ZB00 .
      CLEAR: GS_ZB00_COMPARE .
      GS_ZB00_COMPARE-EX_SITE = GS_ZB00-EX_SITE .
      GS_ZB00_COMPARE-EXTWG = GS_ZB00-EXTWG .
      GS_ZB00_COMPARE-RAUBE = GS_ZB00-RAUBE .
      GS_ZB00_COMPARE-EX_GJS = GS_ZB00-EX_GJS .
      GS_ZB00_COMPARE-DATAB = P_DATAB .
      IF GS_ZB00-DATAB < P_DATAB AND GS_ZB00-DATBI >= P_DATAB .
        " 例如：2015.11.01~2015.12.01 或 2015.11.01~9999.12.31，说明
        " 条件类型在2015.12.01没有发生变化
        GS_ZB00_COMPARE-KBETR_GJS_OLD = GS_ZB00-KBETR .
        GS_ZB00_COMPARE-KBETR_GJS_NEW = GS_ZB00-KBETR .
        MODIFY GT_ZB00_COMPARE FROM GS_ZB00_COMPARE TRANSPORTING KBETR_GJS_OLD KBETR_GJS_NEW
                                                    WHERE EX_SITE = GS_ZB00_COMPARE-EX_SITE
                                                      AND EXTWG   = GS_ZB00_COMPARE-EXTWG
                                                      AND RAUBE   = GS_ZB00_COMPARE-RAUBE
                                                      AND EX_GJS  = GS_ZB00_COMPARE-EX_GJS
                                                      AND DATAB   = GS_ZB00_COMPARE-DATAB.
        IF SY-SUBRC <> 0 .
          APPEND GS_ZB00_COMPARE TO GT_ZB00_COMPARE .
        ENDIF.
      ELSEIF GS_ZB00-DATAB < P_DATAB AND GS_ZB00-DATBI < P_DATAB .
        " 例如：2015.11.01~2015.11.01，说明这条记录是2015.12.01变价前的
        GS_ZB00_COMPARE-KBETR_GJS_OLD = GS_ZB00-KBETR .
        MODIFY GT_ZB00_COMPARE FROM GS_ZB00_COMPARE TRANSPORTING KBETR_GJS_OLD
                                                    WHERE EX_SITE = GS_ZB00_COMPARE-EX_SITE
                                                      AND EXTWG   = GS_ZB00_COMPARE-EXTWG
                                                      AND RAUBE   = GS_ZB00_COMPARE-RAUBE
                                                      AND EX_GJS  = GS_ZB00_COMPARE-EX_GJS
                                                      AND DATAB   = GS_ZB00_COMPARE-DATAB.
        IF SY-SUBRC <> 0 .
          APPEND GS_ZB00_COMPARE TO GT_ZB00_COMPARE .
        ENDIF.
      ELSEIF GS_ZB00-DATAB >= P_DATAB AND GS_ZB00-DATBI >= P_DATAB .
        " 例如：2015.12.01~2015.12.01 或 2015.12.01~9999.12.31，说明
        " 条件类型在2015.12.01变价后的
        GS_ZB00_COMPARE-KBETR_GJS_NEW = GS_ZB00-KBETR .
        MODIFY GT_ZB00_COMPARE FROM GS_ZB00_COMPARE TRANSPORTING KBETR_GJS_NEW
                                                    WHERE EX_SITE = GS_ZB00_COMPARE-EX_SITE
                                                      AND EXTWG   = GS_ZB00_COMPARE-EXTWG
                                                      AND RAUBE   = GS_ZB00_COMPARE-RAUBE
                                                      AND EX_GJS  = GS_ZB00_COMPARE-EX_GJS
                                                      AND DATAB   = GS_ZB00_COMPARE-DATAB.
        IF SY-SUBRC <> 0 .
          APPEND GS_ZB00_COMPARE TO GT_ZB00_COMPARE .
        ENDIF.
      ENDIF.
    ENDLOOP.

    LOOP AT GT_ZB00_COMPARE ASSIGNING FIELD-SYMBOL(<ZB00_COMPARE>) .
      IF <ZB00_COMPARE>-KBETR_GJS_OLD <> <ZB00_COMPARE>-KBETR_GJS_NEW .
        <ZB00_COMPARE>-IS_DIFF_GJS = 'X' .
        GV_HAS_DIFF_GJS = 'X' .
      ENDIF.
    ENDLOOP.
  ENDIF.

  " 取得条件类型ZKGS
  IF P_ZKST = 'X' .
    LV_KSCHL = 'ZKST' .
  ELSE.
    LV_KSCHL = 'ZKSG' .
  ENDIF.
  SELECT A~ZXSKGFD AS ZXSKGFD,
         A~DATBI AS DATBI, A~DATAB AS DATAB,
         P~KBETR AS KBETR
    INTO TABLE @GT_ZKSG
    FROM A911 AS A
    INNER JOIN KONP AS P ON P~KNUMH = A~KNUMH
    WHERE A~KAPPL = 'V'
      AND A~KSCHL = @LV_KSCHL
      AND A~DATAB <= @P_DATAB
      AND A~DATBI >= @LV_DATBI .
  IF SY-SUBRC = 0 .
    LOOP AT GT_ZKSG INTO GS_ZKSG .
      CLEAR: GS_ZKSG_COMPARE .
      GS_ZKSG_COMPARE-ZXSKGFD = GS_ZKSG-ZXSKGFD .
      GS_ZKSG_COMPARE-DATAB = P_DATAB .
      IF GS_ZKSG-DATAB < P_DATAB AND GS_ZKSG-DATBI >= P_DATAB .
        " 例如：2015.11.01~2015.12.01 或 2015.11.01~9999.12.31，说明
        " 条件类型在2015.12.01没有发生变化
        GS_ZKSG_COMPARE-KBETR_GF_OLD = GS_ZKSG-KBETR .
        GS_ZKSG_COMPARE-KBETR_GF_NEW = GS_ZKSG-KBETR .
        MODIFY GT_ZKSG_COMPARE FROM GS_ZKSG_COMPARE TRANSPORTING KBETR_GF_OLD KBETR_GF_NEW
                                                           WHERE ZXSKGFD = GS_ZKSG_COMPARE-ZXSKGFD
                                                             AND DATAB   = GS_ZKSG_COMPARE-DATAB.
        IF SY-SUBRC <> 0 .
          APPEND GS_ZKSG_COMPARE TO GT_ZKSG_COMPARE .
        ENDIF.
      ELSEIF GS_ZKSG-DATAB < P_DATAB AND GS_ZKSG-DATBI < P_DATAB .
        " 例如：2015.11.01~2015.11.01，说明这条记录是2015.12.01变价前的
        GS_ZKSG_COMPARE-KBETR_GF_OLD = GS_ZKSG-KBETR .
        MODIFY GT_ZKSG_COMPARE FROM GS_ZKSG_COMPARE TRANSPORTING KBETR_GF_OLD
                                                           WHERE ZXSKGFD = GS_ZKSG_COMPARE-ZXSKGFD
                                                             AND DATAB   = GS_ZKSG_COMPARE-DATAB.
        IF SY-SUBRC <> 0 .
          APPEND GS_ZKSG_COMPARE TO GT_ZKSG_COMPARE .
        ENDIF.
      ELSEIF GS_ZKSG-DATAB >= P_DATAB AND GS_ZKSG-DATBI >= P_DATAB .
        " 例如：2015.12.01~2015.12.01 或 2015.12.01~9999.12.31，说明
        " 条件类型在2015.12.01变价后的
        GS_ZKSG_COMPARE-KBETR_GF_NEW = GS_ZKSG-KBETR .
        MODIFY GT_ZKSG_COMPARE FROM GS_ZKSG_COMPARE TRANSPORTING KBETR_GF_NEW
                                                           WHERE ZXSKGFD = GS_ZKSG_COMPARE-ZXSKGFD
                                                             AND DATAB   = GS_ZKSG_COMPARE-DATAB.
        IF SY-SUBRC <> 0 .
          APPEND GS_ZKSG_COMPARE TO GT_ZKSG_COMPARE .
        ENDIF.
      ENDIF.
    ENDLOOP.

    LOOP AT GT_ZKSG_COMPARE ASSIGNING FIELD-SYMBOL(<ZKSG_COMPARE>) .
      IF <ZKSG_COMPARE>-KBETR_GF_OLD <> <ZKSG_COMPARE>-KBETR_GF_NEW .
        <ZKSG_COMPARE>-IS_DIFF_GF = 'X' .
        GV_HAS_DIFF_GF = 'X' .
      ENDIF.
    ENDLOOP.
  ENDIF.
ENDFORM.

*&---------------------------------------------------------------------*
*       根据查询条件（结算方式、销售克工费档，供应商代码）查询联营按G销售的批次
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM FRM_GET_BATCH .
  IF P_STOCK = '' .
    SELECT A~*,
           M~EXTWG AS EXTWG, M~RAUBE AS RAUBE
      INTO CORRESPONDING FIELDS OF TABLE @GT_BATCH
      FROM ZMMRP0210_BATCH AS A
      INNER JOIN MARA AS M ON M~MATNR = A~MATNR
      INNER JOIN ZMMDG0010_JSFS AS B ON B~BM = A~ZLYJSFS
      WHERE A~ZLYJSFS IN @S_ZJSFS
        AND A~ZXSKGFD IN @S_ZXSGF
        AND A~ZGYSDM  IN @S_ZGYSD
        AND A~ZGJSCZ  IN @S_GJS     "使用批次上的贵金属材质，而不是款式上的，例如有些杂款没有维护贵金属材质
        AND M~EXTWG   IN @S_EXTWG
        AND M~RAUBE   IN @S_RAUBE
        AND A~ZXSJGJLDW = 'G'
        AND B~SFLY = 'X' .
  ELSEIF P_STOCK = 'X' .
    SELECT A~*,
           M~EXTWG AS EXTWG, M~RAUBE AS RAUBE
      INTO CORRESPONDING FIELDS OF TABLE @GT_BATCH
      FROM ZMMRP0210_BATCH AS A
      INNER JOIN MARA AS M ON M~MATNR = A~MATNR
      INNER JOIN ZMMDG0010_JSFS AS B ON B~BM = A~ZLYJSFS
      INNER JOIN MBEW AS W ON  W~MATNR = A~MATNR
                           AND W~BWTAR = A~CHARG
      WHERE A~ZLYJSFS IN @S_ZJSFS
        AND A~ZXSKGFD IN @S_ZXSGF
        AND A~ZGYSDM  IN @S_ZGYSD
        AND A~ZGJSCZ  IN @S_GJS     "使用批次上的贵金属材质，而不是款式上的，例如有些杂款没有维护贵金属材质
        AND M~EXTWG   IN @S_EXTWG
        AND M~RAUBE   IN @S_RAUBE
        AND A~ZXSJGJLDW = 'G'
        AND B~SFLY = 'X'
        AND W~LBKUM > 0 .
  ENDIF.
ENDFORM.
*&---------------------------------------------------------------------*
*&      Form  FRM_FILT_BATCH
*&---------------------------------------------------------------------*
*       从找到的批次中，筛选价格有变化的批次
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM FRM_FILT_BATCH .

  IF GT_BATCH[] IS INITIAL .
    RETURN .
  ENDIF.
  " 取得被多个供应商使用的销售克工费档
  PERFORM FRM_GET_ZXSKGFD .

  " 开始处理找到的每一个批次，检查其价格是否有变化
  SORT GT_ZXSKGFD BY ZXSKGFD .
  SORT GT_ZB00_COMPARE BY EX_SITE EXTWG RAUBE EX_GJS DATAB .
  SORT GT_ZKSG_COMPARE BY ZXSKGFD DATAB .
  LOOP AT GT_BATCH ASSIGNING FIELD-SYMBOL(<BATCH>) .
    CLEAR: <BATCH>-KBETR_GJS_OLD, <BATCH>-KBETR_GJS_NEW,<BATCH>-IS_DIFF_GJS,
           <BATCH>-KBETR_GF_OLD, <BATCH>-KBETR_GF_NEW, <BATCH>-IS_DIFF_GF,
           <BATCH>-PRICE_OLD, <BATCH>-PRICE_NEW, <BATCH>-IS_DIFF_PRICE,
           <BATCH>-IS_MULTI_GYSDM.
    <BATCH>-EX_SITE = P_WERKS .
    " 检查基础金价
    READ TABLE GT_ZB00_COMPARE INTO GS_ZB00_COMPARE WITH KEY EX_SITE = <BATCH>-EX_SITE
                                                             EXTWG   = <BATCH>-EXTWG
                                                             RAUBE   = <BATCH>-RAUBE
                                                             EX_GJS  = <BATCH>-ZGJSCZ
                                                             DATAB   = P_DATAB
                                                             BINARY SEARCH .
    IF SY-SUBRC = 0.
      <BATCH>-KBETR_GJS_OLD = GS_ZB00_COMPARE-KBETR_GJS_OLD .
      <BATCH>-KBETR_GJS_NEW = GS_ZB00_COMPARE-KBETR_GJS_NEW .
      <BATCH>-IS_DIFF_GJS   = GS_ZB00_COMPARE-IS_DIFF_GJS .
    ENDIF.
    " 检查克工费
    READ TABLE GT_ZKSG_COMPARE INTO GS_ZKSG_COMPARE WITH KEY ZXSKGFD = <BATCH>-ZXSKGFD
                                                             DATAB   = P_DATAB
                                                             BINARY SEARCH .
    IF SY-SUBRC = 0 .
      <BATCH>-KBETR_GF_OLD = GS_ZKSG_COMPARE-KBETR_GF_OLD .
      <BATCH>-KBETR_GF_NEW = GS_ZKSG_COMPARE-KBETR_GF_NEW .
      <BATCH>-IS_DIFF_GF   = GS_ZKSG_COMPARE-IS_DIFF_GF .
    ENDIF.
    " 计算和检查克单价
    <BATCH>-PRICE_OLD = <BATCH>-KBETR_GJS_OLD + <BATCH>-KBETR_GF_OLD + <BATCH>-ZXSKFJGF .
    <BATCH>-PRICE_NEW = <BATCH>-KBETR_GJS_NEW + <BATCH>-KBETR_GF_NEW + <BATCH>-ZXSKFJGF .
    IF <BATCH>-PRICE_OLD <> <BATCH>-PRICE_NEW .
      <BATCH>-IS_DIFF_PRICE = 'X' .
      <BATCH>-PRICE_DIFF = <BATCH>-PRICE_OLD - <BATCH>-PRICE_NEW .
    ELSE.
      <BATCH>-IS_DIFF_PRICE = '' .
      <BATCH>-PRICE_DIFF = 0 .
    ENDIF.
    " 检查克工费档是否被多个供应商使用
    READ TABLE GT_ZXSKGFD INTO GS_ZXSKGFD WITH KEY ZXSKGFD = <BATCH>-ZXSKGFD BINARY SEARCH .
    IF SY-SUBRC = 0 .
      <BATCH>-IS_MULTI_GYSDM = 'X' .
      <BATCH>-REMARK = GS_ZXSKGFD-REMARK .
    ENDIF.
  ENDLOOP.

  " 删除价格没有变化的批次
  IF RB1_1 = 'X' .
    " 只要基础金价不同或工费不同就保留
    DELETE GT_BATCH WHERE IS_DIFF_GJS = '' AND IS_DIFF_GF = '' AND IS_DIFF_PRICE = '' .
  ELSEIF RB1_2 = 'X' .
    " 只保留销售克单价不同的批次
    DELETE GT_BATCH WHERE IS_DIFF_PRICE = '' .
  ENDIF.
ENDFORM.
*&---------------------------------------------------------------------*
*&      Form  FRM_GET_ZXSKGFD
*&---------------------------------------------------------------------*
*       取得被多个供应商使用了的销售克工费档
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM FRM_GET_ZXSKGFD .
  DATA: LT_ZXSKGFD LIKE GT_ZXSKGFD .

  SELECT EX_JJJB AS ZXSKGFD
    INTO CORRESPONDING FIELDS OF TABLE GT_ZXSKGFD
    FROM ZMMEX0010_JJJB .

  SELECT A~ZXSKGFD A~ZGYSDM
    INTO CORRESPONDING FIELDS OF TABLE LT_ZXSKGFD
    FROM ZMMRP0210_BATCH AS A
    GROUP BY A~ZXSKGFD A~ZGYSDM .

  SORT LT_ZXSKGFD BY ZXSKGFD ZGYSDM .
  DELETE ADJACENT DUPLICATES FROM LT_ZXSKGFD COMPARING ZXSKGFD ZGYSDM .

  LOOP AT GT_ZXSKGFD ASSIGNING FIELD-SYMBOL(<GS_ZXSKGFD>) .
    <GS_ZXSKGFD>-REMARK = '该工费档被用于供应商：' .
    LOOP AT LT_ZXSKGFD INTO DATA(LS_ZXSKGFD) WHERE ZXSKGFD = <GS_ZXSKGFD>-ZXSKGFD .
      <GS_ZXSKGFD>-ZCOUNT = <GS_ZXSKGFD>-ZCOUNT + 1 .
      <GS_ZXSKGFD>-REMARK = <GS_ZXSKGFD>-REMARK &&  LS_ZXSKGFD-ZGYSDM && ', ' .
    ENDLOOP.
  ENDLOOP.
  DELETE GT_ZXSKGFD WHERE ZCOUNT < 2 .

ENDFORM.
