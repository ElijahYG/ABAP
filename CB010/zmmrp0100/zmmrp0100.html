FORM GET_DATA1.

  SELECT A~MBLNR     "入库单号
*         A~BUKRS     "公司代码
         A~WERKS     "地点
         A~LGORT     "库存地点
         A~LIFNR     "供应商
         A~BUDAT_MKPF "入库日期
*         B~BUTXT     "公司代码描述
         C~SORTL     "地点描述（简述）
*         C~NAME1      "地点描述
         D~LGOBE     "库存地点描述
         A~BWART     "移动类型
         INTO CORRESPONDING FIELDS OF TABLE IT_TABLE
    FROM MSEG   AS A
         INNER JOIN
         T001   AS B ON A~BUKRS = B~BUKRS
         INNER JOIN
*        TOO1W  AS C ON A~WERKS = C~WERKS
         KNA1   AS C ON A~WERKS = C~WERKS
         INNER JOIN
         T001L  AS D ON A~LGORT = D~LGORT
         INNER JOIN
         EKPO   AS E ON A~EBELN = E~EBELN AND A~EBELP = E~EBELP

         WHERE A~MBLNR IN S_MBLNR
          AND  A~LGORT IN S_LGORT
          AND  A~WERKS IN S_WERKS
          AND  A~BUDAT_MKPF IN S_BUDAT
          AND  A~LIFNR IN S_LIFNR
          AND  A~BWART EQ '101'.

  DATA IT_TABLE1   TYPE STANDARD TABLE OF STRU_0010.
  DATA WA_TABLE1   TYPE                   STRU_0010.

*按照入库单号排序
  SORT IT_TABLE BY MBLNR.
*去除重复的入库单号
  DELETE ADJACENT DUPLICATES FROM IT_TABLE COMPARING MBLNR.

  IT_TABLE1 = IT_TABLE.
*按照供应商排序
  SORT IT_TABLE1 BY LIFNR.
*去除重复的供应商
  DELETE ADJACENT DUPLICATES FROM IT_TABLE1 COMPARING LIFNR.
*取数供应商描述
  SELECT A~LIFNR     "供应商
         B~NAME1     "供应商描述
    FROM MSEG AS A
         INNER JOIN
         LFA1 AS B ON A~LIFNR = B~LIFNR
    INTO CORRESPONDING FIELDS OF TABLE IT_DEL
    FOR ALL ENTRIES IN IT_TABLE1
    WHERE A~LIFNR EQ IT_TABLE1-LIFNR.

*按照供应商排序
  SORT IT_DEL BY LIFNR.
*把供应商描述更新到IT_TABLE
  LOOP AT IT_TABLE INTO WA_TABLE.
    CLEAR WA_DEL.
    READ TABLE IT_DEL INTO WA_DEL WITH KEY LIFNR = WA_TABLE-LIFNR BINARY SEARCH.
    IF SY-SUBRC = 0 .
      WA_TABLE-NAME2 = WA_DEL-NAME1.
*去除供应商字段前置零
      CALL FUNCTION 'CONVERSION_EXIT_ALPHA_OUTPUT'
        EXPORTING
          INPUT  = WA_TABLE-LIFNR   "需要去除前置零的值
        IMPORTING
          OUTPUT = WA_TABLE-LIFNR.  "需要去除前置零的值
    ENDIF.
    MODIFY IT_TABLE FROM WA_TABLE TRANSPORTING NAME2 LIFNR.
  ENDLOOP.
*按照入库单号排序
  SORT IT_TABLE BY MBLNR.

ENDFORM.
