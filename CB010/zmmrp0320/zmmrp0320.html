FORM FM_GET_MTBEZ .
  FIELD-SYMBOLS <FS_OUT_TAB> TYPE TY_OUT_TAB.
  DATA:
    LT_OUT_TAB TYPE TYT_OUT_TAB,
    LW_OUT_TAB TYPE TY_OUT_TAB,
    BEGIN OF LW_T134T,
      MTART TYPE T134T-MTART,
      MTBEZ TYPE T134T-MTBEZ,
    END OF LW_T134T,
    LT_T134T LIKE LW_T134T OCCURS 0.

* 取物料组描述
  LT_OUT_TAB = GT_OUT_TAB.
  SORT LT_OUT_TAB BY MTART.
  DELETE ADJACENT DUPLICATES FROM LT_OUT_TAB COMPARING MTART.
  IF LT_OUT_TAB IS NOT INITIAL.
    SELECT MTART
           MTBEZ
      INTO TABLE LT_T134T
      FROM T134T
       FOR ALL ENTRIES IN LT_OUT_TAB
     WHERE MTART = LT_OUT_TAB-MTART
       AND SPRAS = SY-LANGU.
  ENDIF.
* 将物料组描述追加到输出表中
  SORT LT_T134T BY MTART.
  LOOP AT GT_OUT_TAB ASSIGNING <FS_OUT_TAB>.
    CLEAR LW_T134T.
    READ TABLE LT_T134T INTO LW_T134T
                   WITH KEY MTART = <FS_OUT_TAB>-MTART
                     BINARY SEARCH .
    <FS_OUT_TAB>-MTBEZ = LW_T134T-MTBEZ.
  ENDLOOP.
ENDFORM.
