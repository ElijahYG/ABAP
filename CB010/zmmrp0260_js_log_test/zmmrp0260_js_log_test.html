  SELECT *
        INTO CORRESPONDING FIELDS OF TABLE IT_TAB
        FROM EKKO AS K
  INNER JOIN EKPO AS P ON P~EBELN EQ K~EBELN
    WHERE K~BEDAT IN S_BEDAT
      AND K~RESWK IN S_RESWK
      AND K~BSART =  'ZUB1'
      AND K~VERKF IN S_VERKF
      AND K~EBELN IN S_EBELN.

  IF XZ = '1'."全部查询
    PERFORM DATA_ALL.
  ELSEIF  XZ = '2'."未收货
    PERFORM DATA_WSH.
  ELSEIF  XZ = '3'."已收货
    PERFORM DATA_YSH.
  ENDIF.

  SELECT *
    INTO CORRESPONDING FIELDS OF TABLE LT_LIPS
    FROM LIPS
    FOR ALL ENTRIES IN IT_TAB

  SORT LT_LIPS BY VBELN.
  DELETE ADJACENT DUPLICATES FROM LT_LIPS COMPARING VBELN.

FORM FM_GET_ITEM USING E_ROW E_COLUMN.

  DATA:LT_LIPS2 LIKE LT_LIPS,
       LW_LIPS2 LIKE LW_LIPS.

  READ TABLE IT_TAB INTO WA_TAB INDEX E_ROW.
* 取明细
  SELECT *
        INTO CORRESPONDING FIELDS OF TABLE IT_TAB2
        FROM EKKO AS K
  INNER JOIN EKPO AS P ON P~EBELN EQ K~EBELN
  WHERE K~EBELN = WA_TAB-EBELN
    .

    ENDIF.

    IF <WA_TAB2>-MEINS = 'ST'.
      <WA_TAB2>-MEINS = 'PC'.
    ENDIF.

    CLEAR WA_TAB2.
    READ TABLE IT_TAB INTO WA_TAB WITH KEY EBELN = <WA_TAB2>-EBELN.
    IF SY-SUBRC = 0.
      <WA_TAB2>-VBELN = WA_TAB-VBELN.
    ENDIF.
  ENDLOOP.

  PERFORM COL_COLOR."格颜色
  SORT IT_TAB2 BY EBELN  EBELP.
ENDFORM.

FORM FM_HOTSPOT_CLICK USING E_ROW_ID TYPE LVC_S_ROW
                            E_COLUMN_ID  TYPE LVC_S_COL
                            ES_ROW_NO    TYPE LVC_S_ROID.

  DATA:
    RT_VERKF TYPE RANGE OF EKKO-VERKF,
    RS_VERKF LIKE LINE OF RT_VERKF.
  CLEAR LW_TAB.
  READ TABLE IT_TAB INTO LW_TAB  INDEX  E_ROW_ID-INDEX.

  CLEAR RS_VERKF.
  RS_VERKF-SIGN   = 'I'.
  RS_VERKF-OPTION = 'EQ'.
  RS_VERKF-LOW    = LW_TAB-VERKF.
  APPEND RS_VERKF TO RT_VERKF.

  CASE E_COLUMN_ID.
    WHEN 'ZSH'.

      SUBMIT ZMMRP0260_JS
        WITH S_VERKF IN RT_VERKF
        AND RETURN.

      PERFORM DATA.
      PERFORM FM_REFRESH_ALV.

  ENDCASE.

ENDFORM.
