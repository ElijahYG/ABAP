  DATA:

*&---------------------------------------------------------------------*
*&读取EXCEL内容到任意内表
* I_PATH  : Excel文档路径
* O_EXCEL : 从EXCEL中获取到的内表数据
*&---------------------------------------------------------------------*
FORM FM_READ_EXCEL USING I_PATH O_EXCEL TYPE STANDARD TABLE.
  DATA:
    L_FILE  TYPE RLGRAP-FILENAME,
    LW_LINE TYPE REF TO DATA,
    LT_TAB  TYPE STANDARD TABLE OF ALSMEX_TABLINE,
    LW_TAB  TYPE ALSMEX_TABLINE.
  FIELD-SYMBOLS:
    <LINE>,
    <FS>.
  CREATE DATA LW_LINE LIKE LINE OF O_EXCEL.
  ASSIGN LW_LINE->* TO <LINE>.
  L_FILE = I_PATH.
  TRANSLATE L_FILE TO UPPER CASE. "字符转换成大写
  CALL FUNCTION 'ALSM_EXCEL_TO_INTERNAL_TABLE'
    EXPORTING
      FILENAME                = L_FILE
      I_BEGIN_COL             = 1
      I_BEGIN_ROW             = 1
      I_END_COL               = 250
      I_END_ROW               = 65536
    TABLES
      INTERN                  = LT_TAB
    EXCEPTIONS
      INCONSISTENT_PARAMETERS = 1
      UPLOAD_OLE              = 2
      OTHERS                  = 3.
  IF SY-SUBRC <> 0 .
    MESSAGE E001(00) WITH '上传文件有误！' SPACE SPACE SPACE.
  ENDIF.

  DELETE LT_TAB WHERE ROW = 1.  "删除字段名行

  REFRESH:
    O_EXCEL.
  LOOP AT LT_TAB INTO LW_TAB.

    ON CHANGE OF LW_TAB-ROW.
      IF SY-TABIX NE 1.
        APPEND  <LINE> TO O_EXCEL.
        CLEAR <LINE>.
      ENDIF.
    ENDON.

    ASSIGN COMPONENT LW_TAB-COL OF STRUCTURE <LINE> TO <FS>.
    TRY.
        <FS> = LW_TAB-VALUE.
      CATCH CX_ROOT.
        MESSAGE E001(00) WITH '上传文件的列多于接收内表的列' SPACE SPACE SPACE.
    ENDTRY.

    AT LAST.
      APPEND <LINE> TO O_EXCEL.
    ENDAT.

  ENDLOOP.

  IF O_EXCEL IS INITIAL.
    MESSAGE E001(00) WITH '上传文件为空' SPACE SPACE SPACE.
  ELSE.
    IT_TAB_01[] = O_EXCEL[].
  ENDIF.

ENDFORM.

*& 通过条码取得批次特性
*&---------------------------------------------------------------------*
FORM FM_GET_CHARACTERS.
* 定义结构
  TYPES: BEGIN OF STRU_CABN,
           ATINN TYPE CABN-ATINN,
           ATNAM TYPE CABN-ATNAM,
         END OF STRU_CABN,
         BEGIN OF STRU_AUSP,
           ATINN    TYPE AUSP-ATINN,
           ATWRT    TYPE AUSP-ATWRT,
           OBJEK    TYPE AUSP-OBJEK,
           CUOBJ_BM TYPE MCH1-CUOBJ_BM,
         END OF STRU_AUSP,
         BEGIN OF STRU_MCH1,
           MATNR TYPE MCH1-MATNR,
           CHARG TYPE MCH1-CHARG,
*      CUOBJ_BM TYPE MCH1-CUOBJ_BM,
         END OF STRU_MCH1.

  DATA:
    IT_CABN       TYPE STANDARD TABLE OF STRU_CABN,
    WA_CABN       TYPE                   STRU_CABN,

    IT_AUSP       TYPE STANDARD TABLE OF STRU_AUSP,
    WA_AUSP       TYPE                   STRU_AUSP,

    IT_MCH1       TYPE STANDARD TABLE OF STRU_MCH1,
    WA_MCH1       TYPE                   STRU_MCH1,

    IT_CHARG      TYPE STANDARD TABLE OF ZMM_CHARG,

    IT_CHARACTERS TYPE STANDARD TABLE OF ZMM_CHARACTERS,
    WA_CHARACTERS TYPE                   ZMM_CHARACTERS,

    RT_ATWRT      TYPE RANGE OF AUSP-ATWRT,
    RW_ATWRT      LIKE LINE OF RT_ATWRT.
******1.通过输入的批次特性值取相应的内部对象编号,通过对象编号取物料批次号*******
  LOOP AT IT_TAB_01 INTO WA_TAB_01.
    CLEAR RW_ATWRT.
    RW_ATWRT-SIGN   = 'I'.
    RW_ATWRT-OPTION = 'EQ'.
    RW_ATWRT-LOW    = WA_TAB_01-ZSPTM.
    APPEND RW_ATWRT TO RT_ATWRT.
  ENDLOOP.
*1.取批次特性字段对应的内部特性
  SELECT ATINN
         ATNAM
    FROM CABN
    INTO TABLE IT_CABN
    WHERE ATNAM = 'ZSPTM'.
*通过内部特性取相应的特征值和对象编号
  SELECT ATINN
         ATWRT
         OBJEK
*         OBJEK
    FROM AUSP
    INTO TABLE IT_AUSP
    FOR ALL ENTRIES IN IT_CABN
    WHERE ATINN = IT_CABN-ATINN
      AND ATWRT IN RT_ATWRT.
*通过对象编号取物料批次号
  LOOP AT IT_AUSP INTO WA_AUSP.
    WA_AUSP-CUOBJ_BM = WA_AUSP-OBJEK.
    MODIFY IT_AUSP FROM WA_AUSP TRANSPORTING CUOBJ_BM.
  ENDLOOP.
  SELECT MATNR
         CHARG
*         CUOBJ_BM
    FROM MCH1
    INTO TABLE IT_MCH1
    FOR ALL ENTRIES IN IT_AUSP
    WHERE CUOBJ_BM = IT_AUSP-CUOBJ_BM.

  IF IT_MCH1 IS INITIAL.
    MESSAGE S001(00) WITH '输入条码没有相应的批次号' DISPLAY LIKE 'E'.
    LEAVE LIST-PROCESSING.
  ELSE.
    IT_CHARG = IT_MCH1.
  ENDIF.

  call function 'ZSDRP0020_FM_01'
    TABLES
      T_CHARG      = IT_CHARG
      T_CHARACTERS = IT_CHARACTERS.

  SORT IT_CHARACTERS BY ZSPTM.
  LOOP AT IT_TAB_01 INTO WA_TAB_01.
    CLEAR WA_CHARACTERS.
    READ TABLE IT_CHARACTERS INTO WA_CHARACTERS WITH KEY ZSPTM = WA_TAB_01-ZSPTM BINARY SEARCH.
    IF SY-SUBRC = 0 AND WA_TAB_01-ZBQMC = WA_CHARACTERS-ZBQMC.
      CLEAR WA_ITEM.
      WA_ITEM-MATNR = WA_CHARACTERS-MATNR.
      WA_ITEM-CHARG = WA_CHARACTERS-CHARG.
      WA_ITEM-MENGE = WA_TAB_01-MENGE.
      WA_ITEM-MWSKZ = 'J1'.
      WA_ITEM-WERKS = 'DC80'.
      WA_ITEM-LGORT = '1401'.
      WA_ITEM-NETPR = WA_TAB_01-ZSPDJ.
      WA_ITEM-MEINS = 'ST'.
      APPEND WA_ITEM TO IT_ITEM.
    ELSE.
      WA_NO_CHAEG        = WA_TAB_01.
      WA_NO_CHAEG-ZBQMC1 = WA_CHARACTERS-ZBQMC.
      APPEND WA_TAB_01 TO IT_NO_CHARG.
    ENDIF.
  ENDLOOP.

  IF IT_NO_CHARG IS NOT INITIAL.
    CLEAR GL_CHECK.
    GL_CHECK = 'X'.
    RETURN.
  ENDIF.

ENDFORM.
