FORM FM_GET_DATA.

  SELECT MKPF~MBLNR MKPF~MJAHR MKPF~BUDAT MSEG~WERKS EKPO~WERKS
*         EKKO~EBELN EKKO~BSART
    INTO TABLE IT_OUT_TAB
    FROM MKPF
    INNER JOIN MSEG ON MSEG~MBLNR = MKPF~MBLNR AND MSEG~MJAHR = MKPF~MJAHR
    INNER JOIN EKPO ON MSEG~EBELN = EKPO~EBELN AND MSEG~EBELP = EKPO~EBELP
    INNER JOIN EKKO ON MSEG~EBELN = EKKO~EBELN
    WHERE MSEG~WERKS IN S_WERKS1
      AND MSEG~BUDAT_MKPF IN S_BUDAT
      AND MSEG~MBLNR IN S_MBLNR
      AND EKPO~WERKS IN S_WERKS2
      AND EKKO~BSART IN ('ZSUB','ZSTO')
      AND MSEG~BWART IN ('641','643','647')
      AND MSEG~SHKZG EQ 'H'
      .

  SORT IT_OUT_TAB BY MBLNR WERKS_GI WERKS_GR.  "EBELP.

  DELETE ADJACENT DUPLICATES FROM IT_OUT_TAB COMPARING MBLNR WERKS_GI WERKS_GR.

ENDFORM.
