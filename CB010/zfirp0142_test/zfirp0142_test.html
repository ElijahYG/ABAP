* I_PATH  : Excel文档路径
* O_EXCEL : 从EXCEL中获取到的内表数据
*&---------------------------------------------------------------------*
FORM FM_READ_EXCEL USING I_PATH O_EXCEL TYPE STANDARD TABLE.
  DATA:
    L_FILE  TYPE RLGRAP-FILENAME,
    LW_LINE TYPE REF TO DATA,
    LT_TAB  TYPE STANDARD TABLE OF ALSMEX_TABLINE,
    LW_TAB  TYPE ALSMEX_TABLINE.
  FIELD-SYMBOLS:
    <LINE>,
    <FS>.
  CREATE DATA LW_LINE LIKE LINE OF O_EXCEL.
  ASSIGN LW_LINE->* TO <LINE>.
  L_FILE = I_PATH.
  TRANSLATE L_FILE TO UPPER CASE. "字符转换成大写
  CALL FUNCTION 'ALSM_EXCEL_TO_INTERNAL_TABLE'
    EXPORTING
      FILENAME                = L_FILE
      I_BEGIN_COL             = 1
      I_BEGIN_ROW             = 1
      I_END_COL               = 250
      I_END_ROW               = 65536
    TABLES
      INTERN                  = LT_TAB
    EXCEPTIONS
      INCONSISTENT_PARAMETERS = 1
      UPLOAD_OLE              = 2
      OTHERS                  = 3.
  IF SY-SUBRC <> 0 .
    MESSAGE E001(00) WITH '上传文件有误！' SPACE SPACE SPACE.
  ENDIF.

  DELETE LT_TAB WHERE ROW = 1.  "删除字段名行

  REFRESH:
    O_EXCEL.
  LOOP AT LT_TAB INTO LW_TAB.

    ON CHANGE OF LW_TAB-ROW.
      IF SY-TABIX NE 1.
        APPEND  <LINE> TO O_EXCEL.
        CLEAR <LINE>.
      ENDIF.
    ENDON.

    ASSIGN COMPONENT LW_TAB-COL OF STRUCTURE <LINE> TO <FS>.
    TRY.
        <FS> = LW_TAB-VALUE.
      CATCH CX_ROOT.
        MESSAGE E001(00) WITH '上传文件的列多于接收内表的列' SPACE SPACE SPACE.
    ENDTRY.

    AT LAST.
      APPEND <LINE> TO O_EXCEL.
    ENDAT.


  ENDLOOP.


  IF O_EXCEL IS INITIAL.
    MESSAGE E001(00) WITH '上传文件为空' SPACE SPACE SPACE.
  ELSE.
    IT_OUT_TAB[] = O_EXCEL[].
  ENDIF.

ENDFORM.

FORM FM_FILL_DATA USING I_PATH
                        T_EXCEL TYPE TYT_EXCEL.
  DATA:
    LW_EXCEL   TYPE TY_EXCEL,
    L_ROW      TYPE I,
    L_PATH     TYPE STRING,
    L_EXCEL    TYPE OLE2_OBJECT,  "application
    L_WORKBOOK TYPE OLE2_OBJECT,  "workbook

    L_CELL     TYPE OLE2_OBJECT.  "单元格

**********  STEP1:打开excel程序
  CREATE OBJECT L_EXCEL 'EXCEL.APPLICATION'.
* 0：不显示（后台运行），1：显示
  SET PROPERTY OF L_EXCEL 'VISIBLE' = 0.
* 不显示警告消息
  SET PROPERTY OF L_EXCEL 'DISPLAYALERTS' = 0.

  CALL METHOD OF L_EXCEL 'SHEETS' = L_SHEET
    EXPORTING
      #1  = 1.

* SET PROPERTY OF l_sheet 'NAME' = '应交税金明细表'. "设定sheet名称

        #2      = LW_EXCEL-COL

    L_WORKBOOK,
    L_EXCEL.

FORM FM_DEL_DATA.

  TYPES: BEGIN OF STRU_JSD,
           ZJSD_NO TYPE ZFIRP0140_HEADER-ZJSD_NO,
           MESSAGE TYPE CHAR40,
         END OF STRU_JSD.

  DATA: IT_JSD TYPE STANDARD TABLE OF STRU_JSD,
        WA_JSD TYPE                   STRU_JSD.

  DATA: L_JSD      TYPE STRING,
        L_NUM      TYPE STRING VALUE '1234567890',
        L_LENGTH   TYPE I,
        L_POSITION TYPE I VALUE 0.

  DATA:
    IT_TAB_HEADER1 TYPE STANDARD TABLE OF ZFIRP0140_HEADER,
    IT_TAB_ITEM11  TYPE STANDARD TABLE OF ZFIRP0140_ITEM1.

  LOOP AT IT_OUT_TAB INTO WA_OUT_TAB.
    CLEAR WA_JSD.
*    WA_JSD-MESSAGE = WA_OUT_TAB-MESSAGE.
*    APPEND WA_JSD TO IT_JSD.

    L_LENGTH   = STRLEN( WA_OUT_TAB-MESSAGE ).
    L_POSITION = 0.

    WHILE L_POSITION LT L_LENGTH.

      IF WA_OUT_TAB-MESSAGE+L_POSITION(1) CA L_NUM.
        CONCATENATE WA_JSD-MESSAGE WA_OUT_TAB-MESSAGE+L_POSITION(1) INTO WA_JSD-MESSAGE.
        CONDENSE WA_JSD-MESSAGE NO-GAPS.
      ENDIF.
      L_POSITION = L_POSITION + 1.

    ENDWHILE.

    WA_JSD-ZJSD_NO = WA_JSD-MESSAGE.

    APPEND WA_JSD TO IT_JSD.

  ENDLOOP.

  SELECT * FROM ZFIRP0140_HEADER INTO TABLE IT_TAB_HEADER1
    FOR ALL ENTRIES IN IT_JSD
    WHERE ZJSD_NO = IT_JSD-ZJSD_NO.

  SELECT * FROM ZFIRP0140_ITEM1 INTO TABLE IT_TAB_ITEM11
   FOR ALL ENTRIES IN IT_JSD
   WHERE ZJSD_NO = IT_JSD-ZJSD_NO.

  DELETE ZFIRP0140_HEADER FROM TABLE IT_TAB_HEADER1.
  DELETE ZFIRP0140_ITEM1 FROM TABLE IT_TAB_ITEM11.

ENDFORM.
