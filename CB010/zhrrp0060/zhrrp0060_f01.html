FORM FRM_GET_WERKS .
  SELECT WERKS BUKRS NAME1
    INTO TABLE GT_WERKS
    FROM T001W AS W
    INNER JOIN T001K AS K ON K~BWKEY = W~BWKEY
    WHERE K~BUKRS IN S_BUKRS .
  IF SY-SUBRC = 0 .
    LOOP AT GT_WERKS INTO GS_WERKS .
      GRS_WERKS-SIGN = 'I' .
      GRS_WERKS-OPTION = 'EQ' .
      GRS_WERKS-LOW = GS_WERKS-WERKS .
      GRS_WERKS-HIGH = '' .
      APPEND GRS_WERKS TO GRT_WERKS .
    ENDLOOP.
  ENDIF.
ENDFORM.

FORM FRM_GET_ZYD .
  DATA: LV_DATE TYPE BUDAT,
        LRT_DATE TYPE RANGE OF MKPF-BUDAT,
        LRS_DATE LIKE LINE OF LRT_DATE .

  REFRESH LRT_DATE .

  CLEAR LRS_DATE .
  LRS_DATE-SIGN = 'I' .
  LRS_DATE-OPTION = 'BT' .
  LV_DATE = GV_THIS_M && '01' .
  LRS_DATE-LOW = LV_DATE .
  CALL FUNCTION 'DATE_GET_MONTH_LASTDAY'
    EXPORTING
      I_DATE        = LV_DATE
    IMPORTING
      E_DATE        = LV_DATE .
  LRS_DATE-HIGH = LV_DATE .
  APPEND LRS_DATE TO LRT_DATE .

  CLEAR LRS_DATE .
  LRS_DATE-SIGN = 'I' .
  LRS_DATE-OPTION = 'BT' .
  LV_DATE = GV_LAST_M && '01' .
  LRS_DATE-LOW = LV_DATE .
  CALL FUNCTION 'DATE_GET_MONTH_LASTDAY'
    EXPORTING
      I_DATE        = LV_DATE
    IMPORTING
      E_DATE        = LV_DATE .
  LRS_DATE-HIGH = LV_DATE .
  APPEND LRS_DATE TO LRT_DATE .

  CLEAR LRS_DATE .
  LRS_DATE-SIGN = 'I' .
  LRS_DATE-OPTION = 'BT' .
  LV_DATE = GV_LAST_Y_M && '01' .
  LRS_DATE-LOW = LV_DATE .
  CALL FUNCTION 'DATE_GET_MONTH_LASTDAY'
    EXPORTING
      I_DATE        = LV_DATE
    IMPORTING
      E_DATE        = LV_DATE .
  LRS_DATE-HIGH = LV_DATE .
  APPEND LRS_DATE TO LRT_DATE .

  SELECT H~EX_YERQ H~EX_MDBM H~EX_YWDJLX H~EX_STA
         I~EX_SPID I~CHARG I~EX_SL I~EX_XSKZ I~EX_DJJE
         A~EXTWG
    INTO TABLE GT_ZYD
    FROM ZYDHEAD AS H
    INNER JOIN ZYDITEM AS I ON I~EX_PBH = H~EX_PBH
    INNER JOIN MARA AS A ON A~MATNR = I~EX_SPID
    WHERE H~EX_YERQ IN LRT_DATE
      AND H~EX_MDBM IN GRT_WERKS
      AND H~EX_YWDJLX IN ( '32' , '33' ) .
ENDFORM.
