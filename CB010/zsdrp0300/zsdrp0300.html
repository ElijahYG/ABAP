  CLEAR IT_LJKPZ.

*按照屏幕输入选择界面从VBRK、VBRP中取出商品类目为1104的
FORM GET_DATA.
  DATA: LT_CHARG      TYPE STANDARD TABLE OF ZMM_CHARG,     "批次属性输入
        LT_CHARACTERS TYPE STANDARD TABLE OF ZMM_CHARACTERS, "批次属性输出
        LW_CHARACTERS TYPE                   ZMM_CHARACTERS.
  DATA LT_KZWI6 LIKE IT_TABLE.
*------------------补齐前置零----------------------*
*  CALL FUNCTION 'CONVERSION_EXIT_ALPHA_INPUT'
*    EXPORTING
*      INPUT  = S_ZGYSDM-LOW
*    IMPORTING
*      OUTPUT = S_ZGYSDM-LOW.
*
*  CALL FUNCTION 'CONVERSION_EXIT_ALPHA_INPUT'
*    EXPORTING
*      INPUT  = S_ZGYSDM-HIGH
*    IMPORTING
*      OUTPUT = S_ZGYSDM-HIGH.
*------------------补齐前置零----------------------*

  SELECT K~VBELN
         K~FKDAT
         K~KUNAG
         K~EX_POSNUM
         P~MATNR
         P~CHARG
         P~MATKL
         P~KZWI6
    FROM VBRK AS K
       INNER JOIN
         VBRP AS P ON K~VBELN EQ P~VBELN
    INTO CORRESPONDING FIELDS OF TABLE IT_TABLE
    WHERE K~KUNAG     IN S_KUNAG
     AND  K~FKDAT     IN S_FKDAT
     AND  K~EX_POSNUM IN S_POSNUM
     AND  K~VBELN     IN S_VBELN
     AND  P~MATKL     = '1104'."只取出商品类目为1104的。1104是饰品类项坠。

  DATA ZNUM1 TYPE VBRP-KZWI6.
  DATA ZNUM2 TYPE VBRP-KZWI6.
*取配置表中返卷起步金额
  LOOP AT IT_LJKPZ INTO WA_LJKPZ.
    IF ZNUM1 IS INITIAL.
      ZNUM1 = WA_LJKPZ-ZQSJE.
    ELSE.
      ZNUM2 = WA_LJKPZ-ZQSJE.

      IF ZNUM1 < ZNUM2.
      ELSE.
        ZNUM1 = ZNUM2.
      ENDIF.
    ENDIF.
  ENDLOOP.
*删除不够资格返卷的凭证
  LOOP  AT IT_TABLE ASSIGNING <WA_TABLE>.
    IF ABS( <WA_TABLE>-KZWI6 )  >= ZNUM1.
      APPEND <WA_TABLE> TO LT_KZWI6.
    ENDIF.
  ENDLOOP.

  CLEAR IT_TABLE.
  IT_TABLE = LT_KZWI6.

  SORT IT_TABLE BY MATNR CHARG.
  MOVE-CORRESPONDING IT_TABLE TO LT_CHARG.

  call function 'ZSDRP0020_FM_01'
    TABLES
      T_CHARG      = LT_CHARG
      T_CHARACTERS = LT_CHARACTERS.

  SORT LT_CHARACTERS BY MATNR CHARG.

  LOOP  AT IT_TABLE ASSIGNING <WA_TABLE>.
    CLEAR LW_CHARACTERS.
    READ TABLE LT_CHARACTERS INTO LW_CHARACTERS WITH KEY MATNR = <WA_TABLE>-MATNR CHARG = <WA_TABLE>-CHARG BINARY SEARCH.
    IF  SY-SUBRC = 0.
      <WA_TABLE>-ZGYSDM = LW_CHARACTERS-ZGYSDM .
      <WA_TABLE>-ZSPTM  = LW_CHARACTERS-ZSPTM.
      <WA_TABLE>-ZZPBJ  = LW_CHARACTERS-ZZPBJ.
    ENDIF.
  ENDLOOP.

  DELETE IT_TABLE WHERE ZZPBJ <> '1'."只保留增礼金卡的。1为赠礼金卡2为赠链
  DELETE IT_TABLE WHERE ZGYSDM  NOT IN S_ZGYSDM."删除不在选定范围内的
ENDFORM.

    ENDIF.
