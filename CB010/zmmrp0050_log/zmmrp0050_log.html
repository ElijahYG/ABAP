FORM FM_SET_PRINT.
  FIELD-SYMBOLS <FS> TYPE TY_HEAD.
  DATA:
    L_TABIX TYPE I,
    L_LINE  TYPE I,
    BEGIN OF LW_EKBE,
      EBELN TYPE EKBE-EBELN,
      EBELP TYPE EKBE-EBELP,
      ZEKKN TYPE EKBE-ZEKKN,
      VGABE TYPE EKBE-VGABE,
      GJAHR TYPE EKBE-GJAHR,
      BELNR TYPE EKBE-BELNR,
      BUZEI TYPE EKBE-BUZEI,
    END OF LW_EKBE,
    LW_NEXT LIKE LW_EKBE,
    LT_EKBE LIKE LW_EKBE OCCURS 0,
    LT_HEAD LIKE GT_HEAD.
* 取发货商品凭证
  LT_HEAD = GT_HEAD.
  DELETE LT_HEAD WHERE EBELN IS INITIAL.
  IF LT_HEAD IS NOT INITIAL.
    SELECT EBELN
           EBELP
           ZEKKN
           VGABE
           GJAHR
           BELNR
           BUZEI
      FROM EKBE
      INTO TABLE LT_EKBE
       FOR ALL ENTRIES IN LT_HEAD
     WHERE EBELN = LT_HEAD-EBELN
       AND VGABE = '6'
      .
    SORT LT_EKBE BY EBELN BELNR.
    DELETE ADJACENT DUPLICATES FROM LT_EKBE COMPARING EBELN BELNR.

  ENDIF.
* 发货商品凭证更新到内表
  L_LINE = LINES( LT_EKBE ) .
  LOOP AT GT_HEAD ASSIGNING <FS>.
    CLEAR <FS>-MBLNR_TAB.
    CLEAR <FS>-MBLNR.
    CLEAR LW_EKBE.
    READ TABLE LT_EKBE INTO LW_EKBE
                   WITH KEY EBELN = <FS>-EBELN
                     BINARY SEARCH.
    IF SY-SUBRC = 0.
      L_TABIX = SY-TABIX.
      <FS>-MBLNR = LW_EKBE-BELNR.
      DO.
        APPEND LW_EKBE-BELNR TO <FS>-MBLNR_TAB.
        L_TABIX = L_TABIX + 1.
        IF L_TABIX > L_LINE.
          EXIT.
        ENDIF.
        READ TABLE LT_EKBE INTO LW_NEXT INDEX L_TABIX.
        IF LW_NEXT-EBELN <> <FS>-EBELN.
          EXIT.
        ENDIF.
      ENDDO.
    ENDIF.
*   显示打印按钮
    CLEAR <FS>-PRINT.
    IF <FS>-MBLNR_TAB[] IS NOT INITIAL.
      <FS>-PRINT = '@0X@'.
    ENDIF.
  ENDLOOP.
ENDFORM.

FORM FM_USER_COMMAND USING I_UCOMM TYPE SY-UCOMM.
  DATA:
    LT_HEAD TYPE TYT_HEAD.
* 刷新内表
  CALL METHOD GO_ALV_HEAD->CHECK_CHANGED_DATA.
  CASE I_UCOMM.
    WHEN 'EXEC'.
*     检查
      LT_HEAD = GT_HEAD.
      DELETE LT_HEAD WHERE SEL IS INITIAL.
      IF LT_HEAD IS INITIAL.
        MESSAGE S000(ZCB) WITH '请至少选择一条数据' DISPLAY LIKE 'E'.
        RETURN.
      ENDIF.
*     再执行
      PERFORM FM_EXEC.
*     刷新数据
      PERFORM FM_GET_DATA.

    WHEN 'ALL'.
      PERFORM FM_ALL.
    WHEN 'DALL'.
      PERFORM FM_DALL.

  ENDCASE.
* 刷新ALV
  PERFORM FM_REFRESH_ALV.
ENDFORM.
*&---------------------------------------------------------------------*
*&再执行
*&---------------------------------------------------------------------*
FORM FM_EXEC.
  DATA:
    LT_HEAD   TYPE TYT_HEAD,
    LW_HEAD   TYPE TY_HEAD,
    L_MSG     TYPE CHAR200,
    L_S_COUNT TYPE I,
    L_COUNT   TYPE I,
    L_PER(3)  TYPE P DECIMALS 2,
    L_STR     TYPE CHAR200.
  LT_HEAD = GT_HEAD.
  DELETE LT_HEAD WHERE SEL IS INITIAL.
  L_COUNT = LINES( LT_HEAD ).

  LOOP AT LT_HEAD INTO LW_HEAD.
*   进度条
    L_PER = SY-TABIX * 100 / L_COUNT .
    L_STR = '共' && L_COUNT && '条数据，正在处理第' && SY-TABIX && '条'.
    CALL FUNCTION 'SAPGUI_PROGRESS_INDICATOR'
      EXPORTING
        PERCENTAGE = L_PER
        TEXT       = L_STR.

    call function 'ZMMRP0050_POST'
      EXPORTING
        I_ZFHNO = LW_HEAD-ZFHNO.
  ENDLOOP.

* 汇总处理结果
  IF LT_HEAD IS NOT INITIAL.
    SELECT COUNT(*)
      FROM ZMMRP0050_HEAD
      INTO L_S_COUNT
       FOR ALL ENTRIES IN LT_HEAD
     WHERE ZFHNO  = LT_HEAD-ZFHNO
       AND STATUS = 'S'
       .
    L_COUNT = L_COUNT - L_S_COUNT.
    L_MSG = '处理结果：成功' && L_S_COUNT && '条，失败' && L_COUNT && '条'.
  ENDIF.
  MESSAGE S001(00) WITH L_MSG.
ENDFORM.

*    IMPORTING
*       ET_INDEX_ROWS = LT_ROWS
*      .
*  IF LT_ROWS IS INITIAL.
*    MESSAGE S001(00) WITH '没有选中任何数据' DISPLAY LIKE 'E'.
*    EXIT.
*  ENDIF.
** 数据加入LT_HEAD
*  SORT LT_ROWS BY INDEX DESCENDING.
*  LOOP AT LT_ROWS INTO LW_ROWS.
*    READ TABLE GT_HEAD INTO LW_HEAD INDEX LW_ROWS-INDEX.
*    APPEND LW_HEAD TO LT_HEAD.

    FROM ZMMRP0050_ITEM
    INTO TABLE GT_ITEM
   WHERE ZFHNO = LW_HEAD-ZFHNO
      .
  LOOP AT GT_ITEM INTO LW_ITEM.
    LW_ITEM-P_JS = LW_ITEM-JS.
    MODIFY GT_ITEM FROM LW_ITEM TRANSPORTING P_JS.
  ENDLOOP.

      TABLES
        T_CHARG            = LT_CHARG
        T_CHARG_CHARACTERS = LT_CHARG_CHARACTERS.
*   取物料信息
    LT_ITEM = GT_ITEM.

    DELETE ADJACENT DUPLICATES FROM LT_ITEM COMPARING MATNR.
    IF LT_ITEM IS NOT INITIAL.

      SELECT MATNR
             MEINS
             TEMPB
             EXTWG
        FROM MARA
        INTO TABLE LT_MARA
         FOR ALL ENTRIES IN LT_ITEM
       WHERE MATNR = LT_ITEM-MATNR
        .
*     取扩展物料信息
      SELECT MATNR
             EX_PZ01
             EX_PZ
             EX_ZT01
        INTO TABLE LT_ZMMEX0010
        FROM ZMMEX0010_MARA
         FOR ALL ENTRIES IN LT_ITEM
       WHERE MATNR = LT_ITEM-MATNR
        .
    ENDIF.
*   处理证书号及标枪名称
    SORT LT_CHARG_CHARACTERS BY MATNR CHARG ATNAM.
    SORT LT_MARA BY MATNR.
    SORT LT_ZMMEX0010  BY MATNR.
    LOOP AT GT_ITEM ASSIGNING <FS_ITEM>.
      CLEAR LW_CHARG_CHARACTERS.
      READ TABLE LT_CHARG_CHARACTERS INTO LW_CHARG_CHARACTERS
                                  WITH KEY MATNR = <FS_ITEM>-MATNR
                                           CHARG = <FS_ITEM>-CHARG
                                           ATNAM = 'ZZS1'
                                    BINARY SEARCH.
      IF SY-SUBRC = 0.
        <FS_ITEM>-ZZS1 = LW_CHARG_CHARACTERS-ATWRT.
      ENDIF.

*---------------------modify by linan-----------------------
      CLEAR LW_CHARG_CHARACTERS.

                                  WITH KEY MATNR = <FS_ITEM>-MATNR
                                           CHARG = <FS_ITEM>-CHARG
                                           ATNAM = 'ZZS5'
                                    BINARY SEARCH.
      IF SY-SUBRC = 0.
        <FS_ITEM>-ZZS5 = LW_CHARG_CHARACTERS-ATWRT.
      ENDIF.
*---------------------end-----------------------------------

      CLEAR LW_CHARG_CHARACTERS.
      READ TABLE LT_CHARG_CHARACTERS INTO LW_CHARG_CHARACTERS
                                  WITH KEY MATNR = <FS_ITEM>-MATNR

      ENDIF.
*     标签条码
      CLEAR LW_CHARG_CHARACTERS.
      READ TABLE LT_CHARG_CHARACTERS INTO LW_CHARG_CHARACTERS
                                  WITH KEY MATNR = <FS_ITEM>-MATNR
                                           CHARG = <FS_ITEM>-CHARG
                                           ATNAM = 'ZSPTM'
                                    BINARY SEARCH.
      IF SY-SUBRC = 0.
        <FS_ITEM>-BQTM = LW_CHARG_CHARACTERS-ATWRT.
      ENDIF.
*     标签包码
      CLEAR LW_CHARG_CHARACTERS.
      READ TABLE LT_CHARG_CHARACTERS INTO LW_CHARG_CHARACTERS
                                  WITH KEY MATNR = <FS_ITEM>-MATNR
                                           CHARG = <FS_ITEM>-CHARG

FORM FM_HOTSPOT_CLICK USING E_ROW_ID TYPE LVC_S_ROW
                            E_COLUMN_ID  TYPE LVC_S_COL
                            ES_ROW_NO    TYPE LVC_S_ROID.
  DATA:
    L_MBLNR   TYPE MBLNR,
    LT_SELTAB TYPE RSPARAMS OCCURS 0,
    LW_SELTAB TYPE RSPARAMS,
    LW_HEAD   TYPE TY_HEAD.

  READ TABLE GT_HEAD INTO LW_HEAD  INDEX  E_ROW_ID-INDEX.
  CASE E_COLUMN_ID.
    WHEN 'EBELN'.
      IF LW_HEAD-EBELN IS NOT INITIAL.
        SET PARAMETER ID 'BES' FIELD LW_HEAD-EBELN.
        CALL TRANSACTION 'ME23N' AND SKIP FIRST SCREEN.
      ENDIF.
    WHEN 'VBELN_DN'.
      IF LW_HEAD-VBELN_DN IS NOT INITIAL.
        SET PARAMETER ID 'VL' FIELD LW_HEAD-VBELN_DN.
        CALL TRANSACTION 'VL03N' AND SKIP FIRST SCREEN.
      ENDIF.
    WHEN 'PRINT'.
      IF LW_HEAD-MBLNR_TAB[] IS NOT INITIAL.
*       跳转
        LOOP AT LW_HEAD-MBLNR_TAB INTO L_MBLNR.
          LW_SELTAB-SELNAME      = 'S_MBLNR'.
          LW_SELTAB-KIND         = 'S'.
          LW_SELTAB-SIGN         = 'I'.
          LW_SELTAB-OPTION       = 'EQ'.
          LW_SELTAB-LOW          = L_MBLNR.
          APPEND LW_SELTAB TO LT_SELTAB.
        ENDLOOP.
        LW_SELTAB-SELNAME      = 'S_BUDAT'.
        LW_SELTAB-KIND         = 'S'.
        LW_SELTAB-SIGN         = 'I'.
        LW_SELTAB-OPTION       = 'BT'.
        LW_SELTAB-LOW          = '20150101'.
        LW_SELTAB-HIGH         = '99991231'.
        APPEND LW_SELTAB TO LT_SELTAB.
        SUBMIT ZMMRP0120 WITH SELECTION-TABLE LT_SELTAB AND RETURN.
      ENDIF.
    WHEN 'MBLNR'.
      IF LW_HEAD-MBLNR IS NOT INITIAL.
        SET PARAMETER ID 'MBN' FIELD LW_HEAD-MBLNR.
        SET PARAMETER ID 'MJA' FIELD LW_HEAD-ADDAT+0(4).
        CALL TRANSACTION 'MB03' AND SKIP FIRST SCREEN.
      ENDIF.
  ENDCASE.
ENDFORM.
