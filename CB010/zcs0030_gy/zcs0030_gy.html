FORM FM_GET_ITEM USING E_ROW E_COLUMN.
  DATA:
    LT_HEAD TYPE STANDARD TABLE OF TY1,
    LW_HEAD LIKE LINE OF LT_HEAD,
    LT_ROWS TYPE LVC_T_ROW,
    LW_ROWS TYPE LVC_S_ROW.

  READ TABLE IT_TAB INTO LW_HEAD INDEX E_ROW.
* 取明细
  SELECT *
    FROM MSEG
    INTO CORRESPONDING FIELDS OF TABLE IT_TAB2
   WHERE MBLNR = LW_HEAD-MBLNR
     AND MJAHR = LW_HEAD-MJAHR
      .
ENDFORM.

FORM LIGHT-AND-LINE_COLOR."指示灯 和行颜色
  DATA:L_LAYOUT TYPE DISVARIANT.
  L_LAYOUT-REPORT = SY-REPID.

  LOOP AT IT_TAB INTO WA_TAB.
    IF WA_TAB-USNAM = 'CUICHEN'.
      WA_TAB-TRAFFIC_LIGHT = '1'.     "指示灯
      WA_TAB-LINE_COLOR    = 'C610'. "行颜色
    ELSEIF WA_TAB-USNAM = 'LIJIA'.
      WA_TAB-TRAFFIC_LIGHT = '2'.     "指示灯
    ELSE.
      WA_TAB-TRAFFIC_LIGHT = '3'.
    ENDIF.
    MODIFY IT_TAB FROM WA_TAB.
  ENDLOOP.
ENDFORM.

FORM COL_COLOR. " 格颜色
  DATA L_ALVCOLOR TYPE SLIS_SPECIALCOL_ALV.
  LOOP AT IT_TAB INTO WA_TAB.
    IF WA_TAB-USNAM = 'LIJIA'.
      L_ALVCOLOR-FIELDNAME = 'USNAM'.
      L_ALVCOLOR-COLOR-COL = 6.       "<--- colour number
      L_ALVCOLOR-COLOR-INT = 0.       "<--- intensified flag
      L_ALVCOLOR-NOKEYCOL = 'X'.     "<--- key column flag
      APPEND L_ALVCOLOR TO WA_TAB-COLTAB.
    ELSE.
      CLEAR  WA_TAB-COLTAB[].
    ENDIF.
    MODIFY IT_TAB FROM WA_TAB TRANSPORTING COLTAB.
  ENDLOOP.
ENDFORM.

FORM ADJUST_TABLE .

  DATA LS_STYLEROW TYPE LVC_S_STYL .
  DATA LT_STYLETAB TYPE LVC_T_STYL .
  LOOP AT IT_TAB INTO WA_TAB.
    CLEAR:LS_STYLEROW.
    REFRESH:LT_STYLETAB.
    IF WA_TAB-USNAM = 'LXUJING'. "如果ISUSE标志有值，那么该行的ZRESON字段为不可输入，ISUSE字段为可输入

      INSERT LS_STYLEROW INTO TABLE LT_STYLETAB. "lt_styletab 为sorted table只能通过INSERT语句插入值
*      LS_STYLEROW-FIELDNAME = 'USNAM'.

ENDFORM.
