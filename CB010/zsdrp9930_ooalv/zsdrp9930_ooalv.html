  IF IT_TABLE IS NOT INITIAL.
    LOOP AT IT_TABLE  ASSIGNING <FS_TABLE> .
      IF <FS_TABLE>-AWTYP = 'MKPF' .
        <FS_TABLE>-MBLNR = <FS_TABLE>-AWKEY+0(10) .

      ENDIF.
    ENDLOOP.

    SELECT V~VBELN
           V~EX_POSNUM
           V~ERDAT
      FROM VBRK AS V
      INTO CORRESPONDING FIELDS OF TABLE LT_TABLE1
       FOR ALL ENTRIES IN IT_TABLE

FORM GET_IDOC.

FORM HANDLE_DOUBLE_CLICK USING E_ROW TYPE LVC_S_ROW
                             E_COLUMN TYPE LVC_S_COL
                            ES_ROW_NO TYPE LVC_S_ROID.
  DATA:
    LT_SELTAB TYPE TABLE OF RSPARAMS,
    LW_SELTAB LIKE LINE OF LT_SELTAB.

  FIELD-SYMBOLS <FS> TYPE STRU_0010.
  READ TABLE IT_TABLE ASSIGNING <FS> INDEX ES_ROW_NO-ROW_ID .
  CASE E_COLUMN-FIELDNAME.
    WHEN 'BELNR'."会计凭证编号（BELNR为ALV结构中显示的字段名称）
      SET PARAMETER ID 'BLN' FIELD <FS>-BELNR.
      SET PARAMETER ID 'BUK' FIELD <FS>-BUKRS.
      CALL TRANSACTION 'FB03' AND SKIP FIRST SCREEN.
    WHEN 'IDOC'."IDOC号
      LW_SELTAB-SELNAME      = 'CREDAT'."日期（CREDAT字段名称不能改）
      LW_SELTAB-SIGN         = 'I'.
      LW_SELTAB-OPTION       = 'BT'.
      LW_SELTAB-LOW          = <FS>-ERDAT."插入自己取到日期（字段名称无所谓）
      LW_SELTAB-HIGH         = <FS>-ERDAT.
      APPEND LW_SELTAB TO LT_SELTAB.
      CLEAR LW_SELTAB.
      LW_SELTAB-SELNAME      = 'DOCNUM'."IDOC号（DOCNUM字段名称不能改）
      LW_SELTAB-SIGN         = 'I'.
      LW_SELTAB-OPTION       = 'BT'.
      LW_SELTAB-LOW          = <FS>-IDOC.
      LW_SELTAB-HIGH        = <FS>-IDOC.
      APPEND LW_SELTAB TO LT_SELTAB.
      SUBMIT RSEIDOC2 WITH SELECTION-TABLE LT_SELTAB AND RETURN.
    WHEN 'VBELN'."开票凭证
      IF SY-SUBRC = 0 AND <FS>-VBELN IS NOT INITIAL .
        SET PARAMETER ID 'VF' FIELD <FS>-VBELN .
        CALL TRANSACTION 'VF03' AND SKIP FIRST SCREEN.
      ENDIF.
    WHEN 'MBLNR'."商品凭证
      IF SY-SUBRC = 0 AND <FS>-MBLNR IS NOT INITIAL .
        SET PARAMETER ID 'MBN' FIELD <FS>-MBLNR .
        SET PARAMETER ID 'MJA' FIELD <FS>-MJAHR .
        CALL TRANSACTION 'MB03' AND SKIP FIRST SCREEN.
      ENDIF.
    WHEN 'BELNR2'."会计凭证编号（BELNR为ALV结构中显示的字段名称）
      SET PARAMETER ID 'BLN' FIELD <FS>-BELNR2.
      SET PARAMETER ID 'BUK' FIELD <FS>-BUKRS.

      IF GO_ALV_ITEM IS INITIAL.
        CALL METHOD GO_SPLIT->SET_SASH_POSITION
          EXPORTING
            SASH_POSITION = 60.
        CREATE OBJECT GO_ALV_ITEM
          EXPORTING
            I_PARENT = GO_SPLIT->BOTTOM_RIGHT_CONTAINER.
