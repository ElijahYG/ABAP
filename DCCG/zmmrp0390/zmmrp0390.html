FORM FM_CONTROLL_SCREEN.

FORM FM_GET_DATA.
  DATA:
    LT_ZSDRP0090        TYPE ZSDRP0090 OCCURS 0,
    LT_PRE              TYPE ZSDRP0090 OCCURS 0,
    LT_TMP              TYPE ZSDRP0090 OCCURS 0,
    LT_CHARG            TYPE ZMM_CHARG OCCURS 0,
    LW_CHARG            TYPE ZMM_CHARG,
    LW_PRE              TYPE ZSDRP0090,
    L_TABIX             TYPE I,
    LT_CHARG_CHARACTERS TYPE ZMM_CHARG_CHARACTERS OCCURS 0,
    LW_CHARG_CHARACTERS TYPE ZMM_CHARG_CHARACTERS,
    LW_OUT_TAB          TYPE TY_OUT_TAB,
    LW_ZSDRP0090        TYPE ZSDRP0090.
  CLEAR GT_OUT_TAB.
* 取变价记录
  SELECT *
    FROM ZSDRP0090
    INTO TABLE LT_ZSDRP0090
   WHERE DATAB IN S_DATAB
     AND MATNR IN S_MATNR
     AND CHARG IN S_CHARG
     AND ZGYSDM IN S_LIFNR
     AND UZEIT IN S_UZEIT
    .
* 取之前的纪录
  LT_TMP = LT_ZSDRP0090.
  SORT LT_TMP BY MATNR CHARG.
  DELETE ADJACENT DUPLICATES FROM LT_TMP COMPARING MATNR CHARG.
  IF LT_TMP IS NOT INITIAL.
    SELECT *
      INTO TABLE LT_PRE
      FROM ZSDRP0090
       FOR ALL ENTRIES IN LT_TMP
     WHERE MATNR = LT_TMP-MATNR
       AND CHARG = LT_TMP-CHARG
      .
  ENDIF.
* 取批次特性
  MOVE-CORRESPONDING LT_ZSDRP0090 TO LT_CHARG.
  call function 'ZMM_GET_CHARG_CHARACTERS'
    TABLES
      T_CHARG            = LT_CHARG
      T_CHARG_CHARACTERS = LT_CHARG_CHARACTERS.
* 排序
  SORT LT_PRE BY MATNR CHARG DATAB UZEIT.
  SORT LT_CHARG_CHARACTERS BY MATNR CHARG ATNAM.
  LOOP AT LT_ZSDRP0090 INTO LW_ZSDRP0090.
*   取上一条更改纪录
    CLEAR LW_PRE.
    READ TABLE LT_PRE TRANSPORTING NO FIELDS
                  WITH KEY MATNR = LW_ZSDRP0090-MATNR
                           CHARG = LW_ZSDRP0090-CHARG
                           DATAB = LW_ZSDRP0090-DATAB
                           UZEIT = LW_ZSDRP0090-UZEIT
                    BINARY SEARCH.
    IF SY-SUBRC = 0.
      L_TABIX = SY-TABIX.
      L_TABIX = L_TABIX - 1.
      IF L_TABIX >= 1.
        READ TABLE LT_PRE INTO LW_PRE INDEX L_TABIX.
        IF LW_PRE-MATNR <> LW_ZSDRP0090-MATNR
        OR LW_PRE-CHARG <> LW_ZSDRP0090-CHARG.
          CLEAR LW_PRE.
        ENDIF.
      ENDIF.
    ENDIF.
*   取批次特性
*    CLEAR LW_CHARG_CHARACTERS.
*    READ TABLE LT_CHARG_CHARACTERS INTO LW_CHARG_CHARACTERS
*                               WITH KEY MATNR = LW_ZSDRP0090-MATNR
*                                        CHARG = LW_ZSDRP0090-CHARG
*                                        ATNAM = 'ZGYSDM'
*                                 BINARY SEARCH.
*    LW_OUT_TAB-LIFNR  = LW_CHARG_CHARACTERS-ATWRT.
    LW_OUT_TAB-LIFNR  = LW_ZSDRP0090-ZGYSDM.
    LW_OUT_TAB-MANTR  = LW_ZSDRP0090-MATNR.
    LW_OUT_TAB-CHARG  = LW_ZSDRP0090-CHARG.
    LW_OUT_TAB-UZEIT  = LW_ZSDRP0090-UZEIT.
    LW_OUT_TAB-UNAME  = LW_ZSDRP0090-UNAME.
    CLEAR LW_CHARG_CHARACTERS.
    READ TABLE LT_CHARG_CHARACTERS INTO LW_CHARG_CHARACTERS
                               WITH KEY MATNR = LW_ZSDRP0090-MATNR
                                        CHARG = LW_ZSDRP0090-CHARG
                                        ATNAM = 'ZBQMC'
                                 BINARY SEARCH.
    LW_OUT_TAB-ZBQMC  = LW_CHARG_CHARACTERS-ATWRT .
*----------MODIFY BY LINAN 20160220---------------------------
    CLEAR LW_CHARG_CHARACTERS.
    READ TABLE LT_CHARG_CHARACTERS INTO LW_CHARG_CHARACTERS
                               WITH KEY MATNR = LW_ZSDRP0090-MATNR
                                        CHARG = LW_ZSDRP0090-CHARG
                                        ATNAM = 'ZGYSSPTM'
                                 BINARY SEARCH.
    LW_OUT_TAB-ZGYSSPTM = LW_CHARG_CHARACTERS-ATWRT.
*-------------------END---------------------------------------
    LW_OUT_TAB-DATAB  = LW_ZSDRP0090-DATAB.
    LW_OUT_TAB-BGQJSJ = LW_PRE-ZPUR_COST       * '1.17'.
    LW_OUT_TAB-DQJSJ  = LW_ZSDRP0090-ZPUR_COST * '1.17'.
    LW_OUT_TAB-JSJZJZ = LW_OUT_TAB-DQJSJ - LW_OUT_TAB-BGQJSJ.
    LW_OUT_TAB-ZCKLSJ = LW_PRE-ZCKLSJ.
    LW_OUT_TAB-DQLSJ  = LW_ZSDRP0090-ZCKLSJ.
    LW_OUT_TAB-LSJZJZ = LW_ZSDRP0090-ZCKLSJ - LW_PRE-ZCKLSJ.
*****   MODIFY BY LIQUAN  2015.12.07   START ***************
    IF LW_OUT_TAB-BGQJSJ = LW_OUT_TAB-DQJSJ
   AND LW_OUT_TAB-ZCKLSJ = LW_OUT_TAB-DQLSJ.
      CONTINUE.
    ENDIF.
*****   MODIFY BY LIQUAN  2015.12.07     END ***************
*****   MODIFY BY LIQUAN  2015.12.10   START ***************
    IF LW_OUT_TAB-BGQJSJ IS INITIAL
   AND LW_OUT_TAB-ZCKLSJ IS INITIAL.
      CONTINUE.
    ENDIF.
*****   MODIFY BY LIQUAN  2015.12.10    END ***************
    APPEND LW_OUT_TAB TO GT_OUT_TAB.
  ENDLOOP.

  DELETE GT_OUT_TAB WHERE LSJZJZ EQ 0.

  IF GT_OUT_TAB IS INITIAL.
    MESSAGE S000(ZCB) WITH '未查到相关信息' DISPLAY LIKE 'E'.
    LEAVE LIST-PROCESSING.
  ENDIF.

ENDFORM.
