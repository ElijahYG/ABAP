    SELECT SINGLE  taklv

    call function 'Z_GET_TAX'
      EXPORTING

        IM_BUDAT = IM_BUDAT

        EM_TAXKD = LV_TAXKD.

    IF LV_TAXKD eq '1' and Lv_taklv eq '1'.

      EM_MWSKZ = 'J0'.

    ELSEIF  LV_TAXKD eq '0' and Lv_taklv eq '0'.
      EM_MWSKZ = 'J0'.
    ENDIF.

  ENDIF.



ENDFUNCTION.
