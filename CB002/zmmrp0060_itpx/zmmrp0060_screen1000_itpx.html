*&---------------------------------------------------------------------*

START-OF-SELECTION.
* 检查要货单
  PERFORM FM_CHECK_YHD.
* 处理界面
  PERFORM FM_INIT_SCREEN.
  CALL SCREEN 8000.

*&---------------------------------------------------------------------*
*&检查要货单
*&---------------------------------------------------------------------*
FORM FM_CHECK_YHD.
  DATA:
    LW_YHD TYPE ZSDIF0050_HEAD,
    L_USR  TYPE CHAR30.
* 检查要货单
  SELECT SINGLE *
    INTO LW_YHD
    FROM ZSDIF0050_HEAD
   WHERE DOC_NUMBER IN S_DOCNO
     AND ORD_TYP    = 'LX01'
     AND DOC_IND    = '2'
       .
  IF SY-SUBRC <> 0.
    MESSAGE S011(ZCB) WITH S_DOCNO-LOW DISPLAY LIKE 'E'.
    LEAVE LIST-PROCESSING.
  ENDIF.
* 收货门店和要货门店必须一致
  IF LW_YHD-WERKS <> S_WERKST-LOW.
    MESSAGE S024(ZCB) WITH LW_YHD-WERKS S_WERKST-LOW DISPLAY LIKE 'E'.
    LEAVE LIST-PROCESSING.
  ENDIF.
  ZMMRP0060_SCREEN-FHZG = LW_YHD-ABLAD.
* 锁表
  CALL FUNCTION 'ENQUEUE_EZ_ZSDIF0050_H'
    EXPORTING
      DOC_NUMBER                = S_DOCNO-LOW
    EXCEPTIONS
      FOREIGN_LOCK              = 1
      SYSTEM_FAILURE            = 2
      OTHERS                    = 3
             .
  IF SY-SUBRC <> 0.
    CASE SY-SUBRC.
      WHEN 1.
        L_USR = SY-MSGV1.
        MESSAGE S012(ZCB) WITH L_USR S_DOCNO-LOW .
      WHEN OTHERS.
        MESSAGE ID SY-MSGID
              TYPE 'S'
            NUMBER SY-MSGNO
              WITH SY-MSGV1
                   SY-MSGV2
                   SY-MSGV3
                   SY-MSGV4
           DISPLAY LIKE 'E'
          .
    ENDCASE.
    LEAVE LIST-PROCESSING.
  ENDIF.

* 取配货单
  SELECT *
    FROM ZMM_DELEVRESULT
    INTO TABLE GT_PHD
   WHERE DOC_NUMBER IN S_DOCNO
    .
ENDFORM.

*&---------------------------------------------------------------------*
*&设定屏幕初始值
*&---------------------------------------------------------------------*
FORM FM_INIT_SCREEN.
  ZMMRP0060_SCREEN-WERKS  = S_WERKS-LOW.
  ZMMRP0060_SCREEN-LGORT  = S_LGORT-LOW.
  ZMMRP0060_SCREEN-WERKST = S_WERKST-LOW.
  ZMMRP0060_SCREEN-LGORTT = S_LGORTT-LOW.
  ZMMRP0060_SCREEN-YHDH   = S_DOCNO-LOW.
  ZMMRP0060_SCREEN-BLDAT  = S_BLDAT-LOW.
  ZMMRP0060_SCREEN-BUDAT  = S_BUDAT-LOW.
  ZMMRP0060_SCREEN-EKGRP  = '110'.
  G_POSITION = 0.
* 设定光标位置
  G_CURSOR = 'ZMMRP0060_SCREEN-BARCODE'.
* 1：判断发货类型
  DATA:
    L_BUKRS1 TYPE BUKRS,
    L_BUKRS2 TYPE BUKRS.
* 取发货地点对应的公司
  SELECT SINGLE BUKRS
    INTO L_BUKRS1
    FROM T001K AS A
    INNER JOIN T001W AS B
      ON A~BWKEY = B~BWKEY
   WHERE B~WERKS = S_WERKS-LOW
    .
* 取收货地点对应的公司
  SELECT SINGLE BUKRS
    INTO L_BUKRS2
    FROM T001K AS A
    INNER JOIN T001W AS B
      ON A~BWKEY = B~BWKEY
   WHERE B~WERKS = S_WERKST-LOW
    .
  IF L_BUKRS1 = L_BUKRS2.
    G_BSART = 'ZSUB'.
  ELSE.
    G_BSART = 'ZSTO'.
  ENDIF.

* 2.取采购组织
  SELECT SINGLE EKORG
    INTO G_EKORG                      "采购组织
    FROM T001W
   WHERE WERKS = S_WERKST-LOW
      .
ENDFORM.
