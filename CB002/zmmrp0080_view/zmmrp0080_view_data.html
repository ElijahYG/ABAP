FORM GET_DATA.

  CLEAR:IT_TABLE[].

  SELECT *
    INTO CORRESPONDING FIELDS OF TABLE IT_TABLE
    FROM ZMMRP0080_GJSXX
    WHERE ZJSZBM            IN S_ZJSZBM.

  LOOP AT IT_TABLE INTO WA_TABLE.
    WA_TABLE-REC = SY-TABIX.
    MODIFY  IT_TABLE FROM WA_TABLE TRANSPORTING REC .
  ENDLOOP.
ENDFORM.



FORM DEL_DATA.
  IF S_ZJSZBM IS NOT INITIAL.
    DELETE IT_TABLE
        WHERE ZJSZBM            IN S_ZJSZBM.
    DELETE FROM ZMMRP0080_GJSXX
       WHERE ZJSZBM            IN S_ZJSZBM.
    IF SY-SUBRC = 0.
      MESSAGE '删除成功!' TYPE 'S'.
    ENDIF.

  ELSE.

    L_ERRTXT = '请输入检测站编码'.
    MESSAGE L_ERRTXT TYPE 'E'.
  ENDIF.


ENDFORM.
