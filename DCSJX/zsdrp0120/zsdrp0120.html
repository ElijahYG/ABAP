  DEFINE DF_ADD_BWART.

  S_BWART-OPTION = 'EQ'.
  S_BWART-LOW =  &1.
  APPEND S_BWART.

  DF_ADD_BWART:
  '101','102'
  ,'201','202','301',
  '302','251','252','671','672'
  .

* 获取凭证
  SELECT A~BUDAT
         B~MBLNR
         B~MJAHR
         B~ZEILE
         B~MATNR
         B~WERKS
         B~BWART
         B~SHKZG
         B~DMBTR
         B~MENGE
         B~MEINS
         B~CHARG
         B~EBELN
         B~EBELP
      INTO  TABLE GT_MSEGPF
      FROM MKPF AS A
      JOIN MSEG AS B ON  A~MBLNR = B~MBLNR  AND A~MJAHR = B~MJAHR
      WHERE A~BUDAT IN S_BUDAT
        AND B~WERKS IN S_WERKS
        AND B~BWART IN S_BWART.

  SORT GT_MSEGPF BY MBLNR MJAHR ZEILE WERKS .

  DELETE GT_MSEGPF WHERE MATNR  =  ''.

  IF GT_MSEGPF IS NOT INITIAL .
    SELECT EBELN ABSGR INTO TABLE GT_EKKO
       FROM EKKO
       FOR ALL ENTRIES IN GT_MSEGPF
       WHERE EBELN = GT_MSEGPF-EBELN.
  ENDIF.

  SELECT * INTO TABLE GT_T001W
      FROM T001W
      WHERE WERKS IN S_WERKS.


* 过滤主键

  LT_MSEGPF  = GT_MSEGPF.
*
  SORT LT_MSEGPF BY WERKS.
  DELETE ADJACENT DUPLICATES FROM LT_MSEGPF COMPARING WERKS.
  LOOP AT LT_MSEGPF INTO LS_MSEGPF.
    GS_KEY-WERKS = LS_MSEGPF-WERKS.
    APPEND GS_KEY TO GT_KEY.
  ENDLOOP.

*获取批次特性 金重
  LT_MSEGPF  = GT_MSEGPF.
  CLEAR: LT_CHARG[],LT_CHARACTERS.
  SORT LT_MSEGPF BY MATNR.
  DELETE ADJACENT DUPLICATES FROM LT_MSEGPF COMPARING MATNR.

  LOOP AT LT_MSEGPF INTO LS_MSEGPF.
    GT_CHARG-MATNR  = LS_MSEGPF-MATNR.
    GT_CHARG-CHARG  = LS_MSEGPF-CHARG.
    APPEND GT_CHARG.
  ENDLOOP.

  call function 'ZMM_GET_CHARG_CHARACTERS'
    TABLES
      T_CHARG            = GT_CHARG
      T_CHARG_CHARACTERS = GT_CHARACTERS.

  LT_CHARACTERS[] = GT_CHARACTERS[].
  DELETE LT_CHARACTERS[] WHERE ATNAM <> 'ZJZ'.

  SORT GT_EKKO BY EBELN .

  LOOP AT GT_KEY INTO GS_KEY.
    CLEAR:
     GS_OUT_TAB,
     LT_RK[],LT_XS[],LT_TH[],LT_ZC[],LT_BHG[],
     LT_RK,LT_XS,LT_TH,LT_ZC,LT_BHG.

    LT_MSEGPF  = GT_MSEGPF.
    DELETE LT_MSEGPF WHERE WERKS <> GS_KEY-WERKS.

    LOOP AT LT_MSEGPF INTO LS_MSEGPF.
      CLEAR LV_ZJS.
      " S-H   汇总金额
      IF LS_MSEGPF-SHKZG = 'H'.
        LS_MSEGPF-DMBTR =  LS_MSEGPF-DMBTR * -1.
        LS_MSEGPF-MENGE =  LS_MSEGPF-MENGE * -1.
        LV_ZJS = LV_ZJS * -1.
      ENDIF.

*** G： 克重 =  MENGE   件数 =  MENGE / 单品金重
*** PC： 件数 =  MENGE     克重=  MENGE *  单品金重
***
      IF LS_MSEGPF-MEINS = 'G' .
        LS_MSEGPF-MENGE = LS_MSEGPF-MENGE."克重
        READ TABLE LT_CHARACTERS WITH KEY MATNR = LS_MSEGPF-MATNR CHARG = LS_MSEGPF-CHARG.
        IF SY-SUBRC = 0.
          IF LT_CHARACTERS-ATFLV IS NOT INITIAL.
            L_P = LT_CHARACTERS-ATFLV.
            LV_ZJS  =   LS_MSEGPF-MENGE / L_P. " 件数
          ENDIF.
        ENDIF.
      ELSEIF LS_MSEGPF-MEINS = 'ST'.
        LV_ZJS = LS_MSEGPF-MENGE."  件数
        READ TABLE LT_CHARACTERS WITH KEY MATNR = LS_MSEGPF-MATNR CHARG = LS_MSEGPF-CHARG.
        IF SY-SUBRC = 0.
          IF LT_CHARACTERS-ATFLV IS NOT INITIAL.
            L_P = LT_CHARACTERS-ATFLV.
            LS_MSEGPF-MENGE =   LS_MSEGPF-MENGE * L_P. " 克重
          ENDIF.
        ENDIF.
      ENDIF.


      CASE LS_MSEGPF-BWART .
        WHEN  '101' OR '201' OR '301' OR '102' OR '202' OR '302'.
          MOVE-CORRESPONDING LS_MSEGPF TO LT_RK.
          LT_RK-ZJS  = LV_ZJS.
          COLLECT LT_RK.
        WHEN '251'.
          MOVE-CORRESPONDING LS_MSEGPF TO LT_XS.
          LT_XS-ZJS  = LV_ZJS.
          COLLECT  LT_XS.
        WHEN '252'.
          MOVE-CORRESPONDING LS_MSEGPF TO LT_TH.
          LT_TH-ZJS  = LV_ZJS.
          COLLECT LT_TH.
        WHEN '671' OR '672'.
          READ TABLE GT_EKKO  INTO GS_EKKO WITH KEY EBELN = LS_MSEGPF-EBELN BINARY SEARCH.
          IF SY-SUBRC = 0.
            IF GS_EKKO-ABSGR = '10' .
              MOVE-CORRESPONDING LS_MSEGPF TO LT_ZC.
              LT_ZC-ZJS  = LV_ZJS.
              COLLECT LT_ZC.
            ELSEIF  GS_EKKO-ABSGR = '30' .
              MOVE-CORRESPONDING LS_MSEGPF TO LT_BHG.
              LT_BHG-ZJS  = LV_ZJS.
              COLLECT  LT_BHG.
            ENDIF.
          ENDIF.
        WHEN OTHERS.
      ENDCASE.
    ENDLOOP.

    "入库
    READ TABLE  LT_RK INDEX 1.
    IF SY-SUBRC = 0.
      GS_OUT_TAB-RKJE = LT_RK-DMBTR. "入库金额
      GS_OUT_TAB-RKKG = LT_RK-MENGE. "入库克重
      GS_OUT_TAB-RKJS = LT_RK-ZJS. "入库件数
    ENDIF.

    "销售
    READ TABLE  LT_XS INDEX 1.
    IF SY-SUBRC = 0.
      GS_OUT_TAB-XSJE = LT_XS-DMBTR."销售金额
      GS_OUT_TAB-XSKG = LT_XS-MENGE. "销售克重
      GS_OUT_TAB-XSJS = LT_XS-ZJS. "销售件数
    ENDIF.

    "退货
    READ TABLE  LT_TH INDEX 1.
    IF SY-SUBRC = 0.
      GS_OUT_TAB-THJE = LT_TH-DMBTR."退货金额
      GS_OUT_TAB-THKG = LT_TH-MENGE. "退货克重
      GS_OUT_TAB-THJS = LT_TH-ZJS. "退货件数
    ENDIF.

    "不合格返库
    READ TABLE  LT_BHG INDEX 1.
    IF SY-SUBRC = 0.
      GS_OUT_TAB-BHGFKJE = LT_BHG-DMBTR."退货金额
      GS_OUT_TAB-BHGFKKG = LT_BHG-MENGE. "退货克重
      GS_OUT_TAB-BHGFKJS = LT_BHG-ZJS. "退货件数
    ENDIF.

    "正常返库
    READ TABLE  LT_ZC INDEX 1.
    IF SY-SUBRC = 0.
      GS_OUT_TAB-ZCFKJE = LT_ZC-DMBTR."退货金额
      GS_OUT_TAB-ZCFKKG = LT_ZC-MENGE. "退货克重
      GS_OUT_TAB-ZCFKJS = LT_ZC-ZJS. "退货件数
    ENDIF.

    GS_OUT_TAB-WERKS = GS_KEY-WERKS.
    CLEAR GS_T001W.
    READ TABLE GT_T001W INTO GS_T001W WITH KEY WERKS = GS_KEY-WERKS.
    IF SY-SUBRC = 0.
      GS_OUT_TAB-NAME1 = GS_T001W-NAME1.
    ENDIF.

    APPEND GS_OUT_TAB TO GT_OUT_TAB.

  ENDLOOP.


ENDFORM.

FORM FM_GET_POS .
  RANGES  LT_KUNAG FOR VBRK-KUNAG.
  DATA: LS_KUNAG LIKE LINE OF LT_KUNAG.
  DATA: LT_T001W TYPE TABLE OF T001W,
        LS_T001W TYPE T001W.
  DATA: LV_KUNAG TYPE T001W-KUNNR.

  SELECT * INTO TABLE LT_T001W
     FROM T001W
     WHERE WERKS IN S_WERKS.

  LOOP AT LT_T001W INTO LS_T001W.
    CLEAR LV_KUNAG.
    CALL FUNCTION 'CONVERSION_EXIT_ALPHA_INPUT'
      EXPORTING
        INPUT  = LS_T001W-WERKS
      IMPORTING
        OUTPUT = LV_KUNAG.

    LS_KUNAG-LOW = LV_KUNAG.
    LS_KUNAG-OPTION = 'EQ'.
    LS_KUNAG-SIGN = 'I'.
    APPEND LS_KUNAG TO LT_KUNAG.
  ENDLOOP.

  SELECT * INTO TABLE GT_VBRK
     FROM VBRK
     WHERE FKART = 'FP'
       AND FKDAT IN S_BUDAT
       AND KUNAG IN LT_KUNAG.
  IF GT_VBRK IS NOT INITIAL.
    SELECT * INTO TABLE GT_VBRP
       FROM VBRP
       FOR ALL ENTRIES IN GT_VBRK
       WHERE VBELN = GT_VBRK-VBELN.
  ENDIF.

  SORT GT_VBRK BY VBELN.
  SORT GT_VBRP BY VBELN POSNR.
  LOOP AT GT_VBRP INTO GS_VBRP.
    CLEAR GS_VBRK.
    READ TABLE GT_VBRK INTO GS_VBRK WITH KEY VBELN = GS_VBRP-VBELN.
    IF SY-SUBRC = 0.
      MOVE-CORRESPONDING   GS_VBRK TO GS_ALV.
      MOVE-CORRESPONDING   GS_VBRP TO GS_ALV.

      CALL FUNCTION 'CONVERSION_EXIT_ALPHA_OUTPUT'
        EXPORTING
          INPUT  = GS_ALV-KUNAG
        IMPORTING
          OUTPUT = GS_ALV-KUNAG.

      GS_ALV-VRKME = 'PC'.
      GS_ALV-VRKME2 = 'G'.
      APPEND GS_ALV TO GT_ALV.
    ENDIF.
    CLEAR GS_VBRP.
  ENDLOOP.

  PERFORM FM_ADD_CHARG_CHARCTERS.
*********MODIFY BY LIQUAN  2015.12.09   END *********
  SORT GT_ALV BY KUNAG.
ENDFORM.
