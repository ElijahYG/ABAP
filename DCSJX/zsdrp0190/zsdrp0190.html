FORM FM_GET_DATA_MX.
  DATA: LT_T001W TYPE TABLE OF T001W,
        LS_T001W TYPE T001W.
  DATA: LV_WERKS TYPE T001W-WERKS.
  DATA: LT_ALV_HZ TYPE TABLE OF TY_ALV_HZ,
        LS_ALV_HZ TYPE TY_ALV_HZ.

  RANGES S_WERKS FOR T001W-WERKS OCCURS 0.
  DATA: BEGIN OF LT_KEY OCCURS 0,
          KUNAG     TYPE ZSDIF0101_XFJL-KUNAG,
          EX_POSTYP TYPE ZSDIF0101_XFJL-EX_POSTYP,
          ZSPKH     TYPE ZSDIF0101_XFJL-ZSPKH,
          NAME1     TYPE T001W-NAME1,
        END OF  LT_KEY.

  IF P_ZJSKH IS INITIAL.
    SELECT * INTO TABLE GT_XFJL
       FROM   ZSDIF0101_XFJL
       WHERE  KUNAG IN S_KUNAG
         AND  FKDAT IN S_FKDAT
         AND  ZSPKH IN S_ZSPKH.
*      AND  ZJSKH EQ P_ZJSKH.
  ELSE.
    SELECT * INTO TABLE GT_XFJL
       FROM   ZSDIF0101_XFJL
       WHERE  KUNAG IN S_KUNAG
         AND  FKDAT IN S_FKDAT
         AND  ZSPKH IN S_ZSPKH
         AND  ZJSKH EQ P_ZJSKH.
  ENDIF.


  IF  GT_XFJL IS NOT INITIAL.
    SELECT * INTO TABLE GT_JSK
       FROM  ZSDIF0101_JSK
       FOR ALL ENTRIES IN GT_XFJL
       WHERE KUNAG  EQ GT_XFJL-KUNAG
         AND  ZSPKH EQ GT_XFJL-ZSPKH
         AND  ZJSKH EQ GT_XFJL-ZJSKH.

  ENDIF.

  LOOP AT GT_XFJL INTO GS_XFJL.
    CALL FUNCTION 'CONVERSION_EXIT_ALPHA_OUTPUT'
      EXPORTING
        INPUT  = GS_XFJL-KUNAG
      IMPORTING
        OUTPUT = S_WERKS-LOW.
    S_WERKS-SIGN = 'I'.
    S_WERKS-OPTION = 'EQ'.
    APPEND S_WERKS.

  ENDLOOP.

  IF S_WERKS IS NOT INITIAL.

    SELECT * INTO TABLE LT_T001W
         FROM T001W
         WHERE WERKS IN S_WERKS.
  ENDIF.




  LOOP AT GT_XFJL INTO GS_XFJL.
    MOVE-CORRESPONDING GS_XFJL TO GS_ALV_MX.
    CLEAR GS_JSK.
    READ TABLE GT_JSK INTO GS_JSK WITH KEY kunag = GS_XFJL-KUNAG
                                          ZSPKH  = GS_XFJL-ZSPKH
                                          ZJSKH  = GS_XFJL-ZJSKH.
    IF SY-SUBRC = 0.
      GS_ALV_MX-ZSPKH = GS_JSK-ZSPKH."旧饰商品号
      GS_ALV_MX-ZSPJH = GS_JSK-ZSPJH."收重重量
      GS_ALV_MX-ZXSJE = GS_JSK-ZXSJE."收重金额
      IF GS_XFJL-EX_POSTYP = '20'.
        GS_ALV_MX-ZSPJH = GS_ALV_MX-ZSPJH * -1.
        GS_ALV_MX-ZXSJE = GS_ALV_MX-ZXSJE * -1.
        GS_ALV_MX-ERFMG = GS_ALV_MX-ERFMG * -1.
        GS_ALV_MX-EXBWR = GS_ALV_MX-EXBWR * -1.
      ENDIF.
    ENDIF.
    CLEAR LS_T001W.
    CALL FUNCTION 'CONVERSION_EXIT_ALPHA_OUTPUT'
      EXPORTING
        INPUT  = GS_ALV_MX-kunag
      IMPORTING
        OUTPUT = LV_WERKS.

    READ TABLE LT_T001W INTO LS_T001W WITH KEY WERKS =   LV_WERKS.
    IF SY-SUBRC = 0.
      GS_ALV_MX-NAME1 = LS_T001W-NAME1.
    ENDIF.

    MOVE-CORRESPONDING GS_ALV_MX TO LT_KEY.
    MOVE-CORRESPONDING GS_ALV_MX TO LS_ALV_HZ.
    APPEND LT_KEY.
    APPEND LS_ALV_HZ TO LT_ALV_HZ.

    CALL FUNCTION 'CONVERSION_EXIT_ALPHA_OUTPUT'
      EXPORTING
        INPUT  = GS_ALV_MX-kunag
      IMPORTING
        OUTPUT = GS_ALV_MX-kunag.

    APPEND GS_ALV_MX TO GT_ALV_MX.
    CLEAR GS_ALV_MX.
  ENDLOOP.

  IF r2 eq 'X'.
    DATA: LV_ZSPJH TYPE ZSDIF0101_JSK-ZSPJH,
          LV_ZXSJE TYPE ZSDIF0101_JSK-ZXSJE,
          LV_ERFMG TYPE ZSDIF0101_XFJL-ERFMG,
          LV_EXBWR TYPE ZSDIF0101_XFJL-EXBWR.

    SORT LT_KEY BY  KUNAG EX_POSTYP ZSPKH.
    DELETE ADJACENT DUPLICATES FROM LT_KEY COMPARING ALL FIELDS.

    LOOP AT LT_KEY.
      CLEAR : LV_ZSPJH,LV_ZXSJE,LV_ERFMG,LV_EXBWR.

      LOOP AT LT_ALV_HZ INTO LS_ALV_HZ WHERE kunag = LT_KEY-KUNAG
                                          AND EX_POSTYP = LT_KEY-EX_POSTYP
                                          AND ZSPKH = LT_KEY-ZSPKH.

        LV_ZSPJH = LV_ZSPJH +  LS_ALV_HZ-ZSPJH.
        LV_ZXSJE = LV_ZXSJE +  LS_ALV_HZ-ZXSJE.
        LV_ERFMG = LV_ERFMG +  LS_ALV_HZ-ERFMG.
        LV_EXBWR = LV_EXBWR +  LS_ALV_HZ-EXBWR.

      ENDLOOP.
      MOVE-CORRESPONDING LT_KEY  TO GS_ALV_HZ.
      GS_ALV_HZ-DATAB = S_FKDAT-LOW.
      GS_ALV_HZ-DATBI = S_FKDAT-HIGH.
      GS_ALV_HZ-ZSPJH = LV_ZSPJH.
      GS_ALV_HZ-ZXSJE = LV_ZXSJE.
      GS_ALV_HZ-ERFMG = LV_ERFMG.
      GS_ALV_HZ-EXBWR = LV_EXBWR.
      GS_ALV_HZ-WDHZL =  GS_ALV_HZ-ZSPJH - GS_ALV_HZ-ERFMG.
      GS_ALV_HZ-WDHJE =  GS_ALV_HZ-ZXSJE - GS_ALV_HZ-EXBWR.
      CALL FUNCTION 'CONVERSION_EXIT_ALPHA_OUTPUT'
        EXPORTING
          INPUT  = GS_ALV_HZ-kunag
        IMPORTING
          OUTPUT = GS_ALV_HZ-kunag.

      APPEND GS_ALV_HZ TO GT_ALV_HZ.
    ENDLOOP.
  ENDIF.


ENDFORM.
