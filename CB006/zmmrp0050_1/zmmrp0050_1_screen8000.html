MODULE USER_COMMAND_8000 INPUT.
  DATA:
    L_CODE LIKE OK_CODE.

  L_CODE = OK_CODE.
  CLEAR:
    OK_CODE.

  CASE L_CODE.
    WHEN '&F03'.
      LEAVE TO SCREEN 0.
    WHEN '&F15'.
      LEAVE PROGRAM.
    WHEN 'SRH'.
      PERFORM FM_SEARCH.
    WHEN 'DIS1'.
*     是否显示
      IF G_DISPLAY = 'X'.
        G_DISPLAY = ''.   "不显示
        G_DYNNO = '9000'.
        G_SUB   = '9000'.
      ELSE.
        G_DISPLAY = 'X'.  "显示
        G_DYNNO = '8100'.
        G_SUB   = '8003'.
      ENDIF.
    WHEN '&SAVE'.
*     检查
      CLEAR G_ERROR.
      PERFORM FM_CHECK.
*     过账
      IF G_ERROR IS INITIAL.
        PERFORM FM_POST.
      ENDIF.
    WHEN 'CHK'.
      PERFORM FM_CHECK.
  ENDCASE.
ENDMODULE.

*&---------------------------------------------------------------------*
FORM FM_USER_COMMAND USING I_TYPE I_UCOMM TYPE SY-UCOMM.
  DATA:
    L_ZEILE     TYPE MBLPO,
    LT_OUT_TAB1 TYPE TYT_OUT_TAB1,
    LT_OUT_TAB2 TYPE TYT_OUT_TAB2,
    LW_OUT_TAB1 TYPE TY_OUT_TAB1,
    LW_OUT_TAB2 TYPE TY_OUT_TAB2,
    L_PATH      TYPE STRING,
    LT_ROWS     TYPE LVC_T_ROW,
    LW_ROWS     TYPE LVC_S_ROW.
  CASE I_UCOMM.
    WHEN 'SDEL'.
*     获取选中行
      CALL METHOD GO_ALV_SRH->GET_SELECTED_ROWS
        IMPORTING
           ET_INDEX_ROWS = LT_ROWS
          .
      IF LT_ROWS IS INITIAL.
        MESSAGE S001(00) WITH '没有选中任何数据' DISPLAY LIKE 'E'.
        EXIT.
      ENDIF.
*     删除
      SORT LT_ROWS BY INDEX DESCENDING.
      LOOP AT LT_ROWS INTO LW_ROWS.
        DELETE GT_OUT_TAB1 INDEX LW_ROWS-INDEX.
      ENDLOOP.
    WHEN 'RDEL'.
*     获取选中行
      CALL METHOD GO_ALV_RST->GET_SELECTED_ROWS
        IMPORTING
           ET_INDEX_ROWS = LT_ROWS
          .
      IF LT_ROWS IS INITIAL.
        MESSAGE S001(00) WITH '没有选中任何数据' DISPLAY LIKE 'E'.
        EXIT.
      ENDIF.
*     删除
      SORT LT_ROWS BY INDEX DESCENDING.
      LOOP AT LT_ROWS INTO LW_ROWS.
        DELETE GT_OUT_TAB2 INDEX LW_ROWS-INDEX.
      ENDLOOP.
    WHEN 'SADD'.
*     获取选中行
      CALL METHOD GO_ALV_SRH->GET_SELECTED_ROWS
        IMPORTING
           ET_INDEX_ROWS = LT_ROWS
          .
      IF LT_ROWS IS INITIAL.
        MESSAGE S001(00) WITH '没有选中任何数据' DISPLAY LIKE 'E'.
        EXIT.
      ENDIF.
*     追加到商品清单
      SORT LT_ROWS BY INDEX DESCENDING.
      CLEAR G_ERROR.
      LOOP AT LT_ROWS INTO LW_ROWS.
        CLEAR : LW_OUT_TAB1 ,LW_OUT_TAB2.
        READ TABLE GT_OUT_TAB1 INDEX LW_ROWS-INDEX INTO LW_OUT_TAB1.
        APPEND LW_OUT_TAB1 TO LT_OUT_TAB1.
      ENDLOOP.
*     行项目检查
      IF LT_OUT_TAB1 IS NOT INITIAL.
        PERFORM FM_CHECK_ITEM USING LT_OUT_TAB1.
        CHECK G_ERROR IS INITIAL.
      ENDIF.

      LOOP AT LT_ROWS INTO LW_ROWS.
        CLEAR : LW_OUT_TAB1 ,LW_OUT_TAB2.
        READ TABLE GT_OUT_TAB1 INDEX LW_ROWS-INDEX INTO LW_OUT_TAB1.
        DELETE GT_OUT_TAB1 INDEX LW_ROWS-INDEX.
        MOVE-CORRESPONDING LW_OUT_TAB1 TO LW_OUT_TAB2.
        APPEND LW_OUT_TAB2 TO GT_OUT_TAB2.
      ENDLOOP.
  ENDCASE.

* 刷新内表到屏幕
  PERFORM FM_REFRESH_TABLE          .

ENDFORM.

  IF SY-SUBRC = 0.
**   重新计算克重

FORM FM_CHECK .
  DATA:
    L_EXTWG     TYPE EXTWG,
    L_WERKS     TYPE WERKS_D,
    L_LINENO    TYPE I,
    L_TABIX     TYPE CHAR10,
    L_P         TYPE P LENGTH 10 DECIMALS 3,
    LW_OUT_TAB2 TYPE TY_OUT_TAB2,
    LW_MSG      TYPE  ESP1_MESSAGE_WA_TYPE,
    LT_MSG      TYPE  ESP1_MESSAGE_TAB_TYPE.

  DEFINE DF_ADD_MSG .
    CLEAR LW_MSG.
    LW_MSG-MSGID  = 'ZCB'.
    LW_MSG-MSGTY  = 'E'.
    LW_MSG-MSGNO  = &1.
    L_LINENO  = L_LINENO + 1.
    LW_MSG-LINENO = L_LINENO.
    LW_MSG-MSGV1  = &2.
    LW_MSG-MSGV2  = &3.
    APPEND LW_MSG TO LT_MSG.
  END-OF-DEFINITION.
** 检查收货专柜
*  IF ZMMRP0050_SCREEN-FHZG IS INITIAL.
*    DF_ADD_MSG '016' '' ''.
*  ENDIF.
* 关闭要货单的检查
  IF ZMMRP0050_SCREEN-GBYHD = 'X'.
    IF ZMMRP0050_SCREEN-YHDH IS INITIAL.
      DF_ADD_MSG '008' '' ''.
    ENDIF.
*   检查要货单状态
    IF ZMMRP0050_SCREEN-YHDH IS NOT INITIAL.
      SELECT COUNT(*)
        FROM ZSDIF0050_HEAD
       WHERE DOC_NUMBER = ZMMRP0050_SCREEN-YHDH
         AND ORD_TYP <> 'LX01'
         AND DOC_IND = '0'
        .
      IF SY-SUBRC <> 0.
        DF_ADD_MSG '009' ZMMRP0050_SCREEN-YHDH ''.
      ELSE.
        SELECT COUNT(*)
          FROM ZSDIF0050_ITEM
         WHERE DOC_NUMBER = ZMMRP0050_SCREEN-YHDH
           AND ITEM_IND = '0'
          .
        IF SY-SUBRC <> 0.
          DF_ADD_MSG '009' ZMMRP0050_SCREEN-YHDH ''.
        ENDIF.
      ENDIF.
*     要货门店和收货门店一致性检查
      SELECT SINGLE WERKS
        INTO L_WERKS
        FROM ZSDIF0050_HEAD
       WHERE DOC_NUMBER = ZMMRP0050_SCREEN-YHDH
        .
      IF L_WERKS <> ZMMRP0050_SCREEN-WERKST.
        DF_ADD_MSG '024' L_WERKS ZMMRP0050_SCREEN-WERKST.
      ENDIF.
    ENDIF.
  ENDIF.


  IF GT_OUT_TAB2 IS INITIAL.
    DF_ADD_MSG '010' '' ''.
  ENDIF.
* 如果有商品大类不一致，则报错
  LOOP AT GT_OUT_TAB2 INTO LW_OUT_TAB2.
    IF SY-TABIX <> 1.
      IF L_EXTWG <> LW_OUT_TAB2-EXTWG.
        DF_ADD_MSG '027' '' ''.
      ENDIF.
    ENDIF.
    L_EXTWG = LW_OUT_TAB2-EXTWG.
  ENDLOOP.
* 件数必须为整数
  LOOP AT GT_OUT_TAB2 INTO LW_OUT_TAB2.
    L_TABIX = SY-TABIX.
    CONDENSE L_TABIX NO-GAPS.
*   单位为g时，数量/克重必须是单品克重的整数倍
    IF LW_OUT_TAB2-MEINS = 'G'.

          DF_ADD_MSG '028' L_TABIX ''.
        ENDIF.
      ENDIF.
    ENDIF.
*    单位为PC时，数量/克重必须是整数
    IF LW_OUT_TAB2-MEINS = 'ST'.
      L_P = FRAC( LW_OUT_TAB2-MENGE ).
      IF L_P IS NOT INITIAL.
        DF_ADD_MSG '029' L_TABIX ''.
      ENDIF.
    ENDIF.
  ENDLOOP.
* 弹出窗口
  IF LT_MSG IS NOT INITIAL.
    CALL FUNCTION 'C14Z_MESSAGES_SHOW_AS_POPUP'
      TABLES
        I_MESSAGE_TAB       = LT_MSG
              .
    G_ERROR = 'X'.
  ENDIF.

ENDFORM.
*&---------------------------------------------------------------------*
*&      Form  FM_POST
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM FM_POST .
  DATA:
    LT_SELTAB   TYPE RSPARAMS OCCURS 0,
    LW_SELTAB   TYPE RSPARAMS,
    L_POSNR     TYPE POSNR,
    LW_OUT_TAB2 TYPE TY_OUT_TAB2,
    LW_HEAD     TYPE ZMMRP0050_HEAD,
    LW_ITEM     TYPE ZMMRP0050_ITEM,
    LT_ITEM     TYPE ZMMRP0050_ITEM OCCURS 0.
* 关闭要货单
  IF ZMMRP0050_SCREEN-GBYHD = 'X'.
    UPDATE ZSDIF0050_HEAD
       SET DOC_IND = '9'
     WHERE DOC_NUMBER = ZMMRP0050_SCREEN-YHDH
     .
    IF SY-SUBRC <> 0.
      ROLLBACK WORK.
      MESSAGE S001(00) WITH '关闭要货单时失败'.
      EXIT.
    ENDIF.

    UPDATE ZSDIF0050_ITEM
       SET ITEM_IND = '9'
     WHERE DOC_NUMBER = ZMMRP0050_SCREEN-YHDH
     .
    IF SY-SUBRC <> 0.
      ROLLBACK WORK.
      MESSAGE S001(00) WITH '关闭要货单时失败'.
      EXIT.
    ENDIF.
  ENDIF.
* 生成无配货发货单号
  CALL FUNCTION 'NUMBER_GET_NEXT'
    EXPORTING
      NR_RANGE_NR                   = '01'
      OBJECT                        = 'ZMMFH'
    IMPORTING
      NUMBER                        = LW_HEAD-ZFHNO
    EXCEPTIONS
      INTERVAL_NOT_FOUND            = 1
      NUMBER_RANGE_NOT_INTERN       = 2
      OBJECT_NOT_FOUND              = 3
      QUANTITY_IS_0                 = 4
      QUANTITY_IS_NOT_1             = 5
      INTERVAL_OVERFLOW             = 6
      BUFFER_OVERFLOW               = 7
      OTHERS                        = 8
            .
  IF SY-SUBRC <> 0.
    MESSAGE S001(00) WITH '生成凭证号时出错' DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.
* 编辑抬头
  MOVE-CORRESPONDING ZMMRP0050_SCREEN TO LW_HEAD.
  LW_HEAD-REPID = SY-REPID.
  LW_HEAD-ADNAM = SY-UNAME.
  LW_HEAD-ADDAT = SY-DATUM.
  LW_HEAD-ADTIM = SY-UZEIT.
  LW_HEAD-UPNAM = SY-UNAME.
  LW_HEAD-UPDAT = SY-DATUM.
  LW_HEAD-UPTIM = SY-UZEIT.
* 编辑行项目
  CLEAR L_POSNR.
  LOOP AT GT_OUT_TAB2 INTO LW_OUT_TAB2.
    L_POSNR = L_POSNR + 10.
    CLEAR LW_ITEM.
    MOVE-CORRESPONDING LW_OUT_TAB2 TO LW_ITEM.
    LW_ITEM-ZPOSN = L_POSNR.
    LW_ITEM-ZFHNO = LW_HEAD-ZFHNO.
    APPEND LW_ITEM TO LT_ITEM.
  ENDLOOP.
* 更新抬头表
  MODIFY ZMMRP0050_HEAD FROM LW_HEAD.
  IF SY-SUBRC <> 0.
    ROLLBACK WORK.
    MESSAGE S001(00) WITH '保存失败' DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.
* 更新行项目表
  MODIFY ZMMRP0050_ITEM FROM TABLE LT_ITEM.
  IF SY-SUBRC <> 0.
    ROLLBACK WORK.
    MESSAGE S001(00) WITH '保存失败' DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.
  COMMIT WORK AND WAIT.
* 调用过账函数过账
  call function 'ZMMRP0050_POST'
    EXPORTING
      I_ZFHNO       = LW_HEAD-ZFHNO
            .
  MESSAGE S000(ZCB) WITH '订单已保存'.
* 日志显示
  LW_SELTAB-SELNAME      = 'S_ZFHNO'.
  LW_SELTAB-KIND         = 'S'.
  LW_SELTAB-SIGN         = 'I'.
  LW_SELTAB-OPTION       = 'EQ'.
  LW_SELTAB-LOW          = LW_HEAD-ZFHNO.
  APPEND LW_SELTAB TO LT_SELTAB.
  SUBMIT ZMMRP0050_LOG WITH SELECTION-TABLE LT_SELTAB AND RETURN.
  LEAVE TO SCREEN 0.
ENDFORM.
