FORM GET_FKDH.
  SELECT EBELN
         VERKF
  FROM EKKO
  INTO TABLE GT_OUT_TAB
  WHERE AEDAT IN S_YDAY
  AND   BSART = 'ZUBR'.
ENDFORM.



FORM CHECK_FKDH.
  DATA:
    L_USER TYPE SY-MSGV1,
    LT_LOG TYPE CHAR1.

  LOOP AT GT_OUT_TAB INTO GW_OUT_TAB.

    CLEAR:
    GW_EKKO,
    GT_EKPO,
    G_BELNR.


    LT_LOG = '' .

* 取采购抬头
    SELECT SINGLE *
      INTO GW_EKKO
      FROM EKKO
     WHERE VERKF = GW_OUT_TAB-VERKF    "返库单号
      .
    IF SY-SUBRC <> 0.
      LT_LOG = 'X'.
    ENDIF.

* 判断订单类型
    IF GW_EKKO-BSART <> 'ZUBR'
   AND GW_EKKO-BSART <> 'ZSTR'
   AND GW_EKKO-BSART <> 'ZTR1'.
      LT_LOG = 'X'.
    ENDIF.
* 判断是否有交货单
    SELECT SINGLE BELNR
      INTO G_BELNR
      FROM EKBE
     WHERE EBELN = GW_EKKO-EBELN
       AND VGABE = '8'      "8表示交货单
      .
    IF SY-SUBRC <> 0.
      LT_LOG = 'X'.
    ENDIF.
* 判断交货单是否已过账
    SELECT COUNT(*)
      FROM VBUK
     WHERE VBELN = G_BELNR
       AND WBSTK = 'C'
      .
    IF SY-SUBRC = 0.
      LT_LOG = 'X'.
    ENDIF.
* 判断是否收货
    SELECT COUNT(*)
      FROM EKBE
     WHERE EBELN = GW_EKKO-EBELN
       AND VGABE = '1'      "1表示收货
      .
    IF SY-SUBRC <> 0.
      LT_LOG = 'X'.
    ENDIF.

    IF  LT_LOG <> 'X'.
      GW_OUT_TAB-CHECK = 'X'.
      MODIFY GT_OUT_TAB FROM  GW_OUT_TAB .
    ENDIF.

  ENDLOOP.

  DELETE  GT_OUT_TAB WHERE CHECK = 'X'.

ENDFORM.
