FORM FM_CHECK_DATA.
  DATA:
    L_COUNT   TYPE I.    "计数器
* 检查1：限制公司代码、利润中心、利润中心组只能取一个
  IF S_ZRQ IS NOT INITIAL.
    L_COUNT = L_COUNT + 1.
  ENDIF.
  IF S_ZFPH  IS NOT INITIAL.
    L_COUNT = L_COUNT + 1.
  ENDIF.
  IF S_ZJSZBM  IS NOT INITIAL.
    L_COUNT = L_COUNT + 1.
  ENDIF.

* 计数器不等于1表示不为三选一
  IF L_COUNT <> 1.
    MESSAGE S001(00) WITH '日期、分配号、检测单号，只能且必须选择一个' DISPLAY LIKE 'E'.
    LEAVE LIST-PROCESSING.
  ENDIF.
ENDFORM.

FORM DEL_DATA.

   IF  IT_GET_DATA IS NOT INITIAL.
*    DELETE IT_GET_DATA
*        WHERE ZJSZBM            IN S_ZJSZBM.
*    DELETE FROM ZMMDG0010_01
*       WHERE ZJSZBM            IN S_ZJSZBM.
    LOOP AT IT_GET_DATA INTO WA_GET_DATA.
     DELETE FROM ZMMDG0010_01
         WHERE MATNR = WA_GET_DATA-MATNR
           AND CHARG = WA_GET_DATA-CHARG
           AND WERKS = WA_GET_DATA-WERKS
           AND LGORT = WA_GET_DATA-LGORT
           and ZFPH  = WA_GET_DATA-ZFPH .
    ENDLOOP.
    IF SY-SUBRC = 0.
      MESSAGE '删除成功!' TYPE 'S'.
    ENDIF.
  ENDIF.



ENDFORM.
