FORM FRM_GET_DATA .
  IF LX_2 = 'X' .
    SELECT B~MBLNR,
           B~MJAHR,
           B~ZEILE,
           A~BUDAT,
           B~WERKS,
           B~MATNR,
           B~CHARG,
           B~MEINS,
           B~BWART,
           B~SHKZG,
           B~MENGE,
           1 AS COUNTER
           INTO CORRESPONDING FIELDS OF TABLE @GT_ALV
    FROM MKPF AS A INNER JOIN
         MSEG AS B ON A~MBLNR = B~MBLNR AND A~MJAHR = B~MJAHR
    WHERE    A~BUDAT IN @S_BUDAT
     AND     B~WERKS IN @S_WERKS
     AND     B~BWART IN @S_BWART
     AND     B~MATNR IN @S_MATNR
     AND     B~CHARG IN @S_CHARG.


  ELSEIF LX_3 = 'X' .
SELECT B~MBLNR,
           B~MJAHR,
           B~ZEILE,
           A~BUDAT,
           B~WERKS,
           B~MATNR,
           B~CHARG,
           B~MEINS,
           B~BWART,
           B~SHKZG,
           CASE B~SHKZG
           WHEN 'S' THEN '借'
           WHEN 'H' THEN '贷'
           END as SHKZG_T,
           CASE B~SHKZG
           WHEN   'S' THEN B~MENGE
           WHEN   'H' THEN B~MENGE * -1
           END as MENGE,
           CASE
           WHEN  B~SHKZG = 'S' THEN B~MENGE ELSE 0
           END as MENGE_S,
            CASE
           WHEN  B~SHKZG = 'H' THEN B~MENGE ELSE 0
           END as MENGE_H,
           1 AS COUNTER
           INTO CORRESPONDING FIELDS OF TABLE @GT_ALV
    FROM MKPF AS A INNER JOIN
         MSEG AS B ON A~MBLNR = B~MBLNR AND A~MJAHR = B~MJAHR
    WHERE    A~BUDAT IN @S_BUDAT
     AND     B~WERKS IN @S_WERKS
     AND     B~BWART IN @S_BWART
     AND     B~MATNR IN @S_MATNR
     AND     B~CHARG IN @S_CHARG.

  ELSEIF LX_4 = 'X' .
 SELECT G~WERKS, G~MATNR, G~MEINS, G~BWART, G~SHKZG,
           SUM( CASE WHEN G~SHKZG = 'S' THEN G~MENGE
                     WHEN G~SHKZG = 'H' THEN G~MENGE * -1
                     END ) AS MENGE,
           SUM( CASE WHEN G~SHKZG = 'S' THEN G~MENGE
                     ELSE 0
                     END ) AS MENGE_S,
           SUM( CASE WHEN G~SHKZG = 'H' THEN G~MENGE
                     else 0
                     END )  AS MENGE_H
      INTO CORRESPONDING FIELDS OF TABLE @GT_ALV
      FROM MKPF AS F
      INNER JOIN MSEG AS G ON  G~MBLNR = F~MBLNR AND G~MJAHR = F~MJAHR
      WHERE F~BUDAT IN @S_BUDAT
        AND G~WERKS IN @S_WERKS
        AND G~BWART IN @S_BWART
        AND G~MATNR IN @S_MATNR
        AND G~CHARG IN @S_CHARG
     GROUP BY G~WERKS, G~MATNR, G~MEINS, G~BWART, G~SHKZG .

  ELSEIF LX_5 = 'X' .
   SELECT G~WERKS, G~MATNR, G~MEINS, G~BWART,
           CASE G~SHKZG
                WHEN 'S' THEN '借'
                WHEN 'H' THEN '贷'
                END AS SHKZG_T,
           1 AS COUNTER,
           SUM( CASE WHEN G~SHKZG = 'S' THEN G~MENGE
                     WHEN G~SHKZG = 'H' THEN G~MENGE * -1
                     END ) AS MENGE,
           SUM( CASE WHEN G~SHKZG = 'S' THEN G~MENGE
                     END ) AS MENGE_S,
           SUM( CASE WHEN G~SHKZG = 'H' THEN G~MENGE
                     END ) AS MENGE_H
      INTO CORRESPONDING FIELDS OF TABLE @GT_ALV
      FROM MKPF AS F
      INNER JOIN MSEG AS G ON  G~MBLNR = F~MBLNR
                           AND G~MJAHR = F~MJAHR
      WHERE F~BUDAT IN @S_BUDAT
        AND G~WERKS IN @S_WERKS
        AND G~BWART IN @S_BWART
        AND G~MATNR IN @S_MATNR
        AND G~CHARG IN @S_CHARG
     GROUP BY G~WERKS, G~MATNR, G~MEINS, G~BWART, G~SHKZG.
  ENDIF.

ENDFORM.
