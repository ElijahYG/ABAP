FORM FM_CHECK_DATA.
  DATA:

FORM FM_GET_DATA.
  DATA:
    LT_R_PRCTR TYPE RANGE OF PRCTR,
    LT_R_RACCT TYPE RANGE OF HKONT,
    LW_R_RACCT LIKE LINE OF  LT_R_RACCT.

* 利润中心不为空，通过利润中心取数
  IF S_PRCTR IS NOT INITIAL.
    LT_R_PRCTR = S_PRCTR[].
  ENDIF.
* 利润中心组不为空，则取该利润中心组下所有的利润中心
  IF S_SNAME IS NOT INITIAL.
    PERFORM FM_GET_PRCTR USING S_SNAME[] LT_R_PRCTR.
  ENDIF.
  IF LT_R_PRCTR IS INITIAL AND S_BUKRS IS INITIAL.
    MESSAGE S001(00) WITH '这个利润中心组下没有利润中心' DISPLAY LIKE 'E'.
    LEAVE LIST-PROCESSING.
  ENDIF.
*-------------------------------取数-------------------------------------*

  CONCATENATE S_GJAHR-LOW S_MONAT-LOW+1(2)'01' INTO VAL_DATE_B .
  IF S_MONAT-HIGH IS NOT INITIAL .
    CONCATENATE S_GJAHR-LOW S_MONAT-HIGH+1(2)'01' INTO VAL_DATE_E .
  ELSE.
    VAL_DATE_E = VAL_DATE_B .
  ENDIF.

*取本月的最后一天
  CALL FUNCTION 'FIMA_DATE_CREATE'
    EXPORTING
      I_DATE                  = VAL_DATE_E
      I_SET_LAST_DAY_OF_MONTH = 'X'
    IMPORTING
      E_DATE                  = VAL_DATE_E.

  SELECT
    A~MATNR     AS   MATNR                                            "商品编码
    MA~EXTWG    AS   EXTWG                                            "商品大类
    MA~MATKL    AS   MATKL                                            "商品类目
    P~CHARG     AS   CHARG                                            "批次号
    P~KZWI3     AS   KZWI3                                            "销售额
    P~KZWI2     AS   KZWI2                                            "小计2―销售成本
    P~FKLMG     AS   FKLMG                                            "件数/克重
    P~ZXSJS     AS   ZXSJS                                            "销售件数-增强字段-用于销售凭证类型为PF
    P~ZXSKZ     AS   ZXSKZ                                            "销售克重-增强字段-用于销售凭证类型为PF
    P~NETWR     AS   NETWR                                            "净价值-用于判断件数、克重是否为负
    P~VRKME     AS   VRKME                                            "单位
    P~PRCTR     AS   PRCTR                                            "利润中心，用来判断是不是良乡
    K~FKART     AS   FKART                                            "销售凭证类型-用于判断是否取标准件数、克重类型
    A~EX_GJS    AS   EX_GJS
    INTO CORRESPONDING FIELDS OF TABLE GT_VBRP_ELSE
    FROM           VBRP AS P                         INNER JOIN
                   VBRK AS K  ON P~VBELN EQ K~VBELN  INNER JOIN
         ZMMEX0010_MARA AS A  ON P~MATNR EQ A~MATNR  INNER JOIN
                   MARA AS MA ON A~MATNR EQ MA~MATNR
    WHERE K~FKDAT  BETWEEN VAL_DATE_B AND VAL_DATE_E AND
          K~BUKRS  IN S_BUKRS                        AND
          P~PRCTR  IN S_PRCTR                        AND

FORM FM_DOWNLOAD.
  DATA:
    L_TEXT     TYPE CHAR100,
    L_DATE     TYPE SY-DATUM,
    LW_OUT_TAB TYPE TY_OUT_TAB,
    LW_EXCEL   TYPE TY_EXCEL,
    LT_EXCEL   TYPE TYT_EXCEL,
    L_PATH     TYPE STRING.
* 获取路径
  PERFORM FM_GET_FILEPATH USING L_PATH '计提消费税'.

  IF L_PATH IS NOT INITIAL.
*   下载模板到指定路径
    PERFORM FM_GET_TEMPLATE USING CNS_OBJID L_PATH.
*   数据编辑
*   1.编辑抬头
    IF S_MONAT-HIGH IS NOT INITIAL.
      L_DATE = S_GJAHR-LOW && S_MONAT-HIGH+1(2) && '01'.
    ELSE.
      L_DATE = S_GJAHR-LOW && S_MONAT-LOW+1(2) && '01'.
    ENDIF.
*   取当月最后一天
    CALL FUNCTION 'FIMA_DATE_CREATE'
      EXPORTING
        I_DATE                  = L_DATE
        I_SET_LAST_DAY_OF_MONTH = 'X'
      IMPORTING
        E_DATE                  = L_DATE.

    CLEAR LW_EXCEL.
    LW_EXCEL-ROW = 2.
    LW_EXCEL-COL = 1.
    LW_EXCEL-VALUE = L_DATE+0(4) && '年' && L_DATE+4(2) && '月' && L_DATE+6(2) && '日'.
    APPEND LW_EXCEL TO LT_EXCEL.

*   2.编辑编制单位
    PERFORM FM_GET_TEXT USING L_TEXT.
    CLEAR LW_EXCEL.
    LW_EXCEL-ROW = 3.
    LW_EXCEL-COL = 1.
    LW_EXCEL-VALUE = L_TEXT .
    APPEND LW_EXCEL TO LT_EXCEL.

*   3.编辑数据行
    LOOP AT LT_OUT_SYMBOL INTO LW_OUT_SYMBOL.
      IF LW_OUT-XS IS NOT INITIAL.
        CLEAR LW_EXCEL.
        LW_EXCEL-ROW = LW_OUT_SYMBOL-LINENO + 4.
        LW_EXCEL-COL = 2.
        LW_EXCEL-VALUE = LW_OUT_SYMBOL-XS.
        APPEND LW_EXCEL TO LT_EXCEL.
      ENDIF.
    ENDLOOP.

    LOOP AT LT_OUT_SYMBOL INTO LW_OUT_SYMBOL WHERE LINENO BETWEEN 1 AND 6.
      IF LW_OUT-CB IS NOT INITIAL.
        CLEAR LW_EXCEL.
        LW_EXCEL-ROW = LW_OUT_SYMBOL-LINENO + 4.
        LW_EXCEL-COL = 3.
        LW_EXCEL-VALUE = LW_OUT_SYMBOL-CB.
        APPEND LW_EXCEL TO LT_EXCEL.
      ENDIF.

      IF LW_OUT_SYMBOL-JS IS NOT INITIAL.
        CLEAR LW_EXCEL.
        LW_EXCEL-ROW = LW_OUT_SYMBOL-LINENO + 4.
        LW_EXCEL-COL = 4.
        LW_EXCEL-VALUE = LW_OUT_SYMBOL-JS.
        APPEND LW_EXCEL TO LT_EXCEL.
      ENDIF.

      IF LW_OUT_SYMBOL-KZ IS NOT INITIAL.
        CLEAR LW_EXCEL.
        LW_EXCEL-ROW = LW_OUT_SYMBOL-LINENO + 4.
        LW_EXCEL-COL = 5.
        LW_EXCEL-VALUE = LW_OUT_SYMBOL-KZ.
        APPEND LW_EXCEL TO LT_EXCEL.
      ENDIF.
    ENDLOOP.

*   数据填充
    PERFORM FM_FILL_DATA USING L_PATH LT_EXCEL.
  ELSE.
    MESSAGE S001(00) WITH '取消下载'.
    LEAVE LIST-PROCESSING.
  ENDIF.
ENDFORM.
