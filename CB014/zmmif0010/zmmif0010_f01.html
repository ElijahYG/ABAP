FORM GET_STOCK .
  DATA: LV_LINES_ANY TYPE I .

  SELECT B~WERKS B~LGORT B~MATNR B~CHARG A~MEINS B~CLABS
    INTO CORRESPONDING FIELDS OF TABLE GT_STOCK
    FROM MCHB AS B
    INNER JOIN MARA AS A ON A~MATNR = B~MATNR
    FOR ALL ENTRIES IN GT_ZMM_DELEVLGORT
    WHERE B~MATNR IN S_MATNR
      AND B~WERKS = GT_ZMM_DELEVLGORT-WERKS
      AND B~LGORT = GT_ZMM_DELEVLGORT-LGORT
      AND B~CLABS > 0
      AND A~EXTWG = GT_ZMM_DELEVLGORT-EXTWG .
  IF SY-SUBRC <> 0 .
    CLEAR GS_RETURN .
    GS_RETURN-TYPE = 'E' .
    GS_RETURN-ID = 'ZCB' .
    GS_RETURN-NUMBER = '000' .
    GS_RETURN-MESSAGE_V1 = '没有找到需要输出的库存！' .
    GS_RETURN-MESSAGE_V2 = '' .
    GS_RETURN-MESSAGE_V3 = '' .
    GS_RETURN-MESSAGE_V4 = '' .
    APPEND GS_RETURN TO GT_RETURN .
    PERFORM WRITE_MESSAGE . " 输出消息并退出
  ELSE.
    LV_LINES_ANY = LINES( GT_STOCK ) .
    CLEAR GS_RETURN .
    GS_RETURN-TYPE = 'S' .
    GS_RETURN-ID = 'ZCB' .
    GS_RETURN-NUMBER = '000' .
    GS_RETURN-MESSAGE_V1 = '共找到' && LV_LINES_ANY && '行批次库存！' .
    GS_RETURN-MESSAGE_V2 = '' .
    GS_RETURN-MESSAGE_V3 = '' .
    GS_RETURN-MESSAGE_V4 = '' .
    APPEND GS_RETURN TO GT_RETURN .
  ENDIF.
ENDFORM .


FORM PROCESS_STOCK .
  DATA:BEGIN OF LS_ZMMRP0210_BATCH ,
         MATNR   TYPE MCHB-MATNR,
         CHARG   TYPE MCHB-CHARG,
         EX_XNKH TYPE ZMMRP0210_BATCH-EX_XNKH,
       END OF LS_ZMMRP0210_BATCH .
  DATA: LT_ZMMRP0210_BATCH LIKE TABLE OF LS_ZMMRP0210_BATCH.

  DATA:
*       LT_CHARG      LIKE STANDARD TABLE OF ZMM_CHARG,
*       LS_CHARG      LIKE                   ZMM_CHARG,
       LT_CHARACTERS LIKE STANDARD TABLE OF ZMM_CHARACTERS,
       LS_CHARACTERS LIKE                   ZMM_CHARACTERS.

  DATA: LV_LINES_NO_CHARACTER TYPE I , " 因没有批次特性而不下传库存的批次条数
        LV_LINES_NO_ZCT       TYPE I ,       " 因没有抽屉号而不下传库存的批次条数
        LV_LINES_NO_XNKH      TYPE I ,  "因商品批次在虚拟款号表ZMMRP0210_BATCH不存在而不下传库存的批次条数
        LV_LINES_COLLECT      TYPE I,
        LV_LINES_ANY          TYPE I.
  FIELD-SYMBOLS <LS_STOCK> LIKE LINE OF GT_STOCK .

* 取得虚拟款号
  SELECT *
    INTO CORRESPONDING FIELDS OF TABLE LT_ZMMRP0210_BATCH
    FROM ZMMRP0210_BATCH
    FOR ALL ENTRIES IN GT_STOCK
    WHERE MATNR = GT_STOCK-MATNR
      AND CHARG = GT_STOCK-CHARG .

* 取得批次特性


* ---- Changed by huxiaowei @2016.02.08 Start ----
* 如果GT_STOCK中的数据量特别大，一次性Call Function ZSDRP0020_FM_01 会导致资源不足而DUMP，
* 因此改为按每10000条记录Call一次ZSDRP0020_FM_01
*  LOOP AT GT_STOCK ASSIGNING <LS_STOCK> .
*    CLEAR:LS_CHARG.
*    LS_CHARG-MATNR = <LS_STOCK>-MATNR.
*    LS_CHARG-CHARG = <LS_STOCK>-CHARG.
*    APPEND  LS_CHARG TO LT_CHARG.
*  ENDLOOP .
*  CALL FUNCTION 'ZSDRP0020_FM_01'
*    TABLES
*      T_CHARG      = LT_CHARG
*      T_CHARACTERS = LT_CHARACTERS.
  DATA: LV_BEGIN TYPE I,
        LV_END   TYPE I,
        LV_TIMS  TYPE I,
        LV_LINES TYPE I,
        LV_MAX   TYPE I.
  DATA: LT_CHARG_TEMP TYPE TABLE OF ZMM_CHARG ,
        LS_CHARG_TEMP LIKE LINE OF LT_CHARG_TEMP ,
        LT_CHARACTERS_TEMP LIKE LT_CHARACTERS .

  LV_MAX = 10000 .
  LV_LINES = LINES( GT_STOCK ).
  IF LV_LINES MOD LV_MAX = 0 .
    LV_TIMS = LV_LINES DIV LV_MAX . " 总共需要循环多少次
  ELSE.
    LV_TIMS = LV_LINES DIV LV_MAX + 1 . " 总共需要循环多少次
  ENDIF.

  DO LV_TIMS TIMES.
    REFRESH: LT_CHARACTERS_TEMP,LT_CHARG_TEMP .

    LV_BEGIN = LV_MAX * ( SY-INDEX - 1 ) + 1 . " 本次循环从内表GT_STOCK的哪一行开始

    IF SY-INDEX = LV_TIMS . " 最后一次
      LV_END = LV_LINES . " 本次循环从内表GT_STOCK的哪一行结束
    ELSE.
      LV_END = LV_MAX * SY-INDEX . " 本次循环从内表GT_STOCK的哪一行结束
    ENDIF.

    LOOP AT GT_STOCK ASSIGNING <LS_STOCK> FROM LV_BEGIN TO LV_END .
      CLEAR:LS_CHARG_TEMP.
      LS_CHARG_TEMP-MATNR = <LS_STOCK>-MATNR.
      LS_CHARG_TEMP-CHARG = <LS_STOCK>-CHARG.
      APPEND  LS_CHARG_TEMP TO LT_CHARG_TEMP.
    ENDLOOP .

    SORT LT_CHARG_TEMP BY MATNR CHARG .
    DELETE ADJACENT DUPLICATES FROM LT_CHARG_TEMP COMPARING MATNR CHARG .

    call function 'ZSDRP0020_FM_01'
    TABLES
      T_CHARG      = LT_CHARG_TEMP
      T_CHARACTERS = LT_CHARACTERS_TEMP .
    APPEND LINES OF LT_CHARACTERS_TEMP TO LT_CHARACTERS .
  ENDDO.
* ---- Changed by huxiaowei @2016.02.08 End ----


* Added by huxiaowei @20151127 Begin
* 对内表进行排序，在后面Read语句中使用Binary Search，提高效率
  SORT LT_ZMMRP0210_BATCH BY MATNR CHARG .
  SORT LT_CHARACTERS BY MATNR CHARG .
* Added by huxiaowei @20151127 End

  CLEAR: LV_LINES_NO_CHARACTER, LV_LINES_NO_ZCT,LV_LINES_COLLECT .
  LOOP AT GT_STOCK ASSIGNING <LS_STOCK> .
    READ TABLE LT_ZMMRP0210_BATCH INTO LS_ZMMRP0210_BATCH
          WITH KEY MATNR = <LS_STOCK>-MATNR CHARG = <LS_STOCK>-CHARG BINARY SEARCH .
    IF SY-SUBRC <> 0 .
      LV_LINES_NO_XNKH = LV_LINES_NO_XNKH + 1 . " 没有找到批次虚拟款号
    ELSE.
      <LS_STOCK>-EX_XNKH = LS_ZMMRP0210_BATCH-EX_XNKH .
      READ TABLE LT_CHARACTERS INTO LS_CHARACTERS WITH KEY MATNR = <LS_STOCK>-MATNR
                                                           CHARG = <LS_STOCK>-CHARG
                                                           BINARY SEARCH.
      IF SY-SUBRC <> 0   .
        LV_LINES_NO_CHARACTER = LV_LINES_NO_CHARACTER + 1 .
        CONTINUE . " 没有找到批次特性，不用下传到POS
      ELSEIF LS_CHARACTERS-ZCT2 IS INITIAL .
        LV_LINES_NO_ZCT = LV_LINES_NO_ZCT + 1 .
        CONTINUE . " 抽屉号2为空的，说明还未上架完成，不用下传到POS
      ENDIF.
      IF <LS_STOCK>-MEINS = 'ST' .
        <LS_STOCK>-WESCH = <LS_STOCK>-CLABS .   " 件数
        <LS_STOCK>-NTGEW = <LS_STOCK>-CLABS * LS_CHARACTERS-ZJZ . " 克重
      ELSE.
        IF LS_CHARACTERS-ZJZ <= 0 .
          <LS_STOCK>-WESCH = 1 .
        ELSE.
          <LS_STOCK>-WESCH = <LS_STOCK>-CLABS / LS_CHARACTERS-ZJZ .   " 件数
        ENDIF.
        <LS_STOCK>-NTGEW = <LS_STOCK>-CLABS  . " 克重
      ENDIF.
      MOVE-CORRESPONDING <LS_STOCK> TO GS_SGM_INFO .
      COLLECT GS_SGM_INFO INTO GT_SGM_INFO . " 按键值汇总件数和克重
      LV_LINES_COLLECT = LV_LINES_COLLECT + 1 .
    ENDIF.
  ENDLOOP.

  IF LV_LINES_NO_XNKH > 0 .
    CLEAR GS_RETURN .
    GS_RETURN-TYPE = 'S' .
    GS_RETURN-ID = 'ZCB' .
    GS_RETURN-NUMBER = '000' .
    GS_RETURN-MESSAGE_V1 = '共' &&  LV_LINES_NO_XNKH .
    GS_RETURN-MESSAGE_V2 = '条批次库存因没有在表ZMMRP0210_BATCH中维护虚拟款号不能下传！' .
    GS_RETURN-MESSAGE_V3 = '' .
    GS_RETURN-MESSAGE_V4 = '' .
    APPEND GS_RETURN TO GT_RETURN .
  ENDIF.

  IF LV_LINES_NO_CHARACTER > 0 .
    CLEAR GS_RETURN .
    GS_RETURN-TYPE = 'S' .
    GS_RETURN-ID = 'ZCB' .
    GS_RETURN-NUMBER = '000' .
    GS_RETURN-MESSAGE_V1 = '共' &&  LV_LINES_NO_CHARACTER .
    GS_RETURN-MESSAGE_V2 = '条批次库存因没有批次特性不能下传！' .
    GS_RETURN-MESSAGE_V3 = '' .
    GS_RETURN-MESSAGE_V4 = '' .
    APPEND GS_RETURN TO GT_RETURN .
  ENDIF.

  IF LV_LINES_NO_ZCT > 0 .
    CLEAR GS_RETURN .
    GS_RETURN-TYPE = 'S' .
    GS_RETURN-ID = 'ZCB' .
    GS_RETURN-NUMBER = '000' .
    GS_RETURN-MESSAGE_V1 = '共' &&  LV_LINES_NO_ZCT .
    GS_RETURN-MESSAGE_V2 = '条批次库存因没有抽屉号2不能下传！' .
    GS_RETURN-MESSAGE_V3 = '' .
    GS_RETURN-MESSAGE_V4 = '' .
    APPEND GS_RETURN TO GT_RETURN .
  ENDIF.

  IF LV_LINES_COLLECT > 0 .
    LV_LINES_ANY = LINES( GT_SGM_INFO ) .
    CLEAR GS_RETURN .
    GS_RETURN-TYPE = 'S' .
    GS_RETURN-ID = 'ZCB' .
    GS_RETURN-NUMBER = '000' .
    GS_RETURN-MESSAGE_V1 = '共' &&  LV_LINES_COLLECT .
    GS_RETURN-MESSAGE_V2 = '条批次库存按虚拟款号汇总，汇总后共' && LV_LINES_ANY && '条记录！'  .
    GS_RETURN-MESSAGE_V3 = ''.
    GS_RETURN-MESSAGE_V4 = '' .
    APPEND GS_RETURN TO GT_RETURN .
  ENDIF.

  IF GT_SGM_INFO[] IS INITIAL .
    CLEAR GS_RETURN .
    GS_RETURN-TYPE = 'E' .
    GS_RETURN-ID = 'ZCB' .
    GS_RETURN-NUMBER = '000' .
    GS_RETURN-MESSAGE_V1 = '没有需要输出的库存！' .
    GS_RETURN-MESSAGE_V2 = '' .
    GS_RETURN-MESSAGE_V3 = '' .
    GS_RETURN-MESSAGE_V4 = '' .
    APPEND GS_RETURN TO GT_RETURN .
    PERFORM WRITE_MESSAGE ." 输出消息并退出
  ENDIF.
ENDFORM .

FORM CREATE_IDOC .
  DATA: LV_BEGIN TYPE I,
        LV_END   TYPE I,
        LV_TIMS  TYPE I,
        LV_LINES TYPE I.
  DATA: BEGIN OF LS_IDOC_NUM,
          DOCNUM TYPE EDIDD-DOCNUM,
        END OF LS_IDOC_NUM .
  DATA: LT_IDOC_NUM LIKE TABLE OF LS_IDOC_NUM.

*-----------------------------------------
*  IDOC 定义的结构
*-----------------------------------------
  DATA:
    IT_EDIDD   TYPE STANDARD TABLE OF EDIDD,           "IDOC 数据记录
    WA_EDIDD   TYPE EDIDD,                             "IDOC 数据记录
    IT_CONTROL TYPE STANDARD TABLE OF EDIDC,           "IDOC 通讯
    WA_CONTROL TYPE EDIDC.                             "IDOC 控制记录

*  -----------------------------------------
*    定义段结构的内表
*  -----------------------------------------
  DATA:
    LT_SGM_INFO TYPE TABLE OF ZMMIF0010_SGM_INFO,
    LS_SGM_INFO TYPE          ZMMIF0010_SGM_INFO,
    LT_SGM_HEAD TYPE TABLE OF ZMMIF0010_SGM_HEAD,
    LS_SGM_HEAD TYPE          ZMMIF0010_SGM_HEAD.

  LV_LINES = LINES( GT_SGM_INFO ).
  IF LV_LINES MOD GV_MAX = 0 .
    LV_TIMS = LV_LINES DIV GV_MAX . " 总共需要产生多少个IDoc
  ELSE.
    LV_TIMS = LV_LINES DIV GV_MAX + 1 . " 总共需要产生多少个IDoc
  ENDIF.
  CLEAR GS_RETURN .
  GS_RETURN-TYPE = 'S' .
  GS_RETURN-ID = 'ZCB' .
  GS_RETURN-NUMBER = '000' .
  GS_RETURN-MESSAGE_V1 = '本次输出库存PACK_ID：' && GV_PACK_ID .
  GS_RETURN-MESSAGE_V2 = '，应输出' && LV_LINES && '条库存，应产生' && LV_TIMS && '个IDoc！' .
  GS_RETURN-MESSAGE_V3 = '' .
  GS_RETURN-MESSAGE_V4 = '' .
  APPEND GS_RETURN TO GT_RETURN .

  DO LV_TIMS TIMES.
    CLEAR: WA_CONTROL,WA_EDIDD,LS_SGM_HEAD,LS_SGM_INFO,LS_IDOC_NUM .
    REFRESH: IT_EDIDD,IT_CONTROL .

    LV_BEGIN = GV_MAX * ( SY-INDEX - 1 ) + 1 . " 本次循环从内表GT_SGM_INFO的哪一行开始

    IF SY-INDEX = LV_TIMS . " 最后一次
      LV_END = LV_LINES . " 本次循环从内表GT_SGM_INFO的哪一行结束
    ELSE.
      LV_END = GV_MAX * SY-INDEX . " 本次循环从内表GT_SGM_INFO的哪一行结束
    ENDIF.

    WA_CONTROL-RCVPRT = GS_SYS_CONFIG-RCVPRT.        "接收方伙伴类型
    WA_CONTROL-RCVPRN = GS_SYS_CONFIG-RCVPRN.        "接收方的合作伙伴编号
    WA_CONTROL-RCVPOR = GS_SYS_CONFIG-RCVPOR.        "接收者端口(SAP 系统，EDI 子系统)
    WA_CONTROL-MESTYP = 'ZMMIF0010_MSG'.             "信息类型
    WA_CONTROL-IDOCTP = 'ZMMIF0010_IDOC'.            "IDOC 类型

    LS_SGM_HEAD-PACK_ID = GV_PACK_ID.
    LS_SGM_HEAD-TOTAL_PACK = LV_TIMS .
    LS_SGM_HEAD-CURRENT_PACK = SY-INDEX .
    LS_SGM_HEAD-TOTAL_LINE = LV_LINES .
    LS_SGM_HEAD-LINE_FROM = LV_BEGIN .
    LS_SGM_HEAD-LINE_FROM = LV_END .
    WA_EDIDD-SDATA = LS_SGM_HEAD .
    WA_EDIDD-SEGNAM = 'ZMMIF0010_SGM_HEAD'.
    APPEND WA_EDIDD TO IT_EDIDD.

    LOOP AT  GT_SGM_INFO INTO GS_SGM_INFO FROM LV_BEGIN TO LV_END .
      MOVE-CORRESPONDING GS_SGM_INFO TO LS_SGM_INFO.
      CONDENSE LS_SGM_INFO-NTGEW.
      CONDENSE LS_SGM_INFO-WESCH.
      WA_EDIDD-SEGNAM = 'ZMMIF0010_SGM_INFO'.
      WA_EDIDD-SDATA  = LS_SGM_INFO.
      APPEND WA_EDIDD TO IT_EDIDD.
    ENDLOOP.

    CALL FUNCTION 'MASTER_IDOC_DISTRIBUTE'
      EXPORTING
        MASTER_IDOC_CONTROL            = WA_CONTROL
      TABLES
        COMMUNICATION_IDOC_CONTROL     = IT_CONTROL
        MASTER_IDOC_DATA               = IT_EDIDD
      EXCEPTIONS
        ERROR_IN_IDOC_CONTROL          = 1
        ERROR_WRITING_IDOC_STATUS      = 2
        ERROR_IN_IDOC_DATA             = 3
        SENDING_LOGICAL_SYSTEM_UNKNOWN = 4
        OTHERS                         = 5.
    IF SY-SUBRC <> 0.
      CALL FUNCTION 'BAPI_TRANSACTION_ROLLBACK' .

      CLEAR GS_RETURN .
      GS_RETURN-TYPE = 'E' .
      GS_RETURN-ID = 'ZCB' .
      GS_RETURN-NUMBER = '000' .
      GS_RETURN-MESSAGE_V1 = '创建IDoc失败，没有IDoc产生！' .
      GS_RETURN-MESSAGE_V2 = '' .
      GS_RETURN-MESSAGE_V3 = '' .
      GS_RETURN-MESSAGE_V4 = '' .
      APPEND GS_RETURN TO GT_RETURN .

      CLEAR GS_RETURN .
      GS_RETURN-TYPE = 'E' .
      GS_RETURN-ID = SY-MSGID .
      GS_RETURN-NUMBER = SY-MSGNO .
      GS_RETURN-MESSAGE_V1 = SY-MSGV1.
      GS_RETURN-MESSAGE_V2 = SY-MSGV2.
      GS_RETURN-MESSAGE_V3 = SY-MSGV3.
      GS_RETURN-MESSAGE_V4 = SY-MSGV4.
      APPEND GS_RETURN TO GT_RETURN .

      PERFORM WRITE_MESSAGE . " 输出消息并退出
    ELSE.
      READ TABLE IT_CONTROL INTO WA_CONTROL INDEX 1 .
      LS_IDOC_NUM-DOCNUM = WA_CONTROL-DOCNUM .
      APPEND LS_IDOC_NUM TO LT_IDOC_NUM .
    ENDIF.
  ENDDO.

  LOOP AT LT_IDOC_NUM INTO LS_IDOC_NUM .
    CALL FUNCTION 'EDI_DOCUMENT_DEQUEUE_LATER'
      EXPORTING
        DOCNUM                 = LS_IDOC_NUM-DOCNUM
      EXCEPTIONS
        IDOC_IS_NOT_TO_DEQUEUE = 1
        OTHERS                 = 2.

    CLEAR GS_RETURN .
    GS_RETURN-TYPE = 'S' .
    GS_RETURN-ID = 'ZCB' .
    GS_RETURN-NUMBER = '000' .
    GS_RETURN-MESSAGE_V1 = 'IDoc已创建：' && LS_IDOC_NUM-DOCNUM.
    GS_RETURN-MESSAGE_V2 = '' .

FORM WRITE_MESSAGE .
  DATA: LV_STOP   TYPE C LENGTH 1,
        LV_STRING TYPE STRING.

  CLEAR: LV_STOP,LV_STRING .
  LOOP AT GT_RETURN INTO GS_RETURN .
    IF  GS_RETURN-TYPE = 'E' .
      LV_STOP = 'X' .
    ENDIF.
    MESSAGE ID GS_RETURN-ID TYPE GS_RETURN-TYPE NUMBER GS_RETURN-NUMBER
            INTO LV_STRING
            WITH GS_RETURN-MESSAGE_V1 GS_RETURN-MESSAGE_V2 GS_RETURN-MESSAGE_V3 GS_RETURN-MESSAGE_V4 .
    WRITE: /,GS_RETURN-ID, GS_RETURN-NUMBER, GS_RETURN-TYPE, LV_STRING .
  ENDLOOP.

  IF LV_STOP = 'X' .
    STOP .
  ENDIF.
ENDFORM .
