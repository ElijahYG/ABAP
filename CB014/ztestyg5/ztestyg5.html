FORM FM_GET_DATA.
*定义结构
  TYPES: BEGIN OF STRU_MATNR,
           MATNR TYPE ZMMIF0150_PDITEM-MATNR,
           WERKS TYPE ZMMIF0150_PDHEAD-WERKS,
           CHARG TYPE ZMMIF0150_PDITEM-CHARG,
           ZJS   TYPE ZMMIF0150_PDITEM-ZDPJS,
           MEINS TYPE MARA-MEINS,
           TEMPB TYPE MARA-TEMPB,
           ZJZ   TYPE ZDE_ZJZ,
         END OF STRU_MATNR,
         BEGIN OF STRU_MBEWH,
           BWKEY TYPE MBEWH-BWKEY,   "评估范围(地点)
           MATNR TYPE MBEWH-MATNR,   "物料号
           BWTAR TYPE MBEWH-BWTAR,   "评估类型(批次号)
           SALK3 TYPE MBEWH-SALK3,   "成本
           VERPR TYPE MBEWH-VERPR,   "移动平均价
         END OF STRU_MBEWH,
         BEGIN OF STRU_MARA,
           MATNR TYPE MARA-MATNR,
           MEINS TYPE MARA-MEINS,
           TEMPB TYPE MARA-TEMPB,
         END OF STRU_MARA.
*定义内表
  DATA: LT_MARA  TYPE STANDARD TABLE OF STRU_MARA,
        LS_MARA  TYPE                   STRU_MARA,

        LT_MATNR TYPE STANDARD TABLE OF STRU_MATNR,
        LS_MATNR TYPE                   STRU_MATNR,

        LT_0211  TYPE STANDARD TABLE OF STRU_MATNR,
        LS_0211  TYPE                   STRU_MATNR,

        LT_DC80  TYPE STANDARD TABLE OF STRU_MATNR,
        LS_DC80  TYPE                   STRU_MATNR,

        LT_TMP   TYPE STANDARD TABLE OF STRU_MATNR,
        LS_TMP   TYPE                   STRU_MATNR,

        LT_JSFS  TYPE STANDARD TABLE OF ZMMDG0010_JSFS,
        LS_JSFS  TYPE                   ZMMDG0010_JSFS,

        LT_JXSP  TYPE STANDARD TABLE OF STRU_MATNR,
        LS_JXSP  TYPE                   STRU_MATNR,

        LT_MBEW  TYPE STANDARD TABLE OF STRU_MBEWH,
        LS_MBEW  TYPE                   STRU_MBEWH,

        LT_MBEWH TYPE STANDARD TABLE OF STRU_MBEWH,
        LS_MBEWH TYPE                   STRU_MBEWH.

*定义RANGE表
  DATA: LRT_JSFS TYPE RANGE OF ZDE_JSFSBM,
        LRS_JSFS LIKE LINE OF LRT_JSFS,

        LRT_RQ   TYPE RANGE OF SY-DATUM,
        LRS_RQ   LIKE LINE OF LRT_RQ.
*定义变量
  DATA: L_YEAR   TYPE NUMC4.

*---------1.计算日期--------
  LOOP AT S_YMON.
    CLEAR LRS_RQ.
    LRS_RQ-SIGN   = 'I'.
    LRS_RQ-OPTION = 'BT'.
    LRS_RQ-LOW    = S_YMON-LOW && '20'.
    IF S_YMON-LOW+4(2) NE 12.
      S_YMON-LOW    = S_YMON-LOW + 1.
    ELSE.
      L_YEAR        = S_YMON-LOW+0(4) + 1.
      S_YMON-LOW    = L_YEAR && '01'.
    ENDIF.
    LRS_RQ-HIGH   = S_YMON-LOW && '10'.
    APPEND LRS_RQ TO LRT_RQ.
  ENDLOOP.

  SELECT *
    FROM ZMMDG0010_JSFS
    INTO TABLE LT_JSFS
    WHERE SFLY = ABAP_FALSE.

  LOOP AT LT_JSFS INTO LS_JSFS.
    CLEAR LRS_JSFS.
    LRS_JSFS-SIGN   = 'I'.
    LRS_JSFS-OPTION = 'EQ'.
    LRS_JSFS-LOW    = LS_JSFS-BM.
    APPEND LRS_JSFS TO LRT_JSFS.
  ENDLOOP.

*-----------2.取得商品在自定义表中的数据,筛选出经销商品--------------
  IF P_1 EQ ABAP_TRUE OR P_3 EQ ABAP_TRUE.
    SELECT A~MATNR
           A~WERKS
           B~CHARG
           A~ZSL
           C~MEINS
           C~TEMPB
           B~ZJZ
      INTO TABLE LT_0211
      FROM ZFIRP0211 AS A
      INNER JOIN ZMMRP0210_BATCH AS B ON A~ZSPTM = B~ZSPTM
      INNER JOIN MARA AS C ON A~MATNR = C~MATNR
      WHERE B~ZLYJSFS IN LRT_JSFS
        AND A~DJRQ    IN LRT_RQ
        AND A~WERKS   IN S_WERKS
        .
  ENDIF.

  IF P_2 EQ ABAP_TRUE OR P_3 EQ ABAP_TRUE.
    SELECT A~MATNR
           B~WERKS
           A~CHARG
           A~ZDPJS
           A~MEINS
           A~TEMPB
           A~ZJZ
      INTO TABLE LT_DC80
      FROM ZMMIF0150_PDITEM AS A
      INNER JOIN ZMMIF0150_PDHEAD AS B ON A~ZPDBH = B~ZPDBH
      INNER JOIN ZMMRP0210_BATCH AS C ON A~MATNR = C~MATNR AND A~CHARG = C~CHARG
*      INNER JOIN MARA AS D ON A~MATNR = D~MATNR
      WHERE C~ZLYJSFS IN LRT_JSFS
        AND B~ZDJRQ   IN LRT_RQ
        .

  ENDIF.

  IF LT_0211 IS INITIAL AND LT_DC80 IS INITIAL.
    MESSAGE S000(ZCB) WITH '没有找到相应数据' DISPLAY LIKE 'E'.
    LEAVE LIST-PROCESSING.
  ENDIF.

*-----------3.汇总分店和物流商品数据,取得商品物料主数据----------------
  LT_JXSP[] = LT_DC80[].
  APPEND LINES OF LT_0211 TO LT_JXSP.
  SORT LT_JXSP BY MATNR CHARG WERKS.
  DELETE ADJACENT DUPLICATES FROM LT_JXSP COMPARING MATNR CHARG WERKS.
  CLEAR: LT_DC80[],LT_0211[].
  LT_0211[] = LT_JXSP[].
  LT_DC80[] = LT_JXSP[].
  DELETE LT_0211 WHERE CHARG IS NOT INITIAL.
  IF LT_0211 IS NOT INITIAL.
    DELETE LT_JXSP WHERE CHARG IS INITIAL.
  ENDIF.

*-----------4.汇总数据从MBEW表和MBEWH表中取成本----------------
*4.1 取实时成本
  SELECT  BWKEY
          MATNR
          BWTAR
          SALK3
          VERPR
  INTO TABLE LT_MBEW
  FROM MBEW
  FOR ALL ENTRIES IN LT_JXSP
  WHERE  MATNR EQ LT_JXSP-MATNR
    AND  BWTAR EQ LT_JXSP-CHARG
    AND  BWKEY EQ LT_JXSP-WERKS
    .

*4.2 筛选实时成本为空的商品取历史成本
*  DELETE LT_MBEW WHERE SALK3 EQ 0.
  CLEAR LT_TMP[].
  LT_TMP[] = LT_JXSP.
  SORT LT_TMP BY MATNR CHARG WERKS.
  LOOP AT LT_MBEW INTO LS_MBEW.
    CLEAR LS_TMP.
    READ TABLE LT_TMP INTO LS_TMP WITH KEY MATNR = LS_MBEW-MATNR
                                           CHARG = LS_MBEW-BWTAR
                                           WERKS = LS_MBEW-BWKEY
                                           BINARY SEARCH.
    IF SY-SUBRC = 0.
      DELETE LT_TMP INDEX SY-TABIX.
    ENDIF.
  ENDLOOP.

  SELECT  BWKEY
          MATNR
          BWTAR
          SALK3
*          SALK3
*          LBKUM
          VERPR
  INTO TABLE LT_MBEWH
  FROM MBEWH
  FOR ALL ENTRIES IN LT_TMP
  WHERE  MATNR EQ LT_TMP-MATNR
    AND  BWTAR EQ LT_TMP-CHARG
    AND  BWKEY EQ LT_TMP-WERKS
    .

  IF LT_0211 IS NOT INITIAL. "处理没有批次号的商品
    MESSAGE S000(ZCB) WITH '有商品没有批次号' DISPLAY LIKE 'W'.
  ENDIF.

*4.3 整合数据到OUT表.
  SORT LT_JXSP BY MATNR WERKS CHARG.
  SORT LT_MBEW  BY MATNR BWKEY BWTAR SALK3 DESCENDING.
  SORT LT_MBEWH BY MATNR BWKEY BWTAR SALK3 DESCENDING.
  DELETE ADJACENT DUPLICATES FROM LT_MBEW  COMPARING MATNR BWKEY BWTAR.
  DELETE ADJACENT DUPLICATES FROM LT_MBEWH COMPARING MATNR BWKEY BWTAR.

  CLEAR LT_TMP.
  LT_TMP[]  = LT_JXSP[].
  LT_DC80[] = LT_JXSP[].
  DELETE LT_TMP  WHERE TEMPB NE '01'.  "筛选出普通商品
  DELETE LT_DC80 WHERE TEMPB EQ '01'.  "筛选出批次码商品

*4.3.1 汇总一般商品
  LOOP AT LT_TMP INTO LS_TMP.
    CLEAR WA_OUT_TAB.
    WA_OUT_TAB-MATNR = LS_TMP-MATNR.
    WA_OUT_TAB-CHARG = LS_TMP-CHARG.
    WA_OUT_TAB-MEINS = LS_TMP-MEINS.
    WA_OUT_TAB-TEMPB = LS_TMP-TEMPB.
    WA_OUT_TAB-WERKS = LS_TMP-WERKS.
    WA_OUT_TAB-ZSL   = LS_TMP-ZJS.
    CLEAR LS_MBEW.
    READ TABLE LT_MBEW INTO LS_MBEW WITH KEY MATNR = LS_TMP-MATNR
                                             BWKEY = LS_TMP-WERKS
                                             BWTAR = LS_TMP-CHARG
                                             BINARY SEARCH.
    IF SY-SUBRC = 0.
      IF LS_MBEW-SALK3 <> 0.
        WA_OUT_TAB-ZJE   = LS_MBEW-SALK3.
      ENDIF.
      IF  LS_MBEW-SALK3 = 0.
        IF WA_OUT_TAB-MEINS = 'G'.
          WA_OUT_TAB-ZJE   = LS_MBEW-VERPR * LS_TMP-ZJZ..
        ELSEIF WA_OUT_TAB-MEINS = 'ST'.
          WA_OUT_TAB-ZJE   = LS_MBEW-VERPR * WA_OUT_TAB-ZSL.
        ENDIF.
      ENDIF.
    ELSE.
      CLEAR LS_MBEWH.
      READ TABLE LT_MBEWH INTO LS_MBEWH WITH KEY MATNR = LS_TMP-MATNR
                                                 BWKEY = LS_TMP-WERKS
                                                 BWTAR = LS_TMP-CHARG
                                                 BINARY SEARCH.
      IF SY-SUBRC = 0.
        IF LS_MBEWH-SALK3 <> 0.
          WA_OUT_TAB-ZJE   = LS_MBEWH-SALK3.
        ENDIF.
        IF  LS_MBEWH-SALK3 = 0.
          IF WA_OUT_TAB-MEINS = 'G'.
            WA_OUT_TAB-ZJE   = LS_MBEWH-VERPR * LS_TMP-ZJZ..
          ELSEIF WA_OUT_TAB-MEINS = 'ST'.
            WA_OUT_TAB-ZJE   = LS_MBEWH-VERPR * WA_OUT_TAB-ZSL.
          ENDIF.
        ENDIF.
      ENDIF.
    ENDIF.
    APPEND WA_OUT_TAB TO IT_OUT_TAB.
  ENDLOOP.

*4.3.2 汇总批次码商品
  LOOP AT LT_DC80 INTO LS_DC80.
    CLEAR WA_OUT_TAB.
    WA_OUT_TAB-MATNR = LS_DC80-MATNR.
    WA_OUT_TAB-CHARG = LS_DC80-CHARG.
    WA_OUT_TAB-MEINS = LS_DC80-MEINS.
    WA_OUT_TAB-TEMPB = LS_DC80-TEMPB.
    WA_OUT_TAB-WERKS = LS_DC80-WERKS.
    WA_OUT_TAB-ZSL   = LS_DC80-ZJS.
    CLEAR LS_MBEW.
    READ TABLE LT_MBEW INTO LS_MBEW WITH KEY MATNR = LS_DC80-MATNR
                                             BWKEY = LS_DC80-WERKS
                                             BWTAR = LS_DC80-CHARG
                                             BINARY SEARCH.
    IF SY-SUBRC = 0.
      IF WA_OUT_TAB-MEINS = 'G'.
        WA_OUT_TAB-ZJE   = LS_MBEW-VERPR * LS_DC80-ZJS * LS_DC80-ZJZ.
      ELSEIF WA_OUT_TAB-MEINS = 'ST'.
        WA_OUT_TAB-ZJE   = LS_MBEW-VERPR * LS_DC80-ZJS.
      ENDIF.
    ELSE.
      CLEAR LS_MBEWH.
      READ TABLE LT_MBEWH INTO LS_MBEWH WITH KEY MATNR = LS_DC80-MATNR
                                                 BWKEY = LS_DC80-WERKS
                                                 BWTAR = LS_DC80-CHARG
                                                 BINARY SEARCH.
      IF SY-SUBRC = 0.
        IF WA_OUT_TAB-MEINS = 'G'.
          WA_OUT_TAB-ZJE   = LS_MBEWH-VERPR * LS_DC80-ZJS * LS_DC80-ZJZ.
        ELSEIF WA_OUT_TAB-MEINS = 'ST'.
          WA_OUT_TAB-ZJE   = LS_MBEWH-VERPR * LS_DC80-ZJS.
        ENDIF.
      ENDIF.
    ENDIF.
    APPEND WA_OUT_TAB TO IT_OUT_TAB.
  ENDLOOP.

*-----------5.取描述-------------
  DATA: BEGIN OF LS_WERKS,
          WERKS TYPE T001W-WERKS,
          KUNNR TYPE KNA1-KUNNR,
          SORTL TYPE KNA1-SORTL,
        END OF LS_WERKS ,
        LT_WERKS LIKE TABLE OF LS_WERKS
*        LT_KUNNR LIKE TABLE OF LS_WERKS
        .

  MOVE-CORRESPONDING IT_OUT_TAB TO LT_WERKS.

  SORT LT_WERKS BY WERKS.
  DELETE ADJACENT DUPLICATES FROM LT_WERKS COMPARING WERKS.

  FIELD-SYMBOLS <FS_WERKS> LIKE LINE OF LT_WERKS.
  LOOP AT LT_WERKS ASSIGNING <FS_WERKS>.
    <FS_WERKS>-KUNNR = '000000' && <FS_WERKS>-WERKS.
    SELECT SINGLE SORTL FROM KNA1 INTO <FS_WERKS>-SORTL WHERE KUNNR = <FS_WERKS>-KUNNR.
  ENDLOOP.

  FIELD-SYMBOLS <FS_OUT> LIKE LINE OF IT_OUT_TAB.
  SORT LT_WERKS BY WERKS.
  LOOP AT IT_OUT_TAB ASSIGNING <FS_OUT>.
    CLEAR LS_WERKS.
    READ TABLE LT_WERKS INTO LS_WERKS WITH KEY WERKS = <FS_OUT>-WERKS BINARY SEARCH.
    IF SY-SUBRC = 0.
      <FS_OUT>-SORTL = LS_WERKS-SORTL.
    ENDIF.
  ENDLOOP.

ENDFORM.

  ADD_ALV:
  1    'WERKS'   TEXT-001,

  3    'MATNR'   TEXT-005,
  4    'CHARG'   TEXT-006,
  5    'MEINS'   TEXT-007,
  6    'TEMPB'   TEXT-008,
  7    'ZSL'     TEXT-003,
  8    'ZJE'     TEXT-004
  .

  CLEAR WA_LAYOUT.
  WA_LAYOUT-CWIDTH_OPT = ABAP_TRUE.     "优化列宽设置
  WA_LAYOUT-ZEBRA = 'X'.

* 调用ALV
  CALL FUNCTION 'REUSE_ALV_GRID_DISPLAY_LVC'
    EXPORTING
      I_CALLBACK_PROGRAM = SY-REPID

      IT_FIELDCAT_LVC    = IT_FIELDCAT
*      IS_VARIANT         = G_VARIANT
      I_SAVE              = 'A'
    TABLES
      T_OUTTAB           = IT_OUT_TAB
    EXCEPTIONS
      PROGRAM_ERROR      = 1
      OTHERS             = 2.
  IF SY-SUBRC <> 0.
    MESSAGE E001(00) WITH '系统错误' SPACE SPACE SPACE.
  ENDIF.
ENDFORM.
